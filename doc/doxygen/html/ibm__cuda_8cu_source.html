233 __global__ void ForcesIntoFluid_Kernel(const IBM_CUDA_ParticleDataInput *const particle_input, LB_node_force_gpu node_f, const LB_parameters_gpu *const paraP)
234 {
235  const unsigned int particleIndex = blockIdx.y * gridDim.x * blockDim.x + blockDim.x * blockIdx.x + threadIdx.x;
236  const LB_parameters_gpu &para = *paraP;
237 
238  if (particleIndex < para.number_of_particles && particle_input[particleIndex].isVirtual)
239  {
240 
241  const float factor = powf( para.agrid,2)*para.tau*para.tau;
242  const float particleForce[3] = { particle_input[particleIndex].f[0] * factor, particle_input[particleIndex].f[1] * factor, particle_input[particleIndex].f[2] * factor};
243  const float pos[3] = { particle_input[particleIndex].pos[0], particle_input[particleIndex].pos[1], particle_input[particleIndex].pos[2] };
244 
245  // First part is the same as for interpolation --> merge into a single function
246  float temp_delta[6];
247  float delta[8];
248  int my_left[3];
249  int node_index[8];
250  for(int i=0; i<3; ++i)
251  {
252  const float scaledpos = pos[i]/para.agrid - 0.5f;
253  my_left[i] = (int)(floorf(scaledpos));
254  temp_delta[3+i] = scaledpos - my_left[i];
255  temp_delta[i] = 1.f - temp_delta[3+i];
256  }
257 
258  delta[0] = temp_delta[0] * temp_delta[1] * temp_delta[2];
259  delta[1] = temp_delta[3] * temp_delta[1] * temp_delta[2];
260  delta[2] = temp_delta[0] * temp_delta[4] * temp_delta[2];
261  delta[3] = temp_delta[3] * temp_delta[4] * temp_delta[2];
262  delta[4] = temp_delta[0] * temp_delta[1] * temp_delta[5];
263  delta[5] = temp_delta[3] * temp_delta[1] * temp_delta[5];
264  delta[6] = temp_delta[0] * temp_delta[4] * temp_delta[5];
265  delta[7] = temp_delta[3] * temp_delta[4] * temp_delta[5];
266 
267  // modulo for negative numbers is strange at best, shift to make sure we are positive
268  const int x = my_left[0] + para.dim_x;
269  const int y = my_left[1] + para.dim_y;
270  const int z = my_left[2] + para.dim_z;
271 
272  node_index[0] = x%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
273  node_index[1] = (x+1)%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
274  node_index[2] = x%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
275  node_index[3] = (x+1)%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
276  node_index[4] = x%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
277  node_index[5] = (x+1)%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
278  node_index[6] = x%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
279  node_index[7] = (x+1)%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
280 
281  for(int i=0; i<8; ++i)
282  {
283 
284  // Atomic add is essential because this runs in parallel!
285  atomicadd(&(node_f.force[0*para.number_of_nodes + node_index[i]]), (particleForce[0] * delta[i]));
286  atomicadd(&(node_f.force[1*para.number_of_nodes + node_index[i]]), (particleForce[1] * delta[i]));
287  atomicadd(&(node_f.force[2*para.number_of_nodes + node_index[i]]), (particleForce[2] * delta[i]));
288  }
289  }
290 }

321  {
322  const float scaledpos = pos[i]/para.agrid - 0.5f;
323  my_left[i] = (int)(floorf(scaledpos));
324  temp_delta[3+i] = scaledpos - my_left[i];

331  delta[3] = temp_delta[3] * temp_delta[4] * temp_delta[2];
332  delta[4] = temp_delta[0] * temp_delta[1] * temp_delta[5];
333  delta[5] = temp_delta[3] * temp_delta[1] * temp_delta[5];

339  int y = my_left[1] + para.dim_y;
340  int z = my_left[2] + para.dim_z;
341 
342  node_index[0] = x%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
343  node_index[1] = (x+1)%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
344  node_index[2] = x%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
345  node_index[3] = (x+1)%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*(z%para.dim_z);
346  node_index[4] = x%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
347  node_index[5] = (x+1)%para.dim_x + para.dim_x*(y%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
348  node_index[6] = x%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
349  node_index[7] = (x+1)%para.dim_x + para.dim_x*((y+1)%para.dim_y) + para.dim_x*para.dim_y*((z+1)%para.dim_z);
350 
351  for(int i=0; i<8; ++i)
352  {
353  double local_rho;
354  double local_j[3];
355 #ifdef LB_BOUNDARIES_GPU
356  if (n_curr.boundary[node_index[i]])
357  {
358  // Boundary node
359  const int boundary_index= n_curr.boundary[node_index[i]];
360 
361  // lb_boundary_velocity is given in MD units --> convert to LB and reconvert back at the end of this function
362  local_rho = para.rho[0]*para.agrid*para.agrid*para.agrid;
363  local_j[0] = para.rho[0]*para.agrid*para.agrid*para.agrid*lb_boundary_velocity[3*(boundary_index-1)+0];
364  local_j[1] = para.rho[0]*para.agrid*para.agrid*para.agrid*lb_boundary_velocity[3*(boundary_index-1)+1];
365  local_j[2] = para.rho[0]*para.agrid*para.agrid*para.agrid*lb_boundary_velocity[3*(boundary_index-1)+2];
366 
367  }
368  else
369 #endif
370  {
371  Calc_m_from_n_IBM(n_curr,node_index[i],mode, paraP);
372  local_rho = para.rho[0]*para.agrid*para.agrid*para.agrid + mode[0];
373 
374  // Add the +f/2 contribution!!
375  local_j[0] = mode[1] + node_f.force_buf[0*para.number_of_nodes + node_index[i]]/2.f;
376  local_j[1] = mode[2] + node_f.force_buf[1*para.number_of_nodes + node_index[i]]/2.f;
377  local_j[2] = mode[3] + node_f.force_buf[2*para.number_of_nodes + node_index[i]]/2.f;
378 
379  }
380 
381 
382  // Interpolate velocity
383  v[0] += delta[i]*local_j[0]/(local_rho);
384  v[1] += delta[i]*local_j[1]/(local_rho);
385  v[2] += delta[i]*local_j[2]/(local_rho);
386  }
387 
388 
389  // Rescale and store output
390  particles_output[particleIndex].v[0] = v[0] * para.agrid / para.tau;
391  particles_output[particleIndex].v[1] = v[1] * para.agrid / para.tau;
392  particles_output[particleIndex].v[2] = v[2] * para.agrid / para.tau;
393 
394  }
395 }
