44 /** Calculate lj-cos2 force between particle p1 and p2 */
45 inline void add_ljcos2_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
46  double d[3], double dist, double force[3])
47 {
48  int j;
49  double r_off, frac2, frac6, fac=0.0;
50  if(CUTOFF_CHECK(dist < ia_params->LJCOS2_cut+ia_params->LJCOS2_offset)) {
51  r_off = dist - ia_params->LJCOS2_offset;
52  /* normal case: resulting force/energy smaller than capping. */
53  if(r_off > ia_params->LJCOS2_capradius) {
54  if(r_off < ia_params->LJCOS2_rchange) {
55  frac2 = SQR(ia_params->LJCOS2_sig/r_off);
56  frac6 = frac2*frac2*frac2;
57  fac = 48.0 * ia_params->LJCOS2_eps * frac6*(frac6 - 0.5) / (r_off*dist);
58  }
59  else if (r_off< ia_params->LJCOS2_rchange + ia_params->LJCOS2_w) {
60  fac = -ia_params->LJCOS2_eps*M_PI/2/ia_params->LJCOS2_w/dist * sin(M_PI*(r_off-ia_params->LJCOS2_rchange)/ia_params->LJCOS2_w);
61  }
62 
63  for(j=0;j<3;j++)
64  force[j] += fac * d[j];
65 
66 #ifdef LJ_WARN_WHEN_CLOSE
67  if(fac*dist > 1000) fprintf(stderr,"%d: LJ-Warning: Pair (%d-%d) force=%f dist=%f\n",
68  this_node,p1->p.identity,p2->p.identity,fac*dist,dist);
69 #endif
70  }
71  /* capped part of lj-cos2 potential. */
72  else if(dist > 0.0) {
73  frac2 = SQR(ia_params->LJCOS2_sig/ia_params->LJCOS2_capradius);
74  frac6 = frac2*frac2*frac2;
75  fac = 48.0 * ia_params->LJCOS2_eps * frac6*(frac6 - 0.5) / (ia_params->LJCOS2_capradius * dist);
76  for(j=0;j<3;j++)
77  /* vector d is rescaled to length LJCOS2_capradius */
78  force[j] += fac * d[j];
79  }
80  /* this should not happen! */
81  else {
82  LJ_TRACE(fprintf(stderr, "%d: Lennard-Jones warning: Particles id1=%d id2=%d exactly on top of each other\n",this_node,p1->p.identity,p2->p.identity));
83 
84  frac2 = SQR(ia_params->LJCOS2_sig/ia_params->LJCOS2_capradius);
85  frac6 = frac2*frac2*frac2;
86  fac = 48.0 * ia_params->LJCOS2_eps * frac6*(frac6 - 0.5) / ia_params->LJCOS2_capradius;
87 
88  force[0] += fac * ia_params->LJCOS2_capradius;
89  }
90 
91  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: LJ f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist,fac));
92  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: LJ f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist,fac));
93 
94  LJ_TRACE(fprintf(stderr,"%d: LJ: Pair (%d-%d) dist=%.3f: force+-: (%.3e,%.3e,%.3e)\n",
95  this_node,p1->p.identity,p2->p.identity,dist,fac*d[0],fac*d[1],fac*d[2]));
96  }
97 }

99 /** calculate lj-cos2 energy between particle p1 and p2. */
100 inline double ljcos2_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
101  double d[3], double dist)
102 {
103  double r_off, frac2, frac6;
104 
105  if(CUTOFF_CHECK(dist < ia_params->LJCOS2_cut+ia_params->LJCOS2_offset)) {
106  r_off = dist - ia_params->LJCOS2_offset;
107  /* normal case: resulting force/energy smaller than capping. */
108  if(r_off > ia_params->LJCOS2_capradius) {
109  if (r_off < ia_params->LJCOS2_rchange){
110  frac2 = SQR(ia_params->LJCOS2_sig/r_off);
111  frac6 = frac2*frac2*frac2;
112  return 4.0*ia_params->LJCOS2_eps*(SQR(frac6)-frac6);
113  }
114  else if (r_off < ia_params->LJCOS2_rchange + ia_params->LJCOS2_w){
115  return -ia_params->LJCOS2_eps/2 * (cos(M_PI*(r_off-ia_params->LJCOS2_rchange)/ia_params->LJCOS2_w)+1);
116  }
117  }
118  /* capped part of lj-cos2 potential. */
119  else if(dist > 0.0) {
120  frac2 = SQR(ia_params->LJCOS2_sig/ia_params->LJCOS2_capradius);
121  frac6 = frac2*frac2*frac2;
122  return 4.0*ia_params->LJCOS2_eps*(SQR(frac6)-frac6);
123  }
124  /* this should not happen! */
125  else {
126  frac2 = SQR(ia_params->LJCOS2_sig/ia_params->LJCOS2_capradius);
127  frac6 = frac2*frac2*frac2;
128  return 4.0*ia_params->LJCOS2_eps*(SQR(frac6)-frac6);
129  }
130  }
131  return 0.0;
132 }
