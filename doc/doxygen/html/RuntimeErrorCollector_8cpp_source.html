90 list<string> RuntimeErrorCollector::
91 gather() {
92  int numMessages = this->count();
93 
94  // If no processor encountered an error, return
95  if (numMessages == 0) return list<string>();
96 
97  list<string> allerrors = this->errors;
98 
99  // subtract the number of messages on the master, as they are not to be sent
100  numMessages -= this->errors.size();
101 
102  MPI_Status status;
103  int count;
104  for (int i = 0; i < numMessages; ++i) {
105  // get the next message
106  MPI_Probe(MPI_ANY_SOURCE, MPI_ANY_TAG, this->comm, &status);
107  MPI_Get_count(&status, MPI_CHAR, &count);
108 
109  char buffer[count];
110  MPI_Recv(buffer, count, MPI_CHAR,
111  MPI_ANY_SOURCE, MPI_ANY_TAG, this->comm, MPI_STATUS_IGNORE);
112 
113  allerrors.push_back(string());
114  string &s = allerrors.back();
115  s.assign(buffer, count);
116  }
117 
118  this->clear();
119 
120  return allerrors;
121 }
