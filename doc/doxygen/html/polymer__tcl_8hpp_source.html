30 /** Implementation of the tcl-command <br>
31  polymer <N_P> <MPC> <bond_length> [start <part_id>] [pos <x> <y> <z>] [mode { SAW | RW | PSAW } [<shield> [<max_try>]]]
32  [charge <val_cM>] [distance <cM_dist>] [types <type_nM> [<type_cM>]] [FENE <type_FENE>] [angle <angle>
33  [<angle2> [<x2> <y2> <z2>]]] <br>
34  Creates some polymer chains within the simulation box,
35  and returns how often the attempt to place a monomer failed in the worst case.
36  Parameters: <br>
37  <N_P> = how many polymers to create <br>
38  <MPC> = monomers per chain <br>
39  <bond_length> = length of the bonds between two monomers <br>
40  <part_id> = particle number of the start monomer (defaults to '0') <br>
41  <pos> = sets the position of the start monomer of the first chain (defaults to a randomly chosen value) <br>
42  <mode> = selects setup mode: (Pseudo) self avoiding walk ([P]SAW) or plain random walk (RW) (defaults to 'SAW') <br>
43  <shield> = shield around each particle another particle's position may not enter if using SAW (defaults to '0.0') <br>
44  <max_try> = how often a monomer should be reset if current position collides with a previous particle (defaults to '30000') <br>
45  <val_cM> = valency of charged monomers (defaults to '0.0') <br>
46  <cM_dist> = distance between two charged monomers' indices (defaults to '1') <br>
47  <type_{n|c}P> = type number of {neutral|charged} monomers to be used with "part" (default to '0' and '1') <br>
48  <type_FENE> = type number of the FENE-typed bonded interaction bonds to be set between the monomers (defaults to '0') <br>
49  <angle> = freely rotating bond-angle to be fixed <br>
50  <angle2> = second spherical bond-angle (for setting up helixes or planar polymers) <br>
51  <x2,y2,z2> = sets the position of the 2nd monomer of the first chain <br>
52  <constr> = shall constraints be respected when setting up polymer? (0=no, 1=yes, default: 0)
53  <br>For more informations on the parameters see \ref polymerC. */
54 int tclcommand_polymer (ClientData data, Tcl_Interp *interp, int argc, char **argv);

83  <type_{p|n}S> = type numbers to be used with "part" (default to '3' and '4'); if <type_nS> is not given, <type_nS> = <type_pS> is assumed. <br>

111  Evaluates the current configuration and connects each chain's end to a random monomer of another chain at most <r_catch> away,

120  <chain_dist> = same as <link_dist>, but for monomers of the same bond (defaults to <MPC> => no bonds to the same chain allowed) <br>
121  <type_FENE> = type number of the FENE-typed bonded interaction bonds to be set between the monomers (defaults to '0') <br>
122  <max_try> = how often crosslinks should be removed if they are too close to other links (defaults to '30000') <br>
