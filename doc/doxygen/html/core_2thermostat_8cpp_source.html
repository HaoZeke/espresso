79 void thermo_init_langevin()
80 {
81  langevin_pref1 = -langevin_gamma/time_step;
82 #if defined (FLATNOISE)
83  langevin_pref2 = sqrt(24.0*temperature*langevin_gamma/time_step);
84 #elif defined (GAUSSRANDOMCUT) || defined (GAUSSRANDOM)
85  langevin_pref2 = sqrt(2.0*temperature*langevin_gamma/time_step);
86 #else
87 #error No Noise defined
88 #endif
89 #ifdef MULTI_TIMESTEP
90  if (smaller_time_step > 0.) {
91  langevin_pref1_small = -langevin_gamma/smaller_time_step;
92  #ifndef LANGEVIN_PER_PARTICLE
93  langevin_pref2_small = sqrt(24.0*temperature*langevin_gamma/smaller_time_step);
94  #endif
95  } else {
96  langevin_pref1_small = -langevin_gamma/time_step;
97  #ifndef LANGEVIN_PER_PARTICLE
98  langevin_pref2_small = sqrt(24.0*temperature*langevin_gamma/time_step);
99  #endif
100  }
101 #endif
102 
103 #ifdef ROTATION
104  langevin_gamma_rotation = langevin_gamma/3;
105 #if defined (FLATNOISE)
106  langevin_pref2_rotation = sqrt(24.0*temperature*langevin_gamma_rotation/time_step);
107 #elif defined (GAUSSRANDOMCUT) || defined (GAUSSRANDOM)
108  langevin_pref2_rotation = sqrt(2.0*temperature*langevin_gamma_rotation/time_step);
109 #else
110 #error No Noise defined
111 #endif
112  THERMO_TRACE(fprintf(stderr,"%d: thermo_init_langevin: langevin_gamma_rotation=%f, langevin_pref2_rotation=%f",this_node, langevin_gamma_rotation,langevin_pref2_rotation));
113 #endif

168 void thermo_heat_up()
169 {
170  if(thermo_switch & THERMO_LANGEVIN) {
171  langevin_pref2_buffer = langevin_pref2;
172  langevin_pref2_rotation_buffer = langevin_pref2_rotation;
173  langevin_pref2 *= sqrt(3);
174  langevin_pref2_rotation *= sqrt(3);
175 #ifdef MULTI_TIMESTEP
176  langevin_pref2_small_buffer = langevin_pref2_small;
177  langevin_pref2_small *= sqrt(3);
178 #endif
179  }

181  else if (thermo_switch & THERMO_DPD){dpd_heat_up();}

215 void set_cpu_temp(int temp)
216 {
217  while (temp != get_cpu_temp())
218  {
219  if (temp < get_cpu_temp())
220  {
221  //printf("Cooling down CPU from %dK to %dK\n", get_cpu_temp(), temp);
222  // pause for 1 second to give the CPU time to cool down
223  usleep(1e6);
224  }
225  else
226  {
227  //printf("Heating up CPU from %dK to %dK\n", get_cpu_temp(), temp);
228  // crunch some numbers to heat up the CPU
229  cpu_temp_count = 0;
230  for (int i = 0; i < 1e9; ++i)
231  {
232  cpu_temp_count += i;
233  }
234  }
