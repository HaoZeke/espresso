138 /** This ugly function does the bookkepping which nodes have to
139  * communicate to each other, when you change the node grid.
140  * Changing the domain decomposition requieres communication. This
141  * function finds (hopefully) the best way to do this. As input it
142  * needs the two grids (grid1, grid2) and a linear list (node_list1)
143  * with the node identities for grid1. The linear list (node_list2)
144  * for the second grid is calculated. For the communication group of
145  * the calling node it calculates a list (group) with the node
146  * identities and the positions (pos1, pos2) of that nodes in grid1
147  * and grid2. The return value is the size of the communication
148  * group. It gives -1 if the two grids do not fit to each other
149  * (grid1 and grid2 have to be component wise multiples of each
150  * other. see e.g. \ref calc_2d_grid in \ref grid.cpp for how to do
151  * this.).
152  *
153  * \param grid1 The node grid you start with (Input).
154  * \param grid2 The node grid you want to have (Input).
155  * \param node_list1 Linear node index list for grid1 (Input).
156  * \param node_list2 Linear node index list for grid2 (Output).
157  * \param group communication group (node identity list) for the calling node (Output).
158  * \param pos positions of the nodes in in grid2 (Output).
159  * \param my_pos position of this_node in grid2.
160  * \return Size of the communication group (Output of course!). */

209 /** pack a block (size[3] starting at start[3]) of an input 3d-grid
210  * with dimension dim[3] into an output 3d-block with dimension size[3].
211  *
212  * The block with dimensions (size[0], size[1], size[2]) is stored
213  * in 'row-major-order' or 'C-order', that means the first index is
214  * changing slowest when running through the linear array. The
215  * element (i0 (slow), i1 (mid), i2 (fast)) has the linear index
216  * li = i2 + size[2] * (i1 + (size[1]*i0))
217  *
218  * \param in pointer to input 3d-grid.
219  * \param out pointer to output 3d-grid (block).
220  * \param start start index of the block in the in-grid.
221  * \param size size of the block (=dimension of the out-grid).
222  * \param dim size of the in-grid.
223  * \param element size of a grid element (e.g. 1 for Real, 2 for Complex).
224  */
225 void fft_pack_block(double *in, double *out, int start[3], int size[3],
226  int dim[3], int element);

228 /** pack a block with dimensions (size[0] * size[1] * aize[2]) starting
229  * at start[3] of an input 3d-grid with dimension dim[3] into an
230  * output 3d-grid with dimensions (size[2] * size[0] * size[1]) with
231  * a simulatanous one-fold permutation of the indices.
232  *
233  * The permutation is defined as:
234  * slow_in -> fast_out, mid_in ->slow_out, fast_in -> mid_out
235  *
236  * An element (i0_in , i1_in , i2_in ) is then
237  * (i0_out = i1_in-start[1], i1_out = i2_in-start[2], i2_out = i0_in-start[0]) and
238  * for the linear indices we have: <br>
239  * li_in = i2_in + size[2] * (i1_in + (size[1]*i0_in)) <br>
240  * li_out = i2_out + size[0] * (i1_out + (size[2]*i0_out))
241  *
242  * For index definition see \ref fft_pack_block.
243  *
244  * \param in pointer to input 3d-grid.
245  * \param out pointer to output 3d-grid (block).
246  * \param start start index of the block in the in-grid.
247  * \param size size of the block (=dimension of the out-grid).
248  * \param dim size of the in-grid.
249  * \param element size of a grid element (e.g. 1 for Real, 2 for Complex).
250  */
251 void fft_pack_block_permute1(double *in, double *out, int start[3], int size[3],
252  int dim[3], int element);
