376  get_mi_vector(rcb1, b1->r.p, s1->r.p);

390

392

394  const double alphasb = params.p.hydrogen_bond.alphasb;

396  const double f3 = params.p.hydrogen_bond.f3;
397  const double E0sb = params.p.hydrogen_bond.E0sb;

400  potential += E0sb * exp(-ra)*(1.+ra+f2*ra*ra+f3*ra*ra*ra);
401 
402  ra = (rcb2_l - r0sb)*alphasb;
403  potential += E0sb * exp(-ra)*(1.+ra+f2*ra*ra+f3*ra*ra*ra);
404 
405 
406  /* Hydrogen bond interaction */
407 
408  /* Radial part */
409 
410  ra = (rhb_l - params.p.hydrogen_bond.r0)*params.p.hydrogen_bond.alpha;
411  tau_r = exp(-ra)*(1.+ra);
412 
413  /* Dihedral part */
414 
415  gammad = dot(n1, n2);
416  tau_d = exp(params.p.hydrogen_bond.kd*(gammad - 1));
417 
418  /* Flip part */
419 
420  gamma1 = dot(rcc, rcb1)/(rcc_l*rcb1_l);
421  gamma2 = -dot(rcc, rcb2)/(rcc_l*rcb2_l);
422 
423  /* Avoid illdefined values */
424 
425  if((gamma1 > COS_MAX) || (gamma1 < COS_MIN)) {
426  gamma1 = (gamma1 > COS_MAX)? COS_MAX : COS_MIN;

428  if((gamma2 > COS_MAX) || (gamma2 < COS_MIN))
429  gamma2 = (gamma2 > COS_MAX)? COS_MAX : COS_MIN;
430 
431  psi1 = gamma1 >= 1. ? 0. : (gamma1 <= -1. ? M_PI : acos(gamma1));
432  psi2 = gamma2 >= 1. ? 0. : (gamma2 <= -1. ? M_PI : acos(gamma2));
433 
434  dpsi1 = psi1 - params.p.hydrogen_bond.psi10;
435  dpsi2 = psi2 - params.p.hydrogen_bond.psi20;
436 
437  const double sigma1sqr = (SQR(params.p.hydrogen_bond.sigma1));
438  const double sigma2sqr = (SQR(params.p.hydrogen_bond.sigma2));
439 
440  if(dpsi1 > 0 && dpsi2 > 0) {
441  tau_flip = exp(-(SQR(dpsi1)/(2.*sigma1sqr)+SQR(dpsi2)/(2.*sigma2sqr)));
442  } else if (dpsi1 > 0.) {
443  tau_flip = exp(-(SQR(dpsi1)/(2.*sigma1sqr)));
444  potential += SQR(dpsi2)*(-0.5*E0/sigma2sqr);
445  } else if (dpsi2 > 0.) {
446  tau_flip = exp(-(SQR(dpsi2)/(2.*sigma2sqr)));
447  potential += SQR(dpsi1)*(-0.5*E0/sigma1sqr);
448  } else {
449  tau_flip = 1.;
450  potential += SQR(dpsi2)*(-0.5*E0/sigma2sqr) + SQR(dpsi1)*(-0.5*E0/sigma1sqr);
451  }
452 
453  /* Angle at which the constraint sets in */
454  const double psi_cutoff = PI*140./180.;
455  /* Spring constant for the angle constraint */
456  const double k_constraint = 50.;
457 
458  if(psi1 > psi_cutoff) {
459  potential += 0.5*k_constraint*SQR(psi1-psi_cutoff);
460  }
461  if(psi2 > psi_cutoff) {
462  potential += 0.5*k_constraint*SQR(psi2-psi_cutoff);
463  }
464 
465  potential += tau_r * tau_flip * tau_d * E0;
466 
467  *_energy = potential;
468 
469  return 0;
470 }
