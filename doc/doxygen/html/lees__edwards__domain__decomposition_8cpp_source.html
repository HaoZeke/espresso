76  /* loop all 'conventional' neighbor cells */
77  for(p=o-1; p<=o+1; p++) { /*z-loop*/
78  for(q=n-1; q<=n+1; q++) { /*y-loop*/
79  for(r=m-1; r<=m+2; r++) { /*x-loop*/
80 
81  /* Extra neighbours in x only for some cases */

157  /* find some geometry about this comm, comms order is y-x-z */
158  dir = 1;
