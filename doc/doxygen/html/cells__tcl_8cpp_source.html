42 int tclcommand_cellsystem(ClientData data, Tcl_Interp *interp,
43  int argc, char **argv)
44 {
45  int err = 0;
46 
47  if (argc <= 1) {
48  Tcl_AppendResult(interp, "usage: cellsystem <system> <params>", (char *)NULL);
49  return TCL_ERROR;
50  }
51 
52  if (ARG1_IS_S("domain_decomposition")) {
53  if (argc > 2) {
54  if (ARG_IS_S(2,"-verlet_list"))
55  dd.use_vList = 1;
56  else if(ARG_IS_S(2,"-no_verlet_list"))
57  dd.use_vList = 0;
58  else{
59  Tcl_AppendResult(interp, "wrong flag to",argv[0],
60  " : should be \" -verlet_list or -no_verlet_list \"",
61  (char *) NULL);
62  return (TCL_ERROR);
63  }
64  }
65  /** by default use verlet list */
66  else dd.use_vList = 1;
67  mpi_bcast_cell_structure(CELL_STRUCTURE_DOMDEC);
68  }
69  else if (ARG1_IS_S("nsquare"))
70  mpi_bcast_cell_structure(CELL_STRUCTURE_NSQUARE);
71  else if (ARG1_IS_S("layered")) {
72  if (argc > 2) {
73  if (!ARG_IS_I(2, n_layers))
74  return TCL_ERROR;
75  if (n_layers <= 0) {
76  Tcl_AppendResult(interp, "layer height should be positive", (char *)NULL);
77  return TCL_ERROR;
78  }
79  determine_n_layers = 0;
80  }
81 
82  /* check node grid. All we can do is 1x1xn. */
83  if (node_grid[0] != 1 || node_grid[1] != 1) {
84  node_grid[0] = node_grid[1] = 1;
85  node_grid[2] = n_nodes;
86 
87  err = mpi_bcast_parameter(FIELD_NODEGRID);
88  }
89  else
90  err = 0;
91 
92  if (!err)
93  mpi_bcast_cell_structure(CELL_STRUCTURE_LAYERED);
94  }
95  else {
96  Tcl_AppendResult(interp, "unkown cell structure type \"", argv[1],"\"", (char *)NULL);
97  return TCL_ERROR;
98  }
99  return gather_runtime_errors(interp, TCL_OK);
100 }
