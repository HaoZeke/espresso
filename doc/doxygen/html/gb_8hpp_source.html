43 inline void add_gb_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
44  double d[3], double dist, double force[3],
45  double torque1[3], double torque2[3])
46 
47 {
48  if (!CUTOFF_CHECK(dist < ia_params->GB_cut))
49  return;
50 
51  double a,b,c, X, Xcut,
52  Brack,BrackCut,
53  Bra12,Bra12Cut,
54  u1x, u1y, u1z,
55  u2x, u2y, u2z,
56  E,E1,E2, Sigma,
57  Brhi1,Brhi2,
58  Plus1,Minus1,
59  Plus2,Minus2,
60  Koef1,Koef2, /* mu/E2 and Sigma^3/2 */
61  dU_dr, dU_da, dU_db, dU_dc, /* all derivatives */
62  FikX,FikY,FikZ, /* help for forces */
63  Gx,Gy,Gz; /* help for torques */
64 
65  u1x = p1->r.quatu[0]; u1y = p1->r.quatu[1]; u1z = p1->r.quatu[2];
66  u2x = p2->r.quatu[0]; u2y = p2->r.quatu[1]; u2z = p2->r.quatu[2];
67 
68  a = d[0]*u1x + d[1]*u1y + d[2]*u1z;
69  b = d[0]*u2x + d[1]*u2y + d[2]*u2z;
70  c = u1x*u2x + u1y*u2y + u1z*u2z;
71  E1 = 1/sqrt(1-ia_params->GB_chi1*ia_params->GB_chi1*c*c);
72  Plus1 = (a+b)/(1+ia_params->GB_chi1*c);
73  Plus2 = (a+b)/(1+ia_params->GB_chi2*c);
74  Minus1 = (a-b)/(1-ia_params->GB_chi1*c);
75  Minus2 = (a-b)/(1-ia_params->GB_chi2*c);
76  Brhi2 = (ia_params->GB_chi2/dist/dist)*(Plus2*(a+b) + Minus2*(a-b));
77  E2 = 1-0.5*Brhi2;
78  E = 4*ia_params->GB_eps*pow(E1,ia_params->GB_nu)*pow(E2,ia_params->GB_mu);
79  Brhi1 = (ia_params->GB_chi1/dist/dist)*(Plus1*(a+b) + Minus1*(a-b));
80  Sigma = ia_params->GB_sig/sqrt(1-0.5*Brhi1);
81  Koef1 = ia_params->GB_mu/E2;
82  Koef2 = Sigma*Sigma*Sigma*0.5;
83 
84  X = 1/(dist - Sigma + ia_params->GB_sig);
85  Xcut = 1/(ia_params->GB_cut - Sigma + ia_params->GB_sig);
86 
87  if (X < 1.25) { /* 1.25 corresponds to the interparticle penetration of 0.2 units of length.
88  If they are not that close, the GB forces and torques are calculated */
89 
90  Brack = X*X*X;
91  BrackCut = Xcut*Xcut*Xcut;
92  Brack = Brack*Brack;
93  BrackCut = BrackCut*BrackCut;
94 
95  Bra12 = 6*Brack*X*(2*Brack-1);
96  Bra12Cut = 6*BrackCut*Xcut*(2*BrackCut-1);
97  Brack = Brack*(Brack-1);
98  BrackCut = BrackCut*(BrackCut-1);
99 
100  /*-------- Here we calculate derivatives -----------------------------*/
101 
102  dU_dr = E*(Koef1*Brhi2*(Brack-BrackCut)-Koef2*Brhi1*(Bra12-Bra12Cut)-Bra12*dist)/dist/dist;
103  Koef1 = Koef1*ia_params->GB_chi2/dist/dist;
104  Koef2 = Koef2*ia_params->GB_chi1/dist/dist;
105  dU_da = E*(Koef1*(Minus2+Plus2)*(BrackCut-Brack)+Koef2*(Plus1+Minus1)*(Bra12-Bra12Cut));
106  dU_db = E*(Koef1*(Minus2-Plus2)*(Brack-BrackCut)+Koef2*(Plus1-Minus1)*(Bra12-Bra12Cut));
107  dU_dc = E*((Brack-BrackCut)*(ia_params->GB_nu*E1*E1*ia_params->GB_chi1*ia_params->GB_chi1*c+
108  0.5*Koef1*ia_params->GB_chi2*(Plus2*Plus2-Minus2*Minus2))-
109  (Bra12-Bra12Cut)*0.5*Koef2*ia_params->GB_chi1*(Plus1*Plus1-Minus1*Minus1));
110 
111  /*--------------------------------------------------------------------*/
112 
113  FikX = -dU_dr*d[0] - dU_da*u1x - dU_db*u2x;
114  FikY = -dU_dr*d[1] - dU_da*u1y - dU_db*u2y;
115  FikZ = -dU_dr*d[2] - dU_da*u1z - dU_db*u2z;
116 
117  force[0] += FikX;
118  force[1] += FikY;
119  force[2] += FikZ;
120 
121  if (torque1 != NULL) {
122  /* calculate torque: torque = u_1 x G */
123 
124  Gx = -dU_da*d[0] - dU_dc*u2x;
125  Gy = -dU_da*d[1] - dU_dc*u2y;
126  Gz = -dU_da*d[2] - dU_dc*u2z;
127 
128  torque1[0]+= u1y*Gz - u1z*Gy;
129  torque1[1]+= u1z*Gx - u1x*Gz;
130  torque1[2]+= u1x*Gy - u1y*Gx;
131 
132  if (torque2 != NULL) {
133  /* calculate torque: torque = u_2 x G */
134 
135  Gx = -dU_db*d[0] - dU_dc*u1x;
136  Gy = -dU_db*d[1] - dU_dc*u1y;
137  Gz = -dU_db*d[2] - dU_dc*u1z;
138 
139  torque2[0]+= u2y*Gz - u2z*Gy;
140  torque2[1]+= u2z*Gx - u2x*Gz;
141  torque2[2]+= u2x*Gy - u2y*Gx;
142  }
143  }
144  }
145  else { /* the particles are too close to each other */
146  Koef1 = 100;
147 
148  force[0] += Koef1 * d[0];
149  force[1] += Koef1 * d[1];
150  force[2] += Koef1 * d[2];
151  }
152 }

155  double d[3], double dist)

163  u2x, u2y, u2z,

170  u2x = p2->r.quatu[0]; u2y = p2->r.quatu[1]; u2z = p2->r.quatu[2];
171 
172  a = d[0]*u1x + d[1]*u1y + d[2]*u1z;
173  b = d[0]*u2x + d[1]*u2y + d[2]*u2z;
174  c = u1x*u2x + u1y*u2y + u1z*u2z;
175 
176  Plus1 = (a+b)/(1+ia_params->GB_chi1*c);
177  Plus2 = (a+b)/(1+ia_params->GB_chi2*c);
178  Minus1 = (a-b)/(1-ia_params->GB_chi1*c);
179  Minus2 = (a-b)/(1-ia_params->GB_chi2*c);
180  E1 = 1/sqrt(1-ia_params->GB_chi1*ia_params->GB_chi1*c*c);
181  E2 = 1-0.5*(ia_params->GB_chi2/dist/dist)*(Plus2*(a+b) + Minus2*(a-b));
182  E = 4*ia_params->GB_eps*pow(E1,ia_params->GB_nu)*pow(E2,ia_params->GB_mu);
183  Sigma = ia_params->GB_sig/sqrt(1-0.5*(ia_params->GB_chi1/dist/dist)*(Plus1*(a+b) + Minus1*(a-b)));
184 
185  X = 1/(dist - Sigma + ia_params->GB_sig);
186  Xcut = 1/(ia_params->GB_cut - Sigma + ia_params->GB_sig);
187 
188  Brack = X*X*X;
189  BrackCut = Xcut*Xcut*Xcut;
190  Brack = Brack*Brack;
191  BrackCut = BrackCut*BrackCut;
192  Brack = Brack*(Brack-1);
193  BrackCut = BrackCut*(BrackCut-1);
194
