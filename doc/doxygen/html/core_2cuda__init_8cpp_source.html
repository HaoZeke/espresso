35 struct CompareDevices {
36  bool operator()(const EspressoGpuDevice &a, const EspressoGpuDevice &b) const {
37  const int name_comp = strncmp(a.proc_name, b.proc_name, 63);
38  /* Both devs are from the same node, order by id */
39  if(name_comp == 0)
40  return a.id < b.id;
41  else
42  return name_comp < 0;
43  }

52 std::vector<EspressoGpuDevice> cuda_gather_gpus(void) {
53  int n_gpus = cuda_get_n_gpus();
54  char proc_name[MPI_MAX_PROCESSOR_NAME];
55  int proc_name_len;
56  /* List of local devices */
57  std::vector<EspressoGpuDevice> devices;
58  /* Global unique device list (only relevant on master) */
59  std::vector<EspressoGpuDevice> g_devices;
60  int *n_gpu_array = 0;
61 
62  MPI_Get_processor_name(proc_name, &proc_name_len);
63 
64  /* Truncate to 63 chars to fit struct. */
65  if(strlen(proc_name) > 63)
66  proc_name[63] = 0;
67 
68  for(int i = 0; i < n_gpus; ++i) {
69  /* Check if device has at least mininum compute capability */
70  if(cuda_check_gpu(i) == ES_OK) {
71  EspressoGpuDevice device;
72  if(cuda_get_device_props(i, device) == ES_OK){
73  strncpy(device.proc_name, proc_name, 64);
74  devices.push_back(device);
75  }
76  }
77  }
78 
79  /** Update n_gpus to number of usable devices */
80  n_gpus = devices.size();
81 
82  if(this_node == 0) {
83  std::set<EspressoGpuDevice, CompareDevices> device_set;
84  n_gpu_array = new int[n_nodes];
85  MPI_Gather(&n_gpus, 1, MPI_INT, n_gpu_array, 1, MPI_INT, 0, MPI_COMM_WORLD);
86 
87  /* insert local devices */
88  std::copy(devices.begin(), devices.end(), std::inserter(device_set, device_set.begin()));
89 
90  EspressoGpuDevice device;
91  MPI_Status s;
92  /* Get devices from other nodes */
93  for(int i = 1; i < n_nodes; ++i) {
94  for(int j = 0; j < n_gpu_array[i]; ++j) {
95  MPI_Recv(&device, sizeof(EspressoGpuDevice), MPI_BYTE, i, 0, MPI_COMM_WORLD, &s);
96  device_set.insert(device);
97  }
98  }
99  /* Copy unique devices to result, if any */
100  std::copy(device_set.begin(), device_set.end(), std::inserter(g_devices, g_devices.begin()));
101  delete n_gpu_array;
102  } else {
103  /* Send number of devices to master */
104  MPI_Gather(&n_gpus, 1, MPI_INT, n_gpu_array, 1, MPI_INT, 0, MPI_COMM_WORLD);
105  /* Send devices to maser */
106  for(std::vector<EspressoGpuDevice>::iterator device = devices.begin();
107  device != devices.end(); ++device) {
108  MPI_Send(&(*device), sizeof(EspressoGpuDevice), MPI_BYTE, 0, 0, MPI_COMM_WORLD);
109  }
110  }
111  return g_devices;
112 }
