397 int double_correlation_get_data( double_correlation* self ) {
398  // We must now go through the hierarchy and make sure there is space for the new
399  // datapoint. For every hierarchy level we have to decide if it necessary to move
400  // something
401  int i,j;
402  int highest_level_to_compress;
403  unsigned int index_new, index_old, index_res;
404  int error;
405 
406  self->t++;
407 
408  highest_level_to_compress=-1;
409  i=0;
410  j=1;
411  // Lets find out how far we have to go back in the hierarchy to make space for the new value
412  while (1) {
413  if ( ( (self->t - ((self->tau_lin + 1)*((1<<(i+1))-1) + 1) )% (1<<(i+1)) == 0) ) {
414  if ( i < (int(self->hierarchy_depth) - 1) && self->n_vals[i]> self->tau_lin) {
415 
416  highest_level_to_compress+=1;
417  i++;
418  } else break;
419  } else break;
420  }
421 
422  // Now we know we must make space on the levels 0..highest_level_to_compress
423  // Now lets compress the data level by level.
424 
425  for ( i = highest_level_to_compress; i >= 0; i-- ) {
426  // We increase the index indicating the newest on level i+1 by one (plus folding)
427  self->newest[i+1] = (self->newest[i+1] + 1) % (self->tau_lin+1);
428  self->n_vals[i+1]+=1;
429 // printf("t %d compressing level %d no %d and %d to level %d no %d, nv %d\n",self->t, i, (self->newest[i]+1) % (self->tau_lin+1),
430 //(self->newest[i]+2) % (self->tau_lin+1), i+1, self->newest[i+1], self->n_vals[i]);
431  (*self->compressA)(self->A[i][(self->newest[i]+1) % (self->tau_lin+1)],
432  self->A[i][(self->newest[i]+2) % (self->tau_lin+1)],
433  self->A[i+1][self->newest[i+1]],self->dim_A);
434  if (!self->autocorrelation)
435  (*self->compressB)(self->B[i][(self->newest[i]+1) % (self->tau_lin+1)],
436  self->B[i][(self->newest[i]+2) % (self->tau_lin+1)],
437  self->B[i+1][self->newest[i+1]],self->dim_B);
438  }
439 
440  self->newest[0] = ( self->newest[0] + 1 ) % (self->tau_lin +1);
441  self->n_vals[0]++;
442 
443  if ( observable_calculate(self->A_obs) != 0 )
444  return 1;
445  // copy the result:
446  memmove(self->A[0][self->newest[0]], self->A_obs->last_value, self->dim_A*sizeof(double));
447 
448  if (!self->autocorrelation) {
449  if ( observable_calculate(self->B_obs) != 0 )
450  return 2;
451  memmove(self->B[0][self->newest[0]], self->B_obs->last_value, self->dim_B*sizeof(double));
452  }
453 
454  // Now we update the cumulated averages and variances of A and B
455  self->n_data++;
456  for (unsigned k=0; k<self->dim_A; k++) {
457  self->A_accumulated_average[k]+=self->A[0][self->newest[0]][k];
458  self->A_accumulated_variance[k]+=self->A[0][self->newest[0]][k]*self->A[0][self->newest[0]][k];
459  }
460  // Here we check if it is an autocorrelation
461  if (!self->autocorrelation) {
462  for (unsigned k=0; k<self->dim_B; k++) {
463  self->B_accumulated_average[k]+=self->B[0][self->newest[0]][k];
464  self->B_accumulated_variance[k]+=self->B[0][self->newest[0]][k]*self->B[0][self->newest[0]][k];
465  }
466  }
467 
468  double* temp = (double*)malloc(self->dim_corr*sizeof(double));
469  if (!temp)
470  return 4;
471 // Now update the lowest level correlation estimates
472  for ( j = 0; j < int(MIN(self->tau_lin+1, self->n_vals[0]) ); j++) {
473  index_new = self->newest[0];
474  index_old = (self->newest[0] - j + self->tau_lin + 1) % (self->tau_lin + 1);
475 // printf("old %d new %d\n", index_old, index_new);
476  error = (self->corr_operation)(self->A[0][index_old], self->dim_A, self->B[0][index_new], self->dim_B, temp, self->dim_corr, self->args);
477  if ( error != 0)
478  return error;
479  self->n_sweeps[j]++;
480  for (unsigned k = 0; k < self->dim_corr; k++) {
481  self->result[j][k] += temp[k];
482  }
483  }
484 // Now for the higher ones
485  for ( int i = 1; i < highest_level_to_compress+2; i++) {
486  for ( unsigned j = (self->tau_lin+1)/2+1; j < MIN(self->tau_lin+1, self->n_vals[i]); j++) {
487  index_new = self->newest[i];
488  index_old = (self->newest[i] - j + self->tau_lin + 1) % (self->tau_lin + 1);
489  index_res = self->tau_lin + (i-1)*self->tau_lin/2 + (j - self->tau_lin/2+1) -1;
490  error=(self->corr_operation)(self->A[i][index_old], self->dim_A, self->B[i][index_new], self->dim_B, temp, self->dim_corr, self->args);
491  if ( error != 0)
492  return error;
493  self->n_sweeps[index_res]++;
494  for (unsigned k = 0; k < self->dim_corr; k++) {
495  self->result[index_res][k] += temp[k];
496  }
497  }
498  }
499  free(temp);
500  return 0;
501 }

627 int double_correlation_finalize( double_correlation* self ) {
628  // We must now go through the hierarchy and make sure there is space for the new
629  // datapoint. For every hierarchy level we have to decide if it necessary to move
630  // something
631  int i,j;
632  int ll=0; // current lowest level
633  int vals_ll=0; // number of values remaining in the lowest level
634  int highest_level_to_compress;
635  unsigned int index_new, index_old, index_res;
636  int error;
637  //int compress;
638  unsigned tau_lin=self->tau_lin;
639  int hierarchy_depth=self->hierarchy_depth;
640 
641  double* temp = (double*)malloc(self->dim_corr*sizeof(double));
642 
643  // make a flag that the correlation is finalized
644  self->finalized=1;
645 
646  if (!temp)
647  return 4;
648  //printf ("tau_lin:%d, hierarchy_depth: %d\n",tau_lin,hierarchy_depth);
649  //for(ll=0;ll<hierarchy_depth;ll++) printf("n_vals[l=%d]=%d\n",ll, self->n_vals[ll]);
650  for(ll=0;ll<hierarchy_depth-1;ll++) {
651  if (self->n_vals[ll] > tau_lin+1 ) vals_ll = tau_lin + self->n_vals[ll]%2;
652  else vals_ll=self->n_vals[ll];
653  //printf("\nfinalizing level %d with %d vals initially\n",ll,vals_ll);
654 
655  while (vals_ll) {
656  // Check, if we will want to push the value from the lowest level
657  if (vals_ll % 2) {
658  highest_level_to_compress=ll;
659  } else {
660  highest_level_to_compress=-1;
661  }
662  //printf("have %d values in ll=%d ",vals_ll,ll);
663  //if(highest_level_to_compress<0) printf("Do NOT ");
664  //printf("Compress\n"); fflush(stdout);
665 
666  i=ll+1; // lowest level, for which we have to check for compression
667  j=1;
668  // Lets find out how far we have to go back in the hierarchy to make space for the new value
669  while (highest_level_to_compress>-1) {
670  //printf("test level %d for compression, n_vals=%d ... ",i,self->n_vals[i]);
671  if ( self->n_vals[i]%2 ) {
672  if ( i < (hierarchy_depth-1) && self->n_vals[i]> tau_lin) {
673  //printf("YES\n");
674  highest_level_to_compress+=1;
675  i++;
676  } else {
677  //printf("NO\n");
678  break;
679  }
680  } else {
681  //printf("NO\n");
682  break;
683  }
684  }
685  vals_ll-=1;
686  //printf("ll: %d, highest_lvevel_to_compress:%d\n",ll, highest_level_to_compress); fflush(stdout);
687 
688  // Now we know we must make space on the levels 0..highest_level_to_compress
689  // Now lets compress the data level by level.
690 
691  for ( i = highest_level_to_compress; i >=ll; i-- ) {
692  // We increase the index indicating the newest on level i+1 by one (plus folding)
693  self->newest[i+1] = (self->newest[i+1] + 1) % (tau_lin+1);
694  self->n_vals[i+1]+=1;
695  //printf("compressing level %d no %d and %d to level %d no %d, nv %d\n",i, (self->newest[i]+1) % (tau_lin+1), (self->newest[i]+2) % (tau_lin+1), i+1, self->newest[i+1], self->n_vals[i]);
696  (*self->compressA)(self->A[i][(self->newest[i]+1) % (tau_lin+1)],
697  self->A[i][(self->newest[i]+2) % (tau_lin+1)],
698  self->A[i+1][self->newest[i+1]],self->dim_A);
699  (*self->compressB)(self->B[i][(self->newest[i]+1) % (tau_lin+1)],
700  self->B[i][(self->newest[i]+2) % (tau_lin+1)],
701  self->B[i+1][self->newest[i+1]],self->dim_B);
702  }
703  self->newest[ll] = (self->newest[ll] + 1) % (tau_lin+1);
704 
705  // We only need to update correlation estimates for the higher levels
706  for ( i = ll+1; i < highest_level_to_compress+2; i++) {
707  for ( j = (tau_lin+1)/2+1; j < int(MIN(tau_lin+1, self->n_vals[i])); j++) {
708  index_new = self->newest[i];
709  index_old = (self->newest[i] - j + tau_lin + 1) % (tau_lin + 1);
710  index_res = tau_lin + (i-1)*tau_lin/2 + (j - tau_lin/2+1) -1;
711  error=(self->corr_operation)(self->A[i][index_old], self->dim_A, self->B[i][index_new], self->dim_B, temp, self->dim_corr, self->args);
712  if ( error != 0)
713  return error;
714  self->n_sweeps[index_res]++;
715  for (unsigned k = 0; k < self->dim_corr; k++) {
716  self->result[index_res][k] += temp[k];
717  }
718  }
719  }
720  // lowest level exploited, go upwards
721  if(!vals_ll) {
722  //printf("reached end of level %d, go up\n",ll);
723  //fflush(stdout);
724  }
725  }
726  }
727  free(temp);
728  return 0;
729 }
