83 /** Add a non-bonded pair force by linear interpolation from a table.
84  Needs feature TABULATED compiled in (see \ref config.hpp). */
85 inline void add_tabulated_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
86  double d[3], double dist, double force[3])
87 {
88  if (CUTOFF_CHECK(dist < ia_params->TAB_maxval)){
89  double phi, dindex, fac;
90  int tablepos, table_start,j;
91  double rescaled_force_cap = force_cap/dist;
92 
93  fac = 0.0;
94 
95  table_start = ia_params->TAB_startindex;
96  dindex = (dist-ia_params->TAB_minval)/ia_params->TAB_stepsize;
97  tablepos = (int)(floor(dindex));
98 
99  if ( dist > ia_params->TAB_minval ) {
100  phi = dindex - tablepos;
101  fac = tabulated_forces.e[table_start + tablepos]*(1-phi) + tabulated_forces.e[table_start + tablepos+1]*phi;
102  }
103  else {
104  /* Use an extrapolation beyond the table */
105  if ( dist > 0 ) {
106  tablepos = 0;
107  phi = dindex - tablepos;
108  fac = (tabulated_forces.e[table_start]*ia_params->TAB_minval)*(1-phi) +
109  (tabulated_forces.e[table_start+1]*(ia_params->TAB_minval+ia_params->TAB_stepsize))*phi;
110  fac = fac/dist;
111  }
112  else { /* Particles on top of each other .. leave fac as 0.0 */
113  }
114  }
115 
116  if ( rescaled_force_cap < fac && force_cap > 0.0) {
117  fac = rescaled_force_cap;
118  }
119 
120  for(j=0;j<3;j++)
121  force[j] += fac * d[j];
122  }
123 }

125 /** Add a non-bonded pair energy by linear interpolation from a table.
126  Needs feature TABULATED compiled in (see \ref config.hpp). */
127 inline double tabulated_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
128  double d[3], double dist) {
129  double phi, dindex;
130  int tablepos, table_start;
131  double x0, b;
132 
133  if (CUTOFF_CHECK(dist < ia_params->TAB_maxval)) {
134  dindex = (dist-ia_params->TAB_minval)/ia_params->TAB_stepsize;
135  tablepos = (int)(floor(dindex));
136  table_start = ia_params->TAB_startindex;
137 
138  if ( tablepos < 0 ){
139  /* For distances smaller than the tabulated minimim take quadratic
140  extrapolation from first two values of the force table!
141  This corresponds to the linear extrapolation of the force at that point.
142  This sould not occur too often, since it is quite expensive!
143  */
144  tablepos = 0;
145  b = (tabulated_forces.e[table_start + tablepos + 1]-tabulated_forces.e[table_start + tablepos])/ia_params->TAB_stepsize;
146  x0 = ia_params->TAB_minval-tabulated_forces.e[table_start + tablepos]/b;
147  return ( (tabulated_energies.e[table_start + tablepos]
148  + 0.5*b*SQR(ia_params->TAB_minval-x0))
149  - 0.5*b*SQR(dist-x0) );
150  }
151 
152  phi = (dindex - tablepos);
153 
154  return tabulated_energies.e[table_start + tablepos]*(1-phi)
155  + tabulated_energies.e[table_start + tablepos+1]*phi;
156  }
157  return 0.0;
158 }

174 {

234 /** Calculate and return a tabulated bond length energy with number
235  type_num (see \ref Bonded_ia_parameters) between particles p1 and
236  p2. For distances smaller than the tabulated range it uses a
237  quadratic extrapolation based on the first two tabulated force
238  values and the first tabulated energy value.
239  Needs feature TABULATED compiled in (see \ref config.hpp). */
240 inline int tab_bond_energy(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, double dx[3], double *_energy)
241 {
242  double dist = sqrt(sqrlen(dx));
243 
244  if(dist > iaparams->p.tab.maxval)
245  return 1;
246 
247  /* For distances smaller than the tabulated minimim take quadratic
248  extrapolation from first two values of the force table!
249  This corresponds to the linear extrapolation of the force at that point.
250  This sould not occur too often, since it is quite expensive!
251  */
252  if( dist < iaparams->p.tab.minval) {
253  double x0, b;
254  b = (iaparams->p.tab.f[1]-iaparams->p.tab.f[0])*iaparams->p.tab.invstepsize;
255  x0 = iaparams->p.tab.minval - iaparams->p.tab.f[0]/b;
256  *_energy = ( (iaparams->p.tab.e[0] + 0.5*b*SQR(iaparams->p.tab.minval-x0)) -
257  0.5*b*SQR(dist-x0) );
258  }
259  else
260  *_energy = bonded_tab_energy_lookup(dist, iaparams);
261 
262  return 0;
263 }

265 /** Calculate a tabulated bond angle force with number type_num (see
266  \ref Bonded_ia_parameters) between particles p_left, p_mid and
267  p_right and add it to the particle forces. The force on p_left and
268  p_right acts perpendicular to the connecting vector between the
269  particle and p_mid and in the plane defined by the three
270  particles. The force on the middle particle balances the other two
271  forces. The forces are scaled with the invers length of the
272  connecting vectors. It is assumed that the potential is tabulated
273  for all angles between 0 and Pi.
274  Needs feature TABULATED compiled in (see \ref config.hpp). */
275 inline int calc_tab_angle_force(Particle *p_mid, Particle *p_left,
276  Particle *p_right, Bonded_ia_parameters *iaparams,
277  double force1[3], double force2[3])
278 {
279  double cosine, phi, invsinphi, vec1[3], vec2[3], d1i, d2i, dist2, fac, f1=0.0, f2=0.0;
280  int j;
281 
282  /* vector from p_left to p_mid */
283  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
284  dist2 = sqrlen(vec1);
285  d1i = 1.0 / sqrt(dist2);
286  for(j=0;j<3;j++) vec1[j] *= d1i;
287  /* vector from p_mid to p_right */
288  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
289  dist2 = sqrlen(vec2);
290  d2i = 1.0 / sqrt(dist2);
291  for(j=0;j<3;j++) vec2[j] *= d2i;
292  /* scalar produvt of vec1 and vec2 */
293  cosine = scalar(vec1, vec2);
294 #ifdef TABANGLEMINUS
295  phi = acos(-cosine);
296 #else
297  phi = acos(cosine);
298 #endif
299  invsinphi = sin(phi);
300  if(invsinphi < TINY_SIN_VALUE) invsinphi = TINY_SIN_VALUE;
301  invsinphi = 1.0/invsinphi;
302  /* look up force factor */
303  fac = bonded_tab_force_lookup(phi, iaparams);
304  /* apply bend forces */
305  for(j=0;j<3;j++) {
306  f1 = fac * (cosine * vec1[j] - vec2[j])*invsinphi * d1i;
307  f2 = fac * (cosine * vec2[j] - vec1[j])*invsinphi * d2i;
308  force1[j] = (f1-f2);
309  force2[j] = -f1;
310  }
311 
312  return 0;
313 }

315 /* The force on each particle due to a three-body bonded tabulated
316  potential is computed. */
317 inline void calc_angle_3body_tabulated_forces(Particle *p_mid, Particle *p_left,
318  Particle *p_right, Bonded_ia_parameters *iaparams,
319  double force1[3], double force2[3], double force3[3]) {
320 
321  int j;
322  double pot_dep;
323  double cos_phi;
324  double sin_phi;
325  double vec31[3];
326  double vec21[3];
327  double vec12[3]; // espresso convention
328  double vec21_sqr;
329  double vec31_sqr;
330  double vec21_magn;
331  double vec31_magn;
332  double fj[3];
333  double fk[3];
334  double phi, dU; // bond angle and d/dphi of U(phi)
335 
336  get_mi_vector(vec12, p_mid->r.p, p_left->r.p);
337  for(j = 0; j < 3; j++)
338  vec21[j] = -vec12[j];
339 
340  get_mi_vector(vec31, p_right->r.p, p_mid->r.p);
341  vec21_sqr = sqrlen(vec21);
342  vec21_magn = sqrt(vec21_sqr);
343  vec31_sqr = sqrlen(vec31);
344  vec31_magn = sqrt(vec31_sqr);
345  cos_phi = scalar(vec21, vec31) / (vec21_magn * vec31_magn);
346  sin_phi = sqrt(1.0 - SQR(cos_phi));
347 
348  if(cos_phi < -1.0) cos_phi = -TINY_COS_VALUE;
349  if(cos_phi > 1.0) cos_phi = TINY_COS_VALUE;
350  phi = acos(cos_phi);
351 
352  dU = bonded_tab_force_lookup(phi, iaparams);
353 
354  // potential dependent term (dU/dphi * 1 / sin(phi))
355  pot_dep = dU / sin_phi;
356 
357  for(j = 0; j < 3; j++) {
358  fj[j] = vec31[j] / (vec21_magn * vec31_magn) - cos_phi * vec21[j] / vec21_sqr;
359  fk[j] = vec21[j] / (vec21_magn * vec31_magn) - cos_phi * vec31[j] / vec31_sqr;
360  }
361 
362  // note that F1 = -(F2 + F3) in analytical case
363  for(j = 0; j < 3; j++) {
364  force1[j] = force1[j] - pot_dep * (fj[j] + fk[j]);
365  force2[j] = force2[j] + pot_dep * fj[j];
366  force3[j] = force3[j] + pot_dep * fk[j];
367  }
368 }

396 /** Calculate a tabulated dihedral force with number type_num (see
397  \ref Bonded_ia_parameters) between particles p1. p2, p3 and p4 and
398  add it to the particle forces. This function is not tested yet.
399  Needs feature TABULATED compiled in (see \ref config.hpp). */
400 inline int calc_tab_dihedral_force(Particle *p2, Particle *p1,
401  Particle *p3, Particle *p4, Bonded_ia_parameters *iaparams,
402  double force2[3], double force1[3], double force3[3])
403 {
404  int i;
405  /* vectors for dihedral angle calculation */
406  double v12[3], v23[3], v34[3], v12Xv23[3], v23Xv34[3], l_v12Xv23, l_v23Xv34;
407  double v23Xf1[3], v23Xf4[3], v34Xf4[3], v12Xf1[3];
408  /* dihedral angle, cosine of the dihedral angle, cosine of the bond angles */
409  double phi, cosphi;
410  /* force factors */
411  double fac, f1[3], f4[3];
412 
413  /* dihedral angle */
414  calc_dihedral_angle(p1, p2, p3, p4, v12, v23, v34, v12Xv23, &l_v12Xv23, v23Xv34, &l_v23Xv34, &cosphi, &phi);
415  /* dihedral angle not defined - force zero */
416  if ( phi == -1.0 ) {
417  for(i=0;i<3;i++) { force1[i] = 0.0; force2[i] = 0.0; force3[i] = 0.0; }
418  return 0;
419  }
420 
421  /* calculate force components (directions) */
422  for(i=0;i<3;i++) {
423  f1[i] = (v23Xv34[i] - cosphi*v12Xv23[i])/l_v12Xv23;;
424  f4[i] = (v12Xv23[i] - cosphi*v23Xv34[i])/l_v23Xv34;
425  }
426  vector_product(v23, f1, v23Xf1);
427  vector_product(v23, f4, v23Xf4);
428  vector_product(v34, f4, v34Xf4);
429  vector_product(v12, f1, v12Xf1);
430 
431  /* table lookup */
432  fac = bonded_tab_force_lookup(phi, iaparams);
433 
434  /* store dihedral forces */
435  for(i=0;i<3;i++) {
436  force1[i] = fac*v23Xf1[i];
437  force2[i] = fac*(v34Xf4[i] - v12Xf1[i] - v23Xf1[i]);
438  force3[i] = fac*(v12Xf1[i] - v23Xf4[i] - v34Xf4[i]);
439  }
440 
441  return 0;
442 }
