163 /**
164  This routine calculates the orientational order parameter for a
165  lipid bilayer.
166 */
167 int orient_order(double* result, double* stored_dirs)
168 {
169  double dir[3];
170  double refdir[3] = {0,0,1};
171  double sumdir[3] = {0,0,0};
172  double zref;
173  int bilayer_cnt;
174  int i,atom,tmpzdir;
175  double dp;
176  double len;
177 
178  IntList l_orient;
179  init_intlist(&l_orient);
180  realloc_intlist(&l_orient, n_molecules);
181 
182  bilayer_cnt = 0;
183  *result = 0;
184 
185 
186  // Check to see if the grid has been set and if not then interpret
187  if ( xdir + ydir + zdir == -3 ) {
188  tmpzdir = 2;
189  } else {
190  tmpzdir = zdir;
191  };
192 
193  /* Update particles */
194  updatePartCfg(WITHOUT_BONDS);
195  /* Make sure particles are sorted */
196  if (!sortPartCfg()) {
197  ostringstream msg;
198  msg <<"could not sort partCfg, particles have to start at 0 and have consecutive identities";
199  runtimeError(msg);
200  return ES_ERROR;
201  }
202 
203  /* Calculate the reference z position as its mean.*/
204  zref = calc_zref( tmpzdir );
205 
206 
207  /* Calculate the orientation of all the lipids in turn and include
208  only UP or DOWN lipids in a calculation of the overall orientation
209  direction of the bilayer .. ie the reference vector from which we
210  can calculate the orientational order. */
211 
212  for ( i = 0 ; i < n_molecules ; i++) {
213  atom = topology[i].part.e[0];
214  l_orient.e[i] = lipid_orientation(atom,partCfg,zref,dir,refdir);
215  stored_dirs[i*3] = dir[0];
216  stored_dirs[i*3+1] = dir[1];
217  stored_dirs[i*3+2] = dir[2];
218 
219  if ( l_orient.e[i] == LIPID_UP ) {
220  sumdir[0] += dir[0];
221  sumdir[1] += dir[1];
222  sumdir[2] += dir[2];
223  bilayer_cnt++;
224  }
225  if ( l_orient.e[i] == LIPID_DOWN ) {
226  sumdir[0] -= dir[0];
227  sumdir[1] -= dir[1];
228  sumdir[2] -= dir[2];
229  bilayer_cnt++;
230  }
231  }
232 
233  /* Normalise the bilayer normal vector */
234  len = 0.0;
235  for ( i = 0 ; i < 3 ; i++) {
236  sumdir[i] = sumdir[i]/(double)(bilayer_cnt);
237  len += sumdir[i]*sumdir[i];
238  }
239  for ( i = 0 ; i < 3 ; i++) {
240  sumdir[i] = sumdir[i]/sqrt(len);
241  }
242 
243  /* Calculate the orientational order */
244  for ( i = 0 ; i < n_molecules ; i++ ) {
245  dir[0] = stored_dirs[i*3];
246  dir[1] = stored_dirs[i*3+1];
247  dir[2] = stored_dirs[i*3+2];
248 
249  if ( l_orient.e[i] != LIPID_STRAY && l_orient.e[i] != REAL_LIPID_STRAY ) {
250  dp = scalar(dir,sumdir);
251  *result += dp*dp*1.5-0.5;
252  }
253 
254  }
255 
256 
257  realloc_intlist(&l_orient, 0);
258 
259  *result = *result/(double)(bilayer_cnt);
260  return ES_OK;
261 }

265 /**
266  This routine performs a simple check to see whether a lipid is
267  oriented up or down or if it has escaped the bilayer.
268 
269  \param id The particle identifier
270  \param partCfg An array of sorted particles
271 
272  \param zref The average z position of all particles. This is used
273  to check for stray lipids
274 
275  \param director director
276  \param refdir is a vector indicating the direction indicating
277  up. This is usually the zaxis. If it is not the z axis then lipids
278  will not be returned as stray.
279  */
280 int lipid_orientation( int id, Particle* partCfg , double zref, double director[3], double refdir[3]) {
281  int mol_size, head_id, tail_id, mol_id;
282  int i;
283  int tmpxdir,tmpydir,tmpzdir;
284  double distance;
285  double tailz;
286  double len;
287  double proj;
288 
289  if ( zdir + ydir + zdir != -3 ) {
290  tmpxdir = xdir;
291  tmpydir = ydir;
292  tmpzdir = zdir;
293  } else {
294  tmpxdir = 0;
295  tmpydir = 1;
296  tmpzdir = 2;
297  }
298 
299 
300  /* check molecule information exists */
301  if ( n_molecules < 0 ) return ES_ERROR;
302 
303  /* Get basic molecule parameters */
304  mol_id = partCfg[id].p.mol_id ;
305  mol_size = topology[mol_id].part.n;
306 
307  /* If the head and tail id's were not found above then assume the
308  head atom is the first and tail is the last in the molecule */
309  head_id = topology[mol_id].part.e[0];
310  tail_id = topology[mol_id].part.e[mol_size -1];
311 
312  /* Check for stray lipids only if the system is a flat bilayer and the director is in the zdirection */
313  if ( ( refdir[tmpxdir] == 0.0 ) && ( refdir[tmpydir] == 0 ) ) {
314  tailz = partCfg[tail_id].r.p[tmpzdir];
315  distance = sqrt(pow((tailz - zref),2));
316  /* Check if the lipid is a stray based on this distance */
317  if ( distance > stray_cut_off ) {
318  return REAL_LIPID_STRAY;
319  }
320  }
321 
322  /* Calculate a normalised vector called director that is in the
323  direction of the lipid from tail to head */
324  len = 0;
325  for ( i = 0 ; i < 3 ; i++ ) {
326  director[i] = (partCfg[head_id].r.p[i] -
327  partCfg[tail_id].r.p[i]);
328  len += director[i]*director[i];
329  }
330  for ( i = 0 ; i < 3 ; i++ ) {
331  director[i] = director[i]/sqrt(len);
332  }
333 
334 
335  proj = director[0]*refdir[0] + director[1]*refdir[1] + director[2]*refdir[2];
336  if ( proj > 0.0 ) {
337  /* Lipid is oriented up */
338  return LIPID_UP;
339  } else {
340  return LIPID_DOWN;
341  }
342 }
