161 /** Tune P3M parameters to desired accuracy.
162 
163  Usage:
164  \verbatim inter coulomb <bjerrum> p3m tune accuracy <value> [r_cut <value> mesh <value> cao <value>] \endverbatim
165 
166  The parameters are tuned to obtain the desired accuracy in best
167  time, by running mpi_integrate(0) for several parameter sets.
168 
169  The function utilizes the analytic expression of the error estimate
170  for the P3M method in the book of Hockney and Eastwood (Eqn. 8.23) in
171  order to obtain the rms error in the force for a system of N randomly
172  distributed particles in a cubic box.
173  For the real space error the estimate of Kolafa/Perram is used.
174 
175  Parameter range if not given explicit values: For \ref p3m_parameter_struct::r_cut_iL
176  the function uses the values (\ref min_local_box_l -\ref #skin) /
177  (n * \ref box_l), n being an integer (this implies the assumption that \ref
178  p3m_parameter_struct::r_cut_iL is the largest cutoff in the system!). For \ref
179  p3m_parameter_struct::mesh the function uses the two values which matches best the
180  equation: number of mesh point = number of charged particles. For
181  \ref p3m_parameter_struct::cao the function considers all possible values.
182 
183  For each setting \ref p3m_parameter_struct::alpha_L is calculated assuming that the
184  error contributions of real and reciprocal space should be equal.
185 
186  After checking if the total error fulfils the accuracy goal the
187  time needed for one force calculation (including verlet list
188  update) is measured via \ref mpi_integrate (0).
189 
190  The function returns a log of the performed tuning.
191 
192  The function is based on routines of the program HE_Q.cpp written by M. Deserno.
193  */

211 /** Calculate real space contribution of coulomb pair forces.
212  If NPT is compiled in, it returns the energy, which is needed for NPT. */
213 inline double p3m_add_pair_force(double chgfac, double *d,double dist2,double dist,double force[3])
214 {
215  int j;
216  double fac1,fac2, adist, erfc_part_ri;
217  if(dist < p3m.params.r_cut) {
218  if (dist > 0.0){ //Vincent
219  adist = p3m.params.alpha * dist;
220 #if USE_ERFC_APPROXIMATION
221  erfc_part_ri = AS_erfc_part(adist) / dist;
222  fac1 = coulomb.prefactor * chgfac * exp(-adist*adist);
223  fac2 = fac1 * (erfc_part_ri + 2.0*p3m.params.alpha*wupii) / dist2;
224 #else
225  erfc_part_ri = erfc(adist) / dist;
226  fac1 = coulomb.prefactor * chgfac;
227  fac2 = fac1 * (erfc_part_ri + 2.0*p3m.params.alpha*wupii*exp(-adist*adist)) / dist2;
228 #endif
229  for(j=0;j<3;j++)
230  force[j] += fac2 * d[j];
231  ESR_TRACE(fprintf(stderr,"%d: RSE: Pair dist=%.3f: force (%.3e,%.3e,%.3e)\n",this_node,
232  dist,fac2*d[0],fac2*d[1],fac2*d[2]));
233 #ifdef NPT
234  return fac1 * erfc_part_ri;
235 #endif
236  }
237  }
238  return 0.0;
239 }

254 /** Calculate real space contribution of coulomb pair energy. */
255 inline double p3m_pair_energy(double chgfac, double *d,double dist2,double dist)
256 {
257  double adist, erfc_part_ri;
258 
259  if(dist < p3m.params.r_cut) {
260  adist = p3m.params.alpha * dist;
261 #if USE_ERFC_APPROXIMATION
262  erfc_part_ri = AS_erfc_part(adist) / dist;
263  return coulomb.prefactor*chgfac*erfc_part_ri*exp(-adist*adist);
264 #else
265  erfc_part_ri = erfc(adist) / dist;
266  return coulomb.prefactor*chgfac*erfc_part_ri;
267 #endif
268  }
269  return 0.0;
270 }
