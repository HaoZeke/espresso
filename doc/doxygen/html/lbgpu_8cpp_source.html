252 /** (Re-)initializes the fluid. */
253 void lb_reinit_parameters_gpu() {
254  int ii;
255 
256  lbpar_gpu.time_step = (float)time_step;
257  for(ii=0;ii<LB_COMPONENTS;++ii){
258  lbpar_gpu.mu[ii] = 0.0;
259 
260  if (lbpar_gpu.viscosity[ii] > 0.0) {
261  /* Eq. (80) Duenweg, Schiller, Ladd, PRE 76(3):036704 (2007). */
262  lbpar_gpu.gamma_shear[ii] = 1. - 2./(6.*lbpar_gpu.viscosity[ii]*lbpar_gpu.tau/(lbpar_gpu.agrid*lbpar_gpu.agrid) + 1.);
263  }
264 
265  if (lbpar_gpu.bulk_viscosity[ii] > 0.0) {
266  /* Eq. (81) Duenweg, Schiller, Ladd, PRE 76(3):036704 (2007). */
267  lbpar_gpu.gamma_bulk[ii] = 1. - 2./(9.*lbpar_gpu.bulk_viscosity[ii]*lbpar_gpu.tau/(lbpar_gpu.agrid*lbpar_gpu.agrid) + 1.);
268  }
269 #ifdef SHANCHEN
270  if (lbpar_gpu.mobility[0] > 0.0) {
271  lbpar_gpu.gamma_mobility[0] = 1. - 2./(6.*lbpar_gpu.mobility[0]*lbpar_gpu.tau/(lbpar_gpu.agrid*lbpar_gpu.agrid) + 1.);
272  }
273 #endif
274  if (temperature > 0.0) { /* fluctuating hydrodynamics ? */
275 
276  lbpar_gpu.fluct = 1;
277  LB_TRACE (fprintf(stderr, "fluct on \n"));
278  /* Eq. (51) Duenweg, Schiller, Ladd, PRE 76(3):036704 (2007).*/
279  /* Note that the modes are not normalized as in the paper here! */
280  lbpar_gpu.mu[ii] = (float)temperature*lbpar_gpu.tau*lbpar_gpu.tau/c_sound_sq/(lbpar_gpu.agrid*lbpar_gpu.agrid);
281 
282  /* lb_coupl_pref is stored in MD units (force)
283  * Eq. (16) Ahlrichs and Duenweg, JCP 111(17):8225 (1999).
284  * The factor 12 comes from the fact that we use random numbers
285  * from -0.5 to 0.5 (equally distributed) which have variance 1/12.
286  * time_step comes from the discretization.
287  */
288 
289  lbpar_gpu.lb_coupl_pref[ii] = sqrt(12.f*2.f*lbpar_gpu.friction[ii]*(float)temperature/lbpar_gpu.time_step);
290  lbpar_gpu.lb_coupl_pref2[ii] = sqrt(2.f*lbpar_gpu.friction[ii]*(float)temperature/lbpar_gpu.time_step);
291 
292  } else {
293  /* no fluctuations at zero temperature */
294  lbpar_gpu.fluct = 0;
295  lbpar_gpu.lb_coupl_pref[ii] = 0.0;
296  lbpar_gpu.lb_coupl_pref2[ii] = 0.0;
297  }
298  LB_TRACE (fprintf(stderr,"lb_reinit_prarameters_gpu \n"));
299  }
300 
301 
302 #ifdef ELECTROKINETICS
303  if (ek_initialized) {
304  lbpar_gpu.dim_x = (unsigned int) round(box_l[0] / lbpar_gpu.agrid); //TODO code duplication with lb.c start
305  lbpar_gpu.dim_y = (unsigned int) round(box_l[1] / lbpar_gpu.agrid);
306  lbpar_gpu.dim_z = (unsigned int) round(box_l[2] / lbpar_gpu.agrid);
307 
308  unsigned int tmp[3];
309 
310  tmp[0] = lbpar_gpu.dim_x;
311  tmp[1] = lbpar_gpu.dim_y;
312  tmp[2] = lbpar_gpu.dim_z;
313 
314  /* sanity checks */
315  int dir;
316 
317  for (dir=0;dir<3;dir++) {
318  /* check if box_l is compatible with lattice spacing */
319  if (fabs(box_l[dir] - tmp[dir] * lbpar_gpu.agrid) > 1.0e-3) {
320  ostringstream msg;
321  msg <<"Lattice spacing lbpar_gpu.agrid= "<< lbpar_gpu.agrid << " is incompatible with box_l[" << dir << "]=" << box_l[dir];
322  runtimeError(msg);
323  }
324  }
325 
326  lbpar_gpu.number_of_nodes = lbpar_gpu.dim_x * lbpar_gpu.dim_y * lbpar_gpu.dim_z;
327  lbpar_gpu.tau = (float) time_step; //TODO code duplication with lb.c end
328  }
329 #endif
330 
331  LB_TRACE (fprintf(stderr,"lb_reinit_prarameters_gpu \n"));
332 
333  reinit_parameters_GPU(&lbpar_gpu);
334 }

357 int lb_lbnode_set_extforce_GPU(int ind[3], double f[3])
358 {
359  if ( ind[0] < 0 || ind[0] >= int(lbpar_gpu.dim_x) ||
360  ind[1] < 0 || ind[1] >= int(lbpar_gpu.dim_y) ||
361  ind[2] < 0 || ind[2] >= int(lbpar_gpu.dim_z) )
362  return ES_ERROR;
363 
364  unsigned int index =
365  ind[0] + ind[1]*lbpar_gpu.dim_x + ind[2]*lbpar_gpu.dim_x*lbpar_gpu.dim_y;
366 
367  size_t size_of_extforces = (n_extern_nodeforces+1)*sizeof(LB_extern_nodeforce_gpu);
368  host_extern_nodeforces = (LB_extern_nodeforce_gpu*) realloc(host_extern_nodeforces, size_of_extforces);
369 
370  host_extern_nodeforces[n_extern_nodeforces].force[0] = (float)f[0];
371  host_extern_nodeforces[n_extern_nodeforces].force[1] = (float)f[1];
372  host_extern_nodeforces[n_extern_nodeforces].force[2] = (float)f[2];
373 
374  host_extern_nodeforces[n_extern_nodeforces].index = index;
375  n_extern_nodeforces++;
376 
377  if(lbpar_gpu.external_force == 0)lbpar_gpu.external_force = 1;
378 
379  lb_init_extern_nodeforces_GPU(n_extern_nodeforces, host_extern_nodeforces, &lbpar_gpu);
380 
381  return ES_OK;
382 }
