24 /** \file ljangle.hpp
25  * Routines to calculate the lennard-jones 12-10 with angular dependance.
26  * The potential is a product of a 12-10 LJ potential with two cos^2.
27  * The potential actually relies on 6 particles: the 2 primary beads
28  * and each bead needs two other particles to define an orientation.
29  * We calculate the interaction explicitly *without* the use of the ROTATION feature.
30  *
31  * Optional: simulate two different environments in which the interaction
32  * strengths differ. For example: simulate hydrogen-bonds both in water and
33  * inside a bilayer. The two environments are distinguished by their
34  * z-position. Input: the midplane of the 2nd environment, its total
35  * thickness, the thickness of the interface, and the value of the
36  * interaction strength in this medium. The interaction strengh of the second
37  * environment must be *stronger* than of the first one.
38  *
39  * \ref forces.cpp
40  */

65 /** Calculate lj-angle force between particle p1 and p2
66  Involves 6 particles total */
67 inline void add_ljangle_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
68  double d[3], double dist)
69 {
70  if(!CUTOFF_CHECK(dist < ia_params->LJANGLE_cut))
71  return;
72  int j;
73  double frac2=0.0, frac10=0.0, rad=0.0, radprime=0.0;
74  double r31[3], r41[3], r52[3], r62[3], rij[3], rik[3], rkn[3];
75  double l_rij, l_rik, l_rkn, l_rij2, l_rik2, l_rkn2;
76  double cos_jik, cos_ikn;
77  double angular_jik, angular_ikn, angular_jik_prime, angular_ikn_prime;
78  /* Recreate angular dependence of potential by including 6 particles instead of 2. */
79  Particle *p3=NULL, *p4=NULL, *p5=NULL, *p6=NULL;
80  int part1p, part1n, part2p, part2n;
81  /* Optional 2nd environment */
82  double effective_eps=ia_params->LJANGLE_eps, z1, z2, z_middle, z_ref, z_ref5,
83  localz0=ia_params->LJANGLE_z0, localdz2=ia_params->LJANGLE_dz/2.,
84  localkappa6=pow(1/ia_params->LJANGLE_kappa,6);
85 
86 
87 
88  /* Retrieve the bonded partners from parsing */
89  if (ia_params->LJANGLE_bonded1type == p1->p.type) {
90  part1p = p1->p.identity + ia_params->LJANGLE_bonded1pos;
91  part1n = p1->p.identity + ia_params->LJANGLE_bonded1neg;
92  part2p = p2->p.identity + ia_params->LJANGLE_bonded2pos;
93  part2n = p2->p.identity + ia_params->LJANGLE_bonded2neg;
94  } else {
95  part1p = p1->p.identity + ia_params->LJANGLE_bonded2pos;
96  part1n = p1->p.identity + ia_params->LJANGLE_bonded2neg;
97  part2p = p2->p.identity + ia_params->LJANGLE_bonded1pos;
98  part2n = p2->p.identity + ia_params->LJANGLE_bonded1neg;
99  }
100 
101  if (part1p >= 0 && part1p < n_part &&
102  part1n >= 0 && part1n < n_part &&
103  part2p >= 0 && part2p < n_part &&
104  part2n >= 0 && part2n < n_part ) {
105 
106  p3 = local_particles[part1p];
107  p4 = local_particles[part1n];
108  p5 = local_particles[part2p];
109  p6 = local_particles[part2n];
110 
111  /* Check whether pointers have been allocated.
112  * Otherwise, there's a communication error (verlet skin too small).
113  */
114  if (p3==NULL ||p4==NULL ||p5==NULL ||p6==NULL)
115  fprintf(stderr, "LJANGLE - Communication error, all particles cannot be reached locally.\n");
116 
117 
118 
119  get_mi_vector(r31, p1->r.p, p3->r.p);
120  get_mi_vector(r41, p1->r.p, p4->r.p);
121  get_mi_vector(r52, p2->r.p, p5->r.p);
122  get_mi_vector(r62, p2->r.p, p6->r.p);
123 
124 
125  /* Bead i represents the central particle of monomer 1.
126  * Bead j is the virtual particle that gives the orientation of monomer 1.
127  * Bead k represents the central particle of monomer 2.
128  * Bead n is the virtual particle that gives the orientation of monomer 2.
129  */
130  for(j=0;j<3;++j){
131  rij[j] = r31[j] + r41[j];
132  rik[j] = -d[j]; /* At this point, rik[3] has length dist */
133  rkn[j] = r52[j] + r62[j];
134  }
135 
136  l_rij2 = sqrlen(rij);
137  l_rik2 = dist*dist;
138  l_rkn2 = sqrlen(rkn);
139  l_rij = sqrt(l_rij2);
140  l_rik = dist;
141  l_rkn = sqrt(l_rkn2);
142 
143  cos_jik = scalar(rij,rik)/(l_rij*l_rik);
144  cos_ikn = -scalar(rik,rkn)/(l_rik*l_rkn);
145 
146  if(cos_jik>0. && cos_ikn>0.) {
147  angular_jik = pow(cos_jik,2);
148  angular_ikn = pow(cos_ikn,2);
149  angular_jik_prime = -2*cos_jik;
150  angular_ikn_prime = -2*cos_ikn;
151 
152  /* Optional 2nd environment */
153  if (localdz2 > 0.) {
154  /* calculate center position of the interaction (calculate minimal distance) */
155  z1 = p1->r.p[2];
156  z2 = p2->r.p[2];
157  z_middle = z1 + z2;
158  z_middle /= 2.;
159  if (PERIODIC(2))
160  if (z_middle > box_l[2] || z_middle < 0.)
161  z_middle -= dround(z_middle *box_l_i[2])*box_l[2];
162 
163  /* If we're in the environment #2 region, calculate the new interaction strength */
164  if (z_middle > localz0 - localdz2 && z_middle <= localz0) {
165  z_ref = z_middle - localz0 + localdz2;
166  z_ref5 = pow(z_ref,5);
167  effective_eps += (ia_params->LJANGLE_epsprime - ia_params->LJANGLE_eps) *
168  localkappa6*z_ref5*fabs(z_ref) /
169  (1 + localkappa6*z_ref5*z_ref);
170  } else if (z_middle > localz0 && z_middle < localz0 + localdz2) {
171  z_ref = z_middle - localz0 - localdz2;
172  z_ref5 = pow(z_ref,5);
173  effective_eps -= (ia_params->LJANGLE_epsprime - ia_params->LJANGLE_eps) *
174  localkappa6*z_ref5*fabs(z_ref) /
175  (1 + localkappa6*z_ref5*z_ref);
176  }
177  }
178 
179  /* normal case: resulting force/energy smaller than capping. */
180  if(dist > ia_params->LJANGLE_capradius) {
181  frac2 = SQR(ia_params->LJANGLE_sig/dist);
182  frac10 = frac2*frac2*frac2*frac2*frac2;
183  rad = effective_eps * frac10*(5.0 * frac2 - 6.0);
184  radprime = 60.0 * effective_eps * frac10*(1.0 - frac2) / dist;
185 
186 #ifdef LJ_WARN_WHEN_CLOSE
187  if(radprime > 1000) fprintf(stderr,"%d: LJANGLE-Warning: Pair (%d-%d) force=%f dist=%f\n",
188  this_node,p1->p.identity,p2->p.identity,radprime,dist);
189 #endif
190  }
191  /* capped part of lj-angle potential. */
192  else if(dist > 0.0) {
193  /* set the length of rik to capradius */
194  for (j=0;j<3;++j)
195  rik[j] *= ia_params->LJANGLE_capradius/dist;
196 
197  l_rik = ia_params->LJANGLE_capradius;
198 
199  frac2 = SQR(ia_params->LJANGLE_sig/ia_params->LJANGLE_capradius);
200  frac10 = frac2*frac2*frac2*frac2*frac2;
201 
202  rad = effective_eps * frac10*(5.0 * frac2 - 6.0);
203  radprime = 60.0 * effective_eps * frac10*(1.0 - frac2) / (ia_params->LJANGLE_capradius);
204  }
205 
206  /* Regroup the last two cases in one */
207  /* Propagate all forces in this function rather than in the forces.hpp file */
208  if (dist > 0.0){
209  for(j=0;j<3;++j){
210  p1->f.f[j] += angular_jik * angular_ikn * rik[j]/l_rik *radprime
211  + angular_ikn * rad * angular_jik_prime
212  * ( ( 2*rik[j]-rij[j] )/( l_rij*l_rik )
213  - cos_jik*( 2*rij[j]/l_rij2 - rik[j]/l_rik2 ) )
214  + angular_jik * rad * angular_ikn_prime
215  * ( rkn[j]/( l_rik*l_rkn )
216  + cos_ikn*rik[j]/l_rik2 );
217  p2->f.f[j] += -angular_jik * angular_ikn * rik[j]/l_rik *radprime
218  + angular_ikn * rad * angular_jik_prime
219  * ( rij[j]/( l_rij*l_rik )
220  - cos_jik*rik[j]/l_rik2 )
221  + angular_jik * rad * angular_ikn_prime
222  * ( -(2*rik[j]+rkn[j])/(l_rik*l_rkn)
223  - cos_ikn*( 2*rkn[j]/l_rkn2 + rik[j]/l_rik2 ) );
224  p3->f.f[j] += angular_ikn * rad * angular_jik_prime
225  * ( -rik[j]/(l_rij*l_rik)
226  + cos_jik*rij[j]/l_rij2 );
227  p4->f.f[j] += angular_ikn * rad * angular_jik_prime
228  * ( -rik[j]/(l_rij*l_rik)
229  + cos_jik*rij[j]/l_rij2 );
230  p5->f.f[j] += angular_jik * rad * angular_ikn_prime
231  * ( rik[j]/(l_rik*l_rkn)
232  + cos_ikn*rkn[j]/l_rkn2 );
233  p6->f.f[j] += angular_jik * rad * angular_ikn_prime
234  * ( rik[j]/(l_rik*l_rkn)
235  + cos_ikn*rkn[j]/l_rkn2 );
236  }
237  }
238 
239 
240  /* this should not happen! In this case consider only radial potential*/
241  else {
242  LJ_TRACE(fprintf(stderr, "%d: LJ-angle warning: Particles id1=%d id2=%d exactly on top of each other\n",this_node,p1->p.identity,p2->p.identity));
243 
244  frac2 = SQR(ia_params->LJANGLE_sig/ia_params->LJANGLE_capradius);
245  frac10 = frac2*frac2*frac2*frac2*frac2;
246  rad = effective_eps * frac10*(5.0 * frac2 - 6.0) / ia_params->LJANGLE_capradius;
247 
248  p1->f.f[0] += rad * ia_params->LJANGLE_capradius;
249  }
250 
251 
252  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: LJANGLE f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist,radprime));
253  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: LJANGLE f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist,radprime));
254 
255  LJ_TRACE(fprintf(stderr,"%d: LJANGLE: Pair (%d-%d) dist=%.3f: force+-: (%.3e,%.3e,%.3e)\n",
256  this_node,p1->p.identity,p2->p.identity,dist,rad*d[0],rad*d[1],rad*d[2]));
257  }
258  }
259 }

262 inline double ljangle_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
263  double d[3], double dist)
264 {
265  if(!CUTOFF_CHECK(dist < ia_params->LJANGLE_cut))
266  return 0.0;
267 
268  int j;
269  double frac2, frac10;
270  double r31[3], r41[3], r52[3], r62[3], rij[3], rik[3], rkn[3];
271  double l_rij, l_rik, l_rkn;
272  double cos_jik, cos_ikn;
273  double angular_jik, angular_ikn;
274  /* Recreate angular dependence of potential by including 6 particles instead of 2. */
275  Particle *p3=NULL, *p4=NULL, *p5=NULL, *p6=NULL;
276  int part1p, part1n, part2p, part2n;

279  localz0=ia_params->LJANGLE_z0, localdz2=ia_params->LJANGLE_dz/2.,
280  localkappa6=pow(1/ia_params->LJANGLE_kappa,6);
281 
282 
283  /* Retrieve the bonded partners from parsing */
284  if (ia_params->LJANGLE_bonded1type == p1->p.type) {
285  part1p = p1->p.identity + ia_params->LJANGLE_bonded1pos;
286  part1n = p1->p.identity + ia_params->LJANGLE_bonded1neg;
287  part2p = p2->p.identity + ia_params->LJANGLE_bonded2pos;
288  part2n = p2->p.identity + ia_params->LJANGLE_bonded2neg;
289  } else {
290  part1p = p1->p.identity + ia_params->LJANGLE_bonded2pos;
291  part1n = p1->p.identity + ia_params->LJANGLE_bonded2neg;
292  part2p = p2->p.identity + ia_params->LJANGLE_bonded1pos;
293  part2n = p2->p.identity + ia_params->LJANGLE_bonded1neg;
294  }
295 
296  if (part1p >= 0 && part1p < n_part &&
297  part1n >= 0 && part1n < n_part &&
298  part2p >= 0 && part2p < n_part &&
299  part2n >= 0 && part2n < n_part ) {
300 
301  p3 = local_particles[part1p];
302  p4 = local_particles[part1n];
303  p5 = local_particles[part2p];
304  p6 = local_particles[part2n];
305 
306  get_mi_vector(r31, p1->r.p, p3->r.p);
307  get_mi_vector(r41, p1->r.p, p4->r.p);
308  get_mi_vector(r52, p2->r.p, p5->r.p);
309  get_mi_vector(r62, p2->r.p, p6->r.p);
310 
311  for(j=0;j<3;++j){
312  rij[j] = r31[j] + r41[j];
313  rik[j] = -d[j]; /* At this point, rik[3] has length dist */
314  rkn[j] = r52[j] + r62[j];
315  }
316 
317  l_rij = sqrt(sqrlen(rij));
318  l_rik = dist;
319  l_rkn = sqrt(sqrlen(rkn));
320 
321  cos_jik = scalar(rij,rik)/(dist*l_rij);
322  cos_ikn = -scalar(rik,rkn)/(l_rik*l_rkn);
323 
324  angular_jik = pow(cos_jik,2);
325  angular_ikn = pow(cos_ikn,2);
326 
327  if(cos_jik>0. && cos_ikn>0.) {
328  /* Optional 2nd environment */
329  if (localdz2 > 0.) {
330  /* calculate center position of the interaction (calculate minimal distance) */
331  z1 = p1->r.p[2];
332  z2 = p2->r.p[2];
333  z_middle = z1 + z2;
334  z_middle /= 2.;
335  if (PERIODIC(2))
336  if (z_middle > box_l[2] || z_middle < 0.)
337  z_middle -= dround(z_middle *box_l_i[2])*box_l[2];
338 
339  /* If we're in the environment #2 region, calculate the new interaction strength */
340  if (z_middle > localz0-localdz2 && z_middle <= localz0) {
341  z_ref = z_middle - localz0 + localdz2;
342  z_ref5 = pow(z_ref,5);
343  effective_eps += (ia_params->LJANGLE_epsprime - ia_params->LJANGLE_eps) *
344  localkappa6 * z_ref5 * fabs(z_ref) /
345  (1 + localkappa6 * z_ref5 * z_ref);
346  } else if (z_middle > localz0 && z_middle < localz0+localdz2) {
347  z_ref = z_middle - localz0 - localdz2;
348  z_ref5 = pow(z_ref,5);
349  effective_eps -= (ia_params->LJANGLE_epsprime - ia_params->LJANGLE_eps) *
350  localkappa6 * z_ref5 * fabs(z_ref) /
351  (1 + localkappa6 * z_ref5 * z_ref);
352  }
353  }
354 
355  /* normal case: resulting force/energy smaller than capping. */
356  if(dist > ia_params->LJANGLE_capradius) {
357  frac2 = SQR(ia_params->LJANGLE_sig/dist);
358  frac10 = frac2*frac2*frac2*frac2*frac2;
359  return effective_eps * frac10*(5.0 * frac2 - 6.0) * angular_jik * angular_ikn;
360  }
361  /* capped part of lj potential. */
362  else if(dist > 0.0) {
363  frac2 = SQR(ia_params->LJANGLE_sig/ia_params->LJANGLE_capradius);
364  frac10 = frac2*frac2*frac2*frac2*frac2;
365  return effective_eps * frac10*(5.0 * frac2 - 6.0) * angular_jik * angular_ikn;
366  }
367  /* this should not happen! In this case consider only radial potential*/
368  else {
369  frac2 = SQR(ia_params->LJANGLE_sig/ia_params->LJANGLE_capradius);
370  frac10 = frac2*frac2*frac2*frac2*frac2;
371  return effective_eps * frac10*(5.0 * frac2 - 6.0);
372  }
373  }
374  }
375 
376  return 0.0;
377 }
