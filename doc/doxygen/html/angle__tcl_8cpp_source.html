31 /// parse parameters for the angle potential
32 int tclcommand_inter_parse_angle(Tcl_Interp *interp, int bond_type, int argc, char **argv)
33 {
34  double bend, phi0;
35 
36  /* the optional parameter phi0 is due to backwards compatibility and is set to PI if not given */
37  if (argc != 2 && argc != 3) {
38  Tcl_AppendResult(interp, "angle needs 1 or 2 parameters: "
39  "<bend> [<phi0>]", (char *) NULL);
40  return (TCL_ERROR);
41  }
42 
43  if (! ARG_IS_D(1, bend)) {
44  Tcl_AppendResult(interp, "angle needs a DOUBLE parameter: "
45  "<bend> ", (char *) NULL);
46  return TCL_ERROR;
47  }
48 
49  /* special treatment of the optional parameter phi0 */
50  if (argc == 3) {
51  if (! ARG_IS_D(2, phi0)) {
52  Tcl_AppendResult(interp, "angle needs a DOUBLE parameter: "
53  "<phi0> ", (char *) NULL);
54  return TCL_ERROR;
55  }
56  } else {
57  phi0 = PI;
58  }
59  CHECK_VALUE(angle_set_params(bond_type, bend, phi0), "bond type must be nonnegative");
60 }
