261 void CalcForceGompper(Particle *xi, const int numNeighbors, Particle **const neighbors, const double kb)
262 {
263 // if ( xi->r.p[0] > maxX ) maxX = xi->r.p[0];
264 
265  // DEBUG stuff
266 /* for (int i=0; i < numNeighbors; i++)
267  {
268  Vector3D tmp;
269  Subtr(tmp, xi, neighbors[i]);
270  const double dist = Length(tmp);
271  if ( dist > maxBendingDist ) maxBendingDist = dist;
272  }*/
273 
274  // mainNumerator: Will be set to one of the terms in the numerator.
275  Vector3D mainNumerator;
276  mainNumerator.el[0] = mainNumerator.el[1] = mainNumerator.el[2] = 0;
277 
278  // Will be set to the denominator.
279  double denominator = 0;
280 
281  // We want to differentiate with respect to the i'th node ("virtual" index numDerivatives-1) and with respect to the neighbours of the i'th node (indecies 0<=l<=numDerivatives-2).
282  const int numDerivatives = numNeighbors + 1;
283 
284  // The derivatives of the numerator in the energy with respect to the i'th node and its neighbours.
285  Matrix3D *derivNumerators = new Matrix3D[numDerivatives];
286 
287  // The derivatives of the denominator in the energy with respect to the i'th node and its neighbours.
288  Vector3D *derivDenominators = new Vector3D[numDerivatives];
289 
290  for (int i=0; i < numDerivatives; i++)
291  {
292  for ( int k=0; k < 3; k++)
293  {
294  derivDenominators[i].el[k] = 0;
295  for (int l=0; l < 3; l++)
296  derivNumerators[i].el[k][l] = 0;
297  }
298  }
299 
300  // We need to calculate several sums over the neighbouring sites of i. This is done in the for-loop.
301  for (int curNeighborIdx = 0; curNeighborIdx < numNeighbors; ++curNeighborIdx)
302  {
303  // Get the neighbours of xi.
304  const Particle *xj = neighbors[curNeighborIdx];
305 
306 
307  // Neighbor list with periodic mapping
308  const Particle *xjM1 = (curNeighborIdx == 0) ? neighbors[numNeighbors-1] : neighbors[curNeighborIdx-1];
309  const Particle *xjP1 = (curNeighborIdx == numNeighbors-1) ? neighbors[0] : neighbors[curNeighborIdx+1];
310 
311  Vector3D tmp;
312  Subtr(tmp, xi, xj);
313  const double dxijLen2 = LengthSqr(tmp);
314 
315  // Cosine of the angles between the neighbours.
316  const double cosThetaM1 = CalcCosTheta(xi, xj, xjM1);
317  const double cosThetaP1 = CalcCosTheta(xi, xj, xjP1);
318 
319  // The sum of the two cotangens.
320  const double Tij = CalcCot(cosThetaM1) + CalcCot(cosThetaP1);
321 
322  Vector3D ijDifference;
323  Subtr(ijDifference, xi, xj);
324 
325  // Update the two terms which are independent of the node xl (the derivatives are with respect to node xl).
326  Multiply(tmp, ijDifference, Tij);
327  AddTo( mainNumerator, tmp);
328  denominator += dxijLen2 * Tij;
329 
330 
331  // Now for the other terms, dependent on xl.
332  // TODO: The only terms which are not 0 are jM1, j, jP1 and i
333  for (int gradientNodeIdx = 0; gradientNodeIdx < numDerivatives; ++gradientNodeIdx)
334  {
335 
336  const Particle *const xl = (gradientNodeIdx == numDerivatives-1) ? xi : neighbors[gradientNodeIdx];
337 
338  // Derivatives of the two cotangens.
339  Vector3D TijDeriv;
340  CalcCotDerivativeGompperAnalyt(TijDeriv, xi, xj, xjM1, xl->p.identity);
341  Vector3D tmp;
342  CalcCotDerivativeGompperAnalyt(tmp, xi, xj, xjP1, xl->p.identity);
343  AddTo( TijDeriv, tmp);
344 
345  // Kronecker Deltas.
346  const int ilDelta = (xi->p.identity == xl->p.identity) ? 1 : 0;
347  const int jlDelta = (xj->p.identity == xl->p.identity) ? 1 : 0;
348 
349  // Update
350  Matrix3D tmpM;
351  DyadicProduct(tmpM, ijDifference, TijDeriv);
352  AddTo( derivNumerators[gradientNodeIdx], tmpM );
353  AddScalarTo( derivNumerators[gradientNodeIdx], Tij*(ilDelta-jlDelta) );
354 
355  Multiply(tmp, ijDifference, Tij*2. * (ilDelta-jlDelta) );
356  AddTo( derivDenominators[gradientNodeIdx], tmp );
357  Multiply(tmp, TijDeriv, dxijLen2);
358  AddTo( derivDenominators[gradientNodeIdx], tmp );
359  }
360  }
361 
362 
363 
364  // Calculate the contribution to the force for each node.
365  for (int gradientNodeIdx = 0; gradientNodeIdx < numDerivatives; ++gradientNodeIdx)
366  {
367  Particle *const xl = (gradientNodeIdx == numDerivatives-1) ? xi : neighbors[gradientNodeIdx];
368 
369  // -1: The force is minus the gradient of the energy.
370  // Note: left vector-matrix product: mainNumerator * derivNumerators[gradientNodeIdx]
371  Vector3D force, tmp;
372  LeftVectorMatrix(force, mainNumerator, derivNumerators[gradientNodeIdx]);
373  Multiply(force, force, 4.0 / denominator );
374  Multiply(tmp, derivDenominators[gradientNodeIdx], - 2.0 * LengthSqr(mainNumerator) / (denominator*denominator));
375  AddTo(force, tmp);
376 
377  Multiply(force, force, (-1.0) * (kb / 2.0));
378 
379 // if ( xl->p.identity == 0)
380 // printf(" Adding to node = %d when treating node %d: f = %.12e %.12e %.12e\n", xl->p.identity, xi->p.identity, force.el[0], force.el[1], force.el[2]);
381  xl->f.f[0] += force.el[0];
382  xl->f.f[1] += force.el[1];
383  xl->f.f[2] += force.el[2];
384 
385  // DEBUG stuff
386 /* for (int i=0; i < numNeighbors; i++)
387  {
388  const double f = Length(force);
389  if ( f > maxBendingForce ) maxBendingForce = f;
390  }*/
391 
392  }
393 
394  // Clean up
395  delete []derivNumerators;
396  delete []derivDenominators;
