32 int angledist_set_params(int bond_type, double bend, double phimin, double distmin, double phimax, double distmax)
33 {
34  if(bond_type < 0)
35  return ES_ERROR;
36 
37  make_bond_type_exist(bond_type);
38

40  bonded_ia_params[bond_type].p.angledist.phimin = phimin;
41  bonded_ia_params[bond_type].p.angledist.distmin = distmin;
42  bonded_ia_params[bond_type].p.angledist.phimax = phimax;

59 /** Function to calculate wall distance and phi0(dist).
60  Called by \ref calc_angledist_force */
61 static double calc_angledist_param(Particle *p_mid, Particle *p_left,
62  Particle *p_right, Bonded_ia_parameters *iaparams)
63 {
64  double cosine=0.0, vec1[3], vec2[3], d1i=0.0, d2i=0.0, dist1=0.0, dist2=0.0, phi0=0.0;
65  // double pwdist=0.0, pwdist0=0.0, pwdist1=0.0;
66  double normal, folded_pos[3], phimn=0.0, distmn=0.0, phimx=0.0, distmx=0.0, drange=0.0;
67  double pwdist[n_constraints],pwdistmin=0.0;
68  Constraint_wall wall;
69  int j, k;
70  int img[3];
71 
72  wall.d = 0;
73 
74  /* vector from p_left to p_mid */
75  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
76  dist1 = sqrlen(vec1);
77  d1i = 1.0 / sqrt(dist1);
78  for(j=0;j<3;j++) vec1[j] *= d1i;
79  /* vector from p_mid to p_right */
80  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
81  dist2 = sqrlen(vec2);
82  d2i = 1.0 / sqrt(dist2);
83  for(j=0;j<3;j++) vec2[j] *= d2i;
84  /* vectors are normalised so cosine is just cos(angle_between_vec1_and_vec2) */
85  cosine = scalar(vec1, vec2);
86  if ( cosine > TINY_COS_VALUE) cosine = TINY_COS_VALUE;
87  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
88  // fac = iaparams->p.angledist.bend; /* spring constant from .tcl file */
89  phimn = iaparams->p.angledist.phimin;
90  distmn = iaparams->p.angledist.distmin;
91  phimx = iaparams->p.angledist.phimax;
92  distmx = iaparams->p.angledist.distmax;
93 
94  /* folds coordinates of p_mid into original box */
95  memmove(folded_pos, p_mid->r.p, 3*sizeof(double));
96  memmove(img, p_mid->l.i, 3*sizeof(int));
97  fold_position(folded_pos, img);
98 
99  /* Calculates distance between p_mid and constraint */
100  for(k=0;k<n_constraints;k++) {
101  pwdist[k]=0.0;
102  }
103  for(k=0;k<n_constraints;k++) {
104  if (constraints[k].type == CONSTRAINT_WAL) {
105 
106  /* dist is distance of wall from origin */
107  wall=constraints[k].c.wal;
108 
109  /* check that constraint vector is normalised */
110  normal=0.0;
111  for(j=0;j<3;j++) normal += wall.n[j] * wall.n[j];
112  if (sqrt(normal) != 1.0) {
113  for(j=0;j<3;j++) wall.n[j]=wall.n[j]/normal;
114  }
115 
116  /* pwdist is distance of wall from p_mid */
117  pwdist[k]=-1.0 * constraints[k].c.wal.d;
118  for(j=0;j<3;j++) {
119  pwdist[k] += folded_pos[j] * constraints[k].c.wal.n[j];
120  }
121  if (k==0) {
122  pwdistmin=pwdist[k];
123  }
124  if (pwdist[k] <= pwdistmin) {
125  pwdistmin = pwdist[k];
126  }
127  }
128  }
129 
130  /*get phi0(z)*/
131  if (pwdistmin <= distmn) {
132  phi0 = phimn;
133  // fprintf(stdout,"\nIn angledist_set_params: z_p_mid=%f pwdistmin=%f distmn=%f ",folded_pos[2],pwdistmin,distmn);
134  // fprintf(stdout," phi0=%f\n",phi0*180.0/PI);
135  }
136  else if (pwdistmin >= distmx && pwdistmin <= box_l[2]-wall.d-distmx) {
137  phi0 = phimx;
138  }
139  else {
140  drange = (pwdistmin-distmn)*PI/(distmx-distmn);
141  phi0 = ((cos(drange-PI)+1.0)*(phimx-phimn))*0.5+phimn;
142  // fprintf(stdout,"\nIn angledist_set_params: z_p_mid=%f pwdistmin=%f box_lz/2=%f ",folded_pos[2],pwdistmin,box_l[2]/2.0);
143  // fprintf(stdout," phi0=%f\n",phi0*180.0/PI);
144  }
145 
146  return phi0;
147 }

149 int calc_angledist_force(Particle *p_mid, Particle *p_left,
150  Particle *p_right,
151  Bonded_ia_parameters *iaparams,
152  double force1[3], double force2[3])
153 {
154  double cosine=0.0, vec1[3], vec2[3], d1i=0.0, d2i=0.0, dist2, fac=0.0, f1=0.0, f2=0.0;
155  int j;
156 
157  /* vector from p_left to p_mid */
158  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
159  dist2 = sqrlen(vec1);
160  d1i = 1.0 / sqrt(dist2);
161  for(j=0;j<3;j++) vec1[j] *= d1i;
162  /* vector from p_mid to p_right */
163  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
164  dist2 = sqrlen(vec2);
165  d2i = 1.0 / sqrt(dist2);
166  for(j=0;j<3;j++) vec2[j] *= d2i;
167  /* scalar product of vec1 and vec2 */
168  cosine = scalar(vec1, vec2);
169  fac = iaparams->p.angledist.bend;
170 
171 #ifdef BOND_ANGLEDIST_HARMONIC
172  {
173  double phi,sinphi;
174  /* NOTE: The angledist is ONLY implemented for the HARMONIC case */
175  double phi0 = calc_angledist_param(p_mid, p_left, p_right, iaparams);
176 
177  if ( cosine > TINY_COS_VALUE) cosine = TINY_COS_VALUE;
178  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
179  phi = acos(-cosine);
180  sinphi = sin(phi);
181  if ( sinphi < TINY_SIN_VALUE ) sinphi = TINY_SIN_VALUE;
182  fac *= (phi - phi0)/sinphi;
183  // fprintf(stdout,"\n force: z_pmid=%f, phi0=%f phi=%f fac=%f",p_mid->r.p[2],phi0*180.0/PI,phi*180.0/PI,fac);
184  }
185 #endif
186 
187 #ifdef BOND_ANGLEDIST_COSINE
188  #error angledist ONLY implemented for harmonic case!
189 #endif
190 #ifdef BOND_ANGLEDIST_COSSQUARE
191  #error angledist ONLY implemented for harmonic case!
192 #endif
193 
194  for(j=0;j<3;j++) {
195  f1 = fac * (cosine * vec1[j] - vec2[j]) * d1i;
196  f2 = fac * (cosine * vec2[j] - vec1[j]) * d2i;
197 
198  force1[j] = (f1-f2);
199  force2[j] = -f1;
200  }
201  return 0;
202 }
