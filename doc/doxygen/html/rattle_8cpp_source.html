128 /**Compute positional corrections*/
129 void compute_pos_corr_vec(int *repeat_)
130 {
131  Bonded_ia_parameters *ia_params;
132  int i, j, k, c, np, cnt=-1;
133  Cell *cell;
134  Particle *p, *p1, *p2;
135  double r_ij_t[3], r_ij[3], r_ij_dot, G, pos_corr, r_ij2;
136 
137  for (c = 0; c < local_cells.n; c++) {
138  cell = local_cells.cell[c];
139  p = cell->part;
140  np = cell->n;
141  for(i = 0; i < np; i++) {
142  p1 = &p[i];
143  k=0;
144  while(k<p1->bl.n) {
145  ia_params = &bonded_ia_params[p1->bl.e[k++]];
146  if( ia_params->type == BONDED_IA_RIGID_BOND ) {
147  cnt++;
148  p2 = local_particles[p1->bl.e[k++]];
149  if (!p2) {
150  ostringstream msg;
151  msg <<"rigid bond broken between particles " << p1->p.identity << " and " << p1->bl.e[k-1] << " (particles not stored on the same node)";
152  runtimeError(msg);
153  return;
154  }
155 
156  get_mi_vector(r_ij , p1->r.p , p2->r.p );
157  r_ij2 = sqrlen(r_ij);
158  if(fabs(1.0 - r_ij2/ia_params->p.rigid_bond.d2) > ia_params->p.rigid_bond.p_tol) {
159  get_mi_vector(r_ij_t, p1->r.p_old, p2->r.p_old);
160  r_ij_dot = scalar(r_ij_t, r_ij);
161  G = 0.50*(ia_params->p.rigid_bond.d2 - r_ij2 )/r_ij_dot;
162 #ifdef MASS
163  G /= (PMASS(*p1)+PMASS(*p2));
164 #else
165  G /= 2;
166 #endif
167  for (j=0;j<3;j++) {
168  pos_corr = G*r_ij_t[j];
169  p1->f.f[j] += pos_corr*PMASS(*p2);
170  p2->f.f[j] -= pos_corr*PMASS(*p1);
171  }
172  /*Increase the 'repeat' flag by one */
173  *repeat_ = *repeat_ + 1;
174  }
175  }
176  else
177  /* skip bond partners of nonrigid bond */
178  k+=ia_params->num;
179  } //while loop
180  } //for i loop
181  } //for c loop
182 }

280 /** Velocity correction vectors are computed*/
281 void compute_vel_corr_vec(int *repeat_)
282 {
283  Bonded_ia_parameters *ia_params;
284  int i, j, k, c, np;
285  Cell *cell;
286  Particle *p, *p1, *p2;
287  double v_ij[3], r_ij[3], K, vel_corr;
288 
289  for (c = 0; c < local_cells.n; c++)
290  {
291  cell = local_cells.cell[c];
292  p = cell->part;
293  np = cell->n;
294  for(i = 0; i < np; i++) {
295  p1 = &p[i];
296  k=0;
297  while(k<p1->bl.n)
298  {
299  ia_params = &bonded_ia_params[p1->bl.e[k++]];
300  if( ia_params->type == BONDED_IA_RIGID_BOND )
301  {
302  p2 = local_particles[p1->bl.e[k++]];
303  if (!p2) {
304  ostringstream msg;
305  msg <<"rigid bond broken between particles " << p1->p.identity << " and " << p1->bl.e[k-1] << " (particles not stored on the same node)";
306  runtimeError(msg);
307  return;
308  }
309 
310  vecsub(p1->m.v, p2->m.v, v_ij);
311  get_mi_vector(r_ij, p1->r.p, p2->r.p);
312  if(fabs(scalar(v_ij, r_ij)) > ia_params->p.rigid_bond.v_tol)
313  {
314  K = scalar(v_ij, r_ij)/ia_params->p.rigid_bond.d2;
315 #ifdef MASS
316  K /= (PMASS(*p1) + PMASS(*p2));
317 #else
318  K /= 2.0;
319 #endif
320  for (j=0;j<3;j++)
321  {
322  vel_corr = K*r_ij[j];
323  p1->f.f[j] -= vel_corr*PMASS(*p2);
324  p2->f.f[j] += vel_corr*PMASS(*p1);
325  }
326  *repeat_ = *repeat_ + 1 ;
327  }
328  }
329  else
330  k += ia_params->num;
331  } //while loop
332  } //for i loop
333  } //for c loop
334 }

409  if(this_node == 0)

412  MPI_Reduce(&repeat_, NULL, 1, MPI_INT, MPI_SUM, 0, comm_cart);

415  cnt++;
416  }
417

420  errexit();
421  }
422  /**Puts back the forces from r.p_old to f.f[3]*/
