136 /** Store the x-componenet of the velocity of particle part into the
137  nemd data structure \ref nemddata. */
138 inline void nemd_get_velocity(Particle part)
139 {
140  int slab_num, i;
141  Slab *slab;
142 
143  if(nemd_method == NEMD_METHOD_OFF ) return;
144  if(nemd_method == NEMD_METHOD_EXCHANGE && nemddata.n_exchange==0 ) return;
145 
146  /* calculate slab_num */
147  slab_num = part.r.p[2]*nemddata.invthickness;
148  /* THIS IS A SINGLE NODE CORRECTION !!! */
149  if(slab_num == nemddata.n_slabs) slab_num = 0;
150  if(part.r.p[2] < 0.0) slab_num = nemddata.n_slabs-1;
151  slab = &nemddata.slab[slab_num];
152 
153  /* Add velocity to mean velocity of that slab */
154  slab->v_mean += part.m.v[0];
155  slab->n_parts_in_slab ++;
156 
157  /* Collect the n_exchange particles with largest velocities for top
158  and middle slab if nemd method is exchange */
159  if(nemd_method == NEMD_METHOD_EXCHANGE) {
160  if(slab_num == nemddata.top_slab) {
161  /* top slab: look for largest negative x componenet */
162 
163  /* first fill fastest array as particles com in */
164  if(slab->n_fastest < nemddata.n_exchange) {
165  slab->fastest[slab->n_fastest] = part.p.identity;
166  if( part.m.v[0] > slab->v_min ) {
167  slab->v_min = part.m.v[0];
168  slab->ind_min = slab->n_fastest;
169  }
170  slab->n_fastest++;
171  }
172  /* then replace the slowest with a new one if necessary... */
173  else if ( part.m.v[0] < slab->v_min ) {
174  slab->fastest[slab->ind_min] = part.p.identity;
175  /* ... and find again the slowest one now */
176  slab->v_min = local_particles[slab->fastest[0]]->m.v[0];
177  slab->ind_min = 0;
178  for(i=1;i<slab->n_fastest;i++)
179  if(local_particles[slab->fastest[i]]->m.v[0] > slab->v_min) {
180  slab->v_min = local_particles[slab->fastest[i]]->m.v[0];
181  slab->ind_min = i;
182  }
183  }
184  } else if(slab_num == nemddata.mid_slab) {
185  /* midlle slab: look for largest positive x componenet */
186 
187  /* first fill fastest array as particles com in */
188  if(slab->n_fastest < nemddata.n_exchange) {
189  slab->fastest[slab->n_fastest] = part.p.identity;
190  if( part.m.v[0] < slab->v_min ) {
191  slab->v_min = part.m.v[0];
192  slab->ind_min = slab->n_fastest;
193  }
194  slab->n_fastest++;
195  }
196  /* then replace the slowest with a new one if necessary... */
197  else if ( part.m.v[0] > slab->v_min) {
198  slab->fastest[slab->ind_min] = part.p.identity;
199  /* ... and find again the slowest one now */
200  slab->v_min = local_particles[slab->fastest[0]]->m.v[0];
201  slab->ind_min = 0;
202  for(i=1;i<slab->n_fastest;i++)
203  if(local_particles[slab->fastest[i]]->m.v[0] < slab->v_min) {
204  slab->v_min = local_particles[slab->fastest[i]]->m.v[0];
205  slab->ind_min = i;
206  }
207  }
208  }
209  }
210 }

212 inline void nemd_add_velocity (Particle *part) {
213  if( !(nemd_method == NEMD_METHOD_SHEARRATE) ) {
214  return;
215  } else {
216  int slab_num;
217  Slab *slab;
218 
219  slab_num = part->r.p[2]*nemddata.invthickness;
220  slab = &nemddata.slab[slab_num];
221 
222  if(slab_num == nemddata.top_slab || slab_num == nemddata.mid_slab) {
223  part->m.v[0] += slab->vel_diff;
224  }
225  }
