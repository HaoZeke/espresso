38 /* parser for necklace cluster analyzation:
39  analyze necklace <pearl_treshold> <back_dist> <space_dist> <first> <length>
40  */
41 int tclcommand_analyze_parse_necklace(Tcl_Interp *interp, int argc, char **argv)
42 {
43  double space_dist;
44  int first,length;
45  Particle *part;
46  Cluster *cluster;
47  char buffer[TCL_INTEGER_SPACE];
48  int n_pearls;
49 
50  /* check # of parameters */
51  if (argc < 5) {
52  Tcl_AppendResult(interp, "analyze necklace needs 5 parameters:\n", (char *)NULL);
53  Tcl_AppendResult(interp, "<pearl_treshold> <back_dist> <space_dist> <first> <length>", (char *)NULL);
54  return TCL_ERROR;
55  }
56 
57  /* check parameter types */
58  if( (! ARG_IS_I(0, pearl_treshold)) ||
59  (! ARG_IS_I(1, backbone_distance)) ||
60  (! ARG_IS_D(2, space_dist)) ||
61  (! ARG_IS_I(3, first)) ||
62  (! ARG_IS_I(4, length)) ) {
63  Tcl_AppendResult(interp, "analyze necklace needs 5 parameters of type and meaning:\n", (char *)NULL);
64  Tcl_AppendResult(interp, "INT INT DOUBLE INT INT\n", (char *)NULL);
65  Tcl_AppendResult(interp, "<pearl_treshold> <back_dist> <space_dist> <first> <length>", (char *)NULL);
66  return TCL_ERROR;
67  }
68 
69  /* check parameter values */
70  if( pearl_treshold < 10 ) {
71  Tcl_AppendResult(interp, "analyze necklace: pearl_treshold should be >= 10", (char *)NULL);
72  return TCL_ERROR;
73  }
74  if( backbone_distance < 2 ) {
75  Tcl_AppendResult(interp, "analyze necklace: backbone_dist should be >= 2", (char *)NULL);
76  return TCL_ERROR;
77  }
78  if( space_dist <= 0.0 ) {
79  Tcl_AppendResult(interp, "analyze necklace: space_dist must be positive", (char *)NULL);
80  return TCL_ERROR;
81  }
82  if( first < 0 ) {
83  Tcl_AppendResult(interp, "analyze necklace: identity of first particle can not be negative", (char *)NULL);
84  return TCL_ERROR;
85  }
86  if( first+length > n_part+1) {
87  Tcl_AppendResult(interp, "analyze necklace: identity of last particle out of partCfg array", (char *)NULL);
88  return TCL_ERROR;
89  }
90 
91  /* preparation */
92  space_distance2 = SQR(space_dist);
93  sortPartCfg();
94  part = &partCfg[first];
95 
96  /* perform necklace cluster algorithm */
97  n_pearls = analyze_necklace(part, length) ;
98 
99  /* Append result to tcl interpreter */
100  sprintf(buffer,"%d",n_pearls);
101  Tcl_AppendResult(interp, buffer, " pearls { ", (char *)NULL);
102  if( n_pearls > 0 ) {
103  cluster = first_cluster;
104  sprintf(buffer,"%d",cluster->size);
105  Tcl_AppendResult(interp, buffer, " ",(char *)NULL);
106  cluster = cluster->next;
107  while(cluster->prev != last_cluster) {
108  sprintf(buffer,"%d",cluster->size);
109  Tcl_AppendResult(interp, buffer, " ",(char *)NULL);
110  cluster = cluster->next;
111  }
112  }
113  Tcl_AppendResult(interp, "} ", (char *)NULL);
114 
115  /* free analyzation memory */
116  cluster_free();
117 
118  return (TCL_OK);
119 }

121 /* parser for hole cluster analyzation:
122  analyze holes <prob_part_type_number> <mesh_size>.
123  Needs feature LENNARD_JONES compiled in. */
124 int tclcommand_analyze_parse_holes(Tcl_Interp *interp, int argc, char **argv)
125 {
126  int i,j;
127  int probe_part_type;
128  int mesh_size=1, meshdim[3];
129  double freevol=0.0;
130  char buffer[TCL_INTEGER_SPACE+TCL_DOUBLE_SPACE];
131 
132  IntList mesh;
133 
134  int n_holes;
135  int **holes;
136  int max_size=0;
137  int *surface;
138 
139 #ifndef LENNARD_JONES
140  Tcl_AppendResult(interp, "analyze holes needs feature LENNARD_JONES compiled in.\n", (char *)NULL);
141  return TCL_ERROR;
142 #endif
143 
144  /* check # of parameters */
145  if (argc < 2) {
146  Tcl_AppendResult(interp, "analyze holes needs 2 parameters:\n", (char *)NULL);
147  Tcl_AppendResult(interp, "<prob_part_type_number> <mesh_size>", (char *)NULL);
148  return TCL_ERROR;
149  }
150 
151  /* check parameter types */
152  if( (! ARG_IS_I(0, probe_part_type)) ||
153  (! ARG_IS_I(1, mesh_size)) ) {
154  Tcl_AppendResult(interp, "analyze holes needs 2 parameters of type and meaning:\n", (char *)NULL);
155  Tcl_AppendResult(interp, "INT INT\n", (char *)NULL);
156  Tcl_AppendResult(interp, "<prob_part_type_number> <mesh_size>", (char *)NULL);
157  return TCL_ERROR;
158  }
159 
160  /* check parameter values */
161  if( probe_part_type > n_particle_types || probe_part_type < 0 ) {
162  Tcl_AppendResult(interp, "analyze holes: probe particle type number does not exist", (char *)NULL);
163  return TCL_ERROR;
164  }
165  if( mesh_size < 1 ) {
166  Tcl_AppendResult(interp, "analyze holes: mesh size must be positive (min=1)", (char *)NULL);
167  return TCL_ERROR;
168  }
169 
170  /* preparation */
171  updatePartCfg(WITHOUT_BONDS);
172  meshdim[0]=mesh_size;
173  meshdim[1]=mesh_size;
174  meshdim[2]=mesh_size;
175  alloc_intlist(&mesh, (meshdim[0]*meshdim[1]*meshdim[2]));
176 
177  /* perform free space identification*/
178  create_free_volume_grid(mesh, meshdim, probe_part_type);
179  /* perfrom hole cluster algorithm */
180  n_holes = cluster_free_volume_grid(mesh, meshdim, &holes);
181  /* surface to volume ratio */
182  surface = (int *) malloc(sizeof(int)*(n_holes+1));
183  cluster_free_volume_surface(mesh, meshdim, n_holes, holes, surface);
184  /* calculate accessible volume / max size*/
185  for ( i=0; i<=n_holes; i++ ) {
186  freevol += holes[i][0];
187  if ( holes[i][0]> max_size ) max_size = holes[i][0];
188  }
189 
190  /* Append result to tcl interpreter */
191  Tcl_AppendResult(interp, "{ n_holes mean_hole_size max_hole_size free_volume_fraction { sizes } { surfaces } { element_lists } } ", (char *)NULL);
192 
193  Tcl_AppendResult(interp, "{", (char *)NULL);
194 
195  /* number of holes */
196  sprintf(buffer,"%d ",n_holes+1); Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
197  /* mean hole size */
198  sprintf(buffer,"%f ",freevol/(n_holes+1.0)); Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
199  /* max hole size */
200  sprintf(buffer,"%d ",max_size); Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
201  /* free volume fraction */
202  sprintf(buffer,"%f ",freevol/(meshdim[0]*meshdim[1]*meshdim[2]));
203  Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
204  /* hole sizes */
205  Tcl_AppendResult(interp, "{ ", (char *)NULL);
206  for ( i=0; i<=n_holes; i++ ) {
207  sprintf(buffer,"%d ",holes[i][0]); Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
208  }
209  Tcl_AppendResult(interp, "} ", (char *)NULL);
210  /* hole surfaces */
211  Tcl_AppendResult(interp, "{ ", (char *)NULL);
212  for ( i=0; i<=n_holes; i++ ) {
213  sprintf(buffer,"%d ",surface[i]); Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
214  }
215  Tcl_AppendResult(interp, "} ", (char *)NULL);
216  /* hole elements */
217  Tcl_AppendResult(interp, "{ ", (char *)NULL);
218  for ( i=0; i<=n_holes; i++ ) {
219  Tcl_AppendResult(interp, "{ ", (char *)NULL);
220  for ( j=1; j <= holes[i][0]; j++ ) {
221  sprintf(buffer,"%d",holes[i][j]);
222  Tcl_AppendResult(interp, buffer, " ",(char *)NULL);
223  }
224  Tcl_AppendResult(interp, "} ", (char *)NULL);
225  }
226  Tcl_AppendResult(interp, "} ", (char *)NULL);
227 
228  Tcl_AppendResult(interp, "}", (char *)NULL);
229 
230  /* free allocated memory */
231  realloc_intlist(&mesh, 0);
232  free(surface);
233  for ( i=0; i<=n_holes; i++ ) { free(holes[i]); }
234  free(holes);
235 
236  return (TCL_OK);
237 }
