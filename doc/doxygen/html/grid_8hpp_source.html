23 /** \file grid.hpp Domain decomposition for parallel computing.
24  *
25  * The primary simulation box is divided into orthogonal rectangular
26  * subboxes which are assigned to the different nodes (or processes
27  * or threads if you want). This grid is described in \ref
28  * node_grid. Each node has a number \ref this_node and a position
29  * \ref node_pos in that grid. Each node has also 6 nearest neighbors
30  * \ref node_neighbors which are necessary for the communication
31  * between the nodes (see also \ref ghosts.cpp and \ref p3m.cpp for more
32  * details about the communication.
33  *
34  * For the 6 directions \anchor directions we have the following convention:
35  *
36  * \image html directions.gif "Convention for the order of the directions"
37  *
38  * The Figure illustrates the direction convetion used for arrays
39  * with 6 (e.g. \ref node_neighbors, \ref #boundary) and 3 entries
40  * (e.g \ref node_grid, \ref box_l , \ref my_left,...).
41  *
42  *
43  * For more information on the domain decomposition, see \ref grid.cpp "grid.c".
44 */

183 /** get the minimal distance vector of two vectors in the current bc.
184  * \ref LEES_EDWARDS note: there is no need to add the le_offset here,
185  * any offset should already have been added when the image particle was prepared.
186  * @param a the vector to subtract from
187  * @param b the vector to subtract
188  * @param res where to store the result
189 */
190 inline void get_mi_vector(double res[3], double a[3], double b[3])
191 {
192  int i;
193 
194  for(i=0;i<3;i++) {
195  res[i] = a[i] - b[i];
196  if (PERIODIC(i))
197  res[i] -= dround(res[i]*box_l_i[i])*box_l[i];
198  }
199 }

201 /** fold a coordinate to primary simulation box, including velocity (in case of LEES_EDWARDS,
202  the velocity depends on the image we are in).
203  \param pos the position...
204  \param vel the velocity...
205  \param image_box and the box
206  \param dir the coordinate to fold: dir = 0,1,2 for x, y and z coordinate.
207 
208  Both pos and image_box are I/O,
209  i. e. a previously folded position will be folded correctly.
210 */
211 inline void fold_coordinate(double pos[3], double vel[3], int image_box[3], int dir)
212 {
213  if (PERIODIC(dir)) {
214  int img_count = (int)floor(pos[dir]*box_l_i[dir]);
215  image_box[dir] += img_count;
216  pos[dir] = pos[dir] - img_count*box_l[dir];
217 
218  if(pos[dir]*box_l_i[dir] < -ROUND_ERROR_PREC || pos[dir]*box_l_i[dir] >= 1 + ROUND_ERROR_PREC) {
219  ostringstream msg;
220  msg << "particle coordinate out of range, pos = " << pos[dir] << ", image box = " << image_box[dir];
221  runtimeError(msg);
222  image_box[dir] = 0;
223  pos[dir] = 0;
224  return;
225  }
226 
227 #ifdef LEES_EDWARDS
228 
229  if (dir == 1) {
230  /* must image y and v_x at same time as x */
231  pos[0] -= (lees_edwards_offset * img_count);
232  vel[0] -= (lees_edwards_rate * img_count);
233 
234  /* (re)-image x */
235  img_count = (int)floor(pos[0]*box_l_i[0]);
236  image_box[0] += img_count;
237  pos[0] = pos[0] - img_count*box_l[0];
238  }
239 
240 #endif
241 
242  }
243 }

289 /** unfold coordinates to physical position.
290  \param pos the position
291  \param pos the velocity
292  \param image_box and the box
293 
294  Both pos and image_box are I/O, i.e. image_box will be (0,0,0)
295  afterwards.
296 */
297 inline void unfold_position(double pos[3], double vel[3], int image_box[3])
298 {
299 #ifdef LEES_EDWARDS
300 
301  int y_img_count;
302  y_img_count = (int)floor( pos[1]*box_l_i[1] + image_box[1] );
303 
304  pos[0] += image_box[0]*box_l[0] + y_img_count*lees_edwards_offset;
305  pos[1] += image_box[1]*box_l[1];
306  pos[2] += image_box[2]*box_l[2];
307 
308  vel[0] += y_img_count * lees_edwards_rate;
309 
310  image_box[0] = image_box[1] = image_box[2] = 0;
311 
312 #else
313 
314  int i;
315  for(i=0;i<3;i++) {
316  pos[i] = pos[i] + image_box[i]*box_l[i];
317  image_box[i] = 0;
318  }
319 
320 #endif
321 }
322
