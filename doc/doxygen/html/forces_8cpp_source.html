67 void init_forces()
68 {
69  Cell *cell;
70  Particle *p;
71  int np, c, i;
72 
73  /* The force initialization depends on the used thermostat and the
74  thermodynamic ensemble */
75 
76 #ifdef NPT
77  /* reset virial part of instantaneous pressure */
78  if(integ_switch == INTEG_METHOD_NPT_ISO)
79  nptiso.p_vir[0] = nptiso.p_vir[1] = nptiso.p_vir[2] = 0.0;
80 #endif
81 
82 
83  /* initialize forces with langevin thermostat forces
84  or zero depending on the thermostat
85  set torque to zero for all and rescale quaternions
86  */
87  for (c = 0; c < local_cells.n; c++) {
88  cell = local_cells.cell[c];
89  p = cell->part;
90  np = cell->n;
91  for (i = 0; i < np; i++)
92  init_local_particle_force(&p[i]);
93  }
94 
95  /* initialize ghost forces with zero
96  set torque to zero for all and rescale quaternions
97  */
98  for (c = 0; c < ghost_cells.n; c++) {
99  cell = ghost_cells.cell[c];
100  p = cell->part;
101  np = cell->n;
102  for (i = 0; i < np; i++)
103  init_ghost_force(&p[i]);
104  }
105 
106 #ifdef CONSTRAINTS
107  init_constraint_forces();
108 #endif
109 }

152 void calc_long_range_forces()
153 {
154 #ifdef ELECTROSTATICS
155  /* calculate k-space part of electrostatic interaction. */
156  switch (coulomb.method) {
157 #ifdef P3M
158  case COULOMB_ELC_P3M:
159  if (elc_params.dielectric_contrast_on) {
160  ELC_P3M_modify_p3m_sums_both();
161  ELC_p3m_charge_assign_both();
162  ELC_P3M_self_forces();
163  }
164  else
165  p3m_charge_assign();
166 
167  p3m_calc_kspace_forces(1,0);
168 
169  if (elc_params.dielectric_contrast_on)
170  ELC_P3M_restore_p3m_sums();
171 
172  ELC_add_force();
173 
174  break;
175 #endif
176 #ifdef CUDA
177  case COULOMB_P3M_GPU:
178  if (this_node == 0) {
179  FORCE_TRACE(printf("Computing GPU P3M forces.\n"));
180  p3m_gpu_add_farfield_force();
181  }
182  /* there is no NPT handling here as long as we cannot compute energies.
183  This is checked in integrator_npt_sanity_checks() when integration starts. */
184  break;
185 #endif
186 #ifdef P3M
187  case COULOMB_P3M:
188  FORCE_TRACE(printf("%d: Computing P3M forces.\n", this_node));
189  p3m_charge_assign();
190 #ifdef NPT
191  if (integ_switch == INTEG_METHOD_NPT_ISO)
192  nptiso.p_vir[0] += p3m_calc_kspace_forces(1,1);
193  else
194 #endif
195  p3m_calc_kspace_forces(1, 0);
196  break;
197 #endif
198  case COULOMB_MAGGS:
199  maggs_calc_forces();
200  break;
201  case COULOMB_MMM2D:
202  MMM2D_add_far_force();
203  MMM2D_dielectric_layers_force_contribution();
204  break;
205  default:
206  break;
207  }
208 
209 /* If enabled, calculate electrostatics contribution from electrokinetics species. */
210 #ifdef EK_ELECTROSTATIC_COUPLING
211  ek_calculate_electrostatic_coupling();
212 #endif
213 
214 #endif /*ifdef ELECTROSTATICS */
215 
216 #ifdef DIPOLES
217  /* calculate k-space part of the magnetostatic interaction. */
218  switch (coulomb.Dmethod) {
219 #ifdef DP3M
220  case DIPOLAR_MDLC_P3M:
221  add_mdlc_force_corrections();
222  //fall through
223  case DIPOLAR_P3M:
224  dp3m_dipole_assign();
225 #ifdef NPT
226  if(integ_switch == INTEG_METHOD_NPT_ISO) {
227  nptiso.p_vir[0] += dp3m_calc_kspace_forces(1,1);
228  fprintf(stderr,"dipolar_P3M at this moment is added to p_vir[0]\n");
229  } else
230 #endif
231  dp3m_calc_kspace_forces(1,0);
232 
233  break;
234 #endif
235  case DIPOLAR_ALL_WITH_ALL_AND_NO_REPLICA:
236  dawaanr_calculations(1,0);
237  break;
238  case DIPOLAR_MDLC_DS:
239  add_mdlc_force_corrections();
240  //fall through
241  case DIPOLAR_DS:
242  magnetic_dipolar_direct_sum_calculations(1,0);
243  break;
244  case DIPOLAR_NONE:
245  break;
246  default:
247  ostringstream msg;
248  msg <<"unknown dipolar method";
249  runtimeError(msg);
250  break;
251  }
252 #endif /*ifdef DIPOLES */
253 }
