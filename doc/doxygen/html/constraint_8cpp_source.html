170 void calculate_cylinder_dist(Particle *p1, double ppos[3], Particle *c_p, Constraint_cylinder *c, double *dist, double *vec)
171 {
172  int i;
173  double d_per,d_par,d_real,d_per_vec[3],d_par_vec[3],d_real_vec[3];
174 
175  d_real = 0.0;
176  for(i=0;i<3;i++) {
177  d_real_vec[i] = ppos[i] - c->pos[i];
178  d_real += SQR(d_real_vec[i]);
179  }
180  d_real = sqrt(d_real);
181 
182  d_par=0.;
183  for(i=0;i<3;i++) {
184  d_par += (d_real_vec[i] * c->axis[i]);
185  }
186 
187  for(i=0;i<3;i++) {
188  d_par_vec[i] = d_par * c->axis[i] ;
189  d_per_vec[i] = ppos[i] - (c->pos[i] + d_par_vec[i]) ;
190  }
191 
192  d_per=sqrt(SQR(d_real)-SQR(d_par));
193  d_par = fabs(d_par) ;
194 
195  if ( c->direction == -1 ) {
196  /*apply force towards inside cylinder */
197  d_per = c->rad - d_per ;
198  d_par = c->length - d_par;
199  if (d_per < d_par ) {
200  *dist = d_per ;
201  for (i=0; i<3;i++) {
202  vec[i]= -d_per_vec[i] * d_per / (c->rad - d_per) ;
203  }
204  } else {
205  *dist = d_par ;
206  for (i=0; i<3;i++) {
207  vec[i]= -d_par_vec[i] * d_par / (c->length - d_par) ;
208  }
209  }
210  } else {
211  /*apply force towards outside cylinder */
212  d_per = d_per - c->rad ;
213  d_par = d_par - c->length ;
214  if (d_par < 0 ) {
215  *dist = d_per ;
216  for (i=0; i<3;i++) {
217  vec[i]= d_per_vec[i] * d_per / (d_per + c->rad) ;
218  }
219  } else if ( d_per < 0) {
220  *dist = d_par ;
221  for (i=0; i<3;i++) {
222  vec[i]= d_par_vec[i] * d_par / (d_par + c->length) ;
223  }
224  } else {
225  *dist = sqrt( SQR(d_par) + SQR(d_per)) ;
226  for (i=0; i<3;i++) {
227  vec[i]=
228  d_per_vec[i] * d_per / (d_per + c->rad) +
229  d_par_vec[i] * d_par / (d_par + c->length) ;
230  }
231  }
232  }
233 }
