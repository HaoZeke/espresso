48 /** Calculate umbrella potential force between particle p1 and p2 */
49 inline int calc_umbrella_pair_force(Particle *p1, Particle *p2, Bonded_ia_parameters *ia_params,
50  double d[3], double force[3])
51 {
52  double distn;
53  double fac=0.0;
54  distn = d[ia_params->p.umbrella.dir];
55  fac = -ia_params->p.umbrella.k * (distn - ia_params->p.umbrella.r);
56  force[ia_params->p.umbrella.dir] += fac;

65 /** calculate umbrella energy between particle p1 and p2. */
66 inline int umbrella_pair_energy(Particle *p1, Particle *p2, Bonded_ia_parameters *ia_params,
67  double d[3], double *_energy)
68 {
69  double distn;
70  distn = d[ia_params->p.umbrella.dir];
71  *_energy = 0.5 * ia_params->p.umbrella.k * SQR(distn - ia_params->p.umbrella.r);
