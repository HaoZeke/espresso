37 static void preparePolygammaEven(int n, double binom, Polynom *series)
38 {
39  /* (-0.5 n) psi^2n/2n! (-0.5 n) and psi^(2n+1)/(2n)! series expansions
40  note that BOTH carry 2n! */
41  int order;
42  double deriv;
43  double maxx, x_order, coeff, pref;
44 
45  deriv = 2*n;
46  if (n == 0) {
47  // psi^0 has a slightly different series expansion
48  maxx = 0.25;
49  alloc_doublelist(series, 1);
50  series->e[0] = 2*(1 - C_GAMMA);
51  for (order = 1;; order += 1) {
52  x_order = 2*order;
53  coeff = -2*hzeta(x_order + 1, 2);
54  if (fabs(maxx*coeff)*(4.0/3.0) < ROUND_ERROR_PREC)
55  break;
56  realloc_doublelist(series, order + 1);
57  series->e[order] = coeff;
58  maxx *= 0.25;
59  }
60  series->n = order;
61  }
62  else {
63  // even, n > 0
64  maxx = 1;
65  pref = 2;
66  init_doublelist(series);
67  for (order = 0;; order++) {
68  // only even exponents of x
69  x_order = 2*order;
70  coeff = pref*hzeta(1 + deriv + x_order, 2);
71  if ((fabs(maxx*coeff)*(4.0/3.0) < ROUND_ERROR_PREC) && (x_order > deriv))
72  break;
73  realloc_doublelist(series, order + 1);
74  series->e[order] = -binom*coeff;
75  maxx *= 0.25;
76  pref *= (1.0 + deriv/(x_order + 1));
77  pref *= (1.0 + deriv/(x_order + 2));
78  }
79  series->n = order;
80  }
81 }

83 static void preparePolygammaOdd(int n, double binom, Polynom *series)
84 {
85  int order;
86  double deriv;
87  double maxx, x_order, coeff, pref;
88 
89  deriv = 2*n + 1;
90  maxx = 0.5;
91  // to get 1/(2n)! instead of 1/(2n+1)!
92  pref = 2*deriv*(1 + deriv);
93  init_doublelist(series);
94  for (order = 0;; order++) {
95  // only odd exponents of x
96  x_order = 2*order + 1;
97  coeff = pref*hzeta(1 + deriv + x_order, 2);
98  if ((fabs(maxx*coeff)*(4.0/3.0) < ROUND_ERROR_PREC) && (x_order > deriv))
99  break;
100  realloc_doublelist(series, order + 1);
101  series->e[order] = -binom*coeff;
102  maxx *= 0.25;
103  pref *= (1.0 + deriv/(x_order + 1));
104  pref *= (1.0 + deriv/(x_order + 2));
105  }
106  series->n = order;
107 }

109 void create_mod_psi_up_to(int new_n)
110 {
111  int n;
112  double binom;
113 
114  if (new_n > n_modPsi) {
115  int old = n_modPsi;
116  n_modPsi = new_n;
117  modPsi = (Polynom*)realloc(modPsi, 2*n_modPsi*sizeof(Polynom));
118 
119  binom = 1.0;
120  for (n = 0; n < old; n++)
121  binom *= (-0.5 - n)/(double)(n+1);
122 
123  for (; n < n_modPsi; n++) {
124  preparePolygammaEven(n, binom, &modPsi[2*n]);
125  preparePolygammaOdd(n, binom, &modPsi[2*n + 1]);
126  binom *= (-0.5 - n)/(double)(n+1);
127  }
128  }
