85 {
86  double temp, gamma;
87 
88  /* check number of arguments */
89  if (argc < 4) {
90  Tcl_AppendResult(interp, "wrong # args: should be \n\"",
91  argv[0]," ",argv[1]," <temp> <gamma>\"", (char *)NULL);
92  return (TCL_ERROR);
93  }
94 
95  /* check argument types */
96  if ( !ARG_IS_D(2, temp) || !ARG_IS_D(3, gamma)) {
97  Tcl_AppendResult(interp, argv[0]," ",argv[1]," needs two DOUBLES", (char *)NULL);
98  return (TCL_ERROR);
99  }
100 
101  if (temp < 0 || gamma < 0) {
102  Tcl_AppendResult(interp, "temperature and gamma must be positive", (char *)NULL);
103  return (TCL_ERROR);
104  }
105 
106  /* broadcast parameters */
107  temperature = temp;
108  langevin_gamma = gamma;
109  thermo_switch = ( thermo_switch | THERMO_LANGEVIN );
110  mpi_bcast_parameter(FIELD_THERMO_SWITCH);
111  mpi_bcast_parameter(FIELD_TEMPERATURE);

304 int tclcommand_thermostat_print_all(Tcl_Interp *interp)
305 {
306  char buffer[TCL_DOUBLE_SPACE];
307  /* thermostat not initialized */
308  if(temperature == -1.0) {
309  Tcl_AppendResult(interp,"{ not initialized } ", (char *)NULL);
310  return (TCL_OK);
311  }
312 
313  /* no thermostat on */
314  if(thermo_switch == THERMO_OFF) {
315  Tcl_AppendResult(interp,"{ off } ", (char *)NULL);
316  return (TCL_OK);
317  }
318 
319  /* langevin */
320  if(thermo_switch & THERMO_LANGEVIN ) {
321  Tcl_PrintDouble(interp, temperature, buffer);
322  Tcl_AppendResult(interp,"{ langevin ",buffer, (char *)NULL);
323  Tcl_PrintDouble(interp, langevin_gamma, buffer);
324  Tcl_AppendResult(interp," ",buffer," } ", (char *)NULL);
325  }
326 
327 #ifdef DPD
328  /* dpd */
329  if(thermo_switch & THERMO_DPD) { tclcommand_thermostat_parse_and_print_dpd(interp);}
330 #endif
331 
332 #ifdef NPT
333  /* npt_isotropic */
334  if(thermo_switch & THERMO_NPT_ISO) {
335  Tcl_PrintDouble(interp, temperature, buffer);
336  Tcl_AppendResult(interp,"{ npt_isotropic ",buffer, (char *)NULL);
337  Tcl_PrintDouble(interp, nptiso_gamma0, buffer);
338  Tcl_AppendResult(interp," ",buffer, (char *)NULL);
339  Tcl_PrintDouble(interp, nptiso_gammav, buffer);
340  Tcl_AppendResult(interp," ",buffer, " } ", (char *)NULL);
341  }
342 #endif
343 
344 #ifdef GHMC
345  /* ghmc */
346  if(thermo_switch & THERMO_GHMC) {
347  Tcl_PrintDouble(interp, temperature, buffer);
348  Tcl_AppendResult(interp,"{ ghmc ",buffer, (char *)NULL);
349  Tcl_PrintDouble(interp, ghmc_nmd, buffer);
350  Tcl_AppendResult(interp," ",buffer, (char *)NULL);
351  Tcl_PrintDouble(interp, ghmc_phi, buffer);
352  Tcl_AppendResult(interp," ",buffer, " ", (char *)NULL);
353  switch (ghmc_mflip) {
354  case GHMC_MFLIP_OFF:
355  Tcl_AppendResult(interp, "-no_flip ", (char *)NULL);
356  break;
357  case GHMC_MFLIP_ON:
358  Tcl_AppendResult(interp, "-flip ", (char *)NULL);
359  break;
360  case GHMC_MFLIP_RAND:
361  Tcl_AppendResult(interp, "-random_flip ", (char *)NULL);
362  break;
363  }
364  if (ghmc_tscale == GHMC_TSCALE_ON)
365  Tcl_AppendResult(interp, "-scale }", (char *)NULL);
366  else
367  Tcl_AppendResult(interp, "-no_scale }", (char *)NULL);
368  }
369 #endif
370 
371 #if defined(LB) || defined(LB_GPU)
372  /* lb */
373  if(thermo_switch & THERMO_LB) {
374  Tcl_PrintDouble(interp, temperature, buffer);
375  Tcl_AppendResult(interp,"{ lb ",buffer, " } ", (char *)NULL);
376  }
377 #endif
378 
379 #ifdef INTER_DPD
380  /* inter_dpd */
381  if(thermo_switch & THERMO_INTER_DPD) {
382  Tcl_PrintDouble(interp, temperature, buffer);
383  Tcl_AppendResult(interp,"{ inter_dpd ",buffer, " } ", (char *)NULL);
384  }
385 #endif
386 
387 #ifdef SD
388  if (thermo_switch & THERMO_SD){
389  Tcl_PrintDouble(interp, temperature, buffer);
390  Tcl_AppendResult(interp,"{ sd ",buffer, " } ", (char *)NULL);
391 
392  }
393  if (thermo_switch & THERMO_BD){
394  Tcl_PrintDouble(interp, temperature, buffer);
395  Tcl_AppendResult(interp,"{ bd ",buffer, " } ", (char *)NULL);
396  }
397 #endif
398  return (TCL_OK);
399 }

427 {
428  int err = TCL_OK;
429  THERMO_TRACE(fprintf(stderr,"%d: thermostat:\n",this_node));
430 
431  /* print thermostat status */
432  if(argc == 1) return tclcommand_thermostat_print_all(interp);
433 
434  if ( ARG1_IS_S("set") ) {
435  argc--;
436  argv++;
437 
438  if (argc == 1) {
439  Tcl_AppendResult(interp, "wrong # args: \n", (char *)NULL);

444  err = tclcommand_thermostat_parse_off(interp, argc, argv);

458 #endif
459 #if defined(LB) || defined(LB_GPU)

461  err = tclcommand_thermostat_parse_lb(interp, argc-1, argv+1);

469 #if defined(SD) || defined(BD)
