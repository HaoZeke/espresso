103 /** Metadynamics main function:
104 * - Calculate reaction coordinate
105 * - Update profile and biased force
106 * - apply external force
107 */
108 void meta_perform()
109 {
110  if(meta_switch == META_OFF) return;
111 
112  double ppos1[3] = {0,0,0}, ppos2[3] = {0,0,0}, factor;
113  int img1[3], img2[3], c, i, np, flag1 = 0, flag2 = 0;
114  Particle *p, *p1 = NULL, *p2 = NULL;
115  Cell *cell;
116 
117  for (c = 0; c < local_cells.n; c++) {
118  cell = local_cells.cell[c];
119  p = cell->part;
120  np = cell->n;
121  for(i = 0; i < np; i++) {
122  if (p[i].p.identity == meta_pid1) {
123  flag1 = 1;
124  p1 = &p[i];
125  memmove(ppos1, p[i].r.p, 3*sizeof(double));
126  memmove(img1, p[i].l.i, 3*sizeof(int));
127  unfold_position(ppos1, img1);
128 
129  if (flag1 && flag2) {
130  /* vector r2-r1 - Not a minimal image! Unfolded position */
131  vector_subt(meta_cur_xi,ppos2,ppos1);
132  break;
133  }
134  }
135  if (p[i].p.identity == meta_pid2) {
136  flag2 = 1;
137  p2 = &p[i];
138  memmove(ppos2, p[i].r.p, 3*sizeof(double));
139  memmove(img2, p[i].l.i, 3*sizeof(int));
140  unfold_position(ppos2, img2);
141 
142  if (flag1 && flag2) {
143  /* vector r2-r1 - Not a minimal image! Unfolded position */
144  vector_subt(meta_cur_xi,ppos2,ppos1);
145  break;
146  }
147  }
148  }
149  }
150 
151  if (flag1 == 0 || flag2 == 0) {
152  ostringstream msg;
153  msg <<"Metadynamics: can't find pid1 or pid2.\n";
154  runtimeError(msg);
155  return;
156  }
157 
158  /* Now update free energy profile
159  * Here, we're following the functional form of
160  * Marsili etal., J Comp. Chem, 31 (2009).
161  * Instead of gaussians, we use so-called Lucy's functions */
162 
163  for (i = 0; i < meta_xi_num_bins; ++i) {
164  if (meta_switch == META_DIST) {
165  // reaction coordinate value
166  meta_val_xi = sqrt(sqrlen(meta_cur_xi));
167  // Update free energy profile and biased force
168  if(int(sim_time/time_step)%meta_num_relaxation_steps==0){
169  meta_acc_fprofile[i] -= calculate_lucy(meta_xi_min+i*meta_xi_step,meta_val_xi);
170  meta_acc_force[i] -= calculate_deriv_lucy(meta_xi_min+i*meta_xi_step,meta_val_xi);
171  }
172 
173  // direction of the bias force
174  unit_vector(meta_cur_xi,meta_apply_direction);
175  } else if (meta_switch == META_REL_Z) {
176  // reaction coordinate value: relative height of z_pid1 with respect to z_pid2
177  meta_val_xi = -1.*meta_cur_xi[2];
178  // Update free energy profile and biased force
179  if(int(sim_time/time_step)%meta_num_relaxation_steps==0){
180  meta_acc_fprofile[i] -= calculate_lucy(meta_xi_min+i*meta_xi_step,meta_val_xi);
181  meta_acc_force[i] -= calculate_deriv_lucy(meta_xi_min+i*meta_xi_step,meta_val_xi);
182  }
183  // direction of the bias force (-1 to be consistent with META_DIST: from 1 to 2)
184  meta_apply_direction[0] = meta_apply_direction[1] = 0.;
185  meta_apply_direction[2] = -1.;
186  } else {
187  ostringstream msg;
188  msg <<"Undefined metadynamics scheme.\n";
189  runtimeError(msg);
190  return;
191  }
192  }
193 
194  /** Apply force */
195 
196  // Calculate the strength of the applied force
197  if (meta_val_xi < meta_xi_min) {
198  // below the lower bound
199  factor = -1. * meta_f_bound * (meta_xi_min-meta_val_xi)/meta_xi_step;
200  } else if (meta_val_xi > meta_xi_max) {
201  // above the upper bound
202  factor = meta_f_bound * (meta_val_xi-meta_xi_max)/meta_xi_step;
203  } else {
204  // within the RC interval
205  i = (int) dround((meta_val_xi-meta_xi_min)/meta_xi_step);
206  if (i < 0) i = 0;
207  if (i >= meta_xi_num_bins) i=meta_xi_num_bins-1;
208  factor = meta_acc_force[i];
209  }
210 
211  /* cancel previous force to external force of particle */
212  for (i = 0; i<3; ++i) {
213  p1->f.f[i] += factor * meta_apply_direction[i];
214  p2->f.f[i] += -1. * factor * meta_apply_direction[i];
215  }
216 }

219 /** Calculate Lucy's function */
220 double calculate_lucy(double xi, double xi_0)
221 {
222  double dist = fabs(xi-xi_0);
223  if (dist <= meta_bias_width) {
224  return meta_bias_height
225  * (1+2*dist/meta_bias_width)
226  * pow(1-dist/meta_bias_width,2);
227  } else
228  return 0.;
229 }
230 
231 /** Calculate derivative of Lucy function */
232 double calculate_deriv_lucy(double xi, double xi_0)
233 {
234  double dist = fabs(xi-xi_0);
235  if (dist <= meta_bias_width) {
236  double result = -2*meta_bias_height/meta_bias_width
237  * (pow(1-dist/meta_bias_width,2)
238  -(1+2*dist/meta_bias_width)*(1-dist/meta_bias_width));
239  if (xi < xi_0)
240  result *= -1.;
241  return result;
242  } else
243  return 0.;
244 }
