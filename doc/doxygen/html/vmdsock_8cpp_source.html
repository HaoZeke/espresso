108 int vmdsock_connect(void *v, const char *host, int port) {
109  vmdsocket *s = (vmdsocket *) v;
110  char address[1030];
111  struct hostent *h;
112 
113  h=gethostbyname(host);
114  if (h == NULL)
115  return -1;
116  sprintf(address, "%d.%d.%d.%d",
117  (unsigned char) h->h_addr_list[0][0],
118  (unsigned char) h->h_addr_list[0][1],
119  (unsigned char) h->h_addr_list[0][2],
120  (unsigned char) h->h_addr_list[0][3]);
121 
122  memset(&(s->addr), 0, sizeof(s->addr));
123  s->addr.sin_family = PF_INET;
124  s->addr.sin_addr.s_addr = inet_addr(address);
125  s->addr.sin_port = htons(port);
126 
127  return connect(s->sd, (struct sockaddr *) &s->addr, sizeof(s->addr));
128 }

144 void *vmdsock_accept(void * v) {
145  int rc;
146  vmdsocket *new_s = NULL, *s = (vmdsocket *) v;
147 #ifdef SOCKLEN_T
148  socklen_t len;
149 #else
150  unsigned int len;
151 #endif
152 
153  len = sizeof(s->addr);
154  rc = accept(s->sd, (struct sockaddr *) &s->addr, &len);
155  if (rc >= 0) {
156  new_s = (vmdsocket *) malloc(sizeof(vmdsocket));
157  if (new_s != NULL) {
158  *new_s = *s;
159  new_s->sd = rc;
160  }
161  }
162  return (void *)new_s;
163 }

202

206  tv.tv_sec = sec;
