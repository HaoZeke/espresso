41 /** Calculate Hertzian force between particle p1 and p2 */
42 inline void add_hertzian_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
43  double d[3], double dist, double dist2, double force[3])
44 {
45  double fac;
46  int j;
47  if (CUTOFF_CHECK(dist < ia_params->Hertzian_sig)) {
48  fac = 5./2.*ia_params->Hertzian_eps/ia_params->Hertzian_sig *
49  pow(1 - dist/ia_params->Hertzian_sig, 3./2.)/dist;
50 
51  for(j=0;j<3;j++)
52  force[j] += fac * d[j];
53  }
54 }

56 /** calculate Lennard jones energy between particle p1 and p2. */
57 inline double hertzian_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
58  double d[3], double dist, double dist2)
59 {
60  if (CUTOFF_CHECK(dist < ia_params->Hertzian_sig)) {
61  return ia_params->Hertzian_eps *
62  pow(1 - dist/ia_params->Hertzian_sig, 5./2.);
63  }
64  return 0.0;
65 }
