446 /** Calculate non bonded forces between a pair of particles.
447  @param p1 pointer to particle 1.
448  @param p2 pointer to particle 2.
449  @param d vector between p1 and p2.
450  @param dist distance between p1 and p2.
451  @param dist2 distance squared between p1 and p2. */
452 inline void add_non_bonded_pair_force(Particle *p1, Particle *p2,
453  double d[3], double dist, double dist2)
454 {
455  IA_parameters *ia_params = get_ia_param(p1->p.type,p2->p.type);
456  double force[3] = { 0., 0., 0. };
457  double torque1[3] = { 0., 0., 0. };
458  double torque2[3] = { 0., 0., 0. };
459  int j;
460 
461 
462 #ifdef COLLISION_DETECTION
463  if (collision_params.mode > 0)
464  detect_collision(p1,p2);
465 #endif
466 
467  FORCE_TRACE(fprintf(stderr, "%d: interaction %d<->%d dist %f\n", this_node, p1->p.identity, p2->p.identity, dist));
468 
469  /***********************************************/
470  /* thermostat */
471  /***********************************************/
472 
473 #ifdef DPD
474  /* DPD thermostat forces */
475  if ( thermo_switch & THERMO_DPD ) add_dpd_thermo_pair_force(p1,p2,d,dist,dist2);
476 #endif
477 
478 #ifdef INTER_DPD
479  if ( thermo_switch & THERMO_INTER_DPD ) add_inter_dpd_pair_force(p1,p2,ia_params,d,dist,dist2);
480 #endif
481 
482  /***********************************************/
483  /* non bonded pair potentials */
484  /***********************************************/
485 
486  calc_non_bonded_pair_force(p1,p2,ia_params,d,dist,dist2,force,torque1,torque2);
487 
488  /***********************************************/
489  /* short range electrostatics */
490  /***********************************************/
491 
492 #ifdef ELECTROSTATICS
493  if (coulomb.method == COULOMB_DH)
494  add_dh_coulomb_pair_force(p1,p2,d,dist,force);
495 
496  if (coulomb.method == COULOMB_RF)
497  add_rf_coulomb_pair_force(p1,p2,d,dist,force);
498 #endif
499 
500  /*********************************************************************/
501  /* everything before this contributes to the virial pressure in NpT, */
502  /* but nothing afterwards */
503  /*********************************************************************/
504 #ifdef NPT
505  for (j = 0; j < 3; j++)
506  if(integ_switch == INTEG_METHOD_NPT_ISO)
507  nptiso.p_vir[j] += force[j] * d[j];
508 #endif
509 
510  /***********************************************/
511  /* long range electrostatics */
512  /***********************************************/
513 
514 #ifdef ELECTROSTATICS
515 
516  /* real space coulomb */
517  double q1q2 = p1->p.q*p2->p.q;
518  switch (coulomb.method) {
519 #ifdef P3M
520  case COULOMB_ELC_P3M: {
521  if (q1q2) {
522  p3m_add_pair_force(q1q2,d,dist2,dist,force);
523 
524  // forces from the virtual charges
525  // they go directly onto the particles, since they are not pairwise forces
526  if (elc_params.dielectric_contrast_on)
527  ELC_P3M_dielectric_layers_force_contribution(p1, p2, p1->f.f, p2->f.f);
528  }
529  break;
530  }
531  case COULOMB_P3M_GPU:
532  case COULOMB_P3M: {
533 #ifdef NPT
534  if (q1q2) {
535  double eng = p3m_add_pair_force(q1q2,d,dist2,dist,force);
536  if(integ_switch == INTEG_METHOD_NPT_ISO)
537  nptiso.p_vir[0] += eng;
538  }
539 #else
540  if (q1q2) p3m_add_pair_force(q1q2,d,dist2,dist,force);
541 #endif
542  break;
543  }
544 #endif
545  case COULOMB_MMM1D:
546  if (q1q2) add_mmm1d_coulomb_pair_force(q1q2,d,dist2,dist,force);
547  break;
548  case COULOMB_MMM2D:
549  if (q1q2) add_mmm2d_coulomb_pair_force(q1q2,d,dist2,dist,force);
550  break;
551 #ifdef EWALD_GPU
552  case COULOMB_EWALD_GPU:
553  if (q1q2) add_ewald_gpu_coulomb_pair_force(p1,p2,d,dist,force);
554  break;
555 #endif
556  default:
557  break;
558  }
559 
560 #endif /*ifdef ELECTROSTATICS */
561 
562 
563  /***********************************************/
564  /* long range magnetostatics */
565  /***********************************************/
566 
567 
568 #ifdef DIPOLES
569  /* real space magnetic dipole-dipole */
570  switch (coulomb.Dmethod) {
571 #ifdef DP3M
572  case DIPOLAR_MDLC_P3M:
573  //fall trough
574  case DIPOLAR_P3M: {
575 #ifdef NPT
576  double eng = dp3m_add_pair_force(p1,p2,d,dist2,dist,force);
577  if(integ_switch == INTEG_METHOD_NPT_ISO)
578  nptiso.p_vir[0] += eng;
579 #else
580  dp3m_add_pair_force(p1,p2,d,dist2,dist,force);
581 #endif
582  break;
583  }
584 #endif /*ifdef DP3M */
585  default:
586  break;
587  }
588 #endif /* ifdef DIPOLES */
589 
590  /***********************************************/
591  /* add total nonbonded forces to particle */
592  /***********************************************/
593 
594  for (j = 0; j < 3; j++) {
595  p1->f.f[j] += force[j];
596  p2->f.f[j] -= force[j];
597 #ifdef ROTATION
598  p1->f.torque[j] += torque1[j];
599  p2->f.torque[j] += torque2[j];
600 #endif
601  }
602 }
