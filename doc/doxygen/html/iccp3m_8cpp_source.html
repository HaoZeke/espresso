145 int iccp3m_iteration() {
146  double fdot,hold,hnew,hmax,del_eps,diff=0.0,difftemp=0.0, ex, ey, ez, l_b;
147  Cell *cell;
148  int c,np;
149  Particle *part;
150  int i, j,id;
151  double globalmax;
152  double f1, f2 = 0;
153 
154  iccp3m_sanity_check();
155 
156  l_b = coulomb.bjerrum;
157  if((iccp3m_cfg.eout <= 0)) {
158  ostringstream msg;
159  msg <<"ICCP3M: nonpositive dielectric constant is not allowed. Put a decent tcl error here\n";
160  runtimeError(msg);
161  }
162 
163 
164  iccp3m_cfg.citeration=0;
165  for(j=0;j<iccp3m_cfg.num_iteration;j++) {
166  hmax=0.;
167  force_calc_iccp3m(); /* Calculate electrostatic forces (SR+LR) excluding source source interaction*/
168  diff=0;
169  for(c = 0; c < local_cells.n; c++) {
170  cell = local_cells.cell[c];
171  part = cell->part;
172  np = cell->n;
173  for(i=0 ; i < np; i++) {
174  if( part[i].p.identity < iccp3m_cfg.n_ic+iccp3m_cfg.first_id && part[i].p.identity >= iccp3m_cfg.first_id ) {
175  id = part[i].p.identity - iccp3m_cfg.first_id;
176  /* the dielectric-related prefactor: */
177  del_eps = (iccp3m_cfg.ein[id]-iccp3m_cfg.eout)/(iccp3m_cfg.ein[id] + iccp3m_cfg.eout)/6.283185307;
178  /* calculate the electric field at the certain position */
179  ex=part[i].f.f[0]/part[i].p.q;
180  ey=part[i].f.f[1]/part[i].p.q;
181  ez=part[i].f.f[2]/part[i].p.q;
182 
183  /* let's add the contribution coming from the external field */
184  ex += iccp3m_cfg.extx;
185  ey += iccp3m_cfg.exty;
186  ez += iccp3m_cfg.extz;
187 
188  if (ex == 0 && ey == 0 && ez == 0) {
189  ostringstream msg;
190  msg <<"ICCP3M found zero electric field on a charge. This must never happen";
191  runtimeError(msg);
192  }
193  /* the dot product */
194  fdot = ex*iccp3m_cfg.nvectorx[id]+
195  ey*iccp3m_cfg.nvectory[id]+
196  ez*iccp3m_cfg.nvectorz[id];
197 
198  /* recalculate the old charge density */
199  hold=part[i].p.q/iccp3m_cfg.areas[id];
200  /* determine if it is higher than the previously highest charge density */
201  if(fabs(hold)>hmax)hmax=fabs(hold);
202  f1 = (+del_eps*fdot/l_b);
203 // double f2 = (1- 0.5*(iccp3m_cfg.ein[id]-iccp3m_cfg.eout)/(iccp3m_cfg.eout + iccp3m_cfg.ein[id] ))*(iccp3m_cfg.sigma[id]);
204  if (iccp3m_cfg.sigma!=0) {
205  f2 = (2*iccp3m_cfg.eout)/(iccp3m_cfg.eout + iccp3m_cfg.ein[id] )*(iccp3m_cfg.sigma[id]);
206  }
207 
208  hnew=(1.-iccp3m_cfg.relax)*hold + (iccp3m_cfg.relax)*(f1 + f2);
209  difftemp=fabs( 1*(hnew - hold)/(hmax + fabs(hnew+hold)) ); /* relative variation: never use
210  an estimator which can be negative
211  here */
212  if(difftemp > diff && part[i].p.q > 1e-5)
213  {
214 // if (fabs(difftemp - 1./(1./iccp3m_cfg.relax - 1.)) > 1e-10)
215  diff=difftemp; /* Take the largest error to check for convergence */
216  }
217  part[i].p.q = hnew * iccp3m_cfg.areas[id];
218  /* check if the charge now is more than 1e6, to determine if ICC still leads to reasonable results */
219  /* this is kind a arbitrary measure but, does a good job spotting divergence !*/
220  if(fabs(part[i].p.q) > 1e6) {
221  ostringstream msg;
222  msg <<"too big charge assignment in iccp3m! q >1e6 , assigned charge= " << part[i].p.q << "\n";
223  runtimeError(msg);
224  diff = 1e90; /* A very high value is used as error code */
225  break;
226  }
227  }
228  } /* cell particles */
229  // printf("cell %d w %d particles over (node %d)\n",c,np,this_node); fflush(stdout);
230  } /* local cells */
231  iccp3m_cfg.citeration++;
232  MPI_Allreduce(&diff, &globalmax, 1,MPI_DOUBLE, MPI_MAX, comm_cart);
233 
234  if (globalmax < iccp3m_cfg.convergence)
235  break;
236  if ( diff > 1e89 ) /* Error happened */
237  return iccp3m_cfg.citeration++;
238 
239  } /* iteration */
240  on_particle_change();
241 
242  return iccp3m_cfg.citeration;
243 }
