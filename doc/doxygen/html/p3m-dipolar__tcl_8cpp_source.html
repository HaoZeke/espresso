21 /** \file p3m-dipolar.cpp P3M algorithm for long range magnetic dipole-dipole interaction.
22  *
23  NB: In general the magnetic dipole-dipole functions bear the same
24  name than the charge-charge but, adding in front of the name a D
25  and replacing where "charge" appears by "dipole". In this way one
26  can recognize the similarity of the functions but avoiding nasty
27  confusions in their use.
28 
29  PS: By default the magnetic epsilon is metallic = 0.

38 int tclcommand_inter_magnetic_parse_dp3m_tune_params(Tcl_Interp * interp, int argc, char ** argv)
39 {
40  int mesh = -1, cao = -1, n_interpol = -1;
41  double r_cut = -1, accuracy = -1;
42 
43  while(argc > 0) {
44  if(ARG0_IS_S("r_cut")) {
45  if (! (argc > 1 && ARG1_IS_D(r_cut) && r_cut >= -1)) {
46  Tcl_AppendResult(interp, "r_cut expects a positive double",

48  return TCL_ERROR;
49  }
50 
51  } else if(ARG0_IS_S("mesh")) {
52  if(! (argc > 1 && ARG1_IS_I(mesh) && mesh >= -1)) {
53  Tcl_AppendResult(interp, "mesh expects an integer >= -1",
54  (char *) NULL);
55  return TCL_ERROR;
56  }
57 
58  } else if(ARG0_IS_S("cao")) {
59  if(! (argc > 1 && ARG1_IS_I(cao) && cao >= -1 && cao < 7)) {
60  Tcl_AppendResult(interp, "cao expects an integer between -1 and 7",
61  (char *) NULL);
62  return TCL_ERROR;
63  }
64 
65  } else if(ARG0_IS_S("accuracy")) {
66  if(! (argc > 1 && ARG1_IS_D(accuracy) && accuracy > 0)) {
67  Tcl_AppendResult(interp, "accuracy expects a positive double",
68  (char *) NULL);
69  return TCL_ERROR;
70  }
71 
72  } else if (ARG0_IS_S("n_interpol")) {
73  if (! (argc > 1 && ARG1_IS_I(n_interpol) && n_interpol >= 0)) {
74  Tcl_AppendResult(interp, "n_interpol expects an nonnegative integer",
75  (char *) NULL);
76  return TCL_ERROR;
77  }
78  }
79  /* unknown parameter. Probably one of the optionals */
80  else break;
81 
82  argc -= 2;
83  argv += 2;
84  }
85 
86  dp3m_set_tune_params(r_cut, mesh, cao, -1.0, accuracy, n_interpol);
87 
88  /* check for optional parameters */
89  if (argc > 0) {
90  if (tclcommand_inter_magnetic_parse_dp3m_opt_params(interp, argc, argv) == TCL_ERROR)
91  return TCL_ERROR;
92  }
93 
94  /* do the tuning */
95  char *log = NULL;
96  if (dp3m_adaptive_tune(&log) == ES_ERROR) {
97  Tcl_AppendResult(interp, log, "\nfailed to tune P3M parameters to required accuracy", (char *) NULL);
98  if (log)
99  free(log);
100  return TCL_ERROR;
101  }
102 
103  /* Tell the user about the tuning outcome */
104  Tcl_AppendResult(interp, log, (char *) NULL);
105 
106  if (log)
107  free(log);
108 
109  return TCL_OK;
110 }

115 int tclcommand_inter_magnetic_parse_dp3m(Tcl_Interp * interp, int argc, char ** argv)
116 {
117  double r_cut, alpha, accuracy = -1.0;
118  int mesh, cao, i;
119 
120  if (argc < 1) {
121  Tcl_AppendResult(interp, "expected: inter dipolar <bjerrum> p3m tune | <r_cut> <mesh> <cao> [<alpha> [<accuracy>]]",
122  (char *) NULL);
123  return TCL_ERROR;
124  }
125 
126  if (ARG0_IS_S("tune") || ARG0_IS_S("tunev2"))
127  return tclcommand_inter_magnetic_parse_dp3m_tune_params(interp, argc-1, argv+1);
128 
129  if(! ARG0_IS_D(r_cut))
130  return TCL_ERROR;
131 
132  if(argc < 3 || argc > 5) {
133  Tcl_AppendResult(interp, "wrong # arguments: inter dipolar <bjerrum> p3m <r_cut> <mesh> <cao> [<alpha> [<accuracy>]]",
134  (char *) NULL);
135  return TCL_ERROR;
136  }
137 
138  if((! ARG_IS_I(1, mesh)) || (! ARG_IS_I(2, cao))) {
139  Tcl_AppendResult(interp, "integer expected", (char *) NULL);
140  return TCL_ERROR;
141  }
142 
143  if(argc > 3) {
144  if(! ARG_IS_D(3, alpha))
145  return TCL_ERROR;
146  }
147  else {
148  Tcl_AppendResult(interp, "Automatic p3m tuning not implemented.",
149  (char *) NULL);
150  return TCL_ERROR;
151  }
152 
153  if(argc > 4) {
154  if(! ARG_IS_D(4, accuracy)) {
155  Tcl_AppendResult(interp, "double expected", (char *) NULL);
156  return TCL_ERROR;
157  }
158  }
159 
160  if ((i = dp3m_set_params(r_cut, mesh, cao, alpha, accuracy)) < 0) {
161  switch (i) {
162  case -1:
163  Tcl_AppendResult(interp, "r_cut must be positive", (char *) NULL);
164  break;
165  case -2:
166  Tcl_AppendResult(interp, "mesh must be positive", (char *) NULL);
167  break;
168  case -3:
169  Tcl_AppendResult(interp, "cao must be between 1 and 7 and less than mesh",
170  (char *) NULL);
171  break;
172  case -4:
173  Tcl_AppendResult(interp, "alpha must be positive", (char *) NULL);
174  break;
175  case -5:
176  Tcl_AppendResult(interp, "accuracy must be positive", (char *) NULL);
177  break;
178  default:;
179  Tcl_AppendResult(interp, "unspecified error", (char *) NULL);
180  }
181 
182  return TCL_ERROR;
183 
184  }
185 
186  return TCL_OK;
187 }

194 int tclcommand_inter_magnetic_parse_dp3m_opt_params(Tcl_Interp * interp, int argc, char ** argv)
195 {
196  int i; double d1, d2, d3;
197 
198  Tcl_ResetResult(interp);
199 
200  while (argc > 0) {
201  /* p3m parameter: inter */
202  if (ARG0_IS_S("n_interpol")) {
203 
204  if(argc < 2) {
205  Tcl_AppendResult(interp, argv[0], " needs 1 parameter",
206  (char *) NULL);
207  return TCL_ERROR;
208  }
209 
210  if (! ARG1_IS_I(i)) {
211  Tcl_AppendResult(interp, argv[0], " needs 1 INTEGER parameter",
212  (char *) NULL);
213  return TCL_ERROR;
214  }
215 
216  if (dp3m_set_ninterpol(i) == TCL_ERROR) {
217  Tcl_AppendResult(interp, argv[0], " argument must be positive",
218  (char *) NULL);
219  return TCL_ERROR;
220  }
221 
222  argc -= 2;
223  argv += 2;
224  }
225 
226  /* p3m parameter: mesh_off */
227  else if (ARG0_IS_S("mesh_off")) {
228 
229  if(argc < 4) {
230  Tcl_AppendResult(interp, argv[0], " needs 3 parameters",
231  (char *) NULL);
232  return TCL_ERROR;
233  }
234 
235  if ((! ARG_IS_D(1, d1)) ||
236  (! ARG_IS_D(2, d2)) ||
237  (! ARG_IS_D(3, d3)))
238  {
239  Tcl_AppendResult(interp, argv[0], " needs 3 DOUBLE parameters",
240  (char *) NULL);
241  return TCL_ERROR;
242  }
243 
244  if (dp3m_set_mesh_offset(d1, d2 ,d3) == TCL_ERROR)
245  {
246  Tcl_AppendResult(interp, argv[0], " parameters have to be between 0.0 an 1.0",
247  (char *) NULL);
248  return TCL_ERROR;
249  }
250 
251  argc -= 4;
252  argv += 4;
253  }
254 
255  /* p3m parameter: epsilon */
256  else if(ARG0_IS_S( "epsilon")) {
257 
258  if(argc < 2) {
259  Tcl_AppendResult(interp, argv[0], " needs 1 parameter",
260  (char *) NULL);
261  return TCL_ERROR;
262  }
263 
264  if (ARG1_IS_S("metallic")) {
265  d1 = P3M_EPSILON_METALLIC;
266  }
267  else if (! ARG1_IS_D(d1)) {
268  Tcl_AppendResult(interp, argv[0], " needs 1 DOUBLE parameter or \"metallic\"",
269  (char *) NULL);
270  return TCL_ERROR;
271  }
272 
273  if (dp3m_set_eps(d1) == TCL_ERROR) {
274  Tcl_AppendResult(interp, argv[0], " There is no error msg yet!",
275  (char *) NULL);
276  return TCL_ERROR;
277  }
278 
279  argc -= 2;
280  argv += 2;
281  }
282  else {
283  Tcl_AppendResult(interp, "Unknown coulomb p3m parameter: \"",argv[0],"\"",(char *) NULL);
284  return TCL_ERROR;
285  }
286  }
287 
288  return TCL_OK;
