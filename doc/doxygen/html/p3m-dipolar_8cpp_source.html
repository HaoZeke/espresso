410 double dp3m_average_dipolar_self_energy(double box_l, int mesh) {
411  int i,ind,n[3];
412  double node_phi = 0.0, phi = 0.0;
413  double U2;
414 
415  int end[3];
416  int size=1;
417 
418  for(i=0;i<3;i++) {
419  size *= dfft.plan[3].new_mesh[i];
420  end[i] = dfft.plan[3].start[i] + dfft.plan[3].new_mesh[i];
421  }
422 
423  for(n[0]=dfft.plan[3].start[0]; n[0]<end[0]; n[0]++){
424  for(n[1]=dfft.plan[3].start[1]; n[1]<end[1]; n[1]++){
425  for(n[2]=dfft.plan[3].start[2]; n[2]<end[2]; n[2]++) {
426  ind = (n[2]-dfft.plan[3].start[2]) + dfft.plan[3].new_mesh[2] *
427  ((n[1]-dfft.plan[3].start[1]) + (dfft.plan[3].new_mesh[1]*(n[0]-dfft.plan[3].start[0])));
428 
429  if( (n[0]==0) && (n[1]==0) && (n[2]==0) )
430  node_phi += 0.0;
431  else if( (n[0]%(dp3m.params.mesh[0]/2)==0) &&
432  (n[1]%(dp3m.params.mesh[0]/2)==0) &&
433  (n[2]%(dp3m.params.mesh[0]/2)==0) )
434  node_phi += 0.0;
435  else {
436  U2 = dp3m_perform_aliasing_sums_dipolar_self_energy(n);
437  node_phi += dp3m.g_energy[ind] * U2*(SQR(dp3m.d_op[n[0]])+SQR(dp3m.d_op[n[1]])+SQR(dp3m.d_op[n[2]]));
438  }
439  }}}
440 
441 
442  MPI_Reduce(&node_phi, &phi, 1, MPI_DOUBLE, MPI_SUM, 0, comm_cart);
443 
444  phi*=PI/3./box_l/pow(mesh,3.0);
445 
446 
447  return phi ;
448 }

452 double dp3m_perform_aliasing_sums_dipolar_self_energy(int n[3])
453 {
454  double u_sum = 0.0;
455  /* lots of temporary variables... */
456  double f1,sx,sy,sz,mx,my,mz,nmx,nmy,nmz;
457  int limit=P3M_BRILLOUIN+5;
458 
459  f1 = 1.0/(double)dp3m.params.mesh[0];
460 
461  for(mx = -limit; mx <=limit; mx++) {
462  nmx = dp3m.meshift[n[0]] + dp3m.params.mesh[0]*mx;
463  sx = pow(sinc(f1*nmx),2.0*dp3m.params.cao);
464  for(my = -limit; my <= limit; my++) {
465  nmy = dp3m.meshift[n[1]] + dp3m.params.mesh[0]*my;
466  sy = sx*pow(sinc(f1*nmy),2.0*dp3m.params.cao);
467  for(mz = -limit; mz <=limit; mz++) {
468  nmz = dp3m.meshift[n[2]] + dp3m.params.mesh[0]*mz;
469  sz = sy*pow(sinc(f1*nmz),2.0*dp3m.params.cao);
470  u_sum += sz;
471  }
472  }
473  }
474  return u_sum;
475 }

1489  double n_nm;
1490  double n_nm2;
1491 
1492  nominator[0]=0.0;
1493 
1494  f1 = 1.0/(double)dp3m.params.mesh[0];
1495  f2 = SQR(PI/(dp3m.params.alpha_L));
1496 
1497  for(mx = -P3M_BRILLOUIN; mx <= P3M_BRILLOUIN; mx++) {
1498  nmx = dp3m.meshift[n[0]] + dp3m.params.mesh[0]*mx;
1499  sx = pow(sinc(f1*nmx),2.0*dp3m.params.cao);
1500  for(my = -P3M_BRILLOUIN; my <= P3M_BRILLOUIN; my++) {
1501  nmy = dp3m.meshift[n[1]] + dp3m.params.mesh[0]*my;
1502  sy = sx*pow(sinc(f1*nmy),2.0*dp3m.params.cao);
1503  for(mz = -P3M_BRILLOUIN; mz <= P3M_BRILLOUIN; mz++) {
1504  nmz = dp3m.meshift[n[2]] + dp3m.params.mesh[0]*mz;
1505  sz = sy*pow(sinc(f1*nmz),2.0*dp3m.params.cao);
1506 
1507  nm2 = SQR(nmx)+SQR(nmy)+SQR(nmz);
1508  expo = f2*nm2;
1509  f3 = (expo<limit) ? sz*exp(-expo)/nm2 : 0.0;
1510 
1511  n_nm = dp3m.d_op[n[0]]*nmx + dp3m.d_op[n[1]]*nmy + dp3m.d_op[n[2]]*nmz;
1512  n_nm2 = n_nm*n_nm;
1513  nominator[0] += f3*n_nm2;
1514  denominator += sz;
1515  }
1516  }
1517  }
1518  return denominator;
1519 }

1571  contributes half of the total error, and then the Fourier space error is calculated. Returns the error and the
1572  optimal alpha, or 0 if this combination does not work at all */
1573 double dp3m_get_accuracy(int mesh, int cao, double r_cut_iL, double *_alpha_L, double *_rs_err, double *_ks_err)
1574 {
1575  double rs_err, ks_err;
1576  double alpha_L;
1577  P3M_TRACE(fprintf(stderr, "dp3m_get_accuracy: mesh %d, cao %d, r_cut %f ", mesh, cao, r_cut_iL));
1578

1580 
1581  //Alpha cannot be zero in the dipolar case because real_space formula breaks down
1582  //Idem of the previous function tclcommand_inter_magnetic_dp3m_print_tune_parameters, here we do nothing
1583  rs_err =P3M_DIPOLAR_real_space_error(box_l[0],coulomb.Dprefactor,r_cut_iL,dp3m.sum_dip_part,dp3m.sum_mu2,0.001);
1584

1589  0.0001*box_l[0],5.0*box_l[0],0.0001,dp3m.params.accuracy);
1590 
1591  }
1592 
1593  else
1594  /* even alpha=0 is ok, however, we cannot choose it since it kills the k-space error formula.
1595  Anyways, this very likely NOT the optimal solution */

1600

1606  P3M_TRACE(fprintf(stderr, "dipolar tuning resulting: %f -> %f %f\n", alpha_L, rs_err, ks_err));

1727 /** get the optimal alpha and the corresponding computation time for fixed mesh. *cao
1728  should contain an initial guess, which is then adapted by stepping up and down. Returns the time
1729  upon completion, -1 if the force evaluation does not work, and -2 if the accuracy cannot be met */
1730 static double dp3m_m_time(char **log, int mesh,
1731  int cao_min, int cao_max, int *_cao,
1732  double r_cut_iL_min, double r_cut_iL_max, double *_r_cut_iL,
1733  double *_alpha_L, double *_accuracy)
1734 {
1735  double best_time = -1, tmp_time, tmp_r_cut_iL, tmp_alpha_L=0.0, tmp_accuracy=0.0;
1736  /* in which direction improvement is possible. Initially, we dont know it yet. */
1737  int final_dir = 0;
1738  int cao = *_cao;
1739 
1740  P3M_TRACE(fprintf(stderr, "dp3m_m_time: Dmesh=%d, Dcao_min=%d, Dcao_max=%d, Drmin=%f, Drmax=%f\n",
1741  mesh, cao_min, cao_max, r_cut_iL_min, r_cut_iL_max));
1742  /* the initial step sets a timing mark. If there is no valid r_cut, we can only try
1743  to increase cao to increase the obtainable precision of the far formula. */
1744  do {
1745  tmp_time = dp3m_mc_time(log, mesh, cao, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL, &tmp_alpha_L, &tmp_accuracy);
1746  /* bail out if the force evaluation is not working */
1747  if (tmp_time == -1) return -1;
1748  /* cao is too large for this grid, but still the accuracy cannot be achieved, give up */
1749  if (tmp_time == -3) {
1750  P3M_TRACE(fprintf(stderr, "dp3m_m_time: no possible cao found\n"));
1751  return -2;
1752  }
1753  /* we have a valid time, start optimising from there */
1754  if (tmp_time >= 0) {
1755  best_time = tmp_time;
1756  *_r_cut_iL = tmp_r_cut_iL;
1757  *_alpha_L = tmp_alpha_L;
1758  *_accuracy = tmp_accuracy;
1759  *_cao = cao;
1760  break;
1761  }
1762  /* the required accuracy could not be obtained, try higher caos. Therefore optimisation can only be
1763  obtained with even higher caos, but not lower ones */
1764  P3M_TRACE(fprintf(stderr, "tclcommand_inter_magnetic_p3m_print_m_time: doesn't give precision, step up\n"));
1765  cao++;
1766  final_dir = 1;
1767  }
1768  while (cao <= cao_max);
1769  /* with this mesh, the required accuracy cannot be obtained. */
1770  if (cao > cao_max) return -2;
1771 
1772  /* at the boundaries, only the opposite direction can be used for optimisation */
1773  if (cao == cao_min) final_dir = 1;
1774  else if (cao == cao_max) final_dir = -1;
1775 
1776  P3M_TRACE(fprintf(stderr, "dp3m_m_time: final constraints dir %d\n", final_dir));
1777 
1778  if (final_dir == 0) {
1779  /* check in which direction we can optimise. Both directions are possible */
1780  double dir_times[3];
1781  for (final_dir = -1; final_dir <= 1; final_dir += 2) {
1782  dir_times[final_dir + 1] = tmp_time =
1783  dp3m_mc_time(log, mesh, cao + final_dir, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL, &tmp_alpha_L, &tmp_accuracy);
1784  /* bail out on errors, as usual */
1785  if (tmp_time == -1) return -1;
1786  /* in this direction, we cannot optimise, since we get into precision trouble */
1787  if (tmp_time < 0) continue;
1788 
1789  if (tmp_time < best_time) {
1790  best_time = tmp_time;
1791  *_r_cut_iL = tmp_r_cut_iL;
1792  *_alpha_L = tmp_alpha_L;
1793  *_accuracy = tmp_accuracy;
1794  *_cao = cao + final_dir;
1795  }
1796  }
1797  /* choose the direction which was optimal, if any of the two */
1798  if (dir_times[0] == best_time) { final_dir = -1; }
1799  else if (dir_times[2] == best_time) { final_dir = 1; }
1800  else {
1801  /* no improvement in either direction, however if one is only marginally worse, we can still try*/
1802  /* down is possible and not much worse, while up is either illegal or even worse */
1803  if ((dir_times[0] >= 0 && dir_times[0] < best_time + P3M_TIME_GRAN) &&
1804  (dir_times[2] < 0 || dir_times[2] > dir_times[0]))
1805  final_dir = -1;
1806  /* same for up */
1807  else if ((dir_times[2] >= 0 && dir_times[2] < best_time + P3M_TIME_GRAN) &&
1808  (dir_times[0] < 0 || dir_times[0] > dir_times[2]))
1809  final_dir = 1;
1810  else {
1811  /* really no chance for optimisation */
1812  P3M_TRACE(fprintf(stderr, "dp3m_m_time: Dmesh=%d final Dcao=%d time=%f\n",mesh, cao, best_time));
1813  return best_time;
1814  }
1815  }
1816  /* we already checked the initial cao and its neighbor */
1817  cao += 2*final_dir;
1818  }
1819  else {
1820  /* here some constraint is active, and we only checked the initial cao itself */
1821  cao += final_dir;
1822  }
1823 
1824  P3M_TRACE(fprintf(stderr, "dp3m_m_time: optimise in direction %d\n", final_dir));
1825 
1826  /* move cao into the optimisation direction until we do not gain anymore. */
1827  for (; cao >= cao_min && cao <= cao_max; cao += final_dir) {
1828  tmp_time = dp3m_mc_time(log, mesh, cao, r_cut_iL_min, r_cut_iL_max,
1829  &tmp_r_cut_iL, &tmp_alpha_L, &tmp_accuracy);
1830  /* bail out on errors, as usual */
1831  if (tmp_time == -1) return -1;
1832  /* if we cannot meet the precision anymore, give up */
1833  if (tmp_time < 0) break;
1834 
1835  if (tmp_time < best_time) {
1836  best_time = tmp_time;
1837  *_r_cut_iL = tmp_r_cut_iL;
1838  *_alpha_L = tmp_alpha_L;
1839  *_accuracy = tmp_accuracy;
1840  *_cao = cao;
1841  }
1842  /* no hope of further optimisation */
1843  else if (tmp_time > best_time + P3M_TIME_GRAN)
1844  break;
1845  }
1846  P3M_TRACE(fprintf(stderr, "tclcommand_inter_magnetic_p3m_print_m_time: Dmesh=%d final Dcao=%d Dr_cut=%f time=%f\n",mesh, *_cao, *_r_cut_iL, best_time));
1847  return best_time;
1848 }

1860  possible, i.e. fastest real space cutoff is determined.
