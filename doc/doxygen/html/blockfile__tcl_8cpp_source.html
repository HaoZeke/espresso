78 static int findNonWs(Tcl_Channel f)
79 {
80  int c, cont = 1;
81  while (cont) {
82  switch ((c = readchar(f))) {
83  case 0:
84  c = RETURN_CODE_EOF;
85  break;
86  case -1:
87  c = RETURN_CODE_ERROR;
88  break;
89  default: ;
90  }
91  if (!isspace(c))
92  break;
93  }
94  return c;

97 static int readString(Tcl_Channel f, char *buffer, int size)
98 {
99  int i = 0;
100  char c;
101  while(i < size - 1) {
102  c = (i == 0) ? findNonWs(f) : readchar(f);
103 
104  switch (c) {
105  case RETURN_CODE_EOF:
106  case RETURN_CODE_ERROR:
107  buffer[i] = 0;
108  return c;
109  default: ;
110  }
111  if (c == '}') {
112  buffer[i] = 0;
113  return 1;
114  }
115  else if (isspace(c)) {
116  buffer[i] = 0;
117  return 0;
118  }
119  buffer[i++] = c;
120  }
121  buffer[i] = 0;
122  return ((i == size - 1) ? 1 : 0);
123 }

125 /** read the title of a block.
126  @param f the file
127  @param index where to store the index if round
128  @return the number of open braces on success (1 if data follows or 0
129  if no data was contained in the block) or
130  \ref RETURN_CODE_EOF or \ref RETURN_CODE_ERROR
131  or \ref RETURN_CODE_FILE_FORMAT if no "{" is found. In that case
132  index contains the offending character.
133 */
134 static int block_startread(Tcl_Channel f, char index[MAXBLOCKTITLE])
135 {
136  int c;
137 
138  index[0] = 0;
139 
140  /* find the block start "{" */
141  switch (c = findNonWs(f)) {
142  case '{':
143  break;
144  case RETURN_CODE_EOF: return RETURN_CODE_EOF;
145  case RETURN_CODE_ERROR: return RETURN_CODE_ERROR;
146  default:
147  index[0] = c;
148  index[1] = 0;
149  return RETURN_CODE_FILE_FORMAT;
150  }
151 
152  /* since a block started, we consider from now on eof an error */
153 
154  switch (readString(f, index, MAXBLOCKTITLE)) {
155  case RETURN_CODE_EOF:
156  case RETURN_CODE_ERROR:
157  return RETURN_CODE_ERROR;
158  case 0:
159  return 1;
160  case 1:
161  return 0;
162  }
163  return RETURN_CODE_ERROR;
164 }

166 /** read a portion of the data of a block.
167  @param f the file
168  @param brace_count the number of open braces. If open_braces is one, reading
169  terminates at the matching closing bracket, else open_braces-1 additional
170  braces will be consumed.
171  @param data where to store the contained data
172  @param size the size of the buffer pointed to by "data".
173  @param spacer if this character is read and no brace is open, i. e. open_braces=1,
174  reading also terminates. A spacer of 0 disables this feature. This can be used
175  to read in for example space separated lists:
176  \verbatim {demoblock 1 2 3}\endverbatim
177  @return returns the number of open braces or \ref RETURN_CODE_EOF or \ref RETURN_CODE_ERROR.
178  The number of open_braces will be non-zero if the data space was exhausted.
179 */
180 static int block_continueread(Tcl_Channel f, int brace_count, char *data, int size,
181  char spacer)
182 {
183  char c;
184  int i;
185 
186  if (!data || !size)
187  return RETURN_CODE_ERROR;
188 
189  data[0] = 0;
190 
191  if (brace_count == 0)
192  return 0;
193 
194  /* scan block data until brace_count = 0 or space eaten up */
195  i = 0;
196  while (i < size - 1) {
197  if ((c = readchar(f)) <= 0) {
198  data[i] = 0;
199  return RETURN_CODE_ERROR;
200  }
201 
202  if (c == '{') brace_count++;
203  if (c == '}') {
204  if (--brace_count == 0) {
205  /* read complete block, strip trailing whitespaces */
206  while (i > 1 && isspace(data[i - 1]))
207  i--;
208  data[i] = 0;
209  return 0;
210  }
211  }
212  if (c == spacer && brace_count == 1) {
213  data[i] = 0;
214  return brace_count;
215  }
216  data[i++] = c;
217  }
218  data[i] = 0;
219 
220  return brace_count;
221 }

223 /** write a start tag to the file f.
224  @param f the file
225  @param index the tag to write (if index is a string longer than \ref MAXBLOCKTITLE,
226  an error is returned.
227  @return 0 on success or RETURN_CODE_ERROR
228 */
229 static int block_writestart(Tcl_Channel f, char index[MAXBLOCKTITLE])
230 {
231  if (strlen(index) >= MAXBLOCKTITLE)
232  return RETURN_CODE_ERROR;
233  if (!writestring(f, "{") ||
234  !writestring(f, index) ||
235  !writestring(f, " "))
236  return RETURN_CODE_ERROR;
237  return 0;
238 }
