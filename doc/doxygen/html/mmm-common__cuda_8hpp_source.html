23 // order hardcoded. mmm1d_recalcTables() typically does order less than 30.

38 int modpsi_init()
39 {
40  if (n_modPsi < modpsi_order)
41  {
42  create_mod_psi_up_to(modpsi_order);
43  }
44 
45  // linearize the coefficients array
46  linModPsi_offsets = (int*) realloc(linModPsi_offsets, sizeof(int) * 2*n_modPsi);
47  linModPsi_lengths = (int*) realloc(linModPsi_lengths, sizeof(int) * 2*n_modPsi);
48  for (int i = 0; i < 2*n_modPsi; i++)
49  {
50  if (i == 0)
51  linModPsi_offsets[i] = 0;
52  else
53  linModPsi_offsets[i] = linModPsi_offsets[i-1] + linModPsi_lengths[i-1];
54  linModPsi_lengths[i] = modPsi[i].n;
55  }
56  linModPsi = (mmm1dgpu_real*) realloc(linModPsi, sizeof(mmm1dgpu_real) * (linModPsi_offsets[2*n_modPsi-1] + linModPsi_lengths[2*n_modPsi-1]));
57  for (int i = 0; i < 2*n_modPsi; i++)
58  {
59  for (int j = 0; j < modPsi[i].n; j++)
60  {
61  linModPsi[linModPsi_offsets[i] + j] = (mmm1dgpu_real) modPsi[i].e[j]; // cast to single-precision if necessary
62  }
63  }
64 
65  for (int d = 0; d < deviceCount; d++)
66  {
67  cudaSetDevice(d);
68 
69  // copy to GPU
70  int linModPsiSize = linModPsi_offsets[2*n_modPsi-1] + linModPsi_lengths[2*n_modPsi-1];
71  if (linModPsiSize > modpsi_constant_size)
72  {
73  printf("ERROR: __constant__ device_linModPsi[] is not large enough\n");
74  exit(EXIT_FAILURE);
75  }
76  cuda_safe_mem( cudaMemcpyToSymbol(device_linModPsi_offsets, linModPsi_offsets, 2*n_modPsi*sizeof(int)) );
77  cuda_safe_mem( cudaMemcpyToSymbol(device_linModPsi_lengths, linModPsi_lengths, 2*n_modPsi*sizeof(int)) );
78  cuda_safe_mem( cudaMemcpyToSymbol(device_linModPsi, linModPsi, linModPsiSize*sizeof(mmm1dgpu_real)) );
79  cuda_safe_mem( cudaMemcpyToSymbol(device_n_modPsi, &n_modPsi, sizeof(int)) );
80  }
81 
82  return 0;
83 }
