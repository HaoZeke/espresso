191 int tclcommand_readmd(ClientData dummy, Tcl_Interp *interp,
192  int argc, char **argv)
193 {
194  char *row;
195  int pos_row[3] = { -1 }, v_row[3] = { -1 },
196  #ifdef DIPOLES
197  dip_row[3] = { -1 },
198  #endif
199  f_row[3] = { -1 };
200 
201  int av_pos = 0, av_v = 0,
202 #ifdef DIPOLES
203  av_dip=0,
204 #endif
205 #ifdef MASS
206  av_mass=0,
207 #endif
208 #ifdef SHANCHEN
209  av_solvation=0,
210 #endif
211  av_f = 0,
212 #ifdef ELECTROSTATICS
213  av_q = 0,
214 #endif
215  av_type = 0;
216 
217  int node, i;
218  struct MDHeader header;
219  Particle data;
220  int tcl_file_mode;
221  Tcl_Channel channel;
222 
223  if (argc != 2) {
224  Tcl_AppendResult(interp, "wrong # args: should be \"",
225  argv[0], " <file>\"",
226  (char *) NULL);
227  return (TCL_ERROR);
228  }
229 
230  if ((channel = Tcl_GetChannel(interp, argv[1], &tcl_file_mode)) == NULL)
231  return (TCL_ERROR);
232 
233  /* tune channel to binary translation, e.g. none */
234  Tcl_SetChannelOption(interp, channel, "-translation", "binary");
235 
236  Tcl_Read(channel, (char *)&header, sizeof(header));
237  /* check token */
238  if (strncmp(header.magic, MDMAGIC, 4) || header.n_rows < 0) {
239  Tcl_AppendResult(interp, "data file \"", argv[1],
240  "\" does not contain tcl MD data",
241  (char *) NULL);
242  return (TCL_ERROR);
243  }
244 
245  if (!particle_node)
246  build_particle_node();
247 
248  /* parse rows */
249  row = (char*)malloc(header.n_rows*sizeof(char));
250  for (i = 0; i < header.n_rows; i++) {
251  Tcl_Read(channel, (char *)&row[i], sizeof(char));
252  switch (row[i]) {
253  case POSX: pos_row[0] = i; break;
254  case POSY: pos_row[1] = i; break;
255  case POSZ: pos_row[2] = i; break;
256  case VX: v_row[0] = i; break;
257  case VY: v_row[1] = i; break;
258  case VZ: v_row[2] = i; break;
259 #ifdef DIPOLES
260  case MX: dip_row[0] = i; break;
261  case MY: dip_row[1] = i; break;
262  case MZ: dip_row[2] = i; break;
263 #endif
264  case FX: f_row[0] = i; break;
265  case FY: f_row[1] = i; break;
266  case FZ: f_row[2] = i; break;
267 #ifdef MASS
268  case MASSES: av_mass = 1; break;
269 #endif
270 #ifdef SHANCHEN
271  case SOLVATION: av_solvation = 1; break;
272 #endif
273 #ifdef ELECTROSTATICS
274  case Q: av_q = 1; break;
275 #endif
276  case TYPE: av_type = 1; break;
277  }
278  }
279 
280  /* *_row[0] tells if * data is completely available -
281  * otherwise we ignore it */
282  if (pos_row[0] != -1 && pos_row[1] != -1 && pos_row[2] != -1) {
283  av_pos = 1;
284  }
285  if (v_row[0] != -1 && v_row[1] != -1 && v_row[2] != -1) {
286  av_v = 1;
287  }
288  if (f_row[0] != -1 && f_row[1] != -1 && f_row[2] != -1) {
289  av_f = 1;
290  }
291 
292  #ifdef DIPOLES
293  if (dip_row[0] != -1 && dip_row[1] != -1 && dip_row[2] != -1) {
294  av_dip = 1;
295  }
296  #endif
297 
298 
299  while (!Tcl_Eof(channel)) {
300  Tcl_Read(channel, (char *)&data.p.identity, sizeof(int));
301  if (data.p.identity == -1)
302  break;
303 
304  /* printf("id=%d\n", data.identity); */
305 
306  if (data.p.identity < 0) {
307  Tcl_AppendResult(interp, "illegal data format in data file \"", argv[1],
308  "\", perhaps wrong file?",
309  (char *) NULL);
310  free(row);
311  return (TCL_ERROR);
312  }
313 
314  for (i = 0; i < header.n_rows; i++) {
315  switch (row[i]) {
316  case POSX: Tcl_Read(channel, (char *)&data.r.p[0], sizeof(double)); break;
317  case POSY: Tcl_Read(channel, (char *)&data.r.p[1], sizeof(double)); break;
318  case POSZ: Tcl_Read(channel, (char *)&data.r.p[2], sizeof(double)); break;
319  case VX: Tcl_Read(channel, (char *)&data.m.v[0], sizeof(double)); break;
320  case VY: Tcl_Read(channel, (char *)&data.m.v[1], sizeof(double)); break;
321  case VZ: Tcl_Read(channel, (char *)&data.m.v[2], sizeof(double)); break;
322  case FX: Tcl_Read(channel, (char *)&data.f.f[0], sizeof(double)); break;
323  case FY: Tcl_Read(channel, (char *)&data.f.f[1], sizeof(double)); break;
324  case FZ: Tcl_Read(channel, (char *)&data.f.f[2], sizeof(double)); break;
325  case MASSES:
326 #ifdef MASS
327  Tcl_Read(channel, (char *)&data.p.mass, sizeof(double)); break;
328 #else
329  {
330  double dummy_mass;
331  Tcl_Read(channel, (char *)&dummy_mass, sizeof(double)); break;
332  }
333 #endif
334 #ifdef ELECTROSTATICS
335  case Q: Tcl_Read(channel, (char *)&data.p.q, sizeof(double)); break;
336 #endif
337 #ifdef DIPOLES
338  case MX: Tcl_Read(channel, (char *)&data.r.dip[0], sizeof(double)); break;
339  case MY: Tcl_Read(channel, (char *)&data.r.dip[1], sizeof(double)); break;
340  case MZ: Tcl_Read(channel, (char *)&data.r.dip[2], sizeof(double)); break;
341 #endif
342 
343  case TYPE: Tcl_Read(channel, (char *)&data.p.type, sizeof(int)); break;
344  }
345  }
346 
347  node = (data.p.identity <= max_seen_particle) ? particle_node[data.p.identity] : -1;
348  if (node == -1) {
349  if (!av_pos) {
350  Tcl_AppendResult(interp, "new particle without position data",
351  (char *) NULL);
352  free(row);
353  return (TCL_ERROR);
354  }
355  }
356 
357  if (av_pos)
358  place_particle(data.p.identity, data.r.p);
359 #ifdef MASS
360  if (av_mass)
361  set_particle_mass(data.p.identity, data.p.mass);
362 #endif
363 #ifdef SHANCHEN
364  if (av_solvation)
365  set_particle_solvation(data.p.identity, data.p.solvation);
366 #endif
367 #ifdef ELECTROSTATICS
368  if (av_q)
369  set_particle_q(data.p.identity, data.p.q);
370 #endif
371 #ifdef DIPOLES
372  if (av_dip)
373  set_particle_dip(data.p.identity, data.r.dip);
374 #endif
375  if (av_v)
376  set_particle_v(data.p.identity, data.m.v);
377  if (av_f)
378  set_particle_f(data.p.identity, data.f.f);
379  if (av_type)
380  set_particle_type(data.p.identity, data.p.type);
381  }
382 
383  free(row);
384  return TCL_OK;
385 }
