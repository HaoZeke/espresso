108 /** Calculate cell grid dimensions, cell sizes and number of cells.
109  * Calculates the cell grid, based on \ref local_box_l and \ref
110  * max_range. If the number of cells is larger than \ref
111  * max_num_cells, it increases max_range until the number of cells is
112  * smaller or equal \ref max_num_cells. It sets: \ref
113  * DomainDecomposition::cell_grid, \ref
114  * DomainDecomposition::ghost_cell_grid, \ref
115  * DomainDecomposition::cell_size, \ref
116  * DomainDecomposition::inv_cell_size, and \ref n_cells.
117  */
118 void dd_create_cell_grid()
119 {
120  int i,n_local_cells,new_cells,min_ind;
121  double cell_range[3], min_size, scale, volume;
122  CELL_TRACE(fprintf(stderr, "%d: dd_create_cell_grid: max_range %f\n",this_node,max_range));
123  CELL_TRACE(fprintf(stderr, "%d: dd_create_cell_grid: local_box %f-%f, %f-%f, %f-%f,\n",this_node,my_left[0],my_right[0],my_left[1],my_right[1],my_left[2],my_right[2]));
124 
125  /* initialize */
126  cell_range[0]=cell_range[1]=cell_range[2] = max_range;
127 
128  if (max_range < ROUND_ERROR_PREC*box_l[0]) {
129  /* this is the initialization case */
130 #ifdef LEES_EDWARDS
131  dd.cell_grid[0] = 2;
132  dd.cell_grid[1] = 1;
133  dd.cell_grid[2] = 1;
134  n_local_cells = 2;
135 #else
136  n_local_cells = dd.cell_grid[0] = dd.cell_grid[1] = dd.cell_grid[2]=1;
137 #endif
138  }
139  else {
140  /* Calculate initial cell grid */
141  volume = local_box_l[0];
142  for(i=1;i<3;i++) volume *= local_box_l[i];
143  scale = pow(max_num_cells/volume, 1./3.);
144  for(i=0;i<3;i++) {
145  /* this is at least 1 */
146  dd.cell_grid[i] = (int)ceil(local_box_l[i]*scale);
147  cell_range[i] = local_box_l[i]/dd.cell_grid[i];
148 
149  if ( cell_range[i] < max_range ) {
150  /* ok, too many cells for this direction, set to minimum */
151  dd.cell_grid[i] = (int)floor(local_box_l[i]/max_range);
152  if ( dd.cell_grid[i] < 1 ) {
153  ostringstream msg;
154  msg << "interaction range " << max_range << " in direction "
155  << i << " is larger than the local box size " << local_box_l[i];
156  runtimeError(msg);
157  dd.cell_grid[i] = 1;
158  }
159 #ifdef LEES_EDWARDS
160  if ( (i == 0) && (dd.cell_grid[0] < 2) ) {
161  ostringstream msg;
162  msg << "interaction range " << max_range << " in direction "
163  << i << " is larger than half the local box size " << local_box_l[i] << "/2";
164  runtimeError(msg);
165  dd.cell_grid[0] = 2;
166  }
167 #endif
168  cell_range[i] = local_box_l[i]/dd.cell_grid[i];
169  }
170  }
171 
172  /* It may be necessary to asymmetrically assign the scaling to the coordinates, which the above approach will not do.
173  For a symmetric box, it gives a symmetric result. Here we correct that. */
174  for (;;) {
175 
176  n_local_cells = dd.cell_grid[0] * dd.cell_grid[1] * dd.cell_grid[2];
177 
178  /* done */
179  if (n_local_cells <= max_num_cells)
180  break;
181 
182  /* find coordinate with the smallest cell range */
183  min_ind = 0;
184  min_size = cell_range[0];
185 
186 #ifdef LEES_EDWARDS
187  for (i = 2; i >= 1; i--) {/*preferably have thin slices in z or y... this is more efficient for Lees Edwards*/
188 #else
189  for (i = 1; i < 3; i++) {
190 #endif
191  if (dd.cell_grid[i] > 1 && cell_range[i] < min_size) {
192  min_ind = i;
193  min_size = cell_range[i];
194  }
195  }
196  CELL_TRACE(fprintf(stderr, "%d: minimal coordinate %d, size %f, grid %d\n", this_node,min_ind, min_size, dd.cell_grid[min_ind]));
197 
198  dd.cell_grid[min_ind]--;
199  cell_range[min_ind] = local_box_l[min_ind]/dd.cell_grid[min_ind];
200  }
201  CELL_TRACE(fprintf(stderr, "%d: final %d %d %d\n", this_node, dd.cell_grid[0], dd.cell_grid[1], dd.cell_grid[2]));
202 
203  /* sanity check */
204  if (n_local_cells < min_num_cells) {
205  ostringstream msg;
206  msg << "number of cells "<< n_local_cells << " is smaller than minimum " << min_num_cells <<
207  " (interaction range too large or min_num_cells too large)";
208  runtimeError(msg);
209  }
210  }
211 
212  /* quit program if unsuccesful */
213  if(n_local_cells > max_num_cells) {
214  ostringstream msg;
215  msg << "no suitable cell grid found ";
216  runtimeError(msg);
217  }
218 
219  /* now set all dependent variables */
220  new_cells=1;
221  for(i=0;i<3;i++) {
222  dd.ghost_cell_grid[i] = dd.cell_grid[i]+2;
223 #ifdef LEES_EDWARDS
224  //Hack alert: only the boundary y-layers actually need the extra-thick ghost cell grid,
225  //so some memory (and copies) are wasted in the name of simpler code.
226  if( i == 0 ){dd.ghost_cell_grid[i]++;}
227 #endif
228  new_cells *= dd.ghost_cell_grid[i];
229  dd.cell_size[i] = local_box_l[i]/(double)dd.cell_grid[i];
230  dd.inv_cell_size[i] = 1.0 / dd.cell_size[i];
231  }
232  max_skin = dmin(dmin(dd.cell_size[0],dd.cell_size[1]),dd.cell_size[2]) - max_cut;
233 
234  /* allocate cell array and cell pointer arrays */
235  realloc_cells(new_cells);
236  realloc_cellplist(&local_cells, local_cells.n = n_local_cells);
237  realloc_cellplist(&ghost_cells, ghost_cells.n = new_cells-n_local_cells);
238 
239  CELL_TRACE(fprintf(stderr, "%d: dd_create_cell_grid, n_cells=%d, local_cells.n=%d, ghost_cells.n=%d, dd.ghost_cell_grid=(%d,%d,%d)\n", this_node, n_cells,local_cells.n,ghost_cells.n,dd.ghost_cell_grid[0],dd.ghost_cell_grid[1],dd.ghost_cell_grid[2]));
240 }

294 /** Create communicators for cell structure domain decomposition. (see \ref GhostCommunicator) */
295 void dd_prepare_comm(GhostCommunicator *comm, int data_parts)
296 {
297  int dir,lr,i,cnt, num, n_comm_cells[3];
298  int lc[3],hc[3],done[3]={0,0,0};
299 
300  /* calculate number of communications */
301  num = 0;
302  for(dir=0; dir<3; dir++) {
303  for(lr=0; lr<2; lr++) {
304  /* No communication for border of non periodic direction */
305  if( PERIODIC(dir) || (boundary[2*dir+lr] == 0) )
306  {
307  if(node_grid[dir] == 1 ) num++;
308  else num += 2;
309  }
310  }
311  }
312 
313  /* prepare communicator */
314  CELL_TRACE(fprintf(stderr,"%d Create Communicator: prep_comm data_parts %d num %d\n",this_node,data_parts,num));
315  prepare_comm(comm, data_parts, num);
316 
317  /* number of cells to communicate in a direction */
318  n_comm_cells[0] = dd.cell_grid[1] * dd.cell_grid[2];
319  n_comm_cells[1] = dd.cell_grid[2] * dd.ghost_cell_grid[0];
320  n_comm_cells[2] = dd.ghost_cell_grid[0] * dd.ghost_cell_grid[1];
321 
322  cnt=0;
323  /* direction loop: x, y, z */
324  for(dir=0; dir<3; dir++) {
325  lc[(dir+1)%3] = 1-done[(dir+1)%3];
326  lc[(dir+2)%3] = 1-done[(dir+2)%3];
327  hc[(dir+1)%3] = dd.cell_grid[(dir+1)%3]+done[(dir+1)%3];
328  hc[(dir+2)%3] = dd.cell_grid[(dir+2)%3]+done[(dir+2)%3];
329  /* lr loop: left right */
330  /* here we could in principle build in a one sided ghost
331  communication, simply by taking the lr loop only over one
332  value */
333  for(lr=0; lr<2; lr++) {
334  if(node_grid[dir] == 1) {
335  /* just copy cells on a single node */
336  if( PERIODIC(dir ) || (boundary[2*dir+lr] == 0) )
337  {
338  comm->comm[cnt].type = GHOST_LOCL;
339  comm->comm[cnt].node = this_node;
340 
341  /* Buffer has to contain Send and Recv cells -> factor 2 */
342  comm->comm[cnt].part_lists = (ParticleList**)malloc(2*n_comm_cells[dir]*sizeof(ParticleList *));
343  comm->comm[cnt].n_part_lists = 2*n_comm_cells[dir];
344  /* prepare folding of ghost positions */
345  if((data_parts & GHOSTTRANS_POSSHFTD) && boundary[2*dir+lr] != 0) {
346  comm->comm[cnt].shift[dir] = boundary[2*dir+lr]*box_l[dir];
347  }
348 
349  /* fill send comm cells */
350  lc[dir] = hc[dir] = 1 + lr * (dd.cell_grid[dir]-1);
351 
352  dd_fill_comm_cell_lists(comm->comm[cnt].part_lists,lc,hc);
353  CELL_TRACE(fprintf(stderr,"%d: prep_comm %d copy to grid (%d,%d,%d)-(%d,%d,%d)\n",this_node,cnt,
354  lc[0],lc[1],lc[2],hc[0],hc[1],hc[2]));
355 
356  /* fill recv comm cells */
357  lc[dir] = hc[dir] = 0 +( 1 - lr ) * ( dd.cell_grid[dir] + 1 );
358 
359  /* place recieve cells after send cells */
360  dd_fill_comm_cell_lists(&comm->comm[cnt].part_lists[n_comm_cells[dir]],lc,hc);
361  CELL_TRACE(fprintf(stderr,"%d: prep_comm %d copy from grid (%d,%d,%d)-(%d,%d,%d)\n",this_node,cnt,lc[0],lc[1],lc[2],hc[0],hc[1],hc[2]));
362  cnt++;
363  }
364  }
365  else {
366  /* i: send/recv loop */
367  for(i=0; i<2; i++) {
368  if( PERIODIC(dir) || (boundary[2*dir+lr] == 0) )
369  if((node_pos[dir]+i)%2==0) {
370  comm->comm[cnt].type = GHOST_SEND;
371  comm->comm[cnt].node = node_neighbors[2*dir+lr];
372  comm->comm[cnt].part_lists = (ParticleList**)malloc(n_comm_cells[dir]*sizeof(ParticleList *));
373  comm->comm[cnt].n_part_lists = n_comm_cells[dir];
374  /* prepare folding of ghost positions */
375  if((data_parts & GHOSTTRANS_POSSHFTD) && boundary[2*dir+lr] != 0) {
376  comm->comm[cnt].shift[dir] = boundary[2*dir+lr]*box_l[dir];
377  }
378 
379  lc[dir] = hc[dir] = 1 + lr * ( dd.cell_grid[dir] - 1 );
380 
381  dd_fill_comm_cell_lists(comm->comm[cnt].part_lists,lc,hc);
382 
383  CELL_TRACE(fprintf(stderr,"%d: prep_comm %d send to node %d grid (%d,%d,%d)-(%d,%d,%d)\n",this_node,cnt,
384  comm->comm[cnt].node,lc[0],lc[1],lc[2],hc[0],hc[1],hc[2]));
385  cnt++;
386  }
387  if( PERIODIC(dir) || (boundary[2*dir+(1-lr)] == 0) )
388  if((node_pos[dir]+(1-i))%2==0) {
389  comm->comm[cnt].type = GHOST_RECV;
390  comm->comm[cnt].node = node_neighbors[2*dir+(1-lr)];
391  comm->comm[cnt].part_lists = (ParticleList**)malloc(n_comm_cells[dir]*sizeof(ParticleList *));
392  comm->comm[cnt].n_part_lists = n_comm_cells[dir];
393 
394  lc[dir] = hc[dir] = ( 1 - lr ) * ( dd.cell_grid[dir] + 1 );
395 
396  dd_fill_comm_cell_lists(comm->comm[cnt].part_lists,lc,hc);
397  CELL_TRACE(fprintf(stderr,"%d: prep_comm %d recv from node %d grid (%d,%d,%d)-(%d,%d,%d)\n",this_node,cnt,
398  comm->comm[cnt].node,lc[0],lc[1],lc[2],hc[0],hc[1],hc[2]));
399  cnt++;
400  }
401  }
402  }
403  done[dir]=1;
404  }
405  }

697 void dd_on_geometry_change(int flags) {
698 
699  /* Realignment of comms along the periodic y-direction is needed */
700  if (flags & CELL_FLAG_LEES_EDWARDS) {
701  CELL_TRACE(fprintf(stderr,"%d: dd_on_geometry_change responding to Lees-Edwards offset change.\n", this_node);)
702 
703 #ifdef LEES_EDWARDS
704  le_mgr.update_on_le_offset_change();
705 
706  le_dd_dynamic_update_comm(&le_mgr, &cell_structure.ghost_cells_comm,
707  GHOSTTRANS_PARTNUM,
708  LE_COMM_FORWARDS);
709  le_dd_dynamic_update_comm(&le_mgr, &cell_structure.exchange_ghosts_comm,
710  (GHOSTTRANS_PROPRTS | GHOSTTRANS_POSITION | GHOSTTRANS_POSSHFTD),
711  LE_COMM_FORWARDS);
712  le_dd_dynamic_update_comm(&le_mgr, &cell_structure.update_ghost_pos_comm,
713  (GHOSTTRANS_POSITION | GHOSTTRANS_POSSHFTD),
714  LE_COMM_FORWARDS);
715  le_dd_dynamic_update_comm(&le_mgr, &cell_structure.collect_ghost_force_comm,
716  GHOSTTRANS_FORCE,
717  LE_COMM_BACKWARDS);
718 #ifdef LB
719  le_dd_dynamic_update_comm(&cell_structure.ghost_lbcoupling_comm, GHOSTTRANS_COUPLING);
720 #endif
721 
722  /* prefetch status may have changed? */
723  dd_assign_prefetches(&cell_structure.ghost_cells_comm);
724  dd_assign_prefetches(&cell_structure.exchange_ghosts_comm);
725  dd_assign_prefetches(&cell_structure.update_ghost_pos_comm);
726  dd_assign_prefetches(&cell_structure.collect_ghost_force_comm);
727 #endif
728 
729  }
730 
731  /* check that the CPU domains are still sufficiently large. */
732  for (int i = 0; i < 3; i++)
733  if (local_box_l[i] < max_range) {
734  ostringstream msg;
735  msg <<"box_l in direction " << i << " is too small";
736  runtimeError(msg);
737  }
738 
739  /* A full resorting is necessary if the grid has changed. We simply
740  don't have anything fast for this case. Probably also not
741  necessary. */
742  if (flags & CELL_FLAG_GRIDCHANGED) {
743  CELL_TRACE(fprintf(stderr,"%d: dd_on_geometry_change full redo\n",
744  this_node));
745  cells_re_init(CELL_STRUCTURE_CURRENT);
746  return;
747  }
748 
749  /* otherwise, re-set our geometrical dimensions which have changed
750  (in addition to the general ones that \ref grid_changed_box_l
751  takes care of) */
752  for(int i=0; i<3; i++) {
753  dd.cell_size[i] = local_box_l[i]/(double)dd.cell_grid[i];
754  dd.inv_cell_size[i] = 1.0 / dd.cell_size[i];
755  }
756  double min_cell_size = dmin(dmin(dd.cell_size[0],dd.cell_size[1]),dd.cell_size[2]);
757  max_skin = min_cell_size - max_cut;
758 
759  CELL_TRACE(fprintf(stderr, "%d: dd_on_geometry_change: max_range = %f, min_cell_size = %f, max_skin = %f\n", this_node, max_range, min_cell_size, max_skin));
760 
761  if (max_range > min_cell_size) {
762  /* if new box length leads to too small cells, redo cell structure
763  using smaller number of cells. */
764  cells_re_init(CELL_STRUCTURE_DOMDEC);
765  return;
766  }
767 
768  /* If we are not in a hurry, check if we can maybe optimize the cell
769  system by using smaller cells. */
770  if (!(flags & CELL_FLAG_FAST)) {
771  int i;
772  for(i=0; i<3; i++) {
773  int poss_size = (int)floor(local_box_l[i]/max_range);
774  if (poss_size > dd.cell_grid[i])
775  break;
776  }
777  if (i < 3) {
778  /* new range/box length allow smaller cells, redo cell structure,
779  possibly using smaller number of cells. */
780  cells_re_init(CELL_STRUCTURE_DOMDEC);
781  return;
782  }
783  }
784 #ifdef LEES_EDWARDS
785  le_dd_update_communicators_w_boxl(&le_mgr);
786 #else
787  dd_update_communicators_w_boxl();
788 #endif
789  /* tell other algorithms that the box length might have changed. */
790  on_boxl_change();
791 }

917 /************************************************************/
918 void dd_exchange_and_sort_particles(int global_flag)
919 {
920  int dir, c, p, i, finished=0;
921  ParticleList *cell,*sort_cell, send_buf_l, send_buf_r, recv_buf_l, recv_buf_r;
922  Particle *part;
923  CELL_TRACE(fprintf(stderr,"%d: dd_exchange_and_sort_particles(%d):\n",this_node,global_flag));
924  CELL_TRACE(fprintf(stderr,"%d: node_neighbors are %d %d %d %d %d %d\n",this_node,
925  node_neighbors[0], node_neighbors[1], node_neighbors[2],
926  node_neighbors[3], node_neighbors[4], node_neighbors[5]));
927 
928  init_particlelist(&send_buf_l);
929  init_particlelist(&send_buf_r);
930  init_particlelist(&recv_buf_l);
931  init_particlelist(&recv_buf_r);
932  while(finished == 0 ) {
933  finished=1;
934  /* direction loop: x, y, z */
935  for(dir=0; dir<3; dir++) {
936  if(node_grid[dir] > 1) {
937  /* Communicate particles that have left the node domain */
938  /* particle loop */
939  for(c=0; c<local_cells.n; c++) {
940  cell = local_cells.cell[c];
941  for (p = 0; p < cell->n; p++) {
942  part = &cell->part[p];
943  /* Move particles to the left side */
944  // Without the factor 0.5 in front of ROUND_ERROR_PREC, particles sitting exactly on the boundary
945  // may be accepted (i.e. not sent) here and rejected later on by dd_save_position_to_cell
946  if(part->r.p[dir] - my_left[dir] < -0.5*ROUND_ERROR_PREC*box_l[dir]) {
947  if( PERIODIC(dir) || (boundary[2*dir]==0) )
948  {
949  CELL_TRACE(fprintf(stderr,"%d: dd_ex_and_sort_p: send part left %d\n",this_node,part->p.identity));
950  local_particles[part->p.identity] = NULL;
951  move_indexed_particle(&send_buf_l, cell, p);
952  if(p < cell->n) p--;
953  }
954  }
955  /* Move particles to the right side */
956  // Factor 0.5 see above
957  else if(part->r.p[dir] - my_right[dir] >= 0.5*ROUND_ERROR_PREC*box_l[dir]) {
958  if( PERIODIC(dir) || (boundary[2*dir+1]==0) )
959  {
960  CELL_TRACE(fprintf(stderr,"%d: dd_ex_and_sort_p: send part right %d\n",this_node,part->p.identity));
961  local_particles[part->p.identity] = NULL;
962  move_indexed_particle(&send_buf_r, cell, p);
963  if(p < cell->n) p--;
964  }
965  }
966  /* Sort particles in cells of this node during last direction */
967  else if(dir==2) {
968  sort_cell = dd_save_position_to_cell(part->r.p);
969  if(sort_cell != cell) {
970  if(sort_cell==NULL) {
971  CELL_TRACE(fprintf(stderr,"%d: dd_exchange_and_sort_particles: Take another loop",this_node));
972  CELL_TRACE(fprintf(stderr, "%d: dd_exchange_and_sort_particles: CP1 Particle %d (%f,%f,%f) not inside node domain.\n",
973  this_node,part->p.identity,part->r.p[0],part->r.p[1],part->r.p[2]));
974  finished=0;
975  sort_cell = local_cells.cell[0];
976  if(sort_cell != cell) {
977  move_indexed_particle(sort_cell, cell, p);
978  if(p < cell->n) p--;
979  }
980  }
981  else {
982  move_indexed_particle(sort_cell, cell, p);
983  if(p < cell->n) p--;
984  }
985  }
986  }
987  }
988  }
989 
990  /* Exchange particles */
991 #ifndef LEES_EDWARDS
992  CELL_TRACE(fprintf(stderr,"%d: send receive %d\n",this_node, dir));
993 
994  if(node_pos[dir]%2==0) {
995  send_particles(&send_buf_l, node_neighbors[2*dir]);
996  recv_particles(&recv_buf_r, node_neighbors[2*dir+1]);
997  send_particles(&send_buf_r, node_neighbors[2*dir+1]);
998  recv_particles(&recv_buf_l, node_neighbors[2*dir]);
999  }
1000  else {
1001  recv_particles(&recv_buf_r, node_neighbors[2*dir+1]);
1002  send_particles(&send_buf_l, node_neighbors[2*dir]);
1003  recv_particles(&recv_buf_l, node_neighbors[2*dir]);
1004  send_particles(&send_buf_r, node_neighbors[2*dir+1]);
1005  }
1006 #else
1007  int ii, nn, lr;
1008  for( ii = 0; ii <2; ii++){
1009 
1010  nn = node_neighbors[2*dir+ii];
1011  lr = node_neighbor_lr[2*dir+ii];
1012 
1013  if( lr == 1 ){
1014  if( nn > this_node ){
1015  send_particles(&send_buf_r, nn);
