55 /** Computes the Debye_Hueckel pair force and adds this
56  force to the particle forces (see \ref tclcommand_inter).
57  @param p1 Pointer to first particle.
58  @param p2 Pointer to second/middle particle.
59  @param d Vector pointing from p1 to p2.
60  @param dist Distance between p1 and p2.
61  @param force returns the force on particle 1.
62 */
63 #ifdef COULOMB_DEBYE_HUECKEL
64 inline void add_dh_coulomb_pair_force(Particle *p1, Particle *p2, double d[3], double dist, double force[3]) {
65  double fac;
66 
67  const double q1 = p1->p.q;
68  const double q2 = p2->p.q;
69 
70  if((q1 == 0.0) || (q2 == 0.0))
71  return;
72 
73  if(dist >= dh_params.r_cut)
74  return;
75 
76  if(dist > dh_params.r1) {
77  const double kappa_dist = dh_params.kappa*dist;
78  fac = coulomb.prefactor * q1 * q2 * exp(-kappa_dist)/(dh_params.eps_ext * dist*dist*dist) * (1.0 + kappa_dist);
79  }
80  else if (dist > dh_params.r0)
81  fac = coulomb.prefactor * q1 * q2 * exp(dh_params.alpha*(dist - dh_params.r0)) / (dh_params.eps_int * dist*dist*dist) * (1. - dh_params.alpha*dist);
82  else
83  fac = coulomb.prefactor * q1 * q2 * p2->p.q / (dh_params.eps_int * dist*dist*dist);
84 
85  force[0] += fac * d[0];
86  force[1] += fac * d[1];
87  force[2] += fac * d[2];
88 }
89 #else
90 inline void add_dh_coulomb_pair_force(Particle *p1, Particle *p2, double d[3], double dist, double force[3])
91 {
92  int j;
93  double kappa_dist, fac;
94 
95  if(dist < dh_params.r_cut) {
96  if(dh_params.kappa > 0.0) {
97  /* debye hueckel case: */
98  kappa_dist = dh_params.kappa*dist;
99  fac = coulomb.prefactor * p1->p.q * p2->p.q * (exp(-kappa_dist)/(dist*dist*dist)) * (1.0 + kappa_dist);
100  }
101  else {
102  /* pure coulomb case: */
103  fac = coulomb.prefactor * p1->p.q * p2->p.q / (dist*dist*dist);
104  }
105  for(j=0;j<3;j++)
106  force[j] += fac * d[j];
107 
108  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: DH f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist,fac));

110  }
111 }

115 inline double dh_coulomb_pair_energy(Particle *p1, Particle *p2, double dist) {
116  const double q1 = p1->p.q;
117  const double q2 = p2->p.q;
118 
119  if((q1 == 0.0) || (q2 == 0.0))
120  return 0.0;
121 
122  const double fac = q1 * q2 * coulomb.prefactor;
123 
124  if((dist < dh_params.r_cut)) {
125  if(dist < dh_params.r0) {
126  return fac / (dh_params.eps_int*dist);
127  }
128  if (dist < dh_params.r1) {
129  return fac * exp(dh_params.alpha*(dist - dh_params.r0)) / (dh_params.eps_int * dist);
130  }
131  return fac * exp(-dh_params.kappa*dist) / (dh_params.eps_ext * dist);
132  }
133  return 0.0;
134 }
135 #else
136 inline double dh_coulomb_pair_energy(Particle *p1, Particle *p2, double dist)
137 {
138  if(dist < dh_params.r_cut) {
139  if(dh_params.kappa > 0.0)
140  return coulomb.prefactor * p1->p.q * p2->p.q * exp(-dh_params.kappa*dist) / dist;
141  else
142  return coulomb.prefactor * p1->p.q * p2->p.q / dist;
143  }
144  return 0.0;
145 }
