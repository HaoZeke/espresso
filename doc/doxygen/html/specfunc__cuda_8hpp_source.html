164 __device__ mmm1dgpu_real evaluateAsChebychevSeriesAt(mmm1dgpu_real *c, int n, mmm1dgpu_real x)
165 {
166  int j;
167  mmm1dgpu_real x2 = 2 * x;
168  mmm1dgpu_real dd = c[n - 1];
169  mmm1dgpu_real d = x2*dd + c[n - 2];
170  for(j = n - 3; j >= 1; j--) {
171  mmm1dgpu_real tmp = d;
172  d = x2*d - dd + c[j];
173  dd = tmp;
174  }
175  return x*d - dd + c[0]/2;
176 }

178 __device__ mmm1dgpu_real evaluateAsTaylorSeriesAt(mmm1dgpu_real *c, int n, mmm1dgpu_real x)
179 {
180  int cnt = n - 1;
181  mmm1dgpu_real r = c[cnt];
182  while (--cnt >= 0)
183  r = r*x + c[cnt];
184  return r;
185 }
