87 void p3m_add_block(double *in, double *out, int start[3], int size[3], int dim[3])
88 {
89  /* fast,mid and slow changing indices */
90  int f,m,s;
91  /* linear index of in grid, linear index of out grid */
92  int li_in=0,li_out=0;
93  /* offsets for indizes in output grid */
94  int m_out_offset,s_out_offset;
95 
96  li_out = start[2] + ( dim[2]*( start[1] + (dim[1]*start[0]) ) );
97  m_out_offset = dim[2] - size[2];
98  s_out_offset = (dim[2] * (dim[1] - size[1]));
99 
100  for(s=0 ;s<size[0]; s++) {
101  for(m=0; m<size[1]; m++) {
102  for(f=0; f<size[2]; f++) {
103  out[li_out++] += in[li_in++];
104  }
105  li_out += m_out_offset;
106  }
107  li_out += s_out_offset;
108  }
