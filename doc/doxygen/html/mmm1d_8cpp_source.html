71 static double far_error(int P, double minrad)
72 {
73  // this uses an upper bound to all force components and the potential
74  double rhores = 2*M_PI*uz*minrad;
75  double pref = 4*uz*dmax(1, 2*M_PI*uz);
76 
77  return pref*K1(rhores*P)*exp(rhores)/rhores*(P - 1 + 1/rhores);
78 }

80 static double determine_minrad(double maxPWerror, int P)
81 {
82  // bisection to search for where the error is maxPWerror
83  double rgranularity = MIN_RAD*box_l[2];
84  double rmin = rgranularity;
85  double rmax = dmin(box_l[0], box_l[1]);
86  double errmin = far_error(P, rmin);
87  double errmax = far_error(P, rmax);
88  if (errmin < maxPWerror) {
89  // we can do almost all radii with this P
90  return rmin;
91  }
92  if (errmax > maxPWerror) {
93  // make sure that this switching radius cannot be reached
94  return 2*dmax(box_l[0], box_l[1]);
95  }
96 
97  while (rmax - rmin > rgranularity) {
98  double c = 0.5*(rmin + rmax);
99  double errc = far_error(P, c);
100  if (errc > maxPWerror) {
101  rmin = c;
102  } else {
103  rmax = c;
104  }
105  }
106  return 0.5*(rmin + rmax);
107 }

118 static void prepare_polygamma_series(double maxPWerror, double maxrad2)
119 {
120  /* polygamma, determine order */
121  int n;
122  double err;
123  double rhomax2nm2, rhomax2 = uz2*maxrad2;
124  /* rhomax2 < 1, so rhomax2m2 falls monotonously */
125 
126  n = 1;
127  rhomax2nm2 = 1.0;
128  do {
129  create_mod_psi_up_to(n+1);
130 
131  /* |uz*z| <= 0.5 */
132  err = 2*n*fabs(mod_psi_even(n, 0.5))*rhomax2nm2;
133  rhomax2nm2 *= rhomax2;
134  n++;
135  // fprintf(stderr, "%f\n", err);
136  }
137  while (err > 0.1*maxPWerror);
138 }

187 void add_mmm1d_coulomb_pair_force(double chpref, double d[3], double r2, double r, double force[3])
188 {
189  int dim;
190  double F[3];
191  double rxy2, rxy2_d, z_d;
192  double pref;
193  double Fx, Fy, Fz;
194 
195 
196  rxy2 = d[0]*d[0] + d[1]*d[1];
197  rxy2_d = rxy2*uz2;
198  z_d = d[2]*uz;
199 
200  if (rxy2 <= mmm1d_params.far_switch_radius_2) {
201  /* near range formula */
202  double sr, sz, r2nm1, rt, rt2, shift_z;
203  int n;
204 
205  /* polygamma summation */
206  sr = 0;
207  sz = mod_psi_odd(0, z_d);
208 
209  r2nm1 = 1.0;
210  for (n = 1; n < n_modPsi; n++) {
211  double deriv = 2*n;
212  double mpe = mod_psi_even(n, z_d);
213  double mpo = mod_psi_odd(n, z_d);
214  double r2n = r2nm1*rxy2_d;
215 
216  sz += r2n*mpo;
217  sr += deriv*r2nm1*mpe;
218 
219  if (fabs(deriv*r2nm1*mpe) < mmm1d_params.maxPWerror)
220  break;
221 
222  r2nm1 = r2n;
223  }
224  // fprintf(stderr, "max_n %d\n", n);
225 
226  Fx = prefL3_i*sr*d[0];
227  Fy = prefL3_i*sr*d[1];
228  Fz = prefuz2*sz;
229 
230  /* real space parts */
231 
232  pref = coulomb.prefactor/(r2*r);
233  Fx += pref*d[0];
234  Fy += pref*d[1];
235  Fz += pref*d[2];
236 
237  shift_z = d[2] + box_l[2];
238  rt2 = rxy2 + shift_z*shift_z;
239  rt = sqrt(rt2);
240  pref = coulomb.prefactor/(rt2*rt);
241  Fx += pref*d[0];
242  Fy += pref*d[1];
243  Fz += pref*shift_z;
244 
245  shift_z = d[2] - box_l[2];
246  rt2 = rxy2 + shift_z*shift_z;
247  rt = sqrt(rt2);
248  pref = coulomb.prefactor/(rt2*rt);
249  Fx += pref*d[0];
250  Fy += pref*d[1];
251  Fz += pref*shift_z;
252 
253  F[0] = Fx;
254  F[1] = Fy;
255  F[2] = Fz;
256  }
257  else {
258  /* far range formula */
259  double rxy = sqrt(rxy2);
260  double rxy_d = rxy*uz;
261  double sr = 0, sz = 0;
262  int bp;
263 
264  for (bp = 1; bp < MAXIMAL_B_CUT; bp++) {
265  if (bessel_radii[bp-1] < rxy)
266  break;
267 
268  double fq = C_2PI*bp, k0, k1;
269 #ifdef BESSEL_MACHINE_PREC
270  k0 = K0(fq*rxy_d);
271  k1 = K1(fq*rxy_d);
272 #else
273  LPK01(fq*rxy_d, &k0, &k1);
274 #endif
275  sr += bp*k1*cos(fq*z_d);
276  sz += bp*k0*sin(fq*z_d);
277  }
278  sr *= uz2*4*C_2PI;
279  sz *= uz2*4*C_2PI;
280 
281  pref = coulomb.prefactor*(sr/rxy + 2*uz/rxy2);
282 
283  F[0] = pref*d[0];
284  F[1] = pref*d[1];
285  F[2] = coulomb.prefactor*sz;
286  }
287 
288  for (dim = 0; dim < 3; dim++)
289  force[dim] += chpref * F[dim];
290 }

292 double mmm1d_coulomb_pair_energy(Particle *p1, Particle *p2, double d[3], double r2, double r)
293 {
294  double chpref = p1->p.q*p2->p.q;
295  double rxy2, rxy2_d, z_d;
296  double E;
297 
298  if (chpref == 0)
299  return 0;
300 
301  rxy2 = d[0]*d[0] + d[1]*d[1];
302  rxy2_d = rxy2*uz2;
303  z_d = d[2]*uz;
304 
305  if (rxy2 <= mmm1d_params.far_switch_radius_2) {
306  /* near range formula */
307  double r2n, rt, shift_z;
308  int n;
309 
310  E = -2*C_GAMMA;
311 
312  /* polygamma summation */
313  r2n = 1.0;
314  for (n = 0; n < n_modPsi; n++) {
315  double add = mod_psi_even(n, z_d)*r2n;
316  E -= add;
317 
318  if (fabs(add) < mmm1d_params.maxPWerror)
319  break;
320 
321  r2n *= rxy2_d;
322  }
323  E *= coulomb.prefactor*uz;
324 
325  /* real space parts */
326 
327  E += coulomb.prefactor/r;
328 
329  shift_z = d[2] + box_l[2];
330  rt = sqrt(rxy2 + shift_z*shift_z);
331  E += coulomb.prefactor/rt;
332 
333  shift_z = d[2] - box_l[2];
334  rt = sqrt(rxy2 + shift_z*shift_z);
335  E += coulomb.prefactor/rt;
336  }
337  else {
338  /* far range formula */
339  double rxy = sqrt(rxy2);
340  double rxy_d = rxy*uz;
341  int bp;
342  /* The first Bessel term will compensate a little bit the
343  log term, so add them close together */
344  E = -0.25*log(rxy2_d) + 0.5*(M_LN2 - C_GAMMA);
345  for (bp = 1; bp < MAXIMAL_B_CUT; bp++) {
346  if (bessel_radii[bp-1] < rxy)
347  break;
348 
349  double fq = C_2PI*bp;
350  E += K0(fq*rxy_d)*cos(fq*z_d);
351  }
352  E *= 4*coulomb.prefactor*uz;
353  }
354 
355  return chpref*E;
356 }

358 int mmm1d_tune(char **log)
359 {
360  char buffer[32 + 2*ES_DOUBLE_SPACE + ES_INTEGER_SPACE];
361  double int_time, min_time=1e200, min_rad = -1;
362  double maxrad = box_l[2]; /* N_psi = 2, theta=2/3 maximum for rho */
363  double switch_radius;
364 
365  if (mmm1d_params.far_switch_radius_2 < 0) {
366  /* determine besselcutoff and optimal switching radius. Should be around 0.33 */
367  for (switch_radius = 0.2*maxrad;
368  switch_radius < 0.4*maxrad;
369  switch_radius += 0.025*maxrad) {
370  if (switch_radius <= bessel_radii[MAXIMAL_B_CUT - 1]) {
371  // this switching radius is too small for our Bessel series
372  continue;
373  }
374 
375  mmm1d_params.far_switch_radius_2 = SQR(switch_radius);
376 
377  coulomb.method = COULOMB_MMM1D;
378 
379  /* initialize mmm1d temporary structures */
380  mpi_bcast_coulomb_params();
381 
382  /* perform force calculation test */
383  int_time = time_force_calc(TEST_INTEGRATIONS);
384 
385  /* exit on errors */
386  if (int_time < 0)
387  return ES_ERROR;
388 
389  sprintf(buffer, "r= %f t= %f ms\n", switch_radius, int_time);
390  *log = strcat_alloc(*log, buffer);
391 
392  if (int_time < min_time) {
393  min_time = int_time;
394  min_rad = switch_radius;
395  }
396  /* stop if all hope is vain... */
397  else if (int_time > 2*min_time)
398  break;
399  }
400  switch_radius = min_rad;
401  mmm1d_params.far_switch_radius_2 = SQR(switch_radius);
402  }
403  else {
404  if (mmm1d_params.far_switch_radius_2 <= SQR(bessel_radii[MAXIMAL_B_CUT - 1])) {
405  // this switching radius is too small for our Bessel series
406  *log = strcat_alloc(*log, "could not find reasonable bessel cutoff");
407  return ES_ERROR;
408  }
409  }
410 
411  coulomb.method = COULOMB_MMM1D;
412 
413  mpi_bcast_coulomb_params();
414 
415  return ES_OK;
416 }
