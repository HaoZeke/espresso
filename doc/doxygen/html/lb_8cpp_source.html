1840 /** (Re-)initializes the fluid. */
1841 void lb_reinit_parameters() {
1842  int i;
1843 
1844  if (lbpar.viscosity[0] > 0.0) {
1845  /* Eq. (80) Duenweg, Schiller, Ladd, PRE 76(3):036704 (2007). */
1846  // unit conversion: viscosity
1847  gamma_shear = 1. - 2./(6.*lbpar.viscosity[0]*lbpar.tau/(lbpar.agrid*lbpar.agrid)+1.);
1848  }
1849 
1850  if (lbpar.bulk_viscosity[0] > 0.0) {
1851  /* Eq. (81) Duenweg, Schiller, Ladd, PRE 76(3):036704 (2007). */
1852  // unit conversion: viscosity
1853  gamma_bulk = 1. - 2./(9.*lbpar.bulk_viscosity[0]*lbpar.tau/(lbpar.agrid*lbpar.agrid)+1.);
1854  }
1855 
1856  gamma_odd = lbpar.gamma_odd[0];
1857  gamma_even = lbpar.gamma_even[0];
1858 
1859  double mu = 0.0;
1860 
1861  if (temperature > 0.0)
1862  {
1863  /* fluctuating hydrodynamics ? */
1864  fluct = 1;
1865 
1866  /* Eq. (51) Duenweg, Schiller, Ladd, PRE 76(3):036704 (2007).
1867  * Note that the modes are not normalized as in the paper here! */
1868  mu = temperature/lbmodel.c_sound_sq*lbpar.tau*lbpar.tau/(lbpar.agrid*lbpar.agrid);
1869  //mu *= agrid*agrid*agrid; // Marcello's conjecture
1870 #ifdef D3Q19
1871  double (*e)[19] = d3q19_modebase;
1872 #else // D3Q19
1873  double **e = lbmodel.e;
1874 #endif // D3Q19

1893  lb_coupl_pref2 = 0.0;

2022 {

2024  double local_rho, local_j[3], local_pi[6], trace;

2026

2028

2031  local_j[2] = j[2];
2032 
2033  for (i = 0; i < 6; i++) local_pi[i] = pi[i];
2034

2037 #ifdef D3Q19
2038  double rho_times_coeff;
2039  double tmp1,tmp2;
2040 
2041  /* update the q=0 sublattice */

2043 
2044  /* update the q=1 sublattice */
2045  rho_times_coeff = 1./18. * (local_rho-avg_rho);
2046 
2047  lbfluid[0][1][index] = rho_times_coeff + 1./6.*local_j[0] + 1./4. * local_pi[0] - 1./12.*trace;
2048  lbfluid[0][2][index] = rho_times_coeff - 1./6.*local_j[0] + 1./4. * local_pi[0] - 1./12.*trace;
2049  lbfluid[0][3][index] = rho_times_coeff + 1./6.*local_j[1] + 1./4. * local_pi[2] - 1./12.*trace;
2050  lbfluid[0][4][index] = rho_times_coeff - 1./6.*local_j[1] + 1./4. * local_pi[2] - 1./12.*trace;
2051  lbfluid[0][5][index] = rho_times_coeff + 1./6.*local_j[2] + 1./4. * local_pi[5] - 1./12.*trace;
2052  lbfluid[0][6][index] = rho_times_coeff - 1./6.*local_j[2] + 1./4. * local_pi[5] - 1./12.*trace;
2053 
2054  /* update the q=2 sublattice */
2055  rho_times_coeff = 1./36. * (local_rho-avg_rho);
2056 
2057  tmp1 = local_pi[0] + local_pi[2];
2058  tmp2 = 2.0*local_pi[1];
2059 
2060  lbfluid[0][7][index] = rho_times_coeff + 1./12.*(local_j[0]+local_j[1]) + 1./8.*(tmp1+tmp2) - 1./24.*trace;
2061  lbfluid[0][8][index] = rho_times_coeff - 1./12.*(local_j[0]+local_j[1]) + 1./8.*(tmp1+tmp2) - 1./24.*trace;

2081 #else // D3Q19
