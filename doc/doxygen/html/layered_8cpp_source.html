41 /* Organization: Layers only in one direction.
42  ghost_bottom
43  c1
44  c2
45  c3
46  .
47  .
48  .
49  cn
50  ghost_top
51 
52  First, all nodes send downwards, then upwards. Within these actions,
53  first the odd nodes send. For even n_nodes, this algorithm is straight
54  forward: first the odd nodes send, the even receive, then vice versa.
55  For odd n_nodes, we have
56  1) 1->2 3->4 5->1
57  2) 2->3 4->5
58  So in the first round node 5 has to wait for node 1 to
59  complete the send and get ready to receive. In other words,
60  what physically happens is:
61  1) 1->2 3->4 5->*
62  2) *->1 2->3 4->*
63  3) *->5
64  This means that one pending receive has to be done in addition
65  provided that all other transmissions can happen in parallel.
66 
67 */
68

88 void layered_get_mi_vector(double res[3], double a[3], double b[3])
89 {
90  int i;

100 Cell *layered_position_to_cell(double pos[3])
101 {
102  int cpos = (int)((pos[2] - my_left[2])*layer_h_i) + 1;
103  if (cpos < 1) {

112  cpos = n_layers;

115  }

128 static void layered_prepare_comm(GhostCommunicator *comm, int data_parts)
129 {
130  int even_odd;
131  int c, n;
132 
133  if (n_nodes > 1) {
134  /* more than one node => no local transfers */
135 
136  /* how many communications to do: up even/odd, down even/odd */
137  n = 4;
138  /* one communication missing if not periodic but on border */
139  if (!LAYERED_TOP_NEIGHBOR)
140  n -= 2;
141  if (!LAYERED_BTM_NEIGHBOR)
142  n -= 2;
143 
144  prepare_comm(comm, data_parts, n);
145 
146  /* always sending/receiving 1 cell per time step */
147  for(c = 0; c < n; c++) {
148  comm->comm[c].part_lists = (ParticleList**)malloc(sizeof(ParticleList *));
149  comm->comm[c].n_part_lists = 1;
150  comm->comm[c].mpi_comm = comm_cart;
151  }
152 
153  c = 0;
154 
155  CELL_TRACE(fprintf(stderr, "%d: ghostrec new comm of size %d\n", this_node, n));
156  /* downwards */
157  for (even_odd = 0; even_odd < 2; even_odd++) {
158  /* send */
159  if (this_node % 2 == even_odd && LAYERED_BTM_NEIGHBOR) {
160  comm->comm[c].type = GHOST_SEND;
161  /* round 1 uses prefetched data and stores delayed data */
162  if (c == 1)
163  comm->comm[c].type |= GHOST_PREFETCH | GHOST_PSTSTORE;
164  comm->comm[c].node = btm;
165  if (data_parts == GHOSTTRANS_FORCE) {
166  comm->comm[c].part_lists[0] = &cells[0];
167  CELL_TRACE(fprintf(stderr, "%d: ghostrec send force to %d btmg\n", this_node, btm));
168  }
169  else {
170  comm->comm[c].part_lists[0] = &cells[1];
171 
172  /* if periodic and bottom or top, send shifted */
173  comm->comm[c].shift[0] = comm->comm[c].shift[1] = 0;
174  if (((layered_flags & LAYERED_BTM_MASK) == LAYERED_BTM_MASK) &&
175  (data_parts & GHOSTTRANS_POSITION)) {
176  comm->data_parts |= GHOSTTRANS_POSSHFTD;
177  comm->comm[c].shift[2] = box_l[2];
178  }
179  else
180  comm->comm[c].shift[2] = 0;
181  CELL_TRACE(fprintf(stderr, "%d: ghostrec send to %d shift %f btml\n", this_node, btm, comm->comm[c].shift[2]));
182  }
183  c++;
184  }
185  /* recv. Note we test r_node as we always have to test the sender
186  as for odd n_nodes maybe we send AND receive. */
187  if (top % 2 == even_odd && LAYERED_TOP_NEIGHBOR) {
188  comm->comm[c].type = GHOST_RECV;
189  /* round 0 prefetch send for round 1 and delay recvd data processing */
190  if (c == 0)
191  comm->comm[c].type |= GHOST_PREFETCH | GHOST_PSTSTORE;
192  comm->comm[c].node = top;
193  if (data_parts == GHOSTTRANS_FORCE) {
194  comm->comm[c].part_lists[0] = &cells[n_layers];
195  CELL_TRACE(fprintf(stderr, "%d: ghostrec get force from %d topl\n", this_node, top));
196  }
197  else {
198  comm->comm[c].part_lists[0] = &cells[n_layers + 1];
199  CELL_TRACE(fprintf(stderr, "%d: ghostrec recv from %d topg\n", this_node, top));
200  }
201  c++;
202  }
203  }
204 
205  CELL_TRACE(fprintf(stderr, "%d: ghostrec upwards\n", this_node));
206  /* upwards */
207  for (even_odd = 0; even_odd < 2; even_odd++) {
208  /* send */
209  if (this_node % 2 == even_odd && LAYERED_TOP_NEIGHBOR) {
210  comm->comm[c].type = GHOST_SEND;
211  /* round 1 use prefetched data from round 0.
212  But this time there may already have been two transfers downwards */
213  if (c % 2 == 1)
214  comm->comm[c].type |= GHOST_PREFETCH | GHOST_PSTSTORE;
215  comm->comm[c].node = top;
216  if (data_parts == GHOSTTRANS_FORCE) {
217  comm->comm[c].part_lists[0] = &cells[n_layers + 1];
218  CELL_TRACE(fprintf(stderr, "%d: ghostrec send force to %d topg\n", this_node, top));
219  }
220  else {
221  comm->comm[c].part_lists[0] = &cells[n_layers];
222 
223  /* if periodic and bottom or top, send shifted */
224  comm->comm[c].shift[0] = comm->comm[c].shift[1] = 0;
225  if (((layered_flags & LAYERED_TOP_MASK) == LAYERED_TOP_MASK) &&
226  (data_parts & GHOSTTRANS_POSITION)) {
227  comm->data_parts |= GHOSTTRANS_POSSHFTD;
228  comm->comm[c].shift[2] = -box_l[2];
229  }
230  else
231  comm->comm[c].shift[2] = 0;
232  CELL_TRACE(fprintf(stderr, "%d: ghostrec send to %d shift %f topl\n", this_node, top, comm->comm[c].shift[2]));
233  }
234  c++;
235  }
236  /* recv. Note we test r_node as we always have to test the sender
237  as for odd n_nodes maybe we send AND receive. */
238  if (btm % 2 == even_odd && LAYERED_BTM_NEIGHBOR) {
239  comm->comm[c].type = GHOST_RECV;
240  /* round 0 prefetch. But this time there may already have been two transfers downwards */
241  if (c % 2 == 0)
242  comm->comm[c].type |= GHOST_PREFETCH | GHOST_PSTSTORE;
243  comm->comm[c].node = btm;
244  if (data_parts == GHOSTTRANS_FORCE) {
245  comm->comm[c].part_lists[0] = &cells[1];
246  CELL_TRACE(fprintf(stderr, "%d: ghostrec get force from %d btml\n", this_node, btm));
247  }
248  else {
249  comm->comm[c].part_lists[0] = &cells[0];
250  CELL_TRACE(fprintf(stderr, "%d: ghostrec recv from %d btmg\n", this_node, btm));
251  }
252  c++;
253  }
254  }
255  }
256  else {
257  /* one node => local transfers, either 2 (up and down, periodic) or zero*/
258 
259  n = (layered_flags & LAYERED_PERIODIC) ? 2 : 0;
260 
261  prepare_comm(comm, data_parts, n);
262 
263  if (n != 0) {
264  /* two cells: from and to */
265  for(c = 0; c < n; c++) {
266  comm->comm[c].part_lists = (ParticleList**)malloc(2*sizeof(ParticleList *));
267  comm->comm[c].n_part_lists = 2;
268  comm->comm[c].mpi_comm = comm_cart;
269  comm->comm[c].node = this_node;
270  }
271 
272  c = 0;
273 
274  /* downwards */
275  comm->comm[c].type = GHOST_LOCL;
276  if (data_parts == GHOSTTRANS_FORCE) {
277  comm->comm[c].part_lists[0] = &cells[0];
278  comm->comm[c].part_lists[1] = &cells[n_layers];
279  }
280  else {
281  comm->comm[c].part_lists[0] = &cells[1];
282  comm->comm[c].part_lists[1] = &cells[n_layers + 1];
283  /* here it is periodic */
284  if (data_parts & GHOSTTRANS_POSITION)
285  comm->data_parts |= GHOSTTRANS_POSSHFTD;
286  comm->comm[c].shift[0] = comm->comm[c].shift[1] = 0;
287  comm->comm[c].shift[2] = box_l[2];
288  }
289  c++;
290 
291  /* upwards */
292  comm->comm[c].type = GHOST_LOCL;
293  if (data_parts == GHOSTTRANS_FORCE) {
294  comm->comm[c].part_lists[0] = &cells[n_layers + 1];
295  comm->comm[c].part_lists[1] = &cells[1];
296  }
297  else {
298  comm->comm[c].part_lists[0] = &cells[n_layers];
299  comm->comm[c].part_lists[1] = &cells[0];
300  /* here it is periodic */
301  if (data_parts & GHOSTTRANS_POSITION)
302  comm->data_parts |= GHOSTTRANS_POSSHFTD;
303  comm->comm[c].shift[0] = comm->comm[c].shift[1] = 0;
304  comm->comm[c].shift[2] = -box_l[2];
305  }
306  }
307  }

310 void layered_topology_init(CellPList *old)
311 {
312  Particle *part;
313  int c, p, np;
314 
315  CELL_TRACE(fprintf(stderr, "%d: layered_topology_init, %d old particle lists\n", this_node, old->n));
316 
317  cell_structure.type = CELL_STRUCTURE_LAYERED;
318  cell_structure.position_to_node = map_position_node_array;
319  cell_structure.position_to_cell = layered_position_to_cell;
320 
321  /* check node grid. All we can do is 1x1xn. */
322  if (node_grid[0] != 1 || node_grid[1] != 1) {
323  ostringstream msg;
324  msg <<"selected node grid is not suitable for layered cell structure (needs 1x1x"<< n_nodes << " grid";
325  runtimeError(msg);
326  node_grid[0] = node_grid[1] = 1;
327  node_grid[2] = n_nodes;
328  }
329 
330  if (this_node == 0 && determine_n_layers) {
331  if (max_range > 0) {
332  n_layers = (int)floor(local_box_l[2]/max_range);
333  if (n_layers < 1) {
334  ostringstream msg;
335  msg <<"layered: maximal interaction range " << max_range << " larger than local box length " << local_box_l[2];
336  runtimeError(msg);
337  n_layers = 1;
338  }
339  if (n_layers > max_num_cells - 2)
340  n_layers = imax(max_num_cells - 2, 0);
341  }
342  else
343  n_layers = 1;
344  }
345  MPI_Bcast(&n_layers, 1, MPI_INT, 0, comm_cart);
346 
347  top = this_node + 1;
348  if (top == n_nodes && (layered_flags & LAYERED_PERIODIC))
349  top = 0;
350  btm = this_node - 1;
351  if (btm == -1 && (layered_flags & LAYERED_PERIODIC))
352  btm = n_nodes - 1;
353 
354  layer_h = local_box_l[2]/(double)(n_layers);
355  layer_h_i = 1/layer_h;
356 
357  if (layer_h < max_range) {
358  ostringstream msg;
359  msg <<"layered: maximal interaction range " << max_range << " larger than layer height " << layer_h;
360  runtimeError(msg);
361  }
362 
363  /* check wether node is top and/or bottom */
364  layered_flags = 0;
365  if (this_node == 0)
366  layered_flags |= LAYERED_BOTTOM;
367  if (this_node == n_nodes - 1)
368  layered_flags |= LAYERED_TOP;
369 
370  if (PERIODIC(2))
371  layered_flags |= LAYERED_PERIODIC;
372 
373  CELL_TRACE(fprintf(stderr, "%d: layered_flags tn %d bn %d \n", this_node, LAYERED_TOP_NEIGHBOR, LAYERED_BTM_NEIGHBOR));
374 
375  /* allocate cells and mark them */
376  realloc_cells(n_layers + 2);
377  realloc_cellplist(&local_cells, local_cells.n = n_layers);
378  for (c = 0; c < n_layers; c++)
379  local_cells.cell[c] = &cells[c + 1];
380  realloc_cellplist(&ghost_cells, ghost_cells.n = 2);
381  ghost_cells.cell[0] = &cells[0];
382  ghost_cells.cell[1] = &cells[n_layers + 1];
383 
384  /* create communicators */
385  layered_prepare_comm(&cell_structure.ghost_cells_comm, GHOSTTRANS_PARTNUM);
386  layered_prepare_comm(&cell_structure.exchange_ghosts_comm, GHOSTTRANS_PROPRTS | GHOSTTRANS_POSITION);
387  layered_prepare_comm(&cell_structure.update_ghost_pos_comm, GHOSTTRANS_POSITION);
388  layered_prepare_comm(&cell_structure.collect_ghost_force_comm, GHOSTTRANS_FORCE);
389 
390  /* copy particles */
391  for (c = 0; c < old->n; c++) {
392  part = old->cell[c]->part;
393  np = old->cell[c]->n;
394  for (p = 0; p < np; p++) {
395  Cell *nc = layered_position_to_cell(part[p].r.p);
396  /* particle does not belong to this node. Just stow away
397  somewhere for the moment */
398  if (nc == NULL)
399  nc = local_cells.cell[0];
400  append_unindexed_particle(nc, &part[p]);
401  }
402  }
403  for (c = 1; c <= n_layers; c++)
404  update_local_particles(&cells[c]);
405 
406  CELL_TRACE(fprintf(stderr, "%d: layered_topology_init done\n", this_node));
407 }

409 static void layered_append_particles(ParticleList *pl, ParticleList *up, ParticleList *dn)
410 {
411  int p;
412 
413  CELL_TRACE(fprintf(stderr, "%d: sorting in %d\n", this_node, pl->n));
414  for(p = 0; p < pl->n; p++) {
415  fold_position(pl->part[p].r.p, pl->part[p].m.v, pl->part[p].l.i);
416 
417  if (LAYERED_BTM_NEIGHBOR && pl->part[p].r.p[2] < my_left[2]) {
418  CELL_TRACE(fprintf(stderr, "%d: leaving part %d for node below\n", this_node, pl->part[p].p.identity));
419  move_indexed_particle(dn, pl, p);
420  }
421  else if (LAYERED_TOP_NEIGHBOR && pl->part[p].r.p[2] >= my_right[2]) {
422  CELL_TRACE(fprintf(stderr, "%d: leaving part %d for node above\n", this_node, pl->part[p].p.identity));
423  move_indexed_particle(up, pl, p);
424  }
425  else
426  move_indexed_particle(layered_position_to_cell(pl->part[p].r.p), pl, p);
427  /* same particle again, as this is now a new one */
428  if (p < pl->n) p--;
429  }
430  CELL_TRACE(fprintf(stderr, "%d: left over %d\n", this_node, pl->n));
431 }

433 void layered_exchange_and_sort_particles(int global_flag)
434 {
435  Particle *part;
436  Cell *nc, *oc;
437  int c, p, flag, redo;
438  ParticleList send_buf_dn, send_buf_up;
439  ParticleList recv_buf;
440 
441  CELL_TRACE(fprintf(stderr, "%d:layered exchange and sort %d\n", this_node, global_flag));
442 
443  init_particlelist(&send_buf_dn);
444  init_particlelist(&send_buf_up);
445 
446  init_particlelist(&recv_buf);
447 
448  /* sort local particles */
449  for (c = 1; c <= n_layers; c++) {
450  oc = &cells[c];
451 
452  for (p = 0; p < oc->n; p++) {
453  part = &oc->part[p];
454 
455  if (n_nodes != 1 && LAYERED_BTM_NEIGHBOR && part->r.p[2] < my_left[2]) {
456  CELL_TRACE(fprintf(stderr, "%d: send part %d down\n", this_node, part->p.identity));
457  move_indexed_particle(&send_buf_dn, oc, p);
458  if (p < oc->n) p--;
459  }
460  else if (n_nodes != 1 && LAYERED_TOP_NEIGHBOR && part->r.p[2] >= my_right[2]) {
461  CELL_TRACE(fprintf(stderr, "%d: send part %d up\n", this_node, part->p.identity));
462  move_indexed_particle(&send_buf_up, oc, p);
463  if (p < oc->n) p--;
464  }
465  else {
466  /* particle stays here. Fold anyways to get x,y correct */
467  fold_position(part->r.p, part->m.v, part->l.i);
468 
469  nc = layered_position_to_cell(part->r.p);
470  if (nc != oc) {
471  move_indexed_particle(nc, oc, p);
472  if (p < oc->n) p--;
473  }
474  }
475  }
476  }
477 
478  for (;;) {
479  /* transfer */
480  if (n_nodes > 1) {
481  if (this_node % 2 == 0) {
482  /* send down */
483  if (LAYERED_BTM_NEIGHBOR) {
484  CELL_TRACE(fprintf(stderr,"%d: send dn\n",this_node));
485  send_particles(&send_buf_dn, btm);
486  }
487  if (LAYERED_TOP_NEIGHBOR) {
488  CELL_TRACE(fprintf(stderr,"%d: recv up\n",this_node));
489  recv_particles(&recv_buf, top);
490  }
491  /* send up */
492  if (LAYERED_TOP_NEIGHBOR) {
493  CELL_TRACE(fprintf(stderr,"%d: send up\n",this_node));
494  send_particles(&send_buf_up, top);
495  }
496  if (LAYERED_BTM_NEIGHBOR) {
497  CELL_TRACE(fprintf(stderr,"%d: recv dn\n",this_node));
498  recv_particles(&recv_buf, btm);
499  }
500  }
501  else {
502  if (LAYERED_TOP_NEIGHBOR) {
503  CELL_TRACE(fprintf(stderr,"%d: recv up\n",this_node));
504  recv_particles(&recv_buf, top);
505  }
506  if (LAYERED_BTM_NEIGHBOR) {
507  CELL_TRACE(fprintf(stderr,"%d: send dn\n",this_node));
508  send_particles(&send_buf_dn, btm);
509  }
510  if (LAYERED_BTM_NEIGHBOR) {
511  CELL_TRACE(fprintf(stderr,"%d: recv dn\n",this_node));
512  recv_particles(&recv_buf, btm);
513  }
514  if (LAYERED_TOP_NEIGHBOR) {
515  CELL_TRACE(fprintf(stderr,"%d: send up\n",this_node));
516  send_particles(&send_buf_up, top);
517  }
518  }
519  }
520  else {
521  if (recv_buf.n != 0 || send_buf_dn.n != 0 || send_buf_up.n != 0) {
522  fprintf(stderr, "1 node but transfer buffers are not empty. send up %d, down %d, recv %d\n",
523  send_buf_up.n, send_buf_dn.n, recv_buf.n);
524  errexit();
525  }
526  }
527  layered_append_particles(&recv_buf, &send_buf_up, &send_buf_dn);
528 
529  /* handshake redo */
530  flag = (send_buf_up.n != 0 || send_buf_dn.n != 0);
531 
532  CELL_TRACE(if (flag) fprintf(stderr, "%d: requesting another exchange round\n", this_node));
533 
534  if (global_flag == CELL_GLOBAL_EXCHANGE) {
535  MPI_Allreduce(&flag, &redo, 1, MPI_INT, MPI_MAX, comm_cart);
536  if (!redo)
537  break;
538  CELL_TRACE(fprintf(stderr, "%d: another exchange round\n", this_node));
539  }
540  else {
541  if (flag) {
542  ostringstream msg;
543  msg <<"layered_exchange_and_sort_particles: particle moved more than one cell";
544  runtimeError(msg);
545 
546  /* sort left over particles into border cells */
547  CELL_TRACE(fprintf(stderr, "%d: emergency sort\n", this_node));
548  while (send_buf_up.n > 0)
