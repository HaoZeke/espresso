62  /** Some general remarks:
63  * This file implements the LB D3Q19 method to Espresso. The LB_Model
64  * construction is preserved for historical reasons and might be removed
65  * soon. It is constructed as a multi-relaxation time LB, thus all populations
66  * are converted to modes, then collision is performed and transfered back
67  * to population space, where the streaming is performed.
68  *
69  * For performance reasons it is clever to do streaming and collision at the same time
70  * because every fluid node has to be read and written only once. This increases
71  * mainly cache efficiency.
72  * Two alternatives are implemented: stream_collide and collide_stream.
73  *
74  * The hydrodynamic fields, corresponding to density, velocity and stress, are
75  * stored in LB_FluidNodes in the array lbfields, the populations in lbfluid
76  * which is constructed as 2 x (Nx x Ny x Nz) x 19 array.
77  */

390 /** Calculate the local fluid fields.
391  * The calculation is implemented explicitly for the special case of D3Q19.
392  *
393  * @param index Index of the local lattice site.
394  * @param rho local fluid density
395  * @param j local fluid speed
396  * @param pi local fluid pressure
397  */
398 inline void lb_calc_local_fields(index_t index, double *rho, double *j, double *pi) {
399 
400  if (!(lattice_switch & LATTICE_LB)) {
401  ostringstream msg;
402  msg <<"Error in lb_calc_local_fields in " << __FILE__ << __LINE__ << ": CPU LB not switched on.";
403  runtimeError(msg);
404  *rho=0; j[0]=j[1]=j[2]=0; pi[0]=pi[1]=pi[2]=pi[3]=pi[4]=pi[5]=0;
405  return;
406  }
407 
408 #ifndef D3Q19
409 #error Only D3Q19 is implemened!
410 #endif
411 
412  if (!(lattice_switch & LATTICE_LB)) {
413  ostringstream msg;
414  msg <<"Error in lb_calc_local_pi in " << __FILE__ << __LINE__ << ": CPU LB not switched on.";
415  runtimeError(msg);
416  j[0] = j[1] = j[2] = 0;
417  return;
418  }
419 
420 #ifdef LB_BOUNDARIES
421  if ( lbfields[index].boundary ) {
422  *rho = lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid;
423  j[0] = 0.; j[1] = 0.; j[2] = 0.;
424  if (pi) {pi[0] = 0.; pi[1] = 0.; pi[2] = 0.; pi[3] = 0.; pi[4] = 0.; pi[5] = 0.;}
425  return;
426  }
427 #endif
428  double mode[19];
429  double modes_from_pi_eq[6];
430  lb_calc_modes(index, mode);
431 
432  *rho = mode[0] + lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid;
433 
434  j[0] = mode[1];
435  j[1] = mode[2];
436  j[2] = mode[3];
437 
438 #ifndef EXTERNAL_FORCES
439  if (lbfields[index].has_force)
440 #endif
441  {
442  j[0] += 0.5*lbfields[index].force[0];
443  j[1] += 0.5*lbfields[index].force[1];
444  j[2] += 0.5*lbfields[index].force[2];
445  }
446  if (!pi)
447  return;
448 
449  /* equilibrium part of the stress modes */
450  modes_from_pi_eq[0] = scalar(j,j)/ *rho;
451  modes_from_pi_eq[1] = (SQR(j[0])-SQR(j[1]))/ *rho;
452  modes_from_pi_eq[2] = (scalar(j,j) - 3.0*SQR(j[2]))/ *rho;
453  modes_from_pi_eq[3] = j[0]*j[1]/ *rho;
454  modes_from_pi_eq[4] = j[0]*j[2]/ *rho;
455  modes_from_pi_eq[5] = j[1]*j[2]/ *rho;
456 
457  /* Now we must predict the outcome of the next collision */
458  /* We immediately average pre- and post-collision. */
459  mode[4] = modes_from_pi_eq[0] + (0.5+0.5*gamma_bulk )*(mode[4] - modes_from_pi_eq[0]);
460  mode[5] = modes_from_pi_eq[1] + (0.5+0.5*gamma_shear)*(mode[5] - modes_from_pi_eq[1]);
461  mode[6] = modes_from_pi_eq[2] + (0.5+0.5*gamma_shear)*(mode[6] - modes_from_pi_eq[2]);
462  mode[7] = modes_from_pi_eq[3] + (0.5+0.5*gamma_shear)*(mode[7] - modes_from_pi_eq[3]);
463  mode[8] = modes_from_pi_eq[4] + (0.5+0.5*gamma_shear)*(mode[8] - modes_from_pi_eq[4]);
464  mode[9] = modes_from_pi_eq[5] + (0.5+0.5*gamma_shear)*(mode[9] - modes_from_pi_eq[5]);
465 
466  // Transform the stress tensor components according to the modes that
467  // correspond to those used by U. Schiller. In terms of populations this
468  // expression then corresponds exactly to those in Eqs. 116 - 121 in the
469  // Duenweg and Ladd paper, when these are written out in populations.
470  // But to ensure this, the expression in Schiller's modes has to be different!
471 
472  pi[0] = ( 2.0*(mode[0] + mode[4]) + mode[6] + 3.0*mode[5] )/6.0; // xx
473  pi[1] = mode[7]; // xy
474  pi[2] = ( 2.0*(mode[0] + mode[4]) + mode[6] - 3.0*mode[5] )/6.0; // yy
475  pi[3] = mode[8]; // xz
476  pi[4] = mode[9]; // yz
477  pi[5] = ( mode[0] + mode[4] - mode[6] )/3.0; // zz
478 
479 }
