30 int tabulated_set_params(int part_type_a, int part_type_b, char* filename)
31 {
32  IA_parameters *data;
33  FILE* fp;
34  int npoints;
35  double minval, maxval;
36  int i, newsize;
37  int token;
38  double dummr;
39  token = 0;
40 
41  data = get_ia_param_safe(part_type_a, part_type_b);
42 
43  if (!data)
44  return 1;
45 
46  if (strlen(filename) > MAXLENGTH_TABFILE_NAME-1 )
47  return 2;
48 
49  /* Open the file containing force and energy tables */
50  fp = fopen( filename , "r");
51  if ( !fp )
52  return 3;
53 
54  /* Look for a line starting with # */
55  while ( token != EOF) {
56  token = fgetc(fp);
57  if ( token == '#' ) { break; }
58  }
59  if ( token == EOF ) {
60  fclose(fp);
61  return 4;
62  }
63 
64  /* First read two important parameters we read in the data later*/
65  if (fscanf( fp , "%d %lf %lf", &npoints, &minval, &maxval) != 3) return 5;
66 
67  // Set the newsize to the same as old size : only changed if a new force table is being added.
68  newsize = tabulated_forces.max;
69 
70  if ( data->TAB_npoints == 0){
71  // A new potential will be added so set the number of points, the startindex and newsize
72  data->TAB_npoints = npoints;
73  data->TAB_startindex = tabulated_forces.max;
74  newsize += npoints;
75  } else {
76  // We have existing data for this pair of monomer types check array sizing
77  if ( data->TAB_npoints != npoints ) {
78  fclose(fp);
79  return 6;
80  }
81  }
82 
83  /* Update parameters symmetrically */
84  data->TAB_maxval = maxval;
85  data->TAB_minval = minval;
86  strcpy(data->TAB_filename,filename);
87 
88  /* Calculate dependent parameters */
89  data->TAB_stepsize = (maxval-minval)/(double)(data->TAB_npoints - 1);
90 
91  /* Allocate space for new data */
92  realloc_doublelist(&tabulated_forces,newsize);
93  realloc_doublelist(&tabulated_energies,newsize);
94 
95  /* Read in the new force and energy table data */
96  for (i =0 ; i < npoints ; i++)
97  {
98  if (fscanf(fp,"%lf %lf %lf",&dummr,
99  &(tabulated_forces.e[i+data->TAB_startindex]),
100  &(tabulated_energies.e[i+data->TAB_startindex])) != 3) return 5;
101  }
102 
103  fclose(fp);
104 
105  /* broadcast interaction parameters including force and energy tables*/
106  mpi_bcast_ia_params(part_type_a, part_type_b);
107 
108  mpi_cap_forces(force_cap);
109  return 0;
110 }

112 int tabulated_bonded_set_params(int bond_type, TabulatedBondedInteraction tab_type, char * filename)
113 {
114  int i, token = 0, size;
115  double dummr;
116  FILE* fp;
117 
118  if(bond_type < 0)
119  return 1;
120 
121  make_bond_type_exist(bond_type);
122 
123  fp = fopen( filename , "r");
124  if ( !fp )
125  return 3;
126 
127  /*Look for a line starting with # */
128  while ( token != EOF) {
129  token = fgetc(fp);
130  if ( token == '#' ) { break; } // magic number for # symbol
131  }
132  if ( token == EOF ) {
133  fclose(fp);
134  return 4;
135  }
136 
137  /* set types */
138  bonded_ia_params[bond_type].type = BONDED_IA_TABULATED;
139  bonded_ia_params[bond_type].p.tab.type = tab_type;
140 
141  /* set number of interaction partners */
142  if(tab_type == TAB_BOND_LENGTH) bonded_ia_params[bond_type].num = 1;
143  if(tab_type == TAB_BOND_ANGLE) bonded_ia_params[bond_type].num = 2;
144  if(tab_type == TAB_BOND_DIHEDRAL) bonded_ia_params[bond_type].num = 3;
145 
146  /* copy filename */
147  size = strlen(filename);
148  bonded_ia_params[bond_type].p.tab.filename = (char*)malloc((size+1)*sizeof(char));
149  strcpy(bonded_ia_params[bond_type].p.tab.filename,filename);
150 
151  /* read basic parameters from file */
152  if (fscanf( fp , "%d %lf %lf", &size,
153  &bonded_ia_params[bond_type].p.tab.minval,
154  &bonded_ia_params[bond_type].p.tab.maxval) != 3) return 5;
155  bonded_ia_params[bond_type].p.tab.npoints = size;
156 
157  /* Check interval for angle and dihedral potentials. With adding
158  ROUND_ERROR_PREC to the upper boundary we make sure, that during
159  the calculation we do not leave the defined table!
160  */
161  if(tab_type == TAB_BOND_ANGLE ) {
162  if( bonded_ia_params[bond_type].p.tab.minval != 0.0 ||
163  abs(bonded_ia_params[bond_type].p.tab.maxval-PI) > 1e-5 ) {
164  fclose(fp);
165  return 6;
166  }
167  bonded_ia_params[bond_type].p.tab.maxval = PI+ROUND_ERROR_PREC;
168  }
169  /* check interval for angle and dihedral potentials */
170  if(tab_type == TAB_BOND_DIHEDRAL ) {
171  if( bonded_ia_params[bond_type].p.tab.minval != 0.0 ||
172  abs(bonded_ia_params[bond_type].p.tab.maxval-(2*PI)) > 1e-5 ) {
173  fclose(fp);
174  return 6;
175  }
176  bonded_ia_params[bond_type].p.tab.maxval = (2*PI)+ROUND_ERROR_PREC;
177  }
178 
179  /* calculate dependent parameters */
180  bonded_ia_params[bond_type].p.tab.invstepsize = (double)(size-1)/(bonded_ia_params[bond_type].p.tab.maxval-bonded_ia_params[bond_type].p.tab.minval);
181 
182  /* allocate force and energy tables */
183  bonded_ia_params[bond_type].p.tab.f = (double*)malloc(size*sizeof(double));
184  bonded_ia_params[bond_type].p.tab.e = (double*)malloc(size*sizeof(double));
185 
186  /* Read in the new force and energy table data */
187  for (i =0 ; i < size ; i++) {
188  if (fscanf(fp,"%lf %lf %lf", &dummr,
189  &bonded_ia_params[bond_type].p.tab.f[i],
190  &bonded_ia_params[bond_type].p.tab.e[i]) != 3) return 5;
191  }
192  fclose(fp);
193 
194  mpi_bcast_ia_params(bond_type, -1);
195 
196  return ES_OK;
197 }

199 void check_tab_forcecap(double forcecap)
200 {
201  if( force_cap != -1.0){
202  int i,j,startindex;
203  IA_parameters *params;
204 
205  for(i=0; i<n_particle_types; i++) {
206  for(j=0; j<n_particle_types; j++) {
207  params = get_ia_param(i,j);
208  startindex = params->TAB_startindex;
209  if ( tabulated_forces.max < (params->TAB_npoints + startindex )) { /* Make sure forces are initialized */
210  if(forcecap > 0.0 && params->TAB_maxval > 0.0 && tabulated_forces.e[startindex] > forcecap) {
211  for ( i = 0 ; i < params->TAB_npoints ; i++) {
212  if ( tabulated_forces.e[startindex + i] < forcecap ) {
213  return; /* Everything is OK nothing to say :) */
214  }
215  }
216  if ( i == params->TAB_npoints - 1) {
217  force_cap = -1.0;
218  /* Force cap is below all known forces .. turn force capping off */
219  }
220  }
221  if ( forcecap > tabulated_forces.e[startindex] ) {
222  fprintf(stderr,"calc_tab_cap_radii: Capped region is outside the force table");
223 
224  if ( tabulated_forces.e[startindex] < tabulated_forces.e[startindex+1] ) {
225  fprintf(stderr,"Extrapolation does not make sense outside this table \n");
226  errexit();
227  }
228 
229  fprintf(stderr,", will extrapolate the force outside table \n");
230  fprintf(stderr,"fc: %f \n",force_cap);
231  }
232  }
233  }
234  }
