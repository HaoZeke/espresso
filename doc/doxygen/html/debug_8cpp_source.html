101 {

103  Cell *cell;

105  int cell_part_cnt=0, ghost_part_cnt=0, local_part_cnt=0;

107 
108  /* checks: part_id, part_pos, local_particles id */
109  for (c = 0; c < local_cells.n; c++) {
110  cell = local_cells.cell[c];
111  cell_part_cnt += cell->n;
112  part = cell->part;
113  for(int n=0; n<cell->n ; n++) {
114  if(part[n].p.identity < 0 || part[n].p.identity > max_seen_particle) {
115  fprintf(stderr,"%d: check_particle_consistency: ERROR: Cell %d Part %d has corrupted id=%d\n",
116  this_node,c,n,cell->part[n].p.identity);
117  errexit();
118  }
119  for(dir=0;dir<3;dir++) {
120  if(PERIODIC(dir) && (part[n].r.p[dir] < -ROUND_ERROR_PREC*box_l[dir] || part[n].r.p[dir] - box_l[dir] > ROUND_ERROR_PREC*box_l[dir])) {
121  fprintf(stderr,"%d: check_particle_consistency: ERROR: illegal pos[%d]=%f of part %d id=%d in cell %d\n",
122  this_node,dir,part[n].r.p[dir],n,part[n].p.identity,c);
123  errexit();
124  }
125  }
126  if(local_particles[part[n].p.identity] != &part[n]) {
127  fprintf(stderr,"%d: check_particle_consistency: ERROR: address mismatch for part id %d: local: %p cell: %p in cell %d\n",
128  this_node,part[n].p.identity,local_particles[part[n].p.identity],
129  &part[n],c);
130  errexit();
131 
132  }
133  }
134  }
135 
136  for (c = 0; c < ghost_cells.n; c++) {
137  cell = ghost_cells.cell[c];
138  if(cell->n>0) {
139  ghost_part_cnt += cell->n;
140  fprintf(stderr,"%d: check_particle_consistency: WARNING: ghost_cell %d contains %d particles!\n",
141  this_node,c,cell->n);
142  }
143  }
144  CELL_TRACE(fprintf(stderr,"%d: check_particle_consistency: %d particles in cells, %d particles in ghost_cells.\n",
145  this_node,cell_part_cnt, ghost_part_cnt));

147  for(n=0; n< max_seen_particle+1; n++) {
148  if(local_particles[n] != NULL) {
149  local_part_cnt ++;
150  if(local_particles[n]->p.identity != n) {
151  fprintf(stderr,"%d: check_particle_consistency: ERROR: local_particles part %d has corrupted id %d\n",
152  this_node,n,local_particles[n]->p.identity);
153  errexit();
154  }
155  }
156  }
157  CELL_TRACE(fprintf(stderr,"%d: check_particle_consistency: %d particles in local_particles.\n",
158  this_node,local_part_cnt));
159 
160  /* EXIT on severe errors */
161  if(cell_err_cnt>0) {
162  fprintf(stderr,"%d: check_particle_consistency: %d ERRORS detected in cell structure!\n",this_node,cell_err_cnt);
163  errexit();
164  }
165  if(local_part_cnt != cell_part_cnt) {
166  fprintf(stderr,"%d: check_particle_consistency: ERROR: %d parts in cells but %d parts in local_particles\n",
167  this_node,cell_part_cnt,local_part_cnt);
168 
169  for (c = 0; c < local_cells.n; c++) {
170  for(p = 0; p < local_cells.cell[c]->n; p++)
171  fprintf(stderr, "%d: got particle %d in cell %d\n", this_node, local_cells.cell[c]->part[p].p.identity, c);
172  }
173 
174  for(p = 0; p < n_part; p++)
175  if (local_particles[p])

177 
178  if(ghost_part_cnt==0) errexit();
179  }
180  if(ghost_part_cnt>0) {
181  fprintf(stderr,"%d: check_particle_consistency: ERROR: Found %d illegal ghost particles!\n",
182  this_node,ghost_part_cnt);
183  errexit();
184  }

187 void check_particles()
188 {

190  int *is_here;
191  Cell *cell;
192  int n, dir, c, p;
193  int cell_part_cnt=0, local_part_cnt=0;
194  int cell_err_cnt=0;
195  double skin2 = (skin != -1) ? skin/2 : 0;
196 
197  CELL_TRACE(fprintf(stderr, "%d: entering check_particles\n", this_node));
198 
199  /* check the consistency of particle_nodes */
200  /* to this aim the array is broadcasted temporarily */
201  if (this_node != 0)
202  particle_node = (int*)malloc((max_seen_particle + 1)*sizeof(int));
203  is_here = (int*)malloc((max_seen_particle + 1)*sizeof(int));
204  memset(is_here, 0, (max_seen_particle + 1)*sizeof(int));
205 
206  MPI_Bcast(particle_node, max_seen_particle + 1, MPI_INT, 0, comm_cart);
207 
208  /* checks: part_id, part_pos, local_particles id */
209  for (c = 0; c < local_cells.n; c++) {
210  cell = local_cells.cell[c];
211  cell_part_cnt += cell->n;
212  part = cell->part;
213  for(int n=0; n<cell->n ; n++) {
214  if(part[n].p.identity < 0 || part[n].p.identity > max_seen_particle) {
215  fprintf(stderr,"%d: check_particles: ERROR: Cell %d Part %d has corrupted id=%d\n",
216  this_node,c,n,cell->part[n].p.identity);
217  errexit();
218  }
219 
220  is_here[part[n].p.identity] = 1;
221 
222  for(dir=0;dir<3;dir++) {
223  if(PERIODIC(dir) && (part[n].r.p[dir] < -skin2 || part[n].r.p[dir] > box_l[dir] + skin2)) {
224  fprintf(stderr,"%d: check_particles: ERROR: illegal pos[%d]=%f of part %d id=%d in cell %d\n",
225  this_node,dir,part[n].r.p[dir],n,part[n].p.identity,c);
226  errexit();
227  }
228  }
229  if(local_particles[part[n].p.identity] != &part[n]) {
230  fprintf(stderr,"%d: check_particles: ERROR: address mismatch for part id %d: local: %p cell: %p in cell %d\n",
231  this_node,part[n].p.identity,local_particles[part[n].p.identity],
232  &part[n],c);
233  errexit();
234  }
235  if (particle_node[part[n].p.identity] != this_node) {
236  fprintf(stderr,"%d: check_particles: ERROR: node for particle %d wrong\n",
237  this_node,part[n].p.identity);
238  errexit();
239  }
240  }
241  }
242  CELL_TRACE(fprintf(stderr,"%d: check_particles: %d particles in local cells.\n",
243  this_node,cell_part_cnt));
244 
245  /* checks: local particle id */
246  for(n = 0; n <= max_seen_particle; n++) {
247  if(local_particles[n] != NULL) {
248  local_part_cnt ++;
249  if(local_particles[n]->p.identity != n) {
250  fprintf(stderr,"%d: check_particles: ERROR: local_particles part %d has corrupted id %d\n",
251  this_node,n,local_particles[n]->p.identity);
252  errexit();
253  }
254  }
255  }
256  CELL_TRACE(fprintf(stderr,"%d: check_particles: %d particles in local_particles.\n",
257  this_node,local_part_cnt));
258 
259  /* EXIT on severe errors */
260  if(cell_err_cnt>0) {
261  fprintf(stderr,"%d: check_particles: %d ERRORS detected in cell structure!\n",this_node,cell_err_cnt);
262  errexit();
263  }
264 
265  /* check whether the particles on my node are actually here */
266  for (p = 0; p <= max_seen_particle; p++) {
267  if (particle_node[p] == this_node) {
268  if (!is_here[p]) {
269  fprintf(stderr,"%d: check_particles: ERROR: particle %d on this node, but not in local cell\n", this_node, p);
270  }
271  }
272  }
273 
274  free(is_here);
275 
276  if (this_node != 0) {
277  free(particle_node);
278  particle_node = NULL;
279  }
280  else {
281  /* check whether the total count of particles is ok */
282  c = 0;
283  for (p = 0; p <= max_seen_particle; p++)
284  if (particle_node[p] != -1) c++;
285  if (c != n_part) {
286  fprintf(stderr,"%d: check_particles: #particles in particle_node inconsistent\n", this_node);
287  errexit();
288  }
289  CELL_TRACE(fprintf(stderr,"%d: check_particles: %d particles in particle_node.\n",
290  this_node,c));
291  }
292  CELL_TRACE(fprintf(stderr, "%d: leaving check_particles\n", this_node));
293 }
