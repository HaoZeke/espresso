38 /** Calculates the dihedral angle between particle quadruple p1, p2,
39 p3 and p4. The dihedral angle is the angle between the planes
40 specified by the particle triples (p1,p2,p3) and (p2,p3,p4).
41 Vectors a, b and c are the bond vectors between consequtive particles.
42 If the a,b or b,c are parallel the dihedral angle is not defined in which
43 case the routine returns phi=-1. Calling functions should check for that
44 (Written by: Arijit Maitra) */
45 inline void calc_dihedral_angle(Particle *p1, Particle *p2, Particle *p3, Particle *p4,
46  double a[3], double b[3], double c[3],
47  double aXb[3], double *l_aXb, double bXc[3], double *l_bXc,
48  double *cosphi, double *phi)
49 {
50  int i;
51 
52  get_mi_vector(a, p2->r.p, p1->r.p);
53  get_mi_vector(b, p3->r.p, p2->r.p);
54  get_mi_vector(c, p4->r.p, p3->r.p);
55 
56  /* calculate vector product a X b and b X c */
57  vector_product(a, b, aXb);
58  vector_product(b, c, bXc);
59 
60  /* calculate the unit vectors */
61  *l_aXb = sqrt(sqrlen(aXb));
62  *l_bXc = sqrt(sqrlen(bXc));
63 
64  /* catch case of undefined dihedral angle */
65  if ( *l_aXb <= TINY_LENGTH_VALUE || *l_bXc <= TINY_LENGTH_VALUE ) { *phi = -1.0; *cosphi = 0; return;}
66 
67  for (i=0;i<3;i++) {
68  aXb[i] /= *l_aXb;
69  bXc[i] /= *l_bXc;
70  }
71 
72  *cosphi = scalar(aXb, bXc);
73 
74  if ( fabs(fabs(*cosphi)-1) < TINY_SIN_VALUE ) *cosphi = dround(*cosphi);
75 
76  /* Calculate dihedral angle */
77  *phi = acos(*cosphi);
78  if( scalar(aXb, c) < 0.0 ) *phi = (2.0*PI) - *phi;
79 
80 }

82 /** calculate dihedral force between particles p1, p2 p3 and p4
83  Written by Arijit Maitra, adapted to new force interface by Hanjo,
84  more general new dihedral form by Ana.
85 */
86 inline int calc_dihedral_force(Particle *p2, Particle *p1, Particle *p3, Particle *p4,
87  Bonded_ia_parameters *iaparams, double force2[3],
88  double force1[2], double force3[2])
89 {
90  int i;
91  /* vectors for dihedral angle calculation */
92  double v12[3], v23[3], v34[3], v12Xv23[3], v23Xv34[3], l_v12Xv23, l_v23Xv34;
93  double v23Xf1[3], v23Xf4[3], v34Xf4[3], v12Xf1[3];
94  /* dihedral angle, cosine of the dihedral angle */
95  double phi, cosphi, sinmphi_sinphi;
96  /* force factors */
97  double fac, f1[3], f4[3];
98 
99  /* dihedral angle */
100  calc_dihedral_angle(p1, p2, p3, p4, v12, v23, v34, v12Xv23, &l_v12Xv23, v23Xv34, &l_v23Xv34, &cosphi, &phi);
101  /* dihedral angle not defined - force zero */
102  if ( phi == -1.0 ) {
103  for(i=0;i<3;i++) { force1[i] = 0.0; force2[i] = 0.0; force3[i] = 0.0; }
104  return 0;
105  }
106 
107  /* calculate force components (directions) */
108  for(i=0;i<3;i++) {
109  f1[i] = (v23Xv34[i] - cosphi*v12Xv23[i])/l_v12Xv23;;
110  f4[i] = (v12Xv23[i] - cosphi*v23Xv34[i])/l_v23Xv34;
111  }
112  vector_product(v23, f1, v23Xf1);
113  vector_product(v23, f4, v23Xf4);
114  vector_product(v34, f4, v34Xf4);
115  vector_product(v12, f1, v12Xf1);
116 
117  /* calculate force magnitude */
118 #ifdef OLD_DIHEDRAL
119  fac = iaparams->p.dihedral.bend * iaparams->p.dihedral.phase * iaparams->p.dihedral.mult;
120 #else
121  fac = -iaparams->p.dihedral.bend * iaparams->p.dihedral.mult;
122 #endif
123 
124  if(fabs(sin(phi)) < TINY_SIN_VALUE) {
125 #ifdef OLD_DIHEDRAL
126  sinmphi_sinphi = iaparams->p.dihedral.mult * cos(2.0*PI - iaparams->p.dihedral.mult*phi)/cos(phi);
127 #else
128  /*(comes from taking the first term of the MacLaurin expansion of
129  sin(n*phi - phi0) and sin(phi) and then making the division).
130  The original code had a 2PI term in the cosine (cos(2PI - nPhi))
131  but I removed it because it wasn't doing anything. AnaVV*/
132  sinmphi_sinphi = iaparams->p.dihedral.mult*
133  cos(iaparams->p.dihedral.mult*phi - iaparams->p.dihedral.phase)/cosphi;
134 #endif
135  }
136  else {
137 #ifdef OLD_DIHEDRAL
138  sinmphi_sinphi = sin(iaparams->p.dihedral.mult*phi)/sin(phi);
139 #else
140  sinmphi_sinphi = sin(iaparams->p.dihedral.mult*phi - iaparams->p.dihedral.phase)/sin(phi);
141 
142 #ifdef CONFIGTEMP
143  extern double configtemp[2];
144  double a[3], b[3], c[3], aXb[3], bXc[3];
145  get_mi_vector(a, p2->r.p, p1->r.p);
146  get_mi_vector(b, p3->r.p, p2->r.p);
147  get_mi_vector(c, p4->r.p, p3->r.p);
148  vector_product(a, b, aXb);
149  vector_product(b, c, bXc);
150  if (p1->p.configtemp) {
151  configtemp[0] += SQR(fac*sinmphi_sinphi*sin(phi))*sqrlen(b)/sqrlen(aXb);
152  configtemp[1] -= iaparams->p.dihedral.bend * (SQR(iaparams->p.dihedral.mult) *
153  cos(iaparams->p.dihedral.mult*phi - iaparams->p.dihedral.phase)
154  - iaparams->p.dihedral.mult * sinmphi_sinphi*cosphi)*sqrlen(b)/sqrlen(aXb);
155  }
156  if (p4->p.configtemp) {
157  configtemp[0] += SQR(fac*sinmphi_sinphi*sin(phi))*sqrlen(b)/sqrlen(bXc);
158  configtemp[1] -= iaparams->p.dihedral.bend * (SQR(iaparams->p.dihedral.mult) *
159  cos(iaparams->p.dihedral.mult*phi - iaparams->p.dihedral.phase)
160  - iaparams->p.dihedral.mult * sinmphi_sinphi*cosphi)*sqrlen(b)/sqrlen(bXc);
161  }
162 #endif
163 
164 
165 #endif
166  }
167 
168  fac *= sinmphi_sinphi;
169 
170 
171  /* store dihedral forces */
172  for(i=0;i<3;i++) {
173  force1[i] = fac*v23Xf1[i];
174  force2[i] = fac*(v34Xf4[i] - v12Xf1[i] - v23Xf1[i]);
175  force3[i] = fac*(v12Xf1[i] - v23Xf4[i] - v34Xf4[i]);
176  }
177  return 0;
178 }

180 /** calculate dihedral energy between particles p1, p2 p3 and p4
181  Written by Arijit Maitra, adapted to new force interface by Hanjo */
182 inline int dihedral_energy(Particle *p1, Particle *p2, Particle *p3, Particle *p4,
183  Bonded_ia_parameters *iaparams, double *_energy)
184 {
185  /* vectors for dihedral calculations. */
186  double v12[3], v23[3], v34[3], v12Xv23[3], v23Xv34[3], l_v12Xv23, l_v23Xv34;
187  /* dihedral angle, cosine of the dihedral angle */
188  double phi, cosphi;
189  /* energy factors */
190  double fac;
191 
192  calc_dihedral_angle(p1, p2, p3, p4, v12, v23, v34, v12Xv23, &l_v12Xv23, v23Xv34, &l_v23Xv34, &cosphi, &phi);
193 #ifdef OLD_DIHEDRAL
194  fac = iaparams->p.dihedral.phase * cos(iaparams->p.dihedral.mult*phi);
195 #else
196  fac = -cos(iaparams->p.dihedral.mult*phi -iaparams->p.dihedral.phase);
197 #endif
198  fac += 1.0;
199  fac *= iaparams->p.dihedral.bend;
200 
201  *_energy = fac;
202 
203  return 0;
204 }
