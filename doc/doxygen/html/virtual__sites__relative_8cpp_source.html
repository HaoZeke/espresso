35 // Update the pos of the given virtual particle as defined by the real
36 // particles in the same molecule
37 void update_mol_pos_particle(Particle *p)
38 {
39  // First obtain the real particle responsible for this virtual particle:
40  // Find the 1st real particle in the topology for the virtual particle's mol_id
41  Particle *p_real = vs_relative_get_real_particle(p);
42  // Check, if a real particle was found
43  if (!p_real)
44  {
45  ostringstream msg;
46  msg <<"virtual_sites_relative.cpp - update_mol_pos_particle(): No real particle associated with virtual site.\n";
47  runtimeError(msg);
48  return;
49  }
50 
51  // Calculate the quaternion defining the orientation of the vecotr connectinhg
52  // the virtual site and the real particle
53  // This is obtained, by multiplying the quaternion representing the director
54  // of the real particle with the quaternion of the virtual particle, which
55  // specifies the relative orientation.
56  double q[4];
57  multiply_quaternions(p_real->r.quat,p->r.quat,q);
58  // Calculate the director resulting from the quaternions
59  double director[3];
60  convert_quat_to_quatu(q,director);
61  // normalize
62  double l =sqrt(sqrlen(director));
63  // Division comes in the loop below
64 
65  // Calculate the new position of the virtual sites from
66  // position of real particle + director
67  int i;
68  double new_pos[3];
69  double tmp;
70  for (i=0;i<3;i++)
71  {
72  new_pos[i] =p_real->r.p[i] +director[i]/l*p->p.vs_relative_distance;
73  // Handle the case that one of the particles had gone over the periodic
74  // boundary and its coordinate has been folded
75  if (PERIODIC(i))
76  {
77  tmp =p->r.p[i] -new_pos[i];
78  //printf("%f\n",tmp);
79  if (tmp > box_l[i]/2.) {
80  //printf("greater than box_l/2 %f\n",tmp);
81  p->r.p[i] =new_pos[i] + box_l[i];
82  }
83  else if (tmp < -box_l[i]/2.) {
84  //printf("smaller than box_l/2 %f\n",tmp);
85  p->r.p[i] =new_pos[i] - box_l[i];
86  }
87  else p->r.p[i] =new_pos[i];
88  }
89  else p->r.p[i] =new_pos[i];
90  }
91 }

192 // Setup the virtual_sites_relative properties of a particle so that the given virtaul particle will follow the given real particle
193 int vs_relate_to(int part_num, int relate_to)
194 {
195  // Get the data for the particle we act on and the one we wnat to relate
196  // it to.
197  Particle p_current,p_relate_to;
198  if ((get_particle_data(relate_to,&p_relate_to)!=ES_OK) ||
199  (get_particle_data(part_num,&p_current)!=ES_OK)) {
200  ostringstream msg;
201  msg <<"Could not retrieve particle data for the given id";
202  runtimeError(msg);
203  return ES_ERROR;
204  }
205 
206  // get teh distance between the particles
207  double d[3];
208  get_mi_vector(d, p_current.r.p,p_relate_to.r.p);
209 
210  // Set the particle id of the particle we want to relate to and the distnace
211  if (set_particle_vs_relative(part_num, relate_to, sqrt(sqrlen(d))) == ES_ERROR) {
212  ostringstream msg;
213  msg <<"setting the vs_relative attributes failed";
214  runtimeError(msg);
215  return ES_ERROR;
216  }
217 
218 
219  // Check, if the distance between virtual and non-virtual particles is larger htan minimum global cutoff
220  // If so, warn user
221  double l=sqrt(sqrlen(d));
222  if (l>min_global_cut) {
223  ostringstream msg;
224  msg << "Warning: The distance between virtual and non-virtual particle (" << l << ") is\nlarger than the minimum global cutoff (" << min_global_cut << "). This may lead to incorrect simulations\nunder certain conditions. Use \"setmd min_global_cut\" to increase the minimum cutoff.\n";
225  runtimeWarning(msg);
226  return ES_ERROR;
227  }
228 
229  // Now, calculate the quaternions which specify the angle between
230  // the director of the particel we relate to and the vector
231  // (paritlce_we_relate_to - this_particle)
232  double quat[4];
233  // The vs_relative implemnation later obtains the direcotr by multiplying
234  // the quaternions representing the orientation of the real particle
235  // with those in the virtual particle. The re quulting quaternion is then
236  // converted to a director.
237  // Whe have quat_(real particle) *quat_(virtual particle)
238  // = quat_(obtained from desired director)
239  // Resolving this for the quat_(virtaul particle)
240 
241  //Normalize desired director
242  int i;
243  for (i=0;i<3;i++)
244  d[i]/=l;
245 
246  // Obtain quaternions from desired director
247  double quat_director[4];
248  convert_quatu_to_quat(d, quat_director);
249 
250  // Define quat as described above:
251  double x=0;
252  for (i=0;i<4;i++)
253  x+=p_relate_to.r.quat[i]*p_relate_to.r.quat[i];
254 
255  quat[0]=0;
256  for (i=0;i<4;i++)
257  quat[0] +=p_relate_to.r.quat[i]*quat_director[i];
258 
259  quat[1] =-quat_director[0] *p_relate_to.r.quat[1]
260  +quat_director[1] *p_relate_to.r.quat[0]
261  +quat_director[2] *p_relate_to.r.quat[3]
262  -quat_director[3] *p_relate_to.r.quat[2];
263  quat[2] =p_relate_to.r.quat[1] *quat_director[3]
264  + p_relate_to.r.quat[0] *quat_director[2]
265  - p_relate_to.r.quat[3] *quat_director[1]
266  - p_relate_to.r.quat[2] * quat_director[0];
267  quat[3] =quat_director[3] *p_relate_to.r.quat[0]
268  - p_relate_to.r.quat[3] *quat_director[0]
269  + p_relate_to.r.quat[2] * quat_director[1]
270  - p_relate_to.r.quat[1] *quat_director[2];
271  for (i=0;i<4;i++)
272  quat[i]/=x;
273 
274 
275  // Verify result
276  double qtemp[4];
277  multiply_quaternions(p_relate_to.r.quat,quat,qtemp);
278  for (i=0;i<4;i++)
279  if (fabs(qtemp[i]-quat_director[i])>1E-9)
280  fprintf(stderr, "vs_relate_to: component %d: %f instead of %f\n",
281  i, qtemp[i], quat_director[i]);
282 
283 
284  // Save the quaternions in the particle
285  if (set_particle_quat(part_num, quat) == ES_ERROR) {
286  ostringstream msg;
287  msg <<"set particle position first";
288  runtimeError(msg);
289  return ES_ERROR;
290  }
291 
292  return ES_OK;
293 }
