54 int collision_detection_set_params(int mode, double d, int bond_centers,
55  int bond_vs, int t)
56 {
57  if (mode & COLLISION_MODE_VS)
58  mode |= COLLISION_MODE_BOND;
59 
60  // If we don't have virtual sites, virtual site binding isn't possible.
61 #ifndef VIRTUAL_SITES_RELATIVE
62  if (mode & COLLISION_MODE_VS)
63  return 1;
64 #endif
65 
66  // Binding so far only works on a single cpu
67  if ((mode & COLLISION_MODE_VS) && n_nodes != 1)
68  return 2;
69 
70  // Check if bonded ia exist
71  if ((mode & COLLISION_MODE_BOND) &&
72  (bond_centers >= n_bonded_ia))
73  return 3;
74  if ((mode & COLLISION_MODE_VS) &&
75  (bond_vs >= n_bonded_ia))
76  return 3;
77 
78  if ((mode & COLLISION_MODE_BOND) &&
79  (bonded_ia_params[bond_centers].num != 1))
80  return 4;
81 
82  if ((mode & COLLISION_MODE_VS) && !(bonded_ia_params[bond_vs].num == 1 ||
83  bonded_ia_params[bond_vs].num == 2))
84  return 5;
85 
86  // Set params
87  collision_params.mode=mode;
88  collision_params.bond_centers=bond_centers;
89  collision_params.bond_vs=bond_vs;
90  collision_params.distance=d;
91  collision_params.vs_particle_type=t;
92 
93  make_particle_type_exist(t);
94 
95  mpi_bcast_collision_params();
96 
97  return 0;

100 // Detect a collision between the given particles.
101 // Add it to the queue in case virtual sites should be added at the point of collision
102 void detect_collision(Particle* p1, Particle* p2)
103 {
104  // The check, whether collision detection is actually turned on is performed in forces.hpp
105 
106  double dist_betw_part, vec21[3];
107  int part1, part2, size;
108 
109  TRACE(printf("%d: consider particles %d and %d\n", this_node, p1->p.identity, p2->p.identity));
110 
111  // Obtain distance between particles
112  dist_betw_part = sqrt(distance2vec(p1->r.p, p2->r.p, vec21));
113  if (dist_betw_part > collision_params.distance)
114  return;
115 
116  TRACE(printf("%d: particles %d and %d on bonding distance %lf\n", this_node, p1->p.identity, p2->p.identity, dist_betw_part));
117 
118  part1 = p1->p.identity;
119  part2 = p2->p.identity;
120 
121  // Retrieving the particles from local_particles is necessary, because the particle might be a
122  // ghost, and those can't store bonding info.
123  p1 = local_particles[part1];
124  p2 = local_particles[part2];
125 
126 #ifdef VIRTUAL_SITES_RELATIVE
127  // Ignore virtual particles
128  if ((p1->p.isVirtual) || (p2->p.isVirtual))
129  return;
130 #endif
131 
132  // Check, if there's already a bond between the particles
133  // First check the bonds of p1
134  if (p1->bl.e) {
135  int i = 0;
136  while(i < p1->bl.n) {
137  size = bonded_ia_params[p1->bl.e[i]].num;
138 
139  if (p1->bl.e[i] == collision_params.bond_centers &&
140  p1->bl.e[i + 1] == part2) {
141  // There's a bond, already. Nothing to do for these particles
142  return;
143  }
144  i += size + 1;
145  }
146  }
147  if (p2->bl.e) {
148  // Check, if a bond is already stored in p2
149  int i = 0;
150  while(i < p2->bl.n) {
151  size = bonded_ia_params[p2->bl.e[i]].num;
152 
153  /* COMPARE P2 WITH P1'S BONDED PARTICLES*/
154 
155  if (p2->bl.e[i] == collision_params.bond_centers &&
156  p2->bl.e[i + 1] == part1) {
157  return;
158  }
159  i += size + 1;
160  }
161  }
162 
163  TRACE(printf("%d: no previous bond, binding\n", this_node));
164 
165  /* If we're still here, there is no previous bond between the particles,
166  we have a new collision */
167 
168  /* create marking bond between the colliding particles immediately */
169  if (collision_params.mode & COLLISION_MODE_BOND) {
170  int bondG[2];
171  int primary = part1, secondary = part2;
172  // put the bond to the physical particle; at least one partner always is
173  if (p1->l.ghost) {
174  primary = part2;
175  secondary = part1;
176  }
177  bondG[0]=collision_params.bond_centers;
178  bondG[1]=secondary;
179  local_change_bond(primary, bondG, 0);
180  }
181 
182  if (collision_params.mode & (COLLISION_MODE_VS | COLLISION_MODE_EXCEPTION)) {
183  /* If we also create virtual sites or throw an exception, we add the collision
184  to the queue to process later */
185 
186  // Point of collision
187  double new_position[3];
188  for (int i=0;i<3;i++) {
189  new_position[i] = p1->r.p[i] - vec21[i] * 0.50;
190  }
191 
192  number_of_collisions++;
193 
194  // Allocate mem for the new collision info
195  collision_queue = (collision_struct *) realloc (collision_queue, (number_of_collisions) * sizeof(collision_struct));
196 
197  // Save the collision
198  collision_queue[number_of_collisions-1].pp1 = part1;
199  collision_queue[number_of_collisions-1].pp2 = part2;
200  for (int i=0;i<3;i++) {
201  collision_queue[number_of_collisions-1].point_of_collision[i] = new_position[i];
202  }
203  }
204 }

215 // Handle the collisions stored in the queue
216 void handle_collisions ()
217 {
218  // printf("number of collisions in handle collision are %d\n",number_of_collisions);
219 
220  for (int i = 0; i < number_of_collisions; i++) {
221  // printf("Handling collision of particles %d %d\n", collision_queue[i].pp1, collision_queue[i].pp2);
222  // fflush(stdout);
223 
224  if (collision_params.mode & (COLLISION_MODE_EXCEPTION)) {
225 
226  int id1, id2;
227  if (collision_queue[i].pp1 > collision_queue[i].pp2) {
228  id1 = collision_queue[i].pp2;
229  id2 = collision_queue[i].pp1;
230  }
231  else {
232  id1 = collision_queue[i].pp1;
233  id2 = collision_queue[i].pp2;
234  }
235  ostringstream msg;
236  msg << "collision between particles " << id1 << " and " <<id2;
237  runtimeError(msg);
238  }
239 
240  /* If we don't have virtual_sites_relative, only bonds between centers of
241  colliding particles are possible and nothing is to be done here */
242 #ifdef VIRTUAL_SITES_RELATIVE
243  if (collision_params.mode & COLLISION_MODE_VS) {
244 
245  // add virtual sites placed at the point of collision and bind them
246  int bondG[3];
247 
248  // Virtual site related to first particle in the collision
249  place_particle(max_seen_particle+1,collision_queue[i].point_of_collision);
250  vs_relate_to(max_seen_particle,collision_queue[i].pp1);
251  (local_particles[max_seen_particle])->p.isVirtual=1;
252  (local_particles[max_seen_particle])->p.type=collision_params.vs_particle_type;
253 
254  // Virtual particle related to 2nd particle of the collision
255  place_particle(max_seen_particle+1,collision_queue[i].point_of_collision);
256  vs_relate_to(max_seen_particle,collision_queue[i].pp2);
257  (local_particles[max_seen_particle])->p.isVirtual=1;
258  (local_particles[max_seen_particle])->p.type=collision_params.vs_particle_type;
259 
260  switch (bonded_ia_params[collision_params.bond_vs].num) {
261  case 1: {
262  // Create bond between the virtual particles
263  bondG[0] = collision_params.bond_vs;
264  bondG[1] = max_seen_particle-1;
265  local_change_bond(max_seen_particle, bondG, 0);
266  break;
267  }
268  case 2: {
269  // Create 1st bond between the virtual particles
270  bondG[0] = collision_params.bond_vs;
271  bondG[1] = collision_queue[i].pp1;
272  bondG[2] = collision_queue[i].pp2;
273  local_change_bond(max_seen_particle, bondG, 0);
274  local_change_bond(max_seen_particle-1, bondG, 0);
275  break;
276  }
277  }
278  }
279 #endif
280  }
281 
282  // Reset the collision queue
283  if(collision_queue && (number_of_collisions > 0)) {
284  free(collision_queue);
285  collision_queue = 0;
286  number_of_collisions = 0;
287  }
288 
289  announce_resort_particles();
290 }
