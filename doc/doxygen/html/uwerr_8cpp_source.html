64 static double gammaln(double a)
65 {
66  int j;
67  double x, y, tmp, ser;
68  double cof[6] = { 76.18009172947146,
69  -86.50532032941677,
70  24.01409824083091,
71  -1.231739572450155,
72  0.1208650973866179e-2,
73  -0.5395239384953e-5};

75  y = x = a;
76  tmp = x+5.5;

78  ser = 1.000000000190015;

85 static void gammaser(double * gser, double a, double x)
86 {
87  int n, ITMAX = 100;
88  double sum, del, ap, gln, eps = DBL_EPSILON;
89 
90  gln = gammaln(a);
91 
92  if (x <= 0.0) {
93  if (x < 0.0)
94  puts("uwerr: x less than 0 in gammaser.");
95  *gser = 0.0;
96  return;
97  } else {
98  ap = a;
99  del = sum = 1.0/a;
100  for (n = 0; n < ITMAX; ++n) {
101  ++ap;
102  del *= x/ap;
103  sum += del;
104  if (fabs(del) < fabs(sum)*eps) {
105  *gser = sum*exp(-x+a*log(x)-gln);
106  return;
107  }
108  }
109  puts("uwerr: a too large, ITMAX too small in gammaser.");
110  }
111 }

114 static void gammacf(double * gcf, double a, double x)
115 {
116  int i, ITMAX = 100;
117  double an, b, c, d, del, h, gln, eps = DBL_EPSILON, FPMIN = DBL_MIN;
118 
119  gln=gammaln(a);
120  b = x+1.0-a;
121  c = 1.0/FPMIN;
122  d = 1.0/b;
123  h = d;
124  for (i = 1; i <= ITMAX; ++i) {
125  an = -i*(i-a);
126  b += 2.0;
127  d = an*d+b;
128  if (fabs(d) < FPMIN)
129  d = FPMIN;
130  c = b+an/c;
131  if (fabs(c) < FPMIN)
132  c = FPMIN;
133  d = 1.0/d;
134  del = d*c;
135  h *= del;
136  if (fabs(del-1.0) <= eps)
137  break;
138  }
139  if (i > ITMAX)
140  puts("uwerr: a too large, ITMAX too small in gammacf.");
141  *gcf = exp(-x+a*log(x)-gln)*h;
142 }
