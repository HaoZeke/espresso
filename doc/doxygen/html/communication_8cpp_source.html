331 void mpi_who_has()
332 {
333  Cell *cell;
334  int *sizes = (int*)malloc(sizeof(int)*n_nodes);
335  int *pdata = NULL;
336  int pdata_s = 0;
337 
338  mpi_call(mpi_who_has_slave, -1, 0);
339 
340  int n_part = cells_get_n_particles();
341  /* first collect number of particles on each node */
342  MPI_Gather(&n_part, 1, MPI_INT, sizes, 1, MPI_INT, 0, comm_cart);
343 
344  for (int i = 0; i <= max_seen_particle; i++)
345  particle_node[i] = -1;
346 
347  /* then fetch particle locations */
348  for (int pnode = 0; pnode < n_nodes; pnode++) {
349  COMM_TRACE(fprintf(stderr, "node %d reports %d particles\n",
350  pnode, sizes[pnode]));
351  if (pnode == this_node) {
352  for (int c = 0; c < local_cells.n; c++) {
353  cell = local_cells.cell[c];
354  for (int i = 0; i < cell->n; i++)
355  particle_node[cell->part[i].p.identity] = this_node;
356  }
357  }
358  else if (sizes[pnode] > 0) {
359  if (pdata_s < sizes[pnode]) {
360  pdata_s = sizes[pnode];
361  pdata = (int *)realloc(pdata, sizeof(int)*pdata_s);
362  }
363  MPI_Recv(pdata, sizes[pnode], MPI_INT, pnode, SOME_TAG,
364  comm_cart, MPI_STATUS_IGNORE);
365  for (int i = 0; i < sizes[pnode]; i++)
366  particle_node[pdata[i]] = pnode;
367  }
368  }
369 
370  free(pdata);
371  free(sizes);
372 }
