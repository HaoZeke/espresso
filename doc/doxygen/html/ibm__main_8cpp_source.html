178

183 
184  // Loop over all affected nodes

186  {
187  for (int y = 0; y < 2; y++)
188  {

222 
223  double lbboundary_mindist, distvec[3];
224  double pos[3];
225 
226 #ifdef LB_BOUNDARIES
227  int boundary_no;
228  int boundary_flag=-1; // 0 if more than agrid/2 away from the boundary, 1 if 0<dist<agrid/2, 2 if dist <0
229 
230  lbboundary_mindist_position(p, &lbboundary_mindist, distvec, &boundary_no);
231  if (lbboundary_mindist>lbpar.agrid/2) {
232  boundary_flag=0;
233  pos[0]=p[0];
234  pos[1]=p[1];
235  pos[2]=p[2];
236 
237  } else if (lbboundary_mindist > 0 ) {
238  boundary_flag=1;
239  pos[0]=p[0] - distvec[0]+ distvec[0]/lbboundary_mindist*lbpar.agrid/2.;
240  pos[1]=p[1] - distvec[1]+ distvec[1]/lbboundary_mindist*lbpar.agrid/2.;
241  pos[2]=p[2] - distvec[2]+ distvec[2]/lbboundary_mindist*lbpar.agrid/2.;
242 
243  } else {
244  boundary_flag=2;
245  v[0]= lb_boundaries[boundary_no].velocity[0]*lbpar.agrid/lbpar.tau;
246  v[1]= lb_boundaries[boundary_no].velocity[1]*lbpar.agrid/lbpar.tau;
247  v[2]= lb_boundaries[boundary_no].velocity[2]*lbpar.agrid/lbpar.tau;
248  return; // we can return without interpolating
249  }
250 #else
251  pos[0]=p[0];
252  pos[1]=p[1];
253  pos[2]=p[2];
254 #endif
255 
256  /* determine elementary lattice cell surrounding the particle
257  and the relative position of the particle in this cell */
258  lblattice.map_position_to_lattice(pos,node_index,delta);
259 
260  /* calculate fluid velocity at particle's position
261  this is done by linear interpolation
262  (Eq. (11) Ahlrichs and Duenweg, JCP 111(17):8225 (1999)) */
263  interpolated_u[0] = interpolated_u[1] = interpolated_u[2] = 0.0 ;
264  // This for the f/2 contribution to the velocity
265  forceAdded[0] = forceAdded[1] = forceAdded[2] = 0;
266 
267  for (z=0;z<2;z++) {
268  for (y=0;y<2;y++) {
269  for (x=0;x<2;x++) {
270 
271  index = node_index[(z*2+y)*2+x];
272  f = lbfields[index].force_buf;
273 
274  // This can be done easier withouth copying the code twice
275  // We probably can even set the boundary velocity directly
276 #ifdef LB_BOUNDARIES
277  if (lbfields[index].boundary) {
278  local_rho=lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid;
279  local_j[0] = lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid*lb_boundaries[lbfields[index].boundary-1].velocity[0];
280  local_j[1] = lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid*lb_boundaries[lbfields[index].boundary-1].velocity[1];
281  local_j[2] = lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid*lb_boundaries[lbfields[index].boundary-1].velocity[2];
282  } else
283 #endif
284  {
285  lb_calc_modes(index, modes);
286  local_rho = lbpar.rho[0]*lbpar.agrid*lbpar.agrid*lbpar.agrid + modes[0];
287 
288  // Add the +f/2 contribution!!
289  local_j[0] = modes[1] + f[0]/2;
290  local_j[1] = modes[2] + f[1]/2;
291  local_j[2] = modes[3] + f[2]/2;
292 
293  // Keep track of the forces that we added to the fluid
294  // This is necessary for communication because this part is executed for real and ghost particles
295  // Later on we sum the real and ghost contributions
296  const double fExt[3] = { lbpar.ext_force[0]*pow(lbpar.agrid,2)*lbpar.tau*lbpar.tau, lbpar.ext_force[1]*pow(lbpar.agrid,2)*lbpar.tau*lbpar.tau, lbpar.ext_force[2]*pow(lbpar.agrid,2)*lbpar.tau*lbpar.tau };
297 
298  forceAdded[0] += delta[3*x+0]*delta[3*y+1]*delta[3*z+2]*(f[0]-fExt[0])/2/(local_rho);
299  forceAdded[1] += delta[3*x+0]*delta[3*y+1]*delta[3*z+2]*(f[1]-fExt[1])/2/(local_rho);
300  forceAdded[2] += delta[3*x+0]*delta[3*y+1]*delta[3*z+2]*(f[2]-fExt[2])/2/(local_rho);
301 
302  }
303 
304  // Interpolate velocity
305  interpolated_u[0] += delta[3*x+0]*delta[3*y+1]*delta[3*z+2]*local_j[0]/(local_rho);
306  interpolated_u[1] += delta[3*x+0]*delta[3*y+1]*delta[3*z+2]*local_j[1]/(local_rho);
307  interpolated_u[2] += delta[3*x+0]*delta[3*y+1]*delta[3*z+2]*local_j[2]/(local_rho) ;
308 
309  }
310  }
311  }
312 #ifdef LB_BOUNDARIES
313  if (boundary_flag==1) {
314  v[0]=lbboundary_mindist/(lbpar.agrid/2.)*interpolated_u[0]+(1-lbboundary_mindist/(lbpar.agrid/2.))*lb_boundaries[boundary_no].velocity[0];
315  v[1]=lbboundary_mindist/(lbpar.agrid/2.)*interpolated_u[1]+(1-lbboundary_mindist/(lbpar.agrid/2.))*lb_boundaries[boundary_no].velocity[1];
316  v[2]=lbboundary_mindist/(lbpar.agrid/2.)*interpolated_u[2]+(1-lbboundary_mindist/(lbpar.agrid/2.))*lb_boundaries[boundary_no].velocity[2];
317 
318  } else {
319  v[0] = interpolated_u[0];
320  v[1] = interpolated_u[1];
321  v[2] = interpolated_u[2];
322  }
323 #else
324  v[0] = interpolated_u[0];
325  v[1] = interpolated_u[1];
326  v[2] = interpolated_u[2];
327 #endif
328  v[0] *= lbpar.agrid/lbpar.tau;
329  v[1] *= lbpar.agrid/lbpar.tau;
330  v[2] *= lbpar.agrid/lbpar.tau;
331 }

422  p[j].f.f[0] = force[0] * lbpar.agrid/lbpar.tau;

449  p[j].m.v[1] = p[j].f.f[1];
