312 static void add_dipole_force()
313 {
314  double pref = coulomb.prefactor*4*M_PI*ux*uy*uz;
315  int size = 3;
316 
317  /* for nonneutral systems, this shift gives the background contribution
318  (rsp. for this shift, the DM of the background is zero) */
319  double shift = 0.5*box_l[2];
320  double field_tot = 0;
321 
322  // collect moments
323 
324  gblcblk[0] = 0; // sum q_i (z_i - L/2)
325  gblcblk[1] = 0; // sum q_i z_i
326  gblcblk[2] = 0; // sum q_i
327 
328  for (int c = 0; c < local_cells.n; c++) {
329  int np = local_cells.cell[c]->n;
330  Particle *part = local_cells.cell[c]->part;
331  for (int i = 0; i < np; i++) {
332  gblcblk[0] += part[i].p.q*(part[i].r.p[2] - shift);
333  gblcblk[1] += part[i].p.q*part[i].r.p[2];
334  gblcblk[2] += part[i].p.q;
335 
336  if(elc_params.dielectric_contrast_on) {
337  if(part[i].r.p[2]<elc_params.space_layer) {
338  gblcblk[0] +=elc_params.di_mid_bot*part[i].p.q*(-part[i].r.p[2] - shift);
339  gblcblk[2] +=elc_params.di_mid_bot*part[i].p.q;
340  }
341  if(part[i].r.p[2]>(elc_params.h-elc_params.space_layer)) {
342  gblcblk[0] +=elc_params.di_mid_top*part[i].p.q*(2*elc_params.h-part[i].r.p[2] - shift);
343  gblcblk[2] +=elc_params.di_mid_top*part[i].p.q;
344  }
345  }
346  }
347  }
348  gblcblk[0] *= pref;
349  gblcblk[1] *= pref*height_inverse/uz;
350  gblcblk[2] *= pref;
351 
352  distribute(size);
353 
354  // Yeh+Berkowitz dipole term
355  field_tot = gblcblk[0];
356 
357  // Const. potential contribution
358  if (elc_params.const_pot_on) {
359  field_induced = gblcblk[1];
360  field_applied = elc_params.pot_diff * height_inverse;
361  field_tot -= field_applied + field_induced;
362  }
363 
364  for (int c = 0; c < local_cells.n; c++) {
365  int np = local_cells.cell[c]->n;
366  Particle *part = local_cells.cell[c]->part;
367  for (int i = 0; i < np; i++) {
368  part[i].f.f[2] -= field_tot*part[i].p.q;
369 
370  if (!elc_params.neutralize) {
371  // SUBTRACT the forces of the P3M homogeneous neutralizing background
372  part[i].f.f[2] += gblcblk[2]*part[i].p.q*(part[i].r.p[2] - shift);
373  }
374  }
375  }
376 }

378 static double dipole_energy()
379 {
380  double pref = coulomb.prefactor*2*M_PI*ux*uy*uz;
381  int size = 7;
382  /* for nonneutral systems, this shift gives the background contribution
383  (rsp. for this shift, the DM of the background is zero) */
384  double shift = 0.5*box_l[2];
385 
386  // collect moments
387 
388  gblcblk[0] = 0; // sum q_i primary box
389  gblcblk[1] = 0; // sum q_i boundary layers
390  gblcblk[2] = 0; // sum q_i (z_i - L/2) primary box
391  gblcblk[3] = 0; // sum q_i (z_i - L/2) boundary layers
392  gblcblk[4] = 0; // sum q_i (z_i - L/2)^2 primary box
393  gblcblk[5] = 0; // sum q_i (z_i - L/2)^2 boundary layers
394  gblcblk[6] = 0; // sum q_i z_i primary box
395 
396  for (int c = 0; c < local_cells.n; c++) {
397  int np = local_cells.cell[c]->n;
398  Particle *part = local_cells.cell[c]->part;
399  for (int i = 0; i < np; i++) {
400  gblcblk[0] += part[i].p.q;
401  gblcblk[2] += part[i].p.q*(part[i].r.p[2] - shift);
402  gblcblk[4] += part[i].p.q*(SQR(part[i].r.p[2]- shift));
403  gblcblk[6] += part[i].p.q*part[i].r.p[2];
404 
405  if(elc_params.dielectric_contrast_on) {
406  if(part[i].r.p[2]<elc_params.space_layer) {
407  gblcblk[1] +=elc_params.di_mid_bot*part[i].p.q;
408  gblcblk[3] +=elc_params.di_mid_bot*part[i].p.q*(-part[i].r.p[2] - shift);
409  gblcblk[5] +=elc_params.di_mid_bot*part[i].p.q*(SQR(-part[i].r.p[2] - shift));
410  }
411  if(part[i].r.p[2]>(elc_params.h-elc_params.space_layer)) {
412  gblcblk[1] +=elc_params.di_mid_top*part[i].p.q;
413  gblcblk[3] +=elc_params.di_mid_top*part[i].p.q*(2*elc_params.h-part[i].r.p[2] - shift);
414  gblcblk[5] +=elc_params.di_mid_top*part[i].p.q*(SQR(2*elc_params.h-part[i].r.p[2] - shift));
415  }
416  }
417  }
418  }
419 
420  distribute(size);
421 
422  // Yeh + Berkowitz term
423  double eng = 2*pref*(SQR(gblcblk[2])+gblcblk[2]*gblcblk[3]);
424 
425  if (!elc_params.neutralize) {
426  // SUBTRACT the energy of the P3M homogeneous neutralizing background
427  eng += 2*pref*(-gblcblk[0]*gblcblk[4] - (.25 - .5/3.)*SQR(gblcblk[0]*box_l[2]));
428  }
429 
430  if (elc_params.dielectric_contrast_on) {
431  if (elc_params.const_pot_on) {
432  // zero potential difference contribution
433  eng += pref*height_inverse/uz*SQR(gblcblk[6]);
434  // external potential shift contribution
435  eng += elc_params.pot_diff * height_inverse * gblcblk[6];
436  }
437 
438  /* counter the P3M homogeneous background contribution to the
439  boundaries. We never need that, since a homogeneous background
440  spanning the artifical boundary layers is aphysical. */
441  eng += pref*(-(gblcblk[1]*gblcblk[4] + gblcblk[0]*gblcblk[5])
442  - (1. - 2./3.)*gblcblk[0]*gblcblk[1]*SQR(box_l[2]));
443  }
444 
445  return this_node == 0 ? eng : 0;
446 }

466 /*****************************************************************/
467 static double z_energy()
468 {
469  double pref = coulomb.prefactor*2*M_PI*ux*uy;
470  int size = 4;
471 
472  double eng=0;
473  /* for nonneutral systems, this shift gives the background contribution
474  (rsp. for this shift, the DM of the background is zero) */
475  double shift = 0.5*box_l[2];
476 
477  if(elc_params.dielectric_contrast_on) {
478  if(elc_params.const_pot_on) {
479  clear_vec(gblcblk, size);
480  for (int c = 0; c < local_cells.n; c++) {
481  int np = local_cells.cell[c]->n;
482  Particle *part = local_cells.cell[c]->part;
483  for (int i = 0; i < np; i++) {
484  gblcblk[0] += part[i].p.q;
485  gblcblk[1] += part[i].p.q*(part[i].r.p[2] - shift);
486  if(elc_params.dielectric_contrast_on) {
487  if(part[i].r.p[2]<elc_params.space_layer) {
488  gblcblk[2] -= elc_params.di_mid_bot*part[i].p.q;
489  gblcblk[3] -= elc_params.di_mid_bot*part[i].p.q*(-part[i].r.p[2] - shift);
490  }
491  if (part[i].r.p[2]>(elc_params.h-elc_params.space_layer)) {
492  gblcblk[2] += elc_params.di_mid_top*part[i].p.q;
493  gblcblk[3] += elc_params.di_mid_top*part[i].p.q*(2*elc_params.h - part[i].r.p[2] - shift);
494  }
495  }
496  }
497  }
498  }
499  else {
500  double delta = elc_params.di_mid_top*elc_params.di_mid_bot;
501  double fac_delta_mid_bot = elc_params.di_mid_bot /(1 - delta);
502  double fac_delta_mid_top = elc_params.di_mid_top /(1 - delta);
503  double fac_delta = delta /(1 - delta);
504 
505  clear_vec(gblcblk, size);
506  for (int c = 0; c < local_cells.n; c++) {
507  int np = local_cells.cell[c]->n;
508  Particle *part = local_cells.cell[c]->part;
509  for (int i = 0; i < np; i++) {
510  gblcblk[0] += part[i].p.q;
511  gblcblk[1] += part[i].p.q*(part[i].r.p[2] - shift);
512  if(elc_params.dielectric_contrast_on) {
513  if(part[i].r.p[2]<elc_params.space_layer) {
514  gblcblk[2] += fac_delta*(elc_params.di_mid_bot+1)*part[i].p.q;
515  gblcblk[3] += part[i].p.q*(image_sum_b(elc_params.di_mid_bot*delta,-(2*elc_params.h+part[i].r.p[2]))
516  +image_sum_b(delta,-(2*elc_params.h-part[i].r.p[2])));
517  } else {
518  gblcblk[2] += fac_delta_mid_bot*(1+elc_params.di_mid_top)*part[i].p.q;
519  gblcblk[3] += part[i].p.q*(image_sum_b(elc_params.di_mid_bot,-part[i].r.p[2])
520  +image_sum_b(delta,-(2*elc_params.h-part[i].r.p[2])));
521  }
522  if(part[i].r.p[2]>(elc_params.h-elc_params.space_layer)) {
523  //note the minus sign here which is required due to |z_i-z_j|
524  gblcblk[2] -= fac_delta*(elc_params.di_mid_top+1)*part[i].p.q;
525  gblcblk[3] -= part[i].p.q*(image_sum_t(elc_params.di_mid_top*delta,4*elc_params.h-part[i].r.p[2])
526  +image_sum_t(delta,2*elc_params.h+part[i].r.p[2]));
527  } else {
528  //note the minus sign here which is required due to |z_i-z_j|
529  gblcblk[2] -= fac_delta_mid_top*(1+elc_params.di_mid_bot)*part[i].p.q;
530  gblcblk[3] -= part[i].p.q*(image_sum_t(elc_params.di_mid_top,2*elc_params.h-part[i].r.p[2])
531  +image_sum_t(delta,2*elc_params.h+part[i].r.p[2]));
532  }
533  }
534  }
535  }
536  }
537  }
538  distribute(size);
539 
540  if (this_node == 0)
541  eng -= pref*(gblcblk[1]*gblcblk[2]-gblcblk[0]*gblcblk[3]);
542 
543  return eng;
544 }

1165 int ELC_tune(double error)
1166 {
1167  double err;
1168  double h = elc_params.h, lz = box_l[2];
1169  double min_inv_boxl = dmin(ux, uy);
1170 
1171  if (elc_params.dielectric_contrast_on) {
1172  // adjust lz according to dielectric layer method
1173  lz = elc_params.h + elc_params.space_layer;
1174  }
1175 
1176  if (h < 0)
1177  return ES_ERROR;
1178 
1179  elc_params.far_cut = min_inv_boxl;
1180  do {
1181  err = 0.5*(exp(2*M_PI*elc_params.far_cut*h)/(lz - h)*
1182  (C_2PI*elc_params.far_cut + 2*(ux + uy) + 1/(lz - h))/
1183  (exp(2*M_PI*elc_params.far_cut*lz)- 1) +
1184  exp(-2*M_PI*elc_params.far_cut*h)/(lz + 1)*
1185  (C_2PI*elc_params.far_cut + 2*(ux + uy) + 1/(lz + h))/
1186  (exp(2*M_PI*elc_params.far_cut*lz)- 1));
1187 
1188  elc_params.far_cut += min_inv_boxl;
1189  }
1190  while (err > error && elc_params.far_cut < MAXIMAL_FAR_CUT);
1191  if (elc_params.far_cut >= MAXIMAL_FAR_CUT)
1192  return ES_ERROR;
1193  elc_params.far_cut -= min_inv_boxl;
1194  elc_params.far_cut2 = SQR(elc_params.far_cut);
1195 
1196  return ES_OK;
1197 }
