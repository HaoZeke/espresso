93 int calc_transmit_size(GhostCommunication *gc, int data_parts)
94 {
95  int p, n_buffer_new;
96 
97  if (data_parts & GHOSTTRANS_PARTNUM)
98  n_buffer_new = sizeof(int)*gc->n_part_lists;
99  else {
100  int count = 0;
101  for (p = 0; p < gc->n_part_lists; p++)
102  count += gc->part_lists[p]->n;
103 
104  n_buffer_new = 0;
105  if (data_parts & GHOSTTRANS_PROPRTS) {
106  n_buffer_new += sizeof(ParticleProperties);
107  // sending size of bond/exclusion lists
108 #ifdef GHOSTS_HAVE_BONDS
109  n_buffer_new += sizeof(int);
110 #ifdef EXCLUSIONS
111  n_buffer_new += sizeof(int);
112 #endif
113 #endif
114  }
115  if (data_parts & GHOSTTRANS_POSITION)
116  n_buffer_new += sizeof(ParticlePosition);
117  if (data_parts & GHOSTTRANS_MOMENTUM)
118  n_buffer_new += sizeof(ParticleMomentum);
119  if (data_parts & GHOSTTRANS_FORCE)
120  n_buffer_new += sizeof(ParticleForce);
121 #ifdef LB
122  if (data_parts & GHOSTTRANS_COUPLING)
123  n_buffer_new += sizeof(ParticleLatticeCoupling);
124 #endif
125 #ifdef ENGINE
126  if (data_parts & GHOSTTRANS_SWIMMING)
127  n_buffer_new += sizeof(ParticleParametersSwimming);
128 #endif
129  n_buffer_new *= count;
130  }
131  // also sending length of bond buffer
132  if (data_parts & GHOSTTRANS_PROPRTS)
133  n_buffer_new += sizeof(int);
134  return n_buffer_new;
135 }

137 void prepare_send_buffer(GhostCommunication *gc, int data_parts)
138 {
139  GHOST_TRACE(fprintf(stderr, "%d: prepare sending to/bcast from %d\n", this_node, gc->node));
140 
141  /* reallocate send buffer */
142  n_s_buffer = calc_transmit_size(gc, data_parts);
143  if (n_s_buffer > max_s_buffer) {
144  max_s_buffer = n_s_buffer;
145  s_buffer = (char*)realloc(s_buffer, max_s_buffer);
146  }
147  GHOST_TRACE(fprintf(stderr, "%d: will send %d\n", this_node, n_s_buffer));
148 
149  s_bondbuffer.resize(0);
150 
151  /* put in data */
152  char *insert = s_buffer;
153  for (int pl = 0; pl < gc->n_part_lists; pl++) {
154  int np = gc->part_lists[pl]->n;
155  if (data_parts & GHOSTTRANS_PARTNUM) {
156  *(int *)insert = np;
157  insert += sizeof(int);
158  GHOST_TRACE(fprintf(stderr, "%d: %d particles assigned\n",
159  this_node, np));
160  }
161  else {
162  Particle *part = gc->part_lists[pl]->part;
163  for (int p = 0; p < np; p++) {
164  Particle *pt = &part[p];
165  if (data_parts & GHOSTTRANS_PROPRTS) {
166  memmove(insert, &pt->p, sizeof(ParticleProperties));
167  insert += sizeof(ParticleProperties);
168 #ifdef GHOSTS_HAVE_BONDS
169  *(int *)insert = pt->bl.n;
170  insert += sizeof(int);
171  if (pt->bl.n) {
172  s_bondbuffer.insert(s_bondbuffer.end(), pt->bl.e, pt->bl.e + pt->bl.n);
173  }
174 #ifdef EXCLUSIONS
175  *(int *)insert = pt->el.n;
176  insert += sizeof(int);
177  if (pt->el.n) {
178  s_bondbuffer.insert(s_bondbuffer.end(), pt->el.e, pt->el.e + pt->el.n);
179  }
180 #endif
181 #endif
182  }
183  if (data_parts & GHOSTTRANS_POSSHFTD) {
184  /* ok, this is not nice, but perhaps fast */
185  ParticlePosition *pp = (ParticlePosition *)insert;
186  int i;
187  memmove(pp, &pt->r, sizeof(ParticlePosition));
188  for (i = 0; i < 3; i++)
189  pp->p[i] += gc->shift[i];
190  /* No special wrapping for Lees-Edwards here:
191  * LE wrap-on-receive instead, for convenience in
192  * mapping to local cell geometry. */
193  insert += sizeof(ParticlePosition);
194  }
195  else if (data_parts & GHOSTTRANS_POSITION) {
196  memmove(insert, &pt->r, sizeof(ParticlePosition));
197  insert += sizeof(ParticlePosition);
198  }
199  if (data_parts & GHOSTTRANS_MOMENTUM) {
200  memmove(insert, &pt->m, sizeof(ParticleMomentum));
201  insert += sizeof(ParticleMomentum);
202  }
203  if (data_parts & GHOSTTRANS_FORCE) {
204  memmove(insert, &pt->f, sizeof(ParticleForce));
205  insert += sizeof(ParticleForce);
206  }
207 #ifdef LB
208  if (data_parts & GHOSTTRANS_COUPLING) {
209  memmove(insert, &pt->lc, sizeof(ParticleLatticeCoupling));
210  insert += sizeof(ParticleLatticeCoupling);
211  }
212 #endif
213 #ifdef ENGINE
214  if (data_parts & GHOSTTRANS_SWIMMING) {
215  memmove(insert, &pt->swim, sizeof(ParticleParametersSwimming));
216  insert += sizeof(ParticleParametersSwimming);
217  }
218 #endif
219  }
220  }
221  }
222  if (data_parts & GHOSTTRANS_PROPRTS) {
223  GHOST_TRACE(fprintf(stderr, "%d: bond buffer size is %ld\n",
224  this_node, s_bondbuffer.size()));
225  *(int *)insert = int(s_bondbuffer.size());
226  insert += sizeof(int);
227  }
228 
229  if (insert - s_buffer != n_s_buffer) {
230  fprintf(stderr, "%d: INTERNAL ERROR: send buffer size %d "
231  "differs from what I put in (%ld)\n",
232  this_node, n_s_buffer, insert - s_buffer);
233  errexit();
234  }

285 void put_recv_buffer(GhostCommunication *gc, int data_parts)
286 {
287  /* put back data */
288  char *retrieve = r_buffer;
289 
290  std::vector<int>::const_iterator bond_retrieve = r_bondbuffer.begin();
291 
292  for (int pl = 0; pl < gc->n_part_lists; pl++) {
293  ParticleList *cur_list = gc->part_lists[pl];
294  if (data_parts & GHOSTTRANS_PARTNUM) {
295  GHOST_TRACE(fprintf(stderr, "%d: reallocating cell %p to size %d, assigned to node %d\n",
296  this_node, cur_list, *(int *)retrieve, gc->node));
297  prepare_ghost_cell(cur_list, *(int *)retrieve);
298  retrieve += sizeof(int);
299  }
300  else {
301  int np = cur_list->n;
302  Particle *part = cur_list->part;
303  for (int p = 0; p < np; p++) {
304  Particle *pt = &part[p];
305  if (data_parts & GHOSTTRANS_PROPRTS) {
306  memmove(&pt->p, retrieve, sizeof(ParticleProperties));
307  retrieve += sizeof(ParticleProperties);
308 #ifdef GHOSTS_HAVE_BONDS
309  int n_bonds;
310  memmove(&n_bonds, retrieve, sizeof(int));
311  retrieve += sizeof(int);
312  if (n_bonds) {
313  realloc_intlist(&pt->bl, pt->bl.n = n_bonds);
314  std::copy(bond_retrieve, bond_retrieve + n_bonds, pt->bl.e);
315  bond_retrieve += n_bonds;
316  }
317 #ifdef EXCLUSIONS
318  memmove(&n_bonds, retrieve, sizeof(int));
319  retrieve += sizeof(int);
320  if (n_bonds) {
321  realloc_intlist(&pt->el, pt->el.n = n_bonds);
322  std::copy(bond_retrieve, bond_retrieve + n_bonds, pt->el.e);
323  bond_retrieve += n_bonds;
324  }
325 #endif
326 #endif
327  if (local_particles[pt->p.identity] == NULL) {
328  local_particles[pt->p.identity] = pt;
329  }
330  }
331  if (data_parts & GHOSTTRANS_POSITION) {
332  memmove(&pt->r, retrieve, sizeof(ParticlePosition));
333  retrieve += sizeof(ParticlePosition);
334 #ifdef LEES_EDWARDS
335  /* special wrapping conditions for x component of y LE shift */
336  if( gc->shift[1] != 0.0 ){
337  /* LE transforms are wrapped
338  ---Using this method because its a shortcut to getting a neat-looking verlet list. */
339  if( pt->r.p[0]
340  - (my_left[0] + cur_list->myIndex[0]*dd.cell_size[0]) > 2*dd.cell_size[0] )
341  pt->r.p[0]-=box_l[0];
342  if( pt->r.p[0]
343  - (my_left[0] + cur_list->myIndex[0]*dd.cell_size[0]) < -2*dd.cell_size[0] )
344  pt->r.p[0]+=box_l[0];
345  }
346 #endif
347  }
348  if (data_parts & GHOSTTRANS_MOMENTUM) {
349  memmove(&pt->m, retrieve, sizeof(ParticleMomentum));
350  retrieve += sizeof(ParticleMomentum);
351 #ifdef LEES_EDWARDS
352  /* give ghost particles correct velocity for the main
353  * non-ghost LE reference frame */
354  if( gc->shift[1] > 0.0 )
355  pt->m.v[0] += lees_edwards_rate;
356  else if( gc->shift[1] < 0.0 )
357  pt->m.v[0] -= lees_edwards_rate;
358 #endif
359  }
360  if (data_parts & GHOSTTRANS_FORCE) {
361  memmove(&pt->f, retrieve, sizeof(ParticleForce));
362  retrieve += sizeof(ParticleForce);
363  }
364 #ifdef LB
365  if (data_parts & GHOSTTRANS_COUPLING) {
366  memmove(&pt->lc, retrieve, sizeof(ParticleLatticeCoupling));
367  retrieve += sizeof(ParticleLatticeCoupling);
368  }
369 #endif
370 #ifdef ENGINE
371  if (data_parts & GHOSTTRANS_SWIMMING) {
372  memmove(&pt->swim, retrieve, sizeof(ParticleParametersSwimming));
373  retrieve += sizeof(ParticleParametersSwimming);
374  }
375 #endif
376  }
377  }
378  }
379 
380  if (data_parts & GHOSTTRANS_PROPRTS) {
381  // skip the final information on bonds to be sent in a second round
382  retrieve += sizeof(int);
383  }
384 
385  if (retrieve - r_buffer != n_r_buffer) {
386  fprintf(stderr, "%d: recv buffer size %d differs "
387  "from what I read out (%ld)\n",
388  this_node, n_r_buffer, retrieve - r_buffer);
389  errexit();
390  }
391  if (bond_retrieve != r_bondbuffer.end()) {
392  fprintf(stderr, "%d: recv bond buffer was not used up, %ld elements remain\n",
393  this_node, r_bondbuffer.end() - bond_retrieve );
394  errexit();
395  }
396  r_bondbuffer.resize(0);
397 }

533 void ghost_communicator(GhostCommunicator *gc)
534 {
535  MPI_Status status;
536  int n, n2;
537  int data_parts = gc->data_parts;
538 
539  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm %p, data_parts %d\n", this_node, gc, data_parts));
540 
541  for (n = 0; n < gc->num; n++) {
542  GhostCommunication *gcn = &gc->comm[n];
543  int comm_type = gcn->type & GHOST_JOBMASK;
544  int prefetch = gcn->type & GHOST_PREFETCH;
545  int poststore = gcn->type & GHOST_PSTSTORE;
546  int node = gcn->node;
547 
548  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm round %d, job %x\n", this_node, n, gc->comm[n].type));
549  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm shift %f %f %f\n",this_node, gc->comm[n].shift[0], gc->comm[n].shift[1], gc->comm[n].shift[2]));
550  if (comm_type == GHOST_LOCL)
551  cell_cell_transfer(gcn, data_parts);
552  else {
553  /* prepare send buffer if necessary */
554  if (is_send_op(comm_type, node)) {
555  /* ok, we send this step, prepare send buffer if not yet done */
556  if (!prefetch)
557  prepare_send_buffer(gcn, data_parts);
558  else {
559  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm using prefetched data for operation %d, sending to %d\n", this_node, n, node));
560 #ifdef ADDITIONAL_CHECKS
561  if (n_s_buffer != calc_transmit_size(gcn, data_parts)) {
562  fprintf(stderr, "%d: ghost_comm transmission size and current size of cells to transmit do not match\n", this_node);
563  errexit();
564  }
565 #endif
566  }
567  }
568  else {
569  /* we do not send this time, let's look for a prefetch */
570  if (prefetch) {
571  /* find next action where we send and which has PREFETCH set */
572  for (n2 = n+1; n2 < gc->num; n2++) {
573  GhostCommunication *gcn2 = &gc->comm[n2];
574  int comm_type2 = gcn2->type & GHOST_JOBMASK;
575  int prefetch2 = gcn2->type & GHOST_PREFETCH;
576  int node2 = gcn2->node;
577  if (is_send_op(comm_type2, node2) && prefetch2) {
578  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm prefetch operation %d, is send/bcast to/from %d\n", this_node, n2, node2));
579  prepare_send_buffer(gcn2, data_parts);
580  break;
581  }
582  }
583  }
584  }
585 
586  /* recv buffer for recv and multinode operations to this node */
587  if (is_recv_op(comm_type, node))
588  prepare_recv_buffer(gcn, data_parts);
589 
590  /* transfer data */
591  switch (comm_type) {
592  case GHOST_RECV: {
593  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm receive from %d (%d bytes)\n", this_node, node, n_r_buffer));
594  MPI_Recv(r_buffer, n_r_buffer, MPI_BYTE, node, REQ_GHOST_SEND, comm_cart, &status);
595  if (data_parts & GHOSTTRANS_PROPRTS) {
596  int n_bonds = *(int *)(r_buffer + n_r_buffer - sizeof(int));
597  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm receive from %d (%d bonds)\n", this_node, node, n_bonds));
598  if (n_bonds) {
599  r_bondbuffer.resize(n_bonds);
600  MPI_Recv(&r_bondbuffer[0], n_bonds, MPI_INT, node, REQ_GHOST_SEND, comm_cart, &status);
601  }
602  }
603  break;
604  }
605  case GHOST_SEND: {
606  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm send to %d (%d bytes)\n", this_node, node, n_s_buffer));
607  MPI_Send(s_buffer, n_s_buffer, MPI_BYTE, node, REQ_GHOST_SEND, comm_cart);
608  int n_bonds = s_bondbuffer.size();
609  if (!(data_parts & GHOSTTRANS_PROPRTS) && n_bonds > 0) {
610  fprintf(stderr, "%d: INTERNAL ERROR: not sending properties, but bond buffer not empty\n", this_node);
611  errexit();
612  }
613  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm send to %d (%d ints)\n", this_node, node, n_bonds));
614  if (n_bonds) {
615  MPI_Send(&s_bondbuffer[0], n_bonds, MPI_INT, node, REQ_GHOST_SEND, comm_cart);
616  }
617  break;
618  }
619  case GHOST_BCST:
620  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm bcast from %d (%d bytes)\n", this_node, node,
621  (node == this_node) ? n_s_buffer : n_r_buffer));
622  if (node == this_node) {
623  MPI_Bcast(s_buffer, n_s_buffer, MPI_BYTE, node, comm_cart);
624  int n_bonds = s_bondbuffer.size();
625  if (!(data_parts & GHOSTTRANS_PROPRTS) && n_bonds > 0) {
626  fprintf(stderr, "%d: INTERNAL ERROR: not sending properties, but bond buffer not empty\n", this_node);
627  errexit();
628  }
629  if (n_bonds) {
630  MPI_Bcast(&s_bondbuffer[0], n_bonds, MPI_INT, node, comm_cart);
631  }
632  }
633  else {
634  MPI_Bcast(r_buffer, n_r_buffer, MPI_BYTE, node, comm_cart);
635  if (data_parts & GHOSTTRANS_PROPRTS) {
636  int n_bonds = *(int *)(r_buffer + n_r_buffer - sizeof(int));
637  if (n_bonds) {
638  r_bondbuffer.resize(n_bonds);
639  MPI_Bcast(&r_bondbuffer[0], n_bonds, MPI_INT, node, comm_cart);
640  }
641  }
642  }
643  break;
644  case GHOST_RDCE:
645  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm reduce to %d (%d bytes)\n", this_node, node, n_s_buffer));
646  if (node == this_node)
647  MPI_Reduce(s_buffer, r_buffer, n_s_buffer, MPI_BYTE, MPI_FORCES_SUM, node, comm_cart);
648  else
649  MPI_Reduce(s_buffer, NULL, n_s_buffer, MPI_BYTE, MPI_FORCES_SUM, node, comm_cart);
650  break;
651  }
652  //GHOST_TRACE(MPI_Barrier(comm_cart));
653  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm done\n", this_node));
654 
655  /* recv op; write back data directly, if no PSTSTORE delay is requested. */
656  if (is_recv_op(comm_type, node)) {
657  if (!poststore) {
658  /* forces have to be added, the rest overwritten. Exception is RDCE, where the addition
659  is integrated into the communication. */
660  if (data_parts == GHOSTTRANS_FORCE && comm_type != GHOST_RDCE)
661  add_forces_from_recv_buffer(gcn);
662  else
663  put_recv_buffer(gcn, data_parts);
664  }
665  else {
666  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm delaying operation %d, recv from %d\n", this_node, n, node));
667  }
668  }
669  else {
670  /* send op; write back delayed data from last recv, when this was a prefetch send. */
671  if (poststore) {
672  /* find previous action where we recv and which has PSTSTORE set */
673  for (n2 = n-1; n2 >= 0; n2--) {
674  GhostCommunication *gcn2 = &gc->comm[n2];
675  int comm_type2 = gcn2->type & GHOST_JOBMASK;
676  int poststore2 = gcn2->type & GHOST_PSTSTORE;
677  int node2 = gcn2->node;
678  if (is_recv_op(comm_type2, node2) && poststore2) {
679  GHOST_TRACE(fprintf(stderr, "%d: ghost_comm storing delayed recv, operation %d, from %d\n", this_node, n2, node2));
680 #ifdef ADDITIONAL_CHECKS
681  if (n_r_buffer != calc_transmit_size(gcn2, data_parts)) {
682  fprintf(stderr, "%d: ghost_comm transmission size and current size of cells to transmit do not match\n", this_node);
683  errexit();
684  }
685 #endif
686  /* as above */
687  if (data_parts == GHOSTTRANS_FORCE && comm_type != GHOST_RDCE)
688  add_forces_from_recv_buffer(gcn2);
689  else
690  put_recv_buffer(gcn2, data_parts);
691  break;
692  }
693  }
694  }
695  }
696  }
697  }
698 }
