135 void hamiltonian_calc(int ekin_update_flag)
136 {
137 
138  /* if ekin_update_flag = 0, we calculate all energies with \ref master_energy_calc().
139  if ekin_update_flag = 1, we only updated momenta, so there we only need to recalculate
140  kinetic energy with \ref calc_kinetic(). */
141 
142  int i;
143  double result = 0.0;
144  double ekt, ekr;
145 
146  INTEG_TRACE(fprintf(stderr,"%d: hamiltonian_calc:\n",this_node));
147 
148  //initialize energy struct
149  if (total_energy.init_status == 0) {
150  init_energies(&total_energy);
151  //if we are initializing energy we have to calculate all energies anyway
152  ekin_update_flag = 0;
153  }
154 
155  //calculate energies
156  if (ekin_update_flag == 0)
157  master_energy_calc();
158  else
159  calc_kinetic(&ekt, &ekr);
160 
161  //sum up energies on master node, and update ghmcdata struct
162  if (this_node==0) {
163  ghmcdata.hmlt_old = ghmcdata.hmlt_new;
164  for (i = ekin_update_flag; i < total_energy.data.n; i++) {
165  result += total_energy.data.e[i];
166  }
167  if (ekin_update_flag == 1)
168  result += ekt+ekr;
169  ghmcdata.hmlt_new=result;
170  }
171 
172 }

205 void calc_kinetic(double *ek_trans , double *ek_rot)
206 {
207 
208  int i, c, np;
209  Particle *part;
210  double et = 0.0, er = 0.0;
211 
212  for (c = 0; c < local_cells.n; c++) {
213  np = local_cells.cell[c]->n;
214 
215  part = local_cells.cell[c]->part;
216  for (i = 0; i < np; i++) {
217  #ifdef VIRTUAL_SITES
218  if (ifParticleIsVirtual(&part[i])) continue;
219  #endif
220 
221  /* kinetic energy */
222  et += (SQR(part[i].m.v[0]) + SQR(part[i].m.v[1]) + SQR(part[i].m.v[2]))*PMASS(part[i]);
223 
224  /* rotational energy */
225  #ifdef ROTATION
226  #ifdef ROTATIONAL_INERTIA
227  er += SQR(part[i].m.omega[0])*part[i].p.rinertia[0] +
228  SQR(part[i].m.omega[1])*part[i].p.rinertia[1] +
229  SQR(part[i].m.omega[2])*part[i].p.rinertia[2];
230  #else
231  er += SQR(part[i].m.omega[0]) + SQR(part[i].m.omega[1]) + SQR(part[i].m.omega[2]);
232  #endif
233  #endif
234  }
235  }
236 
237  MPI_Allreduce(MPI_IN_PLACE, &et, 1, MPI_DOUBLE, MPI_SUM, comm_cart);
238  MPI_Allreduce(MPI_IN_PLACE, &er, 1, MPI_DOUBLE, MPI_SUM, comm_cart);
239 
240  et /= (2.0*time_step*time_step);
241  #ifdef ROTATION
242  er /= 2.0;
243  #endif
244 
245  *ek_trans=et;
246  *ek_rot=er;
247 
248 }

333 {
334

346  if (ifParticleIsVirtual(&part[i])) continue;

468 /* monte carlo step of ghmc - evaluation stage */
469 void ghmc_mc()
470 {
471  INTEG_TRACE(fprintf(stderr,"%d: ghmc_mc:\n",this_node));
472 
473  double boltzmann;
474 
475  int ekin_update_flag = 0;
476  hamiltonian_calc(ekin_update_flag);
477 
478  //make MC decision only on the master
479  if (this_node==0) {
480 
481  ghmcdata.att++;
482 
483  //metropolis algorithm
484  boltzmann = ghmcdata.hmlt_new - ghmcdata.hmlt_old;
485  if (boltzmann < 0)
486  boltzmann = 1.0;
487  else if (boltzmann > 30)
488  boltzmann = 0.0;
489  else
490  boltzmann = exp(-beta*boltzmann);
491 
492  //fprintf(stderr,"old hamiltonian : %f, new hamiltonian: % f, boltzmann factor: %f\n",ghmcdata.hmlt_old,ghmcdata.hmlt_new,boltzmann);
493 
494  if ( d_random() < boltzmann) {
495  ghmcdata.acc++;
496  ghmc_mc_res = GHMC_MOVE_ACCEPT;
497  } else {
498  ghmc_mc_res = GHMC_MOVE_REJECT;
499  }
500 
501  }
502 
503  //let all nodes know about the MC decision result
504  mpi_bcast_parameter(FIELD_GHMC_RES);
505 
506  if (ghmc_mc_res == GHMC_MOVE_ACCEPT) {
507  save_last_state();
508  //fprintf(stderr,"%d: mc move accepted\n",this_node);
509  } else {
510  load_last_state();
511  //fprintf(stderr,"%d: mc move rejected\n",this_node);
512 
513  //if the move is rejected we might need to resort particles according to the loaded configurations
514  cells_resort_particles(CELL_GLOBAL_EXCHANGE);
515  invalidate_obs();
516 
517  if (ghmc_mflip == GHMC_MFLIP_ON) {
518  momentum_flip();
519  } else if (ghmc_mflip == GHMC_MFLIP_RAND) {
520  if (d_random() < 0.5) momentum_flip();
521  }
522  }
523 
524  //fprintf(stderr,"%d: temp after mc: %f\n",this_node,calc_local_temp());
525 }
