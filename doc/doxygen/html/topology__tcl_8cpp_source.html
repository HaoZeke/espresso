79 int tclcommand_analyze_set_parse_topo_part_sync(Tcl_Interp *interp) {
80  int i,j,ntopoparts;
81 
82  if (n_molecules <= 0) {
83  Tcl_AppendResult(interp, "Can't sync molecules to particle info: No molecules defined ", (char *)NULL);
84  return TCL_ERROR;
85  }
86  if (n_part <= 0) {
87  Tcl_AppendResult(interp, "Can't sync molecules to particle info: No particles defined ", (char *)NULL);
88  return TCL_ERROR;
89  }
90 
91  /* Check to see that the number of particles in the topology info
92  does not exceed the total number of particles */
93  ntopoparts = 0;
94  for ( i = 0 ; i < n_molecules ; i ++ ) {
95  for ( j = 0 ; j < topology[i].part.n ; j++ ) {
96  ntopoparts += 1;
97  }
98  }
99  if ( ntopoparts > n_part ) {
100  Tcl_AppendResult(interp, "Can't sync molecules to particle info: Topology contains more particles than actually exist ", (char *)NULL);
101  return TCL_ERROR;
102  }
103 
104  if ( !mpi_sync_topo_part_info()) {
105  Tcl_AppendResult(interp, "Error syncronising molecules to particle info", (char *)NULL);
106  return TCL_ERROR;
107  }
108  return TCL_OK;
109 }

112 {
113 
114 #ifdef MOLFORCES
115 #ifdef EXTERNAL_FORCES
116  int trap_flag = 0;
117  int noforce_flag =0;
118  int i;
119 #endif
120 #endif

122  double spring_constant;
123  double drag_constant;
124  int isrelative;
125  DoubleList trap_center;
126  IntList trap_coords;
127  IntList noforce_coords;
128  char usage[] = "trapmol usage: <mol_id> { <xpos> <ypos> <zpos> } <isrelative> <spring_constant> <drag_constant> coords { <trapped_coord> <trapped_coord> <trapped_coord> } noforce_coords {<noforce_coord> <noforce_coord> <noforce_coord>}";
129 
130  init_doublelist(&trap_center);
131  init_intlist(&trap_coords);
132  alloc_intlist(&trap_coords,3);
133  init_intlist(&noforce_coords);
134  alloc_intlist(&noforce_coords,3);
135  /* Unless coords are specified the default is just to trap it completely */
136  trap_coords.e[0] = 1;
137  trap_coords.e[1] = 1;
138  trap_coords.e[2] = 1;
139 
140  Tcl_ResetResult(interp);
141  /* The first argument should be a molecule number */
142  if (!ARG0_IS_I(mol_num)) {
143  Tcl_AppendResult(interp, "first argument should be a molecule id", (char *)NULL);
144  Tcl_AppendResult(interp, usage, (char *)NULL);
145  return TCL_ERROR;
146  } else {
147  /* Sanity checks */
148  if (mol_num > n_molecules) {
149  Tcl_AppendResult(interp, "trapmol: cannot trap mol %d because it does not exist",mol_num , (char *)NULL);
150  return TCL_ERROR;
151  }
152  argc--;
153  argv++;
154  }
155 
156  /* The next argument should be a double list specifying the trap center */
157  if (!ARG0_IS_DOUBLELIST(trap_center)) {
158  Tcl_AppendResult(interp, "second argument should be a double list", (char *)NULL);
159  Tcl_AppendResult(interp, usage , (char *)NULL);
160  return TCL_ERROR;
161  } else {
162  argc -= 1;
163  argv += 1;
164  }
165 
166  /* The next argument should be an integer specifying whether the trap is relative (fraction of box_l) or absolute */
167  if (!ARG0_IS_I(isrelative)) {
168  Tcl_AppendResult(interp, "third argument should be an integer", (char *)NULL);
169  Tcl_AppendResult(interp, usage, (char *)NULL);
170  return TCL_ERROR;
171  } else {
172  argc -= 1;
173  argv += 1;
174  }
175 
176  /* The next argument should be the spring constant for the trap */
177  if (!ARG0_IS_D(spring_constant)) {
178  Tcl_AppendResult(interp, "fourth argument should be a double", (char *)NULL);
179  Tcl_AppendResult(interp, usage, (char *)NULL);
180  return TCL_ERROR;
181  } else {
182  argc -= 1;
183  argv += 1;
184  }
185 
186  /* The next argument should be the drag constant for the trap */
187  if (!ARG0_IS_D(drag_constant)) {
188  Tcl_AppendResult(interp, "fifth argument should be a double", (char *)NULL);
189  Tcl_AppendResult(interp, usage, (char *)NULL);
190  return TCL_ERROR;
191  } else {
192  argc -= 1;
193  argv += 1;
194  }
195 
196  /* Process optional arguments */
197  while ( argc > 0 ) {
198  if ( ARG0_IS_S("coords") ) {
199  if ( !ARG_IS_INTLIST(1,trap_coords) ) {
200  Tcl_AppendResult(interp, "an intlist is required to specify coords", (char *)NULL);
201  Tcl_AppendResult(interp, usage, (char *)NULL);
202  return TCL_ERROR;
203  }
204  argc -= 2;
205  argv += 2;
206  } else if ( ARG0_IS_S("noforce_coords")) {
207  if ( !ARG_IS_INTLIST(1,noforce_coords) ) {
208  Tcl_AppendResult(interp, "an intlist is required to specify coords", (char *)NULL);
209  Tcl_AppendResult(interp, usage, (char *)NULL);
210  return TCL_ERROR;
211  }
212  argc -= 2;
213  argv += 2;
214  } else {
215  Tcl_AppendResult(interp, "an option is not recognised", (char *)NULL);
216  Tcl_AppendResult(interp, usage, (char *)NULL);
217  return TCL_ERROR;
218  }
219  }
220 
221 #ifdef MOLFORCES
222 #ifdef EXTERNAL_FORCES
223  for (i = 0; i < 3; i++) {
224  if (trap_coords.e[i])
225  trap_flag |= COORD_FIXED(i);
226 
227  if (noforce_coords.e[i])
228  noforce_flag |= COORD_FIXED(i);
229  }
230  if (set_molecule_trap(mol_num, trap_flag,&trap_center,spring_constant, drag_constant, noforce_flag, isrelative) == TCL_ERROR) {
231  Tcl_AppendResult(interp, "set topology first", (char *)NULL);
232  return TCL_ERROR;
233  }
234 #else
235  Tcl_AppendResult(interp, "Error: EXTERNAL_FORCES not defined ", (char *)NULL);
236  return TCL_ERROR;
237 #endif
238 #endif
239 
240  realloc_doublelist(&trap_center,0);
241  realloc_intlist(&trap_coords,0);
242  realloc_intlist(&noforce_coords,0);
243  return tclcommand_analyze_set_parse_topo_part_sync(interp);
244 
245 }
