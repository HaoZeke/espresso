207 int tclcommand_metadynamics_print_stat(Tcl_Interp *interp, int argc, char **argv)
208 {
209  int j;
210  char buffer[TCL_DOUBLE_SPACE];
211 
212  /* In case nothing has been initialized yet */
213  if (meta_acc_fprofile == NULL) return (TCL_OK);
214 
215  argc -= 1; argv += 1;
216 
217  if ( ARG1_IS_S("current_coord") ) {
218  /* Current value of the reaction coordinate */
219  Tcl_PrintDouble(interp, meta_val_xi, buffer);
220  Tcl_AppendResult(interp,"",buffer, (char *)NULL);
221  } else if ( ARG1_IS_S("coord_values") ) {
222  /* Possible values of the reaction coordinate */
223  for (j = 0; j < meta_xi_num_bins; ++j) {
224  Tcl_PrintDouble(interp, meta_xi_min+j*meta_xi_step, buffer);
225  Tcl_AppendResult(interp,buffer," ", (char *)NULL);
226  }
227  } else if ( ARG1_IS_S("profile") ) {
228  /* Values of the free energy profile */
229  for (j = 0; j < meta_xi_num_bins; ++j) {
230  Tcl_PrintDouble(interp, meta_acc_fprofile[j], buffer);
231  Tcl_AppendResult(interp,buffer," ", (char *)NULL);
232  }
233  } else if ( ARG1_IS_S("force") ) {
234  /* Values of the biased force */
235  for (j = 0; j < meta_xi_num_bins; ++j) {
236  Tcl_PrintDouble(interp, -1.*meta_acc_force[j], buffer);
237  Tcl_AppendResult(interp,buffer," ", (char *)NULL);
238  }
239  } else {
240  Tcl_AppendResult(interp, "Unknown option for 'metadynamics print_stat'", (char *)NULL);
241  return (TCL_ERROR);
242  }
243  return (TCL_OK);
244 }

246 int tclcommand_metadynamics_parse_load_stat(Tcl_Interp *interp, int argc, char **argv){
247  /* Parse free energy profile and biased force that were provided from an
248  * earlier simulation. Allows one to restart from a loaded state, and can
249  * even be used to allow multiple walkers communicating their data through TCL. */
250 
251  if(meta_switch == META_OFF) {
252  Tcl_AppendResult(interp, "Metadynamics hasn't been initialized yet", (char *)NULL);
253  return (TCL_ERROR);
254  }
255 
256  argc -= 1; argv += 1;
257 
258  // There should be
259  if (argc != 3) {
260  Tcl_AppendResult(interp, "Incorrect number of arguments: 'metadynamics load_stat <profile_list> <force_list>'", (char *)NULL);
261  return (TCL_ERROR);
262  }
263 
264  // load free energy profile
265  int i, tmp_argc, parse_error = 0, empty_line=0;
266  char **tmp_argv;
267  DoubleList profile, force;
268 
269  init_doublelist(&profile);
270  Tcl_ResetResult(interp);
271  Tcl_SplitList(interp, argv[1], &tmp_argc, &tmp_argv);
272  realloc_doublelist(&profile, profile.n = tmp_argc);
273  //printf("profile.n %d, meta_xi_num_bins %d\n",profile.n,meta_xi_num_bins);
274  /* Now check that the number of items parsed is equal to the number of bins */
275  /* If there's one extra line, assume it's an empty line */
276  if (profile.n == meta_xi_num_bins+1)
277  empty_line = 1;
278  else if (profile.n != meta_xi_num_bins) {
279  Tcl_AppendResult(interp, "Size of profile list loaded is different than expected from number of bins", (char *)NULL);
280  return (TCL_ERROR);
281  }
282  /* call meta_init() in case it has been loaded yet */
283  meta_init();
284 
285  for(i = 0 ; i < tmp_argc-empty_line; i++) {
286  int tmp_argc2;
287  char **tmp_argv2;
288  Tcl_SplitList(interp, tmp_argv[i], &tmp_argc2, &tmp_argv2);
289  if (tmp_argc2 != 1) {
290  Tcl_AppendResult(interp, "data set has to be a list of doubles", (char *) NULL);
291  parse_error = 1; break;
292  }
293  if (Tcl_GetDouble(interp, tmp_argv2[0], &(profile.e[i])) == TCL_ERROR) { parse_error = 1; break; }
294  /* Load data into meta_acc_fprofile */
295  meta_acc_fprofile[i] = profile.e[i];
296 
297  Tcl_Free((char *)tmp_argv2);
298  }
299  Tcl_Free((char *)tmp_argv);
300  if (parse_error) return TCL_ERROR;
301 
302 
303  // load force
304  argc -= 1; argv += 1;
305  init_doublelist(&force);
306  Tcl_ResetResult(interp);
307  Tcl_SplitList(interp, argv[1], &tmp_argc, &tmp_argv);
308  realloc_doublelist(&force, force.n = tmp_argc);
309  /* Now check that the number of items parsed is equal to the number of bins */
310  if (profile.n == meta_xi_num_bins+1)
311  empty_line = 1;
312  else if (profile.n != meta_xi_num_bins) {
313  Tcl_AppendResult(interp, "Size of force list loaded is different than expected from number of bins", (char *)NULL);
314  return (TCL_ERROR);
315  }
316  for(i = 0 ; i < tmp_argc-empty_line; i++) {
317  int tmp_argc2;
318  char **tmp_argv2;
319  Tcl_SplitList(interp, tmp_argv[i], &tmp_argc2, &tmp_argv2);
320  if (tmp_argc2 != 1) {
321  Tcl_AppendResult(interp, "data set has to be a list of doubles", (char *) NULL);
322  parse_error = 1; break;
323  }
324  if (Tcl_GetDouble(interp, tmp_argv2[0], &(force.e[i])) == TCL_ERROR) { parse_error = 1; break; }
325  /* Load data into meta_acc_fprofile */
326  meta_acc_force[i] = -1.*force.e[i];
327 
328  Tcl_Free((char *)tmp_argv2);
329  }
330  Tcl_Free((char *)tmp_argv);
331  if (parse_error) return TCL_ERROR;
332 
333  return (TCL_OK);
334 }

336 int tclcommand_metadynamics(ClientData data, Tcl_Interp *interp, int argc, char **argv)
337 {
338  int err = TCL_OK;
339 
340  /* print metadynamics status */
341  if(argc == 1) return tclcommand_metadynamics_print_status(interp);
342 
343  if ( ARG1_IS_S("set") ) {
344  argc--;
345  argv++;
346 
347  if (argc == 1) {
348  Tcl_AppendResult(interp, "wrong # args: \n", (char *)NULL);
349  return tclcommand_metadynamics_print_usage(interp, argc, argv);
350  }
351  }
352  if ( ARG1_IS_S("off") )
353  err = tclcommand_metadynamics_parse_off(interp, argc, argv);
354  else if ( ARG1_IS_S("distance"))
355  err = tclcommand_metadynamics_parse_distance(interp, argc, argv);
356  else if ( ARG1_IS_S("relative_z"))
357  err = tclcommand_metadynamics_parse_relative_z(interp, argc, argv);
358  else if ( ARG1_IS_S("print_stat"))
359  err = tclcommand_metadynamics_print_stat(interp, argc, argv);
360  else if ( ARG1_IS_S("load_stat"))
361  err = tclcommand_metadynamics_parse_load_stat(interp, argc, argv);
362  else {
363  Tcl_AppendResult(interp, "Unknown metadynamics command ", argv[1], "\n", (char *)NULL);
364  return tclcommand_metadynamics_print_usage(interp, argc, argv);
365  }
366  return gather_runtime_errors(interp, err);
367 }
