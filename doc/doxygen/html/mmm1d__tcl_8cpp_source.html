55 int tclcommand_inter_coulomb_parse_mmm1d(Tcl_Interp *interp, int argc, char **argv)
56 {
57  double switch_rad, maxPWerror;
58 
59  if (argc < 2) {
60  Tcl_AppendResult(interp, "wrong # arguments: inter coulomb mmm1d <switch radius> "
61  "{<bessel cutoff>} <maximal error for near formula> | tune <maximal pairwise error>", (char *) NULL);
62  return TCL_ERROR;
63  }
64 
65  if (ARG0_IS_S("tune")) {
66  /* autodetermine bessel cutoff AND switching radius */
67  if (! ARG_IS_D(1, maxPWerror))
68  return TCL_ERROR;
69  switch_rad = -1;
70  }
71  else {
72  if (argc == 2) {
73  /* autodetermine bessel cutoff */
74  if ((! ARG_IS_D(0, switch_rad)) ||
75  (! ARG_IS_D(1, maxPWerror)))
76  return TCL_ERROR;
77  }
78  else {
79  Tcl_AppendResult(interp, "wrong # arguments: inter coulomb mmm1d <switch radius> "
80  "<maximal error for near formula> | tune <maximal pairwise error>", (char *) NULL);
81  return TCL_ERROR;
82  }
83 
84  if (switch_rad <= 0 || switch_rad > box_l[2]) {
85  Tcl_AppendResult(interp, "switching radius is not between 0 and box_l[2]", (char *)NULL);
86  return TCL_ERROR;
87  }
88  }
89 
90  MMM1D_set_params(switch_rad, maxPWerror);
91 
92  char *log = NULL;
93  int result = mmm1d_tune(&log) == ES_OK ? TCL_OK : TCL_ERROR;
94 
95  Tcl_AppendResult(interp, log, NULL);
96  if (log)
97  free(log);
98 
99  return gather_runtime_errors(interp, result);
100 }
