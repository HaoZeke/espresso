32 int Lattice::init(double *agrid, double* offset, int halo_size, size_t dim) {
33  this->dim=dim;
34 
35  /* determine the number of local lattice nodes */
36  for (int d=0; d<3; d++) {
37  this->agrid[d] = agrid[d];
38  this->global_grid[d] = (int)dround(box_l[d]/agrid[d]);
39  this->offset[d]=offset[d];
40  this->local_index_offset[d]=(int) ceil((my_left[d]-this->offset[d])/this->agrid[d]);
41  this->local_offset[d] = this->offset[d] +
42  this->local_index_offset[d]*this->agrid[d];
43  this->grid[d] = (int) ceil ( ( my_right[d] - this->local_offset[d]-ROUND_ERROR_PREC )
44  / this->agrid[d]);
45  }
46 
47  // sanity checks
48  for (int dir=0;dir<3;dir++) {
49  // check if local_box_l is compatible with lattice spacing
50  if (fabs(local_box_l[dir]-this->grid[dir]*agrid[dir]) > ROUND_ERROR_PREC*box_l[dir]) {
51  ostringstream msg;
52  msg << "Lattice spacing agrid["<< dir << "]=" << agrid[dir] \
53  << " is incompatible with local_box_l["<< dir << "]=" << local_box_l[dir]\
54  << " ( box_l["<< dir << "]=" << box_l[dir] \
55  << " node_grid["<< dir << "]=" << node_grid[dir] <<" )";
56  runtimeError(msg);
57  }
58  }
59 
60  this->element_size = this->dim*sizeof(double);
61 
62  LATTICE_TRACE(fprintf(stderr,"%d: box_l (%.3f,%.3f,%.3f) grid (%d,%d,%d) node_neighbors (%d,%d,%d,%d,%d,%d)\n",this_node,local_box_l[0],local_box_l[1],local_box_l[2],this->grid[0],this->grid[1],this->grid[2],node_neighbors[0],node_neighbors[1],node_neighbors[2],node_neighbors[3],node_neighbors[4],node_neighbors[5]));
63 
64  this->halo_size = halo_size;
65  /* determine the number of total nodes including halo */
66  this->halo_grid[0] = this->grid[0] + 2*halo_size ;
67  this->halo_grid[1] = this->grid[1] + 2*halo_size ;
68  this->halo_grid[2] = this->grid[2] + 2*halo_size ;
69 
70  this->grid_volume = this->grid[0]*this->grid[1]*this->grid[2] ;
71  this->halo_grid_volume = this->halo_grid[0]*this->halo_grid[1]*this->halo_grid[2] ;
72  this->halo_grid_surface = this->halo_grid_volume - this->grid_volume ;
73  this->halo_offset = get_linear_index(halo_size,halo_size,halo_size,this->halo_grid) ;
74 
75  this->interpolation_type = INTERPOLATION_LINEAR;
76 
77  allocate_memory();
78  return ES_OK;
79

99 void Lattice::interpolate_linear(double* pos, double* value) {
100  int left_halo_index[3];
101  double d[3];
102  if (this->halo_size <= 0) {
103  ostringstream msg;
104  msg <<"Error in interpolate_linear: halo size is 0";
105  runtimeError(msg);
106  return;
107  }
108  for (int dim = 0; dim<3; dim++) {
109  left_halo_index[dim]=(int) floor((pos[dim]-this->local_offset[dim])/this->agrid[dim]) + this->halo_size;
110  d[dim]=((pos[dim]-this->local_offset[dim])/this->agrid[dim] - floor((pos[dim]-this->local_offset[dim])/this->agrid[dim]));
111  if (left_halo_index[dim] < 0 || left_halo_index[dim] >= this->halo_grid[dim]) {
112  ostringstream msg;
113  msg <<"Error in interpolate_linear: Particle out of range";
114  runtimeError(msg);
115  return;
116  }
117  }
118  double w[8];
119  index_t index[8];
120  w[0] = (1-d[0])*(1-d[1])*(1-d[2]);
121  index[0]=get_linear_index( left_halo_index[0], left_halo_index[1], left_halo_index[2], this->halo_grid);
122  w[1] = ( +d[0])*(1-d[1])*(1-d[2]);
123  index[1]=get_linear_index( left_halo_index[0]+1, left_halo_index[1], left_halo_index[2], this->halo_grid);
124  w[2] = (1-d[0])*( +d[1])*(1-d[2]);
125  index[2]=get_linear_index( left_halo_index[0], left_halo_index[1]+1, left_halo_index[2], this->halo_grid);
126  w[3] = ( +d[0])*( +d[1])*(1-d[2]);
127  index[3]=get_linear_index( left_halo_index[0]+1, left_halo_index[1]+1, left_halo_index[2], this->halo_grid);
128 
129  w[4] = (1-d[0])*(1-d[1])*( +d[2]);
130  index[4]=get_linear_index( left_halo_index[0], left_halo_index[1], left_halo_index[2]+1, this->halo_grid);
131  w[5] = ( +d[0])*(1-d[1])*( +d[2]);
132  index[5]=get_linear_index( left_halo_index[0]+1, left_halo_index[1], left_halo_index[2]+1, this->halo_grid);
133  w[6] = (1-d[0])*( +d[1])*( +d[2]);
134  index[6]=get_linear_index( left_halo_index[0], left_halo_index[1]+1, left_halo_index[2]+1, this->halo_grid);
135  w[7] = ( +d[0])*( +d[1])*( +d[2]);
136  index[7]=get_linear_index( left_halo_index[0]+1, left_halo_index[1]+1, left_halo_index[2]+1, this->halo_grid);
137 
138  for (unsigned int i = 0; i<this->dim; i++) {
139  value[i] = 0;
140  }
141 
142  double* local_value;
143  for (unsigned int i=0; i<8; i++) {
144  get_data_for_linear_index(index[i], (void**) &local_value);
145  for (unsigned int j = 0; j<this->dim; j++) {
146  value[j]+=w[i]*local_value[j];
147  }
148  }

344 void Lattice::map_position_to_lattice(const double pos[3], index_t node_index[8], double delta[6]) {
345 
346  int dir,ind[3] ;
347  double lpos, rel;
348 
349  /* determine the elementary lattice cell containing the particle
350  and the relative position of the particle in this cell */
351  for (dir=0;dir<3;dir++) {
352  lpos = pos[dir] - my_left[dir];
353  rel = lpos/this->agrid[dir] + 0.5; // +1 for halo offset
354  ind[dir] = (int)floor(rel);
355 
356  /* surrounding elementary cell is not completely inside this box,
357  adjust if this is due to round off errors */
358  if (ind[dir] < 0) {
359  if (abs(rel) < ROUND_ERROR_PREC) {
360  ind[dir] = 0; // TODO
361  } else {
362  fprintf(stderr,"%d: map_position_to_lattice: position (%f,%f,%f) not inside a local plaquette in dir %d ind[dir]=%d rel=%f lpos=%f.\n",this_node,pos[0],pos[1],pos[2],dir,ind[dir],rel,lpos);
363  }
364  }
365  else if (ind[dir] > this->grid[dir]) {
366  if (lpos - local_box_l[dir] < ROUND_ERROR_PREC*local_box_l[dir])
367  ind[dir] = this->grid[dir];
368  else
369  fprintf(stderr,"%d: map_position_to_lattice: position (%f,%f,%f) not inside a local plaquette in dir %d ind[dir]=%d rel=%f lpos=%f.\n",this_node,pos[0],pos[1],pos[2],dir,ind[dir],rel,lpos);
370  }
371 
372  delta[3+dir] = rel - ind[dir]; // delta_x/a
373  delta[dir] = 1.0 - delta[3+dir];
374  }
375 
376  node_index[0] = get_linear_index(ind[0],ind[1],ind[2],this->halo_grid);
377  node_index[1] = node_index[0] + 1;
378  node_index[2] = node_index[0] + this->halo_grid[0];
379  node_index[3] = node_index[0] + this->halo_grid[0] + 1;
380  node_index[4] = node_index[0] + this->halo_grid[0]*this->halo_grid[1];
381  node_index[5] = node_index[4] + 1;
382  node_index[6] = node_index[4] + this->halo_grid[0];
383  node_index[7] = node_index[4] + this->halo_grid[0] + 1;
