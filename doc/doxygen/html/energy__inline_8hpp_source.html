186  double dist, double dist2)
187 {
188  IA_parameters *ia_params = get_ia_param(p1->p.type,p2->p.type);
189

264 inline void add_bonded_energy(Particle *p1)
265 {
266  Particle *p2, *p3 = NULL, *p4 = NULL;
267 #ifdef TWIST_STACK
268  Particle *p5 = NULL,*p6 = NULL,*p7 = NULL,*p8 = NULL;
269 #endif
270  Bonded_ia_parameters *iaparams;
271  int i, type_num, type, n_partners, bond_broken;
272  double ret=0, dx[3] = {0, 0, 0};
273 
274  i = 0;
275  while(i<p1->bl.n) {
276  type_num = p1->bl.e[i++];
277  iaparams = &bonded_ia_params[type_num];
278  type = iaparams->type;
279  n_partners = iaparams->num;
280 
281  /* fetch particle 2, which is always needed */
282  p2 = local_particles[p1->bl.e[i++]];
283  if (!p2) {
284  ostringstream msg;
285  msg <<"bond broken between particles " << p1->p.identity << " and " << p1->bl.e[i-1]
286  <<" (particles not stored on the same node)";
287  runtimeError(msg);
288  return;
289  }
290 
291  /* fetch particle 3 eventually */
292  if (n_partners >= 2) {
293  p3 = local_particles[p1->bl.e[i++]];
294  if (!p3) {
295  ostringstream msg;
296  msg <<"bond broken between particles " << p1->p.identity <<", "<< p1->bl.e[i-2] << " and " << p1->bl.e[i-1]
297  <<" (particles not stored on the same node)";
298  runtimeError(msg);
299  return;
300  }
301  }
302 
303  /* fetch particle 4 eventually */
304  if (n_partners >= 3) {
305  p4 = local_particles[p1->bl.e[i++]];
306  if (!p4) {
307  ostringstream msg;
308  msg <<"bond broken between particles " << p1->p.identity <<", "<< p1->bl.e[i-3] <<", "<< p1->bl.e[i-2]
309  << " and " << p1->bl.e[i-1] << " (particles not stored on the same node)";
310  runtimeError(msg);
311  return;
312  }
313  }
314 #ifdef TWIST_STACK
315  if(n_partners >= 7) {
316  p5 = local_particles[p1->bl.e[i++]];
317  p6 = local_particles[p1->bl.e[i++]];
318  p7 = local_particles[p1->bl.e[i++]];
319  p8 = local_particles[p1->bl.e[i++]];
320 
321  if(!p4 || !p5 || !p6 || !p7 || !p8) {
322  ostringstream msg;
323  msg << "bond broken between particles" <<
324  p1->p.identity << ", " << p1->bl.e[i-7] << ", " << p1->bl.e[i-6] << ", " << p1->bl.e[i-5] << ", " << p1->bl.e[i-4] << ", " << p1->bl.e[i-3] << ", " << p1->bl.e[i-2] << ", " << p1->bl.e[i-1] << " (particles not stored on the same node)";
325  return;
326  }
327  }
328 #endif
329  /* similar to the force, we prepare the center-center vector */
330  if (n_partners == 1)
331  get_mi_vector(dx, p1->r.p, p2->r.p);
332 
333  switch(type) {
334  case BONDED_IA_FENE:
335  bond_broken = fene_pair_energy(p1, p2, iaparams, dx, &ret);
336  break;
337 #ifdef ROTATION
338  case BONDED_IA_HARMONIC_DUMBBELL:
339  bond_broken = harmonic_dumbbell_pair_energy(p1, p2, iaparams, dx, &ret);
340  break;
341 #endif
342  case BONDED_IA_HARMONIC:
343  bond_broken = harmonic_pair_energy(p1, p2, iaparams, dx, &ret);
344  break;
345  case BONDED_IA_QUARTIC:
346  bond_broken = quartic_pair_energy(p1, p2, iaparams, dx, &ret);
347  break;
348 #ifdef ELECTROSTATICS
349  case BONDED_IA_BONDED_COULOMB:
350  bond_broken = bonded_coulomb_pair_energy(p1, p2, iaparams, dx, &ret);
351  break;
352 #endif
353 #ifdef LENNARD_JONES
354  case BONDED_IA_SUBT_LJ:
355  bond_broken = subt_lj_pair_energy(p1, p2, iaparams, dx, &ret);
356  break;
357 #endif
358 #ifdef BOND_ANGLE_OLD
359  /* the first case is not needed and should not be called */
360  case BONDED_IA_ANGLE_OLD:
361  bond_broken = angle_energy(p1, p2, p3, iaparams, &ret);
362  break;
363 #endif
364 #ifdef TWIST_STACK
365  case BONDED_IA_CG_DNA_STACKING:
366  bond_broken = calc_twist_stack_energy(p1, p2, p3, p4, p5, p6, p7, p8, iaparams, &ret);
367  break;
368 #endif
369 #ifdef HYDROGEN_BOND
370  case BONDED_IA_CG_DNA_BASEPAIR:
371  bond_broken = calc_hydrogen_bond_energy(p1, p2, p3, p4, iaparams, &ret);
372  break;
373 #endif
374 #ifdef BOND_ANGLE
375  case BONDED_IA_ANGLE_HARMONIC:
376  bond_broken = angle_harmonic_energy(p1, p2, p3, iaparams, &ret);
377  break;
378  case BONDED_IA_ANGLE_COSINE:
379  bond_broken = angle_cosine_energy(p1, p2, p3, iaparams, &ret);
380  break;
381  case BONDED_IA_ANGLE_COSSQUARE:
382  bond_broken = angle_cossquare_energy(p1, p2, p3, iaparams, &ret);
383  break;
384 #endif
385 #ifdef BOND_ANGLEDIST
386  case BONDED_IA_ANGLEDIST:
387  bond_broken = angledist_energy(p1, p2, p3, iaparams, &ret);
388  break;
389 #endif
390 #ifdef BOND_ENDANGLEDIST
391  case BONDED_IA_ENDANGLEDIST:
392  bond_broken = endangledist_pair_energy(p1, p2, iaparams, dx, &ret);
393  break;
394 #endif
395  case BONDED_IA_DIHEDRAL:
396  bond_broken = dihedral_energy(p2, p1, p3, p4, iaparams, &ret);
397  break;
398 #ifdef BOND_CONSTRAINT
399  case BONDED_IA_RIGID_BOND:
400  bond_broken = 0;
401  ret = 0;
402  break;
403 #endif
404 #ifdef TABULATED
405  case BONDED_IA_TABULATED:
406  switch(iaparams->p.tab.type) {
407  case TAB_BOND_LENGTH:
408  bond_broken = tab_bond_energy(p1, p2, iaparams, dx, &ret);
409  break;
410  case TAB_BOND_ANGLE:
411  bond_broken = tab_angle_energy(p1, p2, p3, iaparams, &ret);
412  break;
413  case TAB_BOND_DIHEDRAL:
414  bond_broken = tab_dihedral_energy(p2, p1, p3, p4, iaparams, &ret);
415  break;
416  default :
417  ostringstream msg;
418  msg << "add_bonded_energy: tabulated bond type of atom " << p1->p.identity << " unknown\n";
419  runtimeError(msg);
420  return;
421  }
422  break;
423 #endif
424 #ifdef OVERLAPPED
425  case BONDED_IA_OVERLAPPED:
426  switch(iaparams->p.overlap.type) {
427  case OVERLAP_BOND_LENGTH:
428  bond_broken = overlap_bond_energy(p1, p2, iaparams, dx, &ret);
429  break;
430  case OVERLAP_BOND_ANGLE:
431  bond_broken = overlap_angle_energy(p1, p2, p3, iaparams, &ret);
432  break;
433  case OVERLAP_BOND_DIHEDRAL:
434  bond_broken = overlap_dihedral_energy(p2, p1, p3, p4, iaparams, &ret);
435  break;
436  default :
437  ostringstream msg;
438  msg << "add_bonded_energy: overlapped bond type of atom " << p1->p.identity << " unknown\n";
439  runtimeError(msg);
440  return;
441  }
442  break;
443 #endif
444 #ifdef BOND_VIRTUAL
445  case BONDED_IA_VIRTUAL_BOND:
446  bond_broken = 0;
447  ret = 0;
448  break;
449 #endif
450  default :
451  ostringstream msg;
452  msg <<"add_bonded_energy: bond type of atom "<< p1->p.identity << " unknown\n";
453  runtimeError(msg);
454  return;
455  }
456 
457  if (bond_broken) {
458  switch (n_partners) {
459  case 1: {
460  ostringstream msg;
461  msg <<"bond broken between particles "<< p1->p.identity << " and " << p2->p.identity;
462  runtimeError(msg);
463  break;
464  }
465  case 2: {
466  ostringstream msg;
467  msg <<"bond broken between particles "<< p1->p.identity << ", " << p2->p.identity << " and " << p3->p.identity;

472  ostringstream msg;

476  break;
477  }
478  }
479  // bond broken, don't add whatever we find in the energy
480  continue;
481  }
482 
483  *obsstat_bonded(&energy, type_num) += ret;
484  }
