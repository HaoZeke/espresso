23 /** \file maggs.cpp
24  * Maxwell Equations Molecular Dynamics (MEMD) method for electrostatic
25  * interactions.
26  *
27  * We use a local update scheme to propagate artificial B-fields on a
28  * lattice in the system. In principal, the algorithm simulates full
29  * electrodynamics, but with a tunable speed of light.
30  *
31  * The method is very usable for large particle numbers or highly
32  * parallel architectures, since it is local and scales linearly.
33  * It is not suited for high-precision calculation of forces, since
34  * the simple interpolation scheme produces errors in the order of
35  * 10^-5 in the force.
36  *
37  * The chosen mesh should roughly be of the size of the particles.
38  *
39  * Further reading on the algorithm:
40  * <ul>
41  * <li> I. Pasichnyk and B. Dunweg, Coulomb interaction via local dynamics: a molecular-dynamics algorithm. J. Phys: Condens. Matter, 16 ,p. 1399-4020, (2004).
42  * </ul>
43  *
44  */

56 int tclcommand_localeps(Tcl_Interp* interp, int argc, char** argv)
57 {
58  int mesh = maggs_get_mesh_1D();
59  int node_x, node_y, node_z, direction;
60  double relative_epsilon;
61 
62  /* number of arguments has to be 8 */
63  if(argc != 9) {
64  Tcl_AppendResult(interp, "Wrong number of paramters. Usage: \n", (char *) NULL);
65  Tcl_AppendResult(interp, "inter coulomb <bjerrum> memd localeps node <x> <y> <z> dir <X/Y/Z> eps <epsilon>", (char *) NULL);
66  return TCL_ERROR;
67  }
68 
69  /* first argument should be "node" */
70  if(! ARG_IS_S(1, "node")) return TCL_ERROR;
71 
72  /* arguments 2-4 should be integers */
73  if(! ARG_IS_I(2, node_x)) {
74  Tcl_AppendResult(interp, "integer expected", (char *) NULL);
75  return TCL_ERROR; }
76  if(! ARG_IS_I(3, node_y)) {
77  Tcl_AppendResult(interp, "integer expected", (char *) NULL);
78  return TCL_ERROR; }
79  if(! ARG_IS_I(4, node_z)) {
80  Tcl_AppendResult(interp, "integer expected", (char *) NULL);
81  return TCL_ERROR; }
82  /* check if mesh position is in range */
83  if ( (node_x < 0) || (node_y < 0) || (node_z < 0) || (node_x > mesh) || (node_y > mesh) || (node_z > mesh) ) {
84  char buffer[TCL_INTEGER_SPACE];
85  sprintf(buffer, "%d", mesh);
86  Tcl_AppendResult(interp, "epsilon position out of mesh range. Mesh in each dimension is ", buffer, ".", (char *) NULL);
87  return TCL_ERROR;
88  }
89 
90  /* parse fifth and sixth argument (e.g. dir X) */
91  if(! ARG_IS_S(5, "dir")) return TCL_ERROR;
92  if ( (! ARG_IS_S(6, "X")) && (! ARG_IS_S(6, "Y")) && (! ARG_IS_S(6, "Z")) ) {
93  Tcl_AppendResult(interp, "Parameter dir should be 'X', 'Y' or 'Z'.", (char *) NULL);
94  return TCL_ERROR; }
95  if(ARG_IS_S(6, "X")) direction = 0;
96  if(ARG_IS_S(6, "Y")) direction = 1;
97  if(ARG_IS_S(6, "Z")) direction = 2;
98 
99  /* parse seventh and eight argument (e.g. eps 0.5) */
100  if(! ARG_IS_S(7, "eps")) return TCL_ERROR;
101  if ( (! ARG_IS_D(8, relative_epsilon)) || (relative_epsilon < 0.0) ) {
102  Tcl_AppendResult(interp, "eps expects a positive double", (char *) NULL);
103  return TCL_ERROR; }
104 
105  double eps_before = maggs_set_permittivity(node_x, node_y, node_z, direction, relative_epsilon);
106 
107  if (eps_before == 1.0) return TCL_OK;
108  else return TCL_OK;
109 }

111 /** parse TCL command.
112  number of parameters is checked and maggs_set_parameters function is called.
113  @return zero if successful
114  @param interp TCL interpreter handle
115  @param argc number of arguments given
116  @param argv array of arguments given
117 */
118 int tclcommand_inter_coulomb_parse_maggs(Tcl_Interp * interp, int argc, char ** argv)
119 {
120  int mesh;
121  double f_mass;
122  double epsilon = 1.0;
123  int finite_epsilon_flag = 1;
124 
125  /* if the command is localeps, call function */
126  if ( (argc > 0) && (ARG_IS_S(0, "localeps")) )
127  return tclcommand_localeps(interp, argc, argv);
128 
129  if(argc < 2) {
130  Tcl_AppendResult(interp, "Not enough parameters: inter coulomb <bjerrum> memd <f_mass> <mesh>", (char *) NULL);
131  return TCL_ERROR;
132  }
133 
134  if(! ARG_IS_D(0, f_mass))
135  return TCL_ERROR;
136 
137  if(! ARG_IS_I(1, mesh)) {
138  Tcl_AppendResult(interp, "integer expected", (char *) NULL);
139  return TCL_ERROR;
140  }
141 
142  if(argc > 4) {
143  Tcl_AppendResult(interp, "Too many parameters: inter coulomb memd <f_mass> <mesh> [epsilon <eps>]", (char *) NULL);
144  return TCL_ERROR;
145  }
146  if(argc == 3) {
147  Tcl_AppendResult(interp, "Usage: inter coulomb memd <f_mass> <mesh> [epsilon <eps>]", (char *) NULL);
148  return TCL_ERROR;
149  }
150  if(argc == 4) {
151  if (ARG_IS_S(2, "epsilon")) {
152  if(! (ARG_IS_D(3, epsilon) && epsilon > 0.0)) {
153  Tcl_AppendResult(interp, "epsilon expects a positive double",
154  (char *) NULL);
155  return TCL_ERROR;
156  }
157  }
158  } else finite_epsilon_flag=1;
159 
160  coulomb.method = COULOMB_MAGGS;
161 
162  int res = maggs_set_parameters(coulomb.bjerrum, f_mass, mesh,
163  finite_epsilon_flag, epsilon);
164  switch (res) {
165  case -1:
166  Tcl_AppendResult(interp, "mass of the field is negative", (char *)NULL);
167  return TCL_ERROR;
168  case -2:
169  Tcl_AppendResult(interp, "mesh must be positive", (char *) NULL);
170  return TCL_ERROR;
171  case ES_OK:
172  return TCL_OK;
173  }
174  Tcl_AppendResult(interp, "unknown error", (char *) NULL);
175  return TCL_ERROR;
176 }
