137 void CalcVolumes()
138 {
139 
140  // Partial volumes for each soft particle, to be summed up
141  double tempVol[MaxNumIBM] = {0};
142 
143  // Loop over all particles on local node
144  for (int c = 0; c < local_cells.n; c++)
145  {
146  const Cell *const cell = local_cells.cell[c];
147 
148  for (int i = 0; i < cell->n; i++)
149  {
150  Particle &p1 = cell->part[i];
151 
152  // Check if particle has a BONDED_IA_IBM_TRIEL and a BONDED_IA_IBM_VOLUME_CONSERVATION
153  // Basically this loops over all triangles, not all particles
154  // First round to check for volume conservation and virtual
155  // Loop over all bonds of this particle
156  // Actually j loops over the bond-list, i.e. the bond partners (see particle_data.hpp)
157  int softID = -1;
158  int j = 0;
159  while ( j < p1.bl.n )
160  {
161  const int type_num = p1.bl.e[j];
162  const Bonded_ia_parameters &iaparams = bonded_ia_params[type_num];
163  const int type = iaparams.type;
164  if ( type == BONDED_IA_IBM_VOLUME_CONSERVATION )
165  {
166  if ( p1.p.isVirtual) softID = iaparams.p.ibmVolConsParameters.softID;
167  else { printf("Error. Encountered non-virtual particle with VOLUME_CONSERVATION_IBM\n"); exit(1); }
168  }
169  // Iterate, increase by the number of partners of this bond + 1 for bond type
170  j += iaparams.num+1;
171  }
172 
173 
174  // Second round for triel
175  if ( softID > -1 )
176  {
177  j = 0;
178  while ( j < p1.bl.n)
179  {
180  const int type_num = p1.bl.e[j];
181  const Bonded_ia_parameters &iaparams = bonded_ia_params[type_num];
182  const int type = iaparams.type;
183 
184  if ( type == BONDED_IA_IBM_TRIEL )
185  {
186  // Our particle is the leading particle of a triel
187  // Get second and third particle of the triangle
188  Particle *p2 = local_particles[p1.bl.e[j+1]];
189  if (!p2)
190  {
191  ostringstream msg;
192  msg << "{IBM_CalcVolumes: 078 bond broken between particles "
193  << p1.p.identity << " and " << p1.bl.e[j+1] << " (particles not stored on the same node)} ";
194  runtimeError(msg);
195  return;
196  }
197  Particle *p3 = local_particles[p1.bl.e[j+2]];
198  if (!p3)
199  {
200  ostringstream msg;
201  msg << "{IBM_CalcVolumes: 078 bond broken between particles "
202  << p1.p.identity << " and " << p1.bl.e[j+2] << " (particles not stored on the same node)} ";
203  runtimeError(msg);
204  return;
205  }
206 
207  // Unfold position of first node
208  // this is to get a continuous trajectory with no jumps when box boundaries are crossed
209  double x1[3] = { p1.r.p[0], p1.r.p[1], p1.r.p[2] };
210  int img[3] = { p1.l.i[0], p1.l.i[1], p1.l.i[2] };
211  unfold_position(x1,img);
212 
213  // Unfolding seems to work only for the first particle of a triel
214  // so get the others from relative vectors considering PBC
215  double a12[3];
216  get_mi_vector(a12, p2->r.p, x1);
217  double a13[3];
218  get_mi_vector(a13, p3->r.p, x1);
219 
220  double x2[3];
221  double x3[3];
222 
223  for (int i=0; i < 3; i++)
224  {
225  x2[i] = x1[i] + a12[i];
226  x3[i] = x1[i] + a13[i];
227  }
228 
229  // Volume of this tetrahedron
230  // See Cha Zhang et.al. 2001, doi:10.1109/ICIP.2001.958278
231  // http://research.microsoft.com/en-us/um/people/chazhang/publications/icip01_ChaZhang.pdf
232  // The volume can be negative, but it is not necessarily the "signed volume" in the above paper
233  // (the sign of the real "signed volume" must be calculated using the normal vector; the result of the calculation here
234  // is simply a term in the sum required to calculate the volume of a particle). Again, see the paper.
235  // This should be equivalent to the formulation using vector identities in Krger thesis
236 
237  const double v321 = x3[0] * x2[1] * x1[2];
238  const double v231 = x2[0] * x3[1] * x1[2];
239  const double v312 = x3[0] * x1[1] * x2[2];
240  const double v132 = x1[0] * x3[1] * x2[2];
241  const double v213 = x2[0] * x1[1] * x3[2];
242  const double v123 = x1[0] * x2[1] * x3[2];
243 
244  tempVol[softID] += 1.0/6.0 * (-v321 + v231 + v312 - v132 - v213 + v123);
245  }
246  // Iterate, increase by the number of partners of this bond + 1 for bond type
247  j += iaparams.num+1;
248  }
249  }
250  }
251  }
252 
253  for (int i = 0; i < MaxNumIBM; i++) VolumesCurrent[i] = 0;
254 
255  // Sum up and communicate
256  MPI_Allreduce(tempVol, VolumesCurrent, MaxNumIBM, MPI_DOUBLE, MPI_SUM, comm_cart);
257 
258 }
