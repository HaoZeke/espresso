60 void _cuda_safe_mem(cudaError_t err, const char *file, unsigned int line){
61  if( cudaSuccess != err) {
62  fprintf(stderr, "Cuda Memory error at %s:%u.\n", file, line);
63  printf("CUDA error: %s\n", cudaGetErrorString(err));
64  if ( err == cudaErrorInvalidValue )
65  fprintf(stderr, "You may have tried to allocate zero memory at %s:%u.\n", file, line);
66  errexit();
67  } else {
68  err=cudaGetLastError();
69  if (err != cudaSuccess) {
70  fprintf(stderr, "Error found during memory operation. Possibly however from an failed operation before. %s:%u.\n", file, line);
71  printf("CUDA error: %s\n", cudaGetErrorString(err));
72  if ( err == cudaErrorInvalidValue )
73  fprintf(stderr, "You may have tried to allocate zero memory before %s:%u.\n", file, line);
74  errexit();
75  }
76  }

159

161  global_part_vars_host.number_of_particles = n_part;

164 
165  //if the arrays exists free them to prevent memory leaks
166  if ( particle_forces_host ) cudaFreeHost(particle_forces_host);
167  if ( particle_data_host ) cudaFreeHost(particle_data_host);

172  if ( host_v_cs ) cudaFreeHost(host_v_cs);

181  /**pinned memory mode - use special function to get OS-pinned memory*/

183  cudaHostAlloc((void**)&particle_forces_host, 3 * global_part_vars_host.number_of_particles * sizeof(float), cudaHostAllocWriteCombined);

193 #ifdef ENGINE
194  cudaMallocHost((void**)&host_v_cs, global_part_vars_host.number_of_particles * sizeof(CUDA_v_cs));

198 #endif
199 #endif // __CUDA_ARCH__
200
