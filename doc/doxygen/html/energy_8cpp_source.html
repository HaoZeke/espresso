50 void init_energies(Observable_stat *stat)
51 {
52  int n_pre, n_non_bonded, n_coulomb, n_dipolar;
53 
54  n_pre = 1;
55  n_non_bonded = (n_particle_types*(n_particle_types+1))/2;
56 
57  n_coulomb = 0;
58 #ifdef ELECTROSTATICS
59  switch (coulomb.method) {
60  case COULOMB_NONE: n_coulomb = 0; break;
61 #ifdef P3M
62  case COULOMB_ELC_P3M: n_coulomb = 3; break;
63  case COULOMB_P3M_GPU:
64  case COULOMB_P3M: n_coulomb = 2; break;
65 #endif
66  default: n_coulomb = 1;
67  }
68 #endif
69 
70  n_dipolar = 0;
71 #ifdef DIPOLES
72 
73  switch (coulomb.Dmethod) {
74  case DIPOLAR_NONE: n_dipolar = 1; break;
75 #ifdef DP3M
76  case DIPOLAR_MDLC_P3M: n_dipolar=3; break;
77  case DIPOLAR_P3M: n_dipolar = 2; break;
78 #endif
79  case DIPOLAR_ALL_WITH_ALL_AND_NO_REPLICA: n_dipolar = 2; break;
80  case DIPOLAR_MDLC_DS: n_dipolar=3; break;
81  case DIPOLAR_DS: n_dipolar = 2; break;
82  }
83 
84 #endif
85 
86  obsstat_realloc_and_clear(stat, n_pre, n_bonded_ia, n_non_bonded, n_coulomb, n_dipolar, 0, 1);
87  stat->init_status = 0;
88 
89  external_potential_init_energies();
90 }

102 void energy_calc(double *result)
103 {
104  if (!interactions_sanity_checks())
105  return;
106 
107  init_energies(&energy);
108 
109 #ifdef CUDA
110  clear_energy_on_GPU();
111 #endif
112 
113  EspressoSystemInterface::Instance().update();
114 
115  // Compute the energies from the energyActors
116  for (ActorList::iterator actor= energyActors.begin();
117  actor != energyActors.end(); ++actor)
118  (*actor)->computeEnergy(espressoSystemInterface);
119 
120  on_observable_calc();
121 
122  switch (cell_structure.type) {
123  case CELL_STRUCTURE_LAYERED:
124  layered_calculate_energies();
125  break;
126  case CELL_STRUCTURE_DOMDEC:
127  if(dd.use_vList) {
128  if (rebuild_verletlist)
129  build_verlet_lists();
130  calculate_verlet_energies();
131  }
132  else
133  calculate_link_cell_energies();
134  break;
135  case CELL_STRUCTURE_NSQUARE:
136  nsq_calculate_energies();
137  }
138  /* rescale kinetic energy */
139 #ifdef MULTI_TIMESTEP
140  if (smaller_time_step > 0.)
141  energy.data.e[0] /= (2.0*smaller_time_step*smaller_time_step);
142  else
143 #endif
144  energy.data.e[0] /= (2.0*time_step*time_step);
145 
146  calc_long_range_energies();
147 
148 #ifdef CUDA
149  copy_energy_from_GPU();
150 #endif
151 
152  /* gather data */
153  MPI_Reduce(energy.data.e, result, energy.data.n, MPI_DOUBLE, MPI_SUM, 0, comm_cart);
154 
155  if (n_external_potentials > 0) {
156  double* energies = (double*) malloc(n_external_potentials*sizeof(double));
157  for (int i=0; i<n_external_potentials; i++) {
158  energies[i]=external_potentials[i].energy;
159  }
160  double* energies_sum = (double*) malloc(n_external_potentials*sizeof(double));
161  MPI_Reduce(energies, energies_sum, n_external_potentials, MPI_DOUBLE, MPI_SUM, 0, comm_cart);
162  for (int i=0; i<n_external_potentials; i++) {
163  external_potentials[i].energy=energies_sum[i];
164  }
165  free(energies);
166  free(energies_sum);
167  }
168 
169 }

173 void calc_long_range_energies()
174 {
175 #ifdef ELECTROSTATICS
176  /* calculate k-space part of electrostatic interaction. */
177  switch (coulomb.method) {
178 #ifdef P3M
179  case COULOMB_P3M_GPU:
180  printf("long range energy calculation not implemented for GPU P3M\n"); //TODO make right
181  break;
182  case COULOMB_P3M:
183  p3m_charge_assign();
184  energy.coulomb[1] = p3m_calc_kspace_forces(0,1);
185  break;
186  case COULOMB_ELC_P3M:
187  // assign the original charges first
188  // they may not have been assigned yet
189  p3m_charge_assign();
190  if(!elc_params.dielectric_contrast_on)
191  energy.coulomb[1] = p3m_calc_kspace_forces(0,1);
192  else {
193  energy.coulomb[1] = 0.5*p3m_calc_kspace_forces(0,1);
194  energy.coulomb[1]+= 0.5*ELC_P3M_dielectric_layers_energy_self();
195 
196  // assign both original and image charges now
197  ELC_p3m_charge_assign_both();
198  ELC_P3M_modify_p3m_sums_both();
199 
200  energy.coulomb[1] += 0.5*p3m_calc_kspace_forces(0,1);
201 
202  //assign only the image charges now
203  ELC_p3m_charge_assign_image();
204  ELC_P3M_modify_p3m_sums_image();
205 
206  energy.coulomb[1]-= 0.5*p3m_calc_kspace_forces(0,1);
207  }
208  energy.coulomb[2] = ELC_energy();
209  break;
210 #endif
211  case COULOMB_MMM2D:
212  *energy.coulomb += MMM2D_far_energy();
213  *energy.coulomb += MMM2D_dielectric_layers_energy_contribution();
214  break;
215  /* calculate electric part of energy (only for MAGGS) */
216  case COULOMB_MAGGS:
217  *energy.coulomb += maggs_electric_energy();
218  break;
219  default: break;
220  }
221 #endif /* ifdef ELECTROSTATICS */
222 
223 #ifdef DIPOLES
224  switch (coulomb.Dmethod) {
225 #ifdef DP3M
226  case DIPOLAR_P3M:
227  dp3m_dipole_assign();
228  energy.dipolar[1] = dp3m_calc_kspace_forces(0,1);
229  break;
230  case DIPOLAR_MDLC_P3M:
231  dp3m_dipole_assign();
232  energy.dipolar[1] = dp3m_calc_kspace_forces(0,1);
233  energy.dipolar[2] = add_mdlc_energy_corrections();
234  break;
235 #endif
236  case DIPOLAR_ALL_WITH_ALL_AND_NO_REPLICA:
237  energy.dipolar[1] = dawaanr_calculations(0,1);
238  break;
239  case DIPOLAR_MDLC_DS:
240  energy.dipolar[1] = magnetic_dipolar_direct_sum_calculations(0,1);
241  energy.dipolar[2] = add_mdlc_energy_corrections();
242  break;
243  case DIPOLAR_DS:
244  energy.dipolar[1] = magnetic_dipolar_direct_sum_calculations(0,1);
245  break;
246  case DIPOLAR_NONE:
247  break;
248  default:
249  ostringstream msg;
250  msg <<"unknown dipolar method";
251  runtimeError(msg);
252  break;
253  }
254 #endif /* ifdef DIPOLES */
255 
256 }
