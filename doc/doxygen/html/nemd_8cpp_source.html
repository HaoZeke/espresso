137 void nemd_change_momentum()
138 {
139  int i;
140  double tmp_v0;
141  Slab *mid_slab, *top_slab;
142 
143  if(nemd_method == NEMD_METHOD_OFF) return;
144 
145  INTEG_TRACE(fprintf(stderr,"%d: nemd_change_momentum: Method %d\n",this_node,nemd_method));
146 
147  mid_slab = &nemddata.slab[nemddata.mid_slab];
148  top_slab = &nemddata.slab[nemddata.top_slab];
149 
150  if(nemd_method == NEMD_METHOD_EXCHANGE ) {
151  /* exit if there are not enough particles */
152  INTEG_TRACE(fprintf(stderr,"%d: parts_in_slabs: top %d mid %d\n",this_node,top_slab->n_parts_in_slab,mid_slab->n_parts_in_slab));
153  if(mid_slab->n_fastest != nemddata.n_exchange ||
154  top_slab->n_fastest != nemddata.n_exchange) {
155  ostringstream msg;
156  msg <<"nemd_exchange_momentum: Not enough particles in slab!";
157  runtimeError(msg);
158  /* cannot continue */
159  return;
160  }
161 
162  /* perform momentum exchange */
163  for(i=0;i<nemddata.n_exchange;i++) {
164  /* store momentum change */
165  nemddata.momentum += local_particles[top_slab->fastest[i]]->m.v[0];
166  nemddata.momentum -= local_particles[mid_slab->fastest[i]]->m.v[0];
167  tmp_v0 = local_particles[mid_slab->fastest[i]]->m.v[0];
168  local_particles[mid_slab->fastest[i]]->m.v[0] = local_particles[top_slab->fastest[i]]->m.v[0];
169  local_particles[top_slab->fastest[i]]->m.v[0] = tmp_v0;
170  }
171 
172  /* prepare next round */
173  top_slab->n_fastest = 0;
174  top_slab->v_min = -1e10;
175  mid_slab->n_fastest = 0;
176  mid_slab->v_min = 1e10;
177  }
178  else if (nemd_method == NEMD_METHOD_SHEARRATE) {
179  double vel_mean;
180  /* calculate velocity difference vel_diff = vel_required - vel_actual */
181  vel_mean = top_slab->v_mean/(double)top_slab->n_parts_in_slab;
182  top_slab->vel_diff = nemddata.slab_vel - vel_mean;
183  vel_mean = mid_slab->v_mean/(double)mid_slab->n_parts_in_slab;
184  mid_slab->vel_diff = - (nemddata.slab_vel + vel_mean);
185  /* store momentum change */
186  nemddata.momentum += top_slab->n_parts_in_slab*top_slab->vel_diff;
187  nemddata.momentum -= mid_slab->n_parts_in_slab*mid_slab->vel_diff;
188  }
189  nemddata.momentum_norm ++;
190 }
