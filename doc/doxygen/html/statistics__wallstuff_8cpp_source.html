30 void wall_sort_particles()
31 {
32  // 1. reallocate the boxes for the particle identities
33  // save current number of bins
34  int n_part_in_bin = wallstuff_boundaries.n-1;
35  // free old boxes
36  for (int i = 0; i < n_part_in_bin; ++i) {
37  realloc_intlist(&wallstuff_part_in_bin[i], wallstuff_part_in_bin[i].n = 0);
38  }
39  wallstuff_part_in_bin = (IntList*)realloc(wallstuff_part_in_bin,
40  (wallstuff_boundaries.n-1)*sizeof(IntList));
41  // initialize new ones
42  for (int i = n_part_in_bin; i < wallstuff_boundaries.n-1; ++i) {
43  init_intlist(&wallstuff_part_in_bin[i]);
44  }
45 
46  // 2. for each particle, find the box and put its
47  // identity there
48  for(int i=0; i<n_part; i++) {
49  double x = partCfg[i].r.p[0];
50  // ignore particles outside the wallstuff_boundaries
51  if (x < wallstuff_boundaries.e[0] || x > wallstuff_boundaries.e[wallstuff_boundaries.n-1])
52  continue;
53  // simple bisection on the particle's x-coordinate
54  int s = 0;
55  int e = wallstuff_boundaries.n - 1;
56  while (e - s > 1) {
57  int c = (e + s)/2;
58  if (x >= wallstuff_boundaries.e[c]) s = c; else e = c;
59  }
60  // and add the particle to the resulting list
61  realloc_grained_intlist(&wallstuff_part_in_bin[s], wallstuff_part_in_bin[s].n + 1, 8);
62  wallstuff_part_in_bin[s].e[wallstuff_part_in_bin[s].n++] = i;
63  }
64 }

66 void calc_wallmsdyz(double *g, int bin)
67 {
68  // loop over all stored configurations
69  // and calculate the MSD with respect to the current configuration
70  // MSD of configuration with itself is always 0
71  g[0] = 0;
72  for(int k = 1; k < n_configs; k++) {
73  g[k] = 0.0;
74  // loop over all particles in the specified bin and add up MSD
75  for (int i = 0; i < wallstuff_part_in_bin[bin].n; ++i) {
76  int p = wallstuff_part_in_bin[bin].e[i];
77  g[k] +=
78  + SQR(configs[n_configs-1][3*p + 1]-configs[n_configs-1-k][3*p + 1])
79  + SQR(configs[n_configs-1][3*p + 2]-configs[n_configs-1-k][3*p + 2]);
80  }
81  // normalize
82  g[k] /= wallstuff_part_in_bin[bin].n;
83  }
