24 /** \file domain_decomposition.hpp
25  *
26  * This file contains everything related to the cell system: domain decomposition.
27  *
28  * The simulation box is split into spatial domains for each node
29  * according to a cartesian node grid (\ref node_grid).
30  *
31  * The domain of a node is split into a 3D cell grid with dimension
32  * \ref DomainDecomposition::cell_grid. Together with one ghost cell
33  * layer on each side the overall dimension of the ghost cell grid is
34  * \ref DomainDecomposition::ghost_cell_grid. The domain
35  * decomposition enables one the use of the linked cell algorithm
36  * which is in turn used for setting up the verlet list for the
37  * system. You can see a 2D graphical representation of the linked
38  * cell grid below.
39  *
40  * \image html linked_cells.gif "Linked cells structure"
41  *
42  * 2D representation of a linked cell grid: cell_grid =
43  * {4,4}, ghost_cell_grid = {6,6}
44  *
45  * Each cell has 3^D neighbor cells (For cell 14 they are
46  * marked). Since we deal with pair forces, it is sufficient to
47  * calculate only half of the interactions (Newtons law: actio =
48  * reactio). We have chosen the upper half e.g. all neighbor cells with
49  * a higher linear index (For cell 14 they are marked in light
50  * blue). Caution: This implementation needs double sided ghost
51  * communication! For single sided ghost communication one would need
52  * some ghost-ghost cell interaction as well, which we do not need!
53  *
54  * For more information on cells,
55  * see \ref cells.hpp
56 */

81 typedef struct {
82  /** number of interacting neighbor cells .
83 
84  A word about the interacting neighbor cells:
85 
86  In a 3D lattice each cell has 27 neighbors (including
87  itself!). Since we deal with pair forces, it is sufficient to
88  calculate only half of the interactions (Newtons law: actio =
89  reactio). For each cell 13+1=14 neighbors. This has only to be
90  done for the inner cells.
91 
92  Caution: This implementation needs double sided ghost
93  communication! For single sided ghost communication one would
94  need some ghost-ghost cell interaction as well, which we do not
95  need!
96 
97  It follows: inner cells: n_neighbors = 14
98  ghost cells: n_neighbors = 0
99  */
100  int n_neighbors;
101  /** Interacting neighbor cell list */
102  IA_Neighbor *nList;
103 } IA_Neighbor_List;
