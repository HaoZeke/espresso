45 /** Computes the three body angle interaction force and adds this
46  force to the particle forces (see \ref tclcommand_inter).
47  @param p_mid Pointer to second/middle particle.
48  @param p_left Pointer to first/left particle.
49  @param p_right Pointer to third/right particle.
50  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).
51  @param force1 returns force of particle 1
52  @param force2 returns force of particle 2
53  @return 0
54 */
55 inline int calc_angle_cosine_force(Particle *p_mid, Particle *p_left, Particle *p_right,
56  Bonded_ia_parameters *iaparams, double force1[3], double force2[3])
57 {
58  double cosine, vec1[3], vec2[3], d1i, d2i, dist2, fac, f1=0.0, f2=0.0;
59  int j;
60 
61  cosine=0.0;
62  /* vector from p_left to p_mid */
63  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
64  dist2 = sqrlen(vec1);
65  d1i = 1.0 / sqrt(dist2);
66  for(j=0;j<3;j++) vec1[j] *= d1i;
67  /* vector from p_mid to p_right */
68  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
69  dist2 = sqrlen(vec2);
70  d2i = 1.0 / sqrt(dist2);
71  for(j=0;j<3;j++) vec2[j] *= d2i;
72  /* scalar produvt of vec1 and vec2 */
73  cosine = scalar(vec1, vec2);
74  fac = iaparams->p.angle_cosine.bend;
75 
76  if ( cosine > TINY_COS_VALUE ) cosine = TINY_COS_VALUE;
77  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
78  fac *= iaparams->p.angle_cosine.sin_phi0 * (cosine/sqrt(1-SQR(cosine))) + iaparams->p.angle_cosine.cos_phi0;
79 
80  for(j=0;j<3;j++) {
81  f1 = fac * (cosine * vec1[j] - vec2[j]) * d1i;
82  f2 = fac * (cosine * vec2[j] - vec1[j]) * d2i;
83 
84  force1[j] = (f1-f2);
85  force2[j] = -f1;
86  }
87  return 0;
88 }

90 /* The force on each particle due to a three-body bonded potential
91  is computed. */
92 inline void calc_angle_cosine_3body_forces(Particle *p_mid, Particle *p_left,
93  Particle *p_right, Bonded_ia_parameters *iaparams,
94  double force1[3], double force2[3], double force3[3]) {
95 
96  int j;
97  double pot_dep;
98  double cos_phi;
99  double sin_phi;
100  double vec31[3];
101  double vec21[3];
102  double vec12[3]; // espresso convention
103  double vec21_sqr;
104  double vec31_sqr;
105  double vec21_magn;
106  double vec31_magn;
107  double fj[3];
108  double fk[3];
109  double fac;
110 
111  get_mi_vector(vec12, p_mid->r.p, p_left->r.p);
112  for(j = 0; j < 3; j++)
113  vec21[j] = -vec12[j];
114 
115  get_mi_vector(vec31, p_right->r.p, p_mid->r.p);
116  vec21_sqr = sqrlen(vec21);
117  vec21_magn = sqrt(vec21_sqr);
118  vec31_sqr = sqrlen(vec31);
119  vec31_magn = sqrt(vec31_sqr);
120  cos_phi = scalar(vec21, vec31) / (vec21_magn * vec31_magn);
121  sin_phi = sqrt(1.0 - SQR(cos_phi));
122 
123  /* uncomment this block if interested in the angle
124  if(cos_phi < -1.0) cos_phi = -TINY_COS_VALUE;
125  if(cos_phi > 1.0) cos_phi = TINY_COS_VALUE;
126  phi = acos(cos_phi);
127  */
128 
129  {
130  double K, sin_phi0, cos_phi0;
131  K = iaparams->p.angle_cosine.bend;
132  sin_phi0 = iaparams->p.angle_cosine.sin_phi0;
133  cos_phi0 = iaparams->p.angle_cosine.cos_phi0;
134 
135  // potential dependent term [dU/dphi = K * sin(phi - phi0)]
136  // trig identity: sin(a - b) = sin(a)cos(b) - cos(a)sin(b)
137  pot_dep = K * (sin_phi * cos_phi0 - cos_phi * sin_phi0);
138  }
139 
140  fac = pot_dep / sin_phi;
141 
142  for(j = 0; j < 3; j++) {
143  fj[j] = vec31[j] / (vec21_magn * vec31_magn) - cos_phi * vec21[j] / vec21_sqr;
144  fk[j] = vec21[j] / (vec21_magn * vec31_magn) - cos_phi * vec31[j] / vec31_sqr;
145  }
146 
147  // note that F1 = -(F2 + F3)
148  for(j = 0; j < 3; j++) {
149  force1[j] = force1[j] - fac * (fj[j] + fk[j]);
150  force2[j] = force2[j] + fac * fj[j];
151  force3[j] = force3[j] + fac * fk[j];
152  }
153 }

156 /** Computes the three body angle interaction energy (see \ref tclcommand_inter, \ref tclcommand_analyze).
157  @param p_mid Pointer to first particle.
158  @param p_left Pointer to second/middle particle.
159  @param p_right Pointer to third particle.
160  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).

162  @return 0.
163 */
164 inline int angle_cosine_energy(Particle *p_mid, Particle *p_left, Particle *p_right,
165  Bonded_ia_parameters *iaparams, double *_energy)
166 {
167  double cosine, vec1[3], vec2[3], d1i, d2i, dist2;
168  int j;
169 
170  cosine=0.0;
171  /* vector from p_mid to p_left */
172  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
173  dist2 = sqrlen(vec1);
174  d1i = 1.0 / sqrt(dist2);
175  for(j=0;j<3;j++) vec1[j] *= d1i;
176  /* vector from p_right to p_mid */
177  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
178  dist2 = sqrlen(vec2);
179  d2i = 1.0 / sqrt(dist2);
180  for(j=0;j<3;j++) vec2[j] *= d2i;
181  /* scalar produvt of vec1 and vec2 */
182  cosine = scalar(vec1, vec2);
183  if ( cosine > TINY_COS_VALUE) cosine = TINY_COS_VALUE;
184  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
185  /* bond angle energy */
186 
187  *_energy = iaparams->p.angle_cosine.bend*(cosine*iaparams->p.angle_cosine.cos_phi0 - sqrt(1-SQR(cosine))*iaparams->p.angle_cosine.sin_phi0+1);
188 
189  return 0;
