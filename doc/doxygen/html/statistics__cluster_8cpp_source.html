54 void cluster_init(Particle *part, int size) {

78 /* join two clusters (Join cluster2 to cluster1 and eliminate cluster2) */
79 void clusters_join(Cluster *cluster1, Cluster *cluster2)
80 {
81  ClusterElement *local_element;
82 
83  /* join cluster2 to cluster1 */
84  local_element = cluster1->first;
85  while(local_element->next != NULL) local_element = local_element->next;
86  local_element->next = cluster2->first;
87  cluster1->size += cluster2->size;
88  /* remove cluster2 */
89  cluster2->prev->next = cluster2->next;
90  cluster2->next->prev = cluster2->prev;
91  if(cluster2 == last_cluster) last_cluster = cluster2->prev;
92  if(cluster2 == first_cluster) first_cluster = cluster2->next;
93 }

95 /* test wether two clusters are connected (criterion of step 2 of the necklace cluster algorithm) */
96 int clusters_connected(Particle *part, Cluster cluster1, Cluster cluster2)
97 {
98  int id1,id2;
99  ClusterElement *element1,*element2;
100 
101  element1 = cluster1.first;
102  while ( element1 != NULL ) {
103  id1 = element1->id;
104  element2 = cluster2.first;
105  while ( element2 != NULL ) {
106  id2 = element2->id;
107  if ( abs( id1 - id2 ) > backbone_distance &&
108  distance2(part[id1].r.p,part[id2].r.p) < space_distance2 ) return 1;
109  element2 = element2->next;
110  }
111  element1 = element1->next;
112  }
113  return 0;
114 }

116 /* test wether two clusters interpenetrate (criterion 4 of the necklace cluster algorithm) */
117 int clusters_interpenetrate(Cluster cluster1, Cluster cluster2)
118 {
119  int min1,min2,max1,max2;
120  ClusterElement *local_element;
121 
122  local_element = cluster1.first;
123  min1 = max1 = local_element->id;
124  while(local_element->next != NULL) {
125  local_element = local_element->next;
126  if(local_element->id < min1) min1 = local_element->id;
127  if(local_element->id > max1) max1 = local_element->id;
128  }
129 
130  local_element = cluster2.first;
131  min2 = max2 = local_element->id;
132  while(local_element->next != NULL) {
133  local_element = local_element->next;
134  if(local_element->id < min2) min2 = local_element->id;
135  if(local_element->id > max2) max2 = local_element->id;
136  }
137 
138  /* test if one cluster is completelly on left side of the other ... */
139  if( ((min1 < min2) && (max1 < min2)) ||
140  ((min2 < min1) && (max2 < min1)) ) return 0;
141  /* ... otherwise they interpenetrate. */
142  return 1;
143 }

145 /* perform step 2 to 4 of the necklace cluster algorithm */
146 int cluster_joincicle(Particle *part) {
147  int joins=1, cicles=0;
148  Cluster *cluster1,*cluster2;
149 
150  while(joins > 0) {
151  joins =0;
152  cicles ++;
153  cluster1 = first_cluster;
154  /* loop cluster 1 until last cluster but one */
155  while(cluster1->next != first_cluster) {
156  cluster2 = cluster1->next;
157  /* loop cluster 2 until lust cluster */
158  while(cluster2->prev != last_cluster) {
159  if( clusters_connected(part,*cluster1,*cluster2) ||
160  clusters_interpenetrate(*cluster1,*cluster2) ) {
161  clusters_join(cluster1,cluster2);
162  joins++;
163  }
164  cluster2 = cluster2->next;
165  }
166  cluster1 = cluster1->next;
167  }
168  }
169 
170  return cicles;
171 }

173 /* perform step 5 to 7 of the necklace cluster algorithm
174  \return number of pearls
175 */
176 int cluster_join_to_substructures()
177 {
178  int n_pearls = 0, previous_is_pearl=1;
179  Cluster *cluster1,*cluster2;
180 
181  /* search first pearl */
182  cluster1 = first_cluster;
183  while(cluster1->next != last_cluster &&
184  cluster1->size < pearl_treshold ) cluster1 = cluster1->next;
185 
186  /* if no pearl found join all clusters and return 0 ... */
187  if(cluster1->size < pearl_treshold) {
188  cluster2 = cluster1->next;
189  while(cluster2->next != last_cluster) {
190  clusters_join(cluster1,cluster2);
191  cluster2 = cluster2->next;
192  }
193  return 0;
194  }
195 
196  /* ... join all previous clusters to first pearl */
197  n_pearls++;
198  cluster2 = cluster1->prev;
199  while(cluster2->next != first_cluster) {
200  clusters_join(cluster1,cluster2);
201  cluster2 = cluster2->prev;
202  }
203 
204  /* now go through the rest */
205  cluster1 = cluster1->next;
206  while(cluster1->prev != last_cluster) {
207  /* join actual cluster to previous cluster, if they are of the same type */
208  if(cluster1->size < pearl_treshold) {
209  /* actual cluster is not a pearl */
210  if(previous_is_pearl == 0) clusters_join(cluster1->prev,cluster1);
211  else previous_is_pearl = 0;
212  }
213  else {
214  /* actual cluster is a pearl */
215  if(previous_is_pearl == 1) clusters_join(cluster1->prev,cluster1);
216  else {
217  previous_is_pearl = 1;
218  n_pearls++;
219  }
220  }
221  cluster1 = cluster1->next;
222  }
223 
224  /* if last cluster is not a pearl join it to previous cluster */
225  while(last_cluster->size < pearl_treshold) clusters_join(last_cluster->prev,last_cluster);
226  return n_pearls;
227 }

319 /** hole cluster algorithm.
320  returns the number of holes and a list of mesh points belonging to each of them */
321 int cluster_free_volume_grid(IntList mesh, int dim[3], int ***holes)
322 {
323  int i=0,j, k, n=-1, li;
324  int neighbors[6];
325 
326  int *tmp = (int *) malloc( sizeof(int)* (dim[0]*dim[1]*dim[2]));
327  int *sizes = (int *) malloc( sizeof(int)* (dim[0]*dim[1]*dim[2]));
328 
329 
330  // step 1 go through all mesh points
331  while ( i < (dim[0]*dim[1]*dim[2]) ) {
332  // step 2 test if mesh point is occupied or allready assigned
333  if ( mesh.e[i] == -2 || mesh.e[i] >= 0 ) { i++; }
334  else {
335  // step 3 mesh point is free, create a new cluster
336  n++;
337  mesh.e[i] = n;
338  // start new cluster, put mesh point as first element, and set list pointer on first element
339  sizes[n] = 1;
340  tmp[0] = i;
341  li = 0;
342  // step 4 go through all elements of the cluster
343  while ( li < sizes[n] ) {
344  // step 5 go through all neighbors
345  j = tmp[li];
346  cluster_neighbors(j, dim, neighbors);
347  for ( k=0; k<6; k++ ) {
348  // step 5 b test if status is free and append it to the cluster
349  if ( mesh.e[ neighbors[k] ] == -1 ) {
350  mesh.e[ neighbors[k] ] = n;
351  // append mesh point as element in the list
352  tmp[ sizes[n] ] = neighbors[k];
353  sizes[n]++;
354  }
355  if ( mesh.e[ neighbors[k] ] > 0 && mesh.e[ neighbors[k] ]<n ) {
356  fprintf(stderr,"cfvg:error: i=%d, li=%d, n=%d, mesh.e[x]=%d, x=%d\n",i,li,n,mesh.e[ neighbors[k] ],neighbors[k]); fflush(stderr);
357  }
358  }
359  li++;
360  }
361  }
362 
363  }
364 
365 
366  // allocate list space
367  (*holes) = (int **) malloc ( sizeof(int *)*(n+1) );
368  for ( i=0; i<=n; i++ ) {
369  (*holes)[i] = (int *) malloc ( sizeof(int)*(sizes[i]+1) );
370  (*holes)[i][0] = 0;
371  }
372 
373  for ( i=0; i<(dim[0]*dim[1]*dim[2]); i++ ) {
374  j = mesh.e[i];
375  if ( j >= 0 ) {
376  (*holes)[j][0] ++;
377  (*holes)[j][ (*holes)[j][0] ] = i;
378  }
379  }
380 
381  free(tmp);
382  free(sizes);
383 
384  return n;
385 }
