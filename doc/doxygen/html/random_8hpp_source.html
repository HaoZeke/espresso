80 /** classical RAN1 random number generator */
81 inline long l_random(void)
82 {
83  /*
84  * N O T E T H A T T H E R E A R E N O S A F E T Y C H E C K S !!!
85  */
86  int j;
87  long k;
88 
89  k = (idum) / IQ;
90  idum = IA * (idum - k * IQ) - IR * k;
91  if (idum < 0) idum += IM;
92  j = iy / NDIV;
93  iy = iv[j];
94  iv[j] = idum;
95  return iy;
96 }

110 inline double d_random(void)
111 {

122 /** Generator for Gaussian random numbers. Uses the Box-Muller
123  * transformation to generate two Gaussian random numbers from two
124  * uniform random numbers.
125  *
126  * @return Gaussian random number.
127  *
128  */
129 inline double gaussian_random(void) {
130  double x1, x2, r2, fac;
131  static int calc_new = 1;
132  static double save;
133 
134  /* On every second call two gaussian random numbers are calculated
135  via the Box-Muller transformation. One is returned as the result
136  and the second one is stored for use on the next call.
137  */
138 
139  if (calc_new) {
140 
141  /* draw two uniform random numbers in the unit circle */
142  do {
143  x1 = 2.0*d_random()-1.0;
144  x2 = 2.0*d_random()-1.0;
145  r2 = x1*x1 + x2*x2;
146  } while (r2 >= 1.0 || r2 == 0.0);
147 
148  /* perform Box-Muller transformation */
149  fac = sqrt(-2.0*log(r2)/r2);
150 
151  /* save one number for later use */
152  save = x1*fac;
153  calc_new = 0;
154 
155  /* return the second number */
156  return x2*fac;
157 
158  } else {
159 
160  calc_new = 1;
161 
162  /* return the stored gaussian random number */
163  return save;
164 
165  }
