59 static double calc_pwdist(Particle *p1, Bonded_ia_parameters *iaparams, int *clconstr)
60 {
61  int j,k,img[3];
62  double distwallmin=0.0, normal=0.0;
63  double folded_pos_p1[3];
64  double pwdist[n_constraints];
65  Constraint_wall wall;
66 
67  /*fprintf(stdout," Entering calc_pwdist:\n");*/
68 
69  /* folds coordinates of p_left into original box */
70  memmove(folded_pos_p1, p1->r.p, 3*sizeof(double));
71  memmove(img, p1->l.i, 3*sizeof(int));
72  fold_position(folded_pos_p1, img);
73  /*fprintf(stdout," p1= %9.6f %9.6f %9.6f\n",p1->r.p[0],p1->r.p[1],p1->r.p[2]);*/
74 
75  /* Gets and tests wall data */
76  for(k=0;k<n_constraints;k++) {
77  if (constraints[k].type == CONSTRAINT_WAL) {
78  wall=constraints[k].c.wal;
79  /* check that constraint wall normal is normalised */
80  for(j=0;j<3;j++) normal += wall.n[j] * wall.n[j];
81  if (sqrt(normal) != 1.0) {
82  for(j=0;j<3;j++) wall.n[j]=wall.n[j]/normal;
83  }
84  }
85  }
86 
87  /* Calculate distance of end particle from closest wall */
88  for(k=0;k<n_constraints;k++) {
89  if (constraints[k].type == CONSTRAINT_WAL) {
90  wall=constraints[k].c.wal;
91  /* distwallmin is distance of closest wall from p1 */
92  pwdist[k]=-1.0 * wall.d;
93  for(j=0;j<3;j++) {
94  pwdist[k] += folded_pos_p1[j] * wall.n[j];
95  }
96  if (k==0) {
97  distwallmin=pwdist[k];
98  } else {
99  if (pwdist[k] < distwallmin) {
100  distwallmin = pwdist[k];
101  *clconstr = k;
102  }
103  }
104  /*fprintf(stdout," k=%d clconstr=%d\n",k,*clconstr);*/
105  }
106  }
107  /*
108  if (distwallmin <= iaparams->p.endangledist.distmax) {
109  fprintf(stdout," clconstr=%d distwallmin=%f distmx=%f\n",*clconstr,distwallmin,distmx);
110  }
111  */
112  return distwallmin;
113 }

115 /// Calculate angle that p1--p2 makes with wall constraint
116 static double calc_pwangle(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, int *constr)
117 {
118  int j;
119  double dist,di,cosine,phi;
120  double vec[3];
121 
122  /* vector from p1 to p2 */
123  get_mi_vector(vec, p2->r.p, p1->r.p);
124  dist = sqrlen(vec);
125  di = 1.0/sqrt(dist);
126  for(j=0;j<3;j++) vec[j] *= di;
127  /*
128  fprintf(stdout,"Normalised: p1= %9.6f %9.6f %9.6f p1= %9.6f %9.6f %9.6f vec= %9.6f %9.6f %9.6f\n",p1->r.p[0],p1->r.p[1],p1->r.p[2],p2->r.p[0],p2->r.p[1],p2->r.p[2],vec[0],vec[1],vec[2]);
129  */
130  /* vectors are normalised so cosine is just cos(angle_between_vec1_and_vec2)
131  * Wall is closest wall to particle
132  */
133 
134  cosine = scalar(vec, constraints[*constr].c.wal.n);
135  if ( cosine > TINY_COS_VALUE) cosine = TINY_COS_VALUE;
136  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
137  phi=acos(cosine);
138 
139  /*
140  fprintf(stdout,"Angle with wall 0=%f ",(acos(scalar(vec, constraints[0].c.wal.n)))*180.0/PI);
141  fprintf(stdout,"Angle with wall 1=%f ",(acos(scalar(vec, constraints[1].c.wal.n)))*180.0/PI);
142  fprintf(stdout,"dxy=%f dz=%f angle=%f\n",sqrt(vec[0]*vec[0]+vec[1]*vec[1]),vec[2],atan(sqrt(vec[0]*vec[0]+vec[1]*vec[1])/vec[2])*180.0/PI);
143  fprintf(stdout,"Angle with closest wall %d=%f ",*constr,(acos(scalar(vec, constraints[*constr].c.wal.n)))*180.0/PI);
144  */
145  return phi;
146 }

150 int calc_endangledist_pair_force(Particle *p1, Particle *p2,
151  Bonded_ia_parameters *iaparams,
152  double dx[3], double force1[3], double force2[3])
153 {
154 
155  int i=0;
156  int clconstr=0;
157  // double distwallmin=0.0, distmx, distmn;
158  double bend=0.0,phieq=0.0,phi=0.0,distwallmin=0.0, distmx, distmn, dist, di;
159  double smooth, sinphi, cosphi, fac_a, fac_b, gradharm1, gradharm2;
160  double vec[3],dsmooth[3],f1a[3],f1b[3],f2a[3];
161 
162  /*fprintf(stdout,"\nEntering calc_endangledist_pair_force:\n");*/
163 
164  distwallmin = calc_pwdist(p1, iaparams, &clconstr);
165  distmx = iaparams->p.endangledist.distmax;
166  distmn = iaparams->p.endangledist.distmin;
167 
168  if (distwallmin < distmx) {
169  /* function which goes smoothly from 0 to 1 as z goes from distmax to distmin */
170  if (distwallmin < distmn) {
171  smooth = 1.0;
172  for(i=0;i<3;i++) {
173  dsmooth[i] = 0.0;
174  }
175  } else {
176  smooth = 0.5*(cos((distwallmin-distmn)/(distmx-distmn)*PI)+1.0);
177  for(i=0;i<3;i++) {
178  dsmooth[i] = -0.5*PI/(distmx-distmn)*sin((distwallmin-distmn)/(distmx-distmn)*PI)*constraints[clconstr].c.wal.n[i];
179  }
180  }
181  /* Get vector from particle 1 to particle 2 */
182  get_mi_vector(vec, p2->r.p, p1->r.p);
183  dist = sqrlen(vec);
184  di = 1.0/sqrt(dist);
185  /*
186  for(j=0;j<3;j++) vec[j] *= di;
187  */
188  /* Calculate angle that p1-p2 makes with wall */
189  phi = calc_pwangle(p1, p2, iaparams, &clconstr);
190 
191  sinphi = sin(phi);
192  cosphi = cos(phi);
193  bend = iaparams->p.endangledist.bend;
194  phieq = iaparams->p.endangledist.phi0;
195 
196  /*
197  fprintf(stdout," Bead %4d: Cl.wall=%2d distwallmin=%9.6f \n",p1->p.identity,clconstr,distwallmin);
198  fprintf(stdout," vector=(%f %f %f)\n",vec[0],vec[1],vec[2]);
199  fprintf(stdout,"pos1=(%f %f %f) pos2=(%f %f %f) distwallmin=%9.6f angle=%9.6f\n",p1->r.p[0],p1->r.p[1],p1->r.p[2],p2->r.p[0],p2->r.p[1],p2->r.p[2],distwallmin,phi*180.0/PI);
200  */
201 
202 #ifdef BOND_ENDANGLEDIST_HARMONIC
203  /* Force = -dU/dr_i= k*smooth*(phi-phi0)/sin(phi)(cosphi*vec + n)/|vec| */
204  fac_a = bend*(phi-phieq)/sinphi;
205  fac_b = 0.5*bend*SQR(phi-phieq);
206  for(i=0;i<3;i++) {
207  gradharm1 = -1.0*fac_a*(cosphi*vec[i]-constraints[clconstr].c.wal.n[i])*di;
208  gradharm2 = -1.0*gradharm1;
209  f1a[i] = smooth*gradharm1;
210  f1b[i] = dsmooth[i]*fac_b;
211  f2a[i] = smooth*gradharm2;
212  force1[i] = -1.0*(f1a[i]+f1b[i]);
213  force2[i] = -1.0*f2a[i];
214  }
215  /*
216  fprintf(stdout," f1=(% 9.6f % 9.6f % 9.6f) ",f1[0],f1[1],f1[2]);
217  fprintf(stdout," f2=(% 9.6f % 9.6f % 9.6f)\n",f2[0],f2[1],f2[2]);
218  fprintf(stdout," force=(% 9.6f % 9.6f % 9.6f) ",force1[0],force1[1],force1[2]);
219  fprintf(stdout," force2=(% 9.6f % 9.6f % 9.6f)\n",force2[0],force2[1],force2[2]);
220  */
221  } else if (distwallmin >= distmx) {
222  for(i=0;i<3;i++) {
223  force1[i] = 0.0;
224  force2[i] = 0.0;
225  }
226  }
227 #endif
228 
229  ONEPART_TRACE(if(p1->p.identity==check_id) \
230  fprintf(stderr,"%d: OPT: ENDANGLEDIST f = (%.3e,%.3e,%.3e) " \
231  "with part id=%d at dist %f fac %.3e %.3e\n", \
232  this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2], \
233  p2->p.identity,dist,fac_a,fac_b) \
234  );
235  ONEPART_TRACE(if(p2->p.identity==check_id) \
236  fprintf(stderr,"%d: OPT: ENDANGLEDIST f = (%.3e,%.3e,%.3e) " \
237  "with part id=%d at dist %f fac %.3e %.3e\n", \
238  this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2], \
239  p1->p.identity,dist,fac_a,fac_b));
240 
241  return 0;
242 }

256  distmn = iaparams->p.endangledist.distmin;

258 #ifdef BOND_ENDANGLEDIST_HARMONIC

260  /* function which goes smoothly from 0 to 1 as z goes from distmax to distmin */
261  if (distwallmin < distmn) {
262  smooth = 1.0;
263  } else {
264  smooth = 0.5*(cos((distwallmin-distmn)/(distmx-distmn)*PI)+1);

271  *_energy = 0.5*bend*smooth*SQR(phi - phieq);
