23 /** \file iccp3m.hpp
24 
25  ICCP3M is a method that allows to take into account the influence
26  of arbitrarliy shaped dielectric interfaces. The dielectric
27  properties of a dielectric medium in the bulk of the simulation
28  box are taken into account by reproducing the jump in the electric
29  field at the inface with charge surface segments. The charge
30  density of the surface segments have to be determined
31  self-consistently using an iterative scheme. It can at presently
32  - despite its name - be used with P3M, ELCP3M, MMM2D and MMM1D.
33  For details see:<br> S. Tyagi, M. Suzen, M. Sega, C. Holm,
34  M. Barbosa: A linear-scaling method for computing induced charges
35  on arbitrary dielectric boundaries in large system simulations
36  (Preprint)
37 
38  To set up ICCP3M first the dielectric boundary has to be modelled
39  by espresso particles 0..n where n has to be passed as a parameter
40  to ICCP3M. This is still a bit inconvenient, as it forces the user
41  to reserve the first n particle ids to wall charges, but as the
42  other parts of espresso do not suffer from a limitation like this,
43  it can be tolerated.
44 
45  For the determination of the induced charges only the forces
46  acting on the induced charges has to be determined. As P3M an the
47  other coulomb solvers calculate all mutual forces, the force
48  calculation was modified to avoid the calculation of the short
49  range part of the source-source force calculation. For different
50  particle data organisation schemes this is performed differently.
51  */

145 /** Variant of add_non_bonded_pair_force where only coulomb
146  * contributions are calculated */
147 inline void add_non_bonded_pair_force_iccp3m(Particle *p1, Particle *p2,
148  double d[3], double dist, double dist2)
149 {
150  /* IA_parameters *ia_params = get_ia_param(p1->p.type,p2->p.type);*/
151  double force[3] = { 0, 0, 0 };
152  int j;
153 
154  FORCE_TRACE(fprintf(stderr, "%d: interaction %d<->%d dist %f\n", this_node, p1->p.identity, p2->p.identity, dist));
155 
156  /***********************************************/
157  /* long range electrostatics */
158  /***********************************************/
159 
160  /* real space coulomb */
161  double q1q2 = p1->p.q*p2->p.q;
162  switch (coulomb.method) {
163 #ifdef P3M
164  case COULOMB_ELC_P3M:
165  if (q1q2) p3m_add_pair_force(q1q2,d,dist2,dist,force);
166  break;
167  case COULOMB_P3M_GPU:
168  case COULOMB_P3M:
169  if (q1q2) p3m_add_pair_force(q1q2,d,dist2,dist,force);
170  break;
171 #endif /* P3M */
172  case COULOMB_MMM1D:
173  if (q1q2) add_mmm1d_coulomb_pair_force(q1q2,d,dist2,dist,force);
174  break;
175  case COULOMB_MMM2D:
176  if (q1q2) add_mmm2d_coulomb_pair_force(q1q2,d,dist2,dist,force);
177  break;
178  default:
179  break;
180  }
181 
182  /***********************************************/
183  /* add total nonbonded forces to particle */
184  /***********************************************/
185  for (j = 0; j < 3; j++) {
186  p1->f.f[j] += force[j];
187  p2->f.f[j] -= force[j];
188  }
189  /***********************************************/
190 }
191 #endif /* ELECTROSTATICS */
