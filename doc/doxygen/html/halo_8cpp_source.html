228 /** Preparation of the halo parallelization scheme. Sets up the
229  * necessary datastructures for \ref halo_communication
230  * @param hc halo communicator beeing created (Input/Output)

235 void prepare_halo_communication(HaloCommunicator *hc, Lattice *lattice, Fieldtype fieldtype, MPI_Datatype datatype) {
236  int k, n, dir, lr, cnt, num = 0 ;
237  int *grid = lattice->grid ;
238  int *period = lattice->halo_grid ;
239 
240  for (n=0; n<hc->num; n++) {
241  MPI_Type_free(&(hc->halo_info[n].datatype));
242  }
243 
244  num = 2*3; /* two communications in each space direction */
245 
246  hc->num = num ;
247  hc->halo_info = (HaloInfo*) realloc(hc->halo_info,num*sizeof(HaloInfo)) ;
248 
249  int extent = fieldtype->extent;
250 
251  cnt = 0 ;
252  for (dir=0; dir<3; dir++) {
253  for (lr=0; lr<2; lr++) {
254 
255  HaloInfo *hinfo = &(hc->halo_info[cnt]) ;
256 
257  int nblocks = 1 ;
258  for (k=dir+1;k<3;k++) {
259  nblocks *= period[k] ;
260  }
261  int stride = 1 ;
262  for (k=0;k<dir;k++) {
263  stride *= period[k] ;
264  }
265  int skip = 1 ;
266  for (k=0;k<dir+1 && k<2;k++) {
267  skip *= period[k] ;
268  }
269 
270  if (lr==0) {
271  /* send to left, recv from right */
272  hinfo->s_offset = extent * stride * 1;
273  hinfo->r_offset = extent * stride * (grid[dir]+1);
274  } else {
275  /* send to right, recv from left */
276  hinfo->s_offset = extent * stride * grid[dir];
277  hinfo->r_offset = extent * stride * 0;
278 
279  }
280 
281  hinfo->source_node = node_neighbors[2*dir+1-lr];
282  hinfo->dest_node = node_neighbors[2*dir+lr];
283 
284  halo_create_field_vector(nblocks, stride, skip, fieldtype, &hinfo->fieldtype);
285 
286  MPI_Type_vector(nblocks, stride, skip, datatype, &hinfo->datatype);
287  MPI_Type_commit(&hinfo->datatype);
288 
289 #ifdef PARTIAL_PERIODIC
290  if ( !PERIODIC(dir) && (boundary[2*dir+lr] != 0 || boundary[2*dir+1-lr] != 0) ) {
291  if (node_grid[dir] == 1) {
292  hinfo->type = HALO_OPEN;
293  }
294  else if (lr == 0) {
295  if (boundary[2*dir+lr] == 1) {
296  hinfo->type = HALO_RECV;
297  } else {
298  hinfo->type = HALO_SEND;
299  }
300  }
301  else {
302  if (boundary[2*dir+lr] == -1) {
303  hinfo->type = HALO_RECV;
304  } else {
305  hinfo->type = HALO_SEND;
306  }
307  }
308  } else
309 #endif
310  {
311  if (node_grid[dir] == 1) {
312  hc->halo_info[cnt].type = HALO_LOCL;
313  }
314  else {
315  hc->halo_info[cnt].type = HALO_SENDRECV;
316  }
317  }
318 
319  HALO_TRACE(fprintf(stderr,"%d: prepare_halo_communication dir=%d lr=%d s_offset=%ld r_offset=%ld s_node=%d d_node=%d type=%d\n",this_node,dir,lr,hinfo->s_offset,hinfo->r_offset,hinfo->source_node,hinfo->dest_node,hinfo->type)) ;
320 
321  cnt++;
322 
323  }
324  }

342 /** Perform communication according to the parallelization scheme
343  * described by the halo communicator
344  * @param hc halo communicator describing the parallelization scheme
345  * @param base base plane of local node
346  */
347 void halo_communication(HaloCommunicator *hc, char *base) {
348  int n, comm_type, s_node, r_node;
349  char *s_buffer, *r_buffer ;
350 
351  Fieldtype fieldtype;
352  MPI_Datatype datatype;
353  MPI_Request request;
354  MPI_Status status;
355 
356  HALO_TRACE(fprintf(stderr, "%d: halo_comm base=%p num=%d\n", this_node, base, hc->num)) ;
357 
358  for (n = 0; n < hc->num; n++) {
359 
360  HALO_TRACE(fprintf(stderr, "%d: halo_comm round %d\n", this_node, n)) ;
361 
362  comm_type = hc->halo_info[n].type ;
363  s_buffer = (char *)base + hc->halo_info[n].s_offset;
364  r_buffer = (char *)base + hc->halo_info[n].r_offset;
365 
366  switch (comm_type) {
367 
368  case HALO_LOCL:
369  fieldtype = hc->halo_info[n].fieldtype;
370  halo_dtcopy(r_buffer,s_buffer,1,fieldtype);
371  break ;
372 
373  case HALO_SENDRECV:
374  datatype = hc->halo_info[n].datatype;
375  s_node = hc->halo_info[n].source_node ;
376  r_node = hc->halo_info[n].dest_node ;
377 
378  HALO_TRACE(fprintf(stderr,"%d: halo_comm sendrecv %d to %d (%d) (%p)\n",this_node,s_node,r_node,REQ_HALO_SPREAD,&datatype));
379 
380  MPI_Sendrecv(s_buffer, 1, datatype, r_node, REQ_HALO_SPREAD,
381  r_buffer, 1, datatype, s_node, REQ_HALO_SPREAD,
382  comm_cart, &status);
383  break ;
384 
385  case HALO_SEND:
386  datatype = hc->halo_info[n].datatype;
387  fieldtype = hc->halo_info[n].fieldtype;
388  s_node = hc->halo_info[n].source_node ;
389  r_node = hc->halo_info[n].dest_node ;
390 
391  HALO_TRACE(fprintf(stderr,"%d: halo_comm send to %d.\n",this_node,r_node));
392 
393  MPI_Isend(s_buffer, 1, datatype, r_node, REQ_HALO_SPREAD, comm_cart, &request);
394  halo_dtset(r_buffer,0,fieldtype);
395  MPI_Wait(&request,&status);
396  break;
397 
398  case HALO_RECV:
399  datatype = hc->halo_info[n].datatype;
400  s_node = hc->halo_info[n].source_node ;
401  r_node = hc->halo_info[n].dest_node ;
402 
403  HALO_TRACE(fprintf(stderr,"%d: halo_comm recv from %d.\n",this_node,s_node));
404 
405  MPI_Irecv(r_buffer, 1, datatype, s_node, REQ_HALO_SPREAD, comm_cart, &request);
406  MPI_Wait(&request,&status);
407  break;
408 
409  case HALO_OPEN:
410  fieldtype = hc->halo_info[n].fieldtype;
411 
412  HALO_TRACE(fprintf(stderr,"%d: halo_comm open boundaries\n",this_node));
413 
414  /* \todo this does not work for the n_i - <n_i> */
415  halo_dtset(r_buffer,0,fieldtype);
416  break;
417 
418  }
419 
420  }
