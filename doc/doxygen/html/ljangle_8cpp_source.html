22 /** \file ljangle.hpp
23  * Routines to calculate the lennard-jones 12-10 with angular dependance.
24  * The potential is a product of a 12-10 LJ potential with two cos^2.
25  * The potential actually relies on 6 particles: the 2 primary beads
26  * and each bead needs two other particles to define an orientation.
27  * We calculate the interaction explicitly *without* the use of the ROTATION feature.
28  *
29  * Optional: simulate two different environments in which the interaction
30  * strengths differ. For example: simulate hydrogen-bonds both in water and
31  * inside a bilayer. The two environments are distinguished by their
32  * z-position. Input: the midplane of the 2nd environment, its total
33  * thickness, the thickness of the interface, and the value of the
34  * interaction strength in this medium. The interaction strengh of the second
35  * environment must be *stronger* than of the first one.
36  *
37  * \ref forces.cpp
38  */

87 /** calculate ljangle_capradius from force_cap */
88 /* This routine does not take the optional 2nd environment into account. */
89 void calc_ljangle_cap_radii()
90 {
91  if( force_cap != -1.0){
92  int i,j,cnt=0;
93  IA_parameters *params;
94  double force=0.0, rad=0.0, step, frac2, frac10;
95 
96  for(i=0; i<n_particle_types; i++) {
97  for(j=0; j<n_particle_types; j++) {
98  params = get_ia_param(i,j);
99  if(force_cap > 0.0 && params->LJANGLE_eps > 0.0) {
100  /* I think we have to solve this numerically... and very crude as well */
101  cnt=0;
102  rad = params->LJANGLE_sig;
103  step = -0.1 * params->LJANGLE_sig;
104  force=0.0;
105 
106  while(step != 0) {
107  frac2 = SQR(params->LJANGLE_sig/rad);
108  frac10 = frac2*frac2*frac2*frac2*frac2;
109  force = 60.0 * params->LJANGLE_eps * frac10*(frac2 - 1.0) / rad;
110  if((step < 0 && force_cap < force) || (step > 0 && force_cap > force)) {
111  step = - (step/2.0);
112  }
113  if(fabs(force-force_cap) < 1.0e-6) step=0;
114  rad += step; cnt++;
115  }
116  params->LJANGLE_capradius = rad;
117  }
118  else {
119  params->LJANGLE_capradius = 0.0;
120  }
121  FORCE_TRACE(fprintf(stderr,"%d: LJANGLE Ptypes %d-%d have cap_radius %f and cap_force %f (iterations: %d)\n",
122  this_node,i,j,rad,force,cnt));
123  }
124  }
125  }
126 }
