38 /** Computes the local area force (Dupin2007 eqn. 15) and adds this
39  force to the particle forces (see \ref tclcommand_inter).
40  @param p1,p2,p3 Pointers to triangle particles.
41  @param iaparams elastic area modulus ka, initial area A0 (see \ref tclcommand_inter).

45  @return 0

58  //unfold non-ghost particle using image, because for physical particles, the structure p->l.i is correctly set
59  memmove(p11, p1->r.p, 3*sizeof(double));

64  get_mi_vector(BB, p3->r.p, p11);
65  for (int i=0; i < 3; i++) { p22[i] = p11[i] + AA[i]; p33[i] = p11[i] + BB[i]; }
66  } else {
67  // in case the first particle is a ghost particle
68  if (p2->l.ghost != 1) {

75  } else {
