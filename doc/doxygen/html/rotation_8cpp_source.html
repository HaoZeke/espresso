74 /** convert quaternions to the director */
75 /** Convert director to quaternions */
76 int convert_quatu_to_quat(double d[3], double quat[4])
77 {
78  double d_xy, dm;
79  double theta2, phi2;
80 
81  // Calculate magnitude of the given vector
82  dm = sqrt(d[0]*d[0] + d[1]*d[1] + d[2]*d[2]);
83 
84  // The vector needs to be != 0 to be converted into a quaternion
85  if (dm < ROUND_ERROR_PREC) {
86  return 1;
87  }
88  else {
89  // Calculate angles
90  d_xy = sqrt(d[0]*d[0] + d[1]*d[1]);
91  // If dipole points along z axis:
92  if (d_xy == 0){
93  // We need to distinguish between (0,0,d_z) and (0,0,d_z)
94  if (d[2]>0)
95  theta2 = 0;
96  else
97  theta2 = PI/2.;
98  phi2 = 0;
99  }
100  else {
101  // Here, we take care of all other directions
102  //Here we suppose that theta2 = 0.5*theta and phi2 = 0.5*(phi - PI/2),
103  //where theta and phi - angles are in spherical coordinates
104  theta2 = 0.5*acos(d[2]/dm);
105  if (d[1] < 0) phi2 = -0.5*acos(d[0]/d_xy) - PI*0.25;
106  else phi2 = 0.5*acos(d[0]/d_xy) - PI*0.25;
107  }
108 
109  // Calculate the quaternion from the angles
110  quat[0] = cos(theta2) * cos(phi2);
111  quat[1] = -sin(theta2) * cos(phi2);
112  quat[2] = -sin(theta2) * sin(phi2);
113  quat[3] = cos(theta2) * sin(phi2);
114  }
115  return 0;
116 }

118 /** Here we use quaternions to calculate the rotation matrix which
119  will be used then to transform torques from the laboratory to
120  the body-fixed frames */
121 void define_rotation_matrix(Particle *p, double A[9])
122 {
123  double q0q0 =p->r.quat[0];
124  q0q0 *=q0q0;
125 
126  double q1q1 =p->r.quat[1];
127  q1q1 *=q1q1;
128 
129  double q2q2 =p->r.quat[2];
130  q2q2 *=q2q2;
131 
132  double q3q3 =p->r.quat[3];
133  q3q3 *=q3q3;
134 
135  A[0 + 3*0] = q0q0 + q1q1 - q2q2 - q3q3;
136  A[1 + 3*1] = q0q0 - q1q1 + q2q2 - q3q3;
137  A[2 + 3*2] = q0q0 - q1q1 - q2q2 + q3q3;
138 
139  A[0 + 3*1] = 2*(p->r.quat[1]*p->r.quat[2] + p->r.quat[0]*p->r.quat[3]);
140  A[0 + 3*2] = 2*(p->r.quat[1]*p->r.quat[3] - p->r.quat[0]*p->r.quat[2]);
141  A[1 + 3*0] = 2*(p->r.quat[1]*p->r.quat[2] - p->r.quat[0]*p->r.quat[3]);
142 
143  A[1 + 3*2] = 2*(p->r.quat[2]*p->r.quat[3] + p->r.quat[0]*p->r.quat[1]);
144  A[2 + 3*0] = 2*(p->r.quat[1]*p->r.quat[3] + p->r.quat[0]*p->r.quat[2]);
145  A[2 + 3*1] = 2*(p->r.quat[2]*p->r.quat[3] - p->r.quat[0]*p->r.quat[1]);
146 }

148 /** calculate the second derivative of the quaternion of a given particle
149  as well as Wd vector which is the angular acceleration of this particle */
150 void define_Qdd(Particle *p, double Qd[4], double Qdd[4], double S[3], double Wd[3])
151 {
152  double S1;
153 
154  /* calculate the first derivative of the quaternion */
155  Qd[0] = 0.5 * ( -p->r.quat[1] * p->m.omega[0] -
156  p->r.quat[2] * p->m.omega[1] -
157  p->r.quat[3] * p->m.omega[2] );
158 
159  Qd[1] = 0.5 * ( p->r.quat[0] * p->m.omega[0] -
160  p->r.quat[3] * p->m.omega[1] +
161  p->r.quat[2] * p->m.omega[2] );
162 
163  Qd[2] = 0.5 * ( p->r.quat[3] * p->m.omega[0] +
164  p->r.quat[0] * p->m.omega[1] -
165  p->r.quat[1] * p->m.omega[2] );
166 
167  Qd[3] = 0.5 * ( -p->r.quat[2] * p->m.omega[0] +
168  p->r.quat[1] * p->m.omega[1] +
169  p->r.quat[0] * p->m.omega[2] );
170 
171  /* calculate the second derivative of the quaternion */
172 
173 #ifdef ROTATIONAL_INERTIA
174  Wd[0] = (p->f.torque[0] + p->m.omega[1]*p->m.omega[2]*(I[1]-I[2]))/I[0]/p->p.rinertia[0];
175  Wd[1] = (p->f.torque[1] + p->m.omega[2]*p->m.omega[0]*(I[2]-I[0]))/I[1]/p->p.rinertia[1];
176  Wd[2] = (p->f.torque[2] + p->m.omega[0]*p->m.omega[1]*(I[0]-I[1]))/I[2]/p->p.rinertia[2];
177 #else
178  Wd[0] = (p->f.torque[0] + p->m.omega[1]*p->m.omega[2]*(I[1]-I[2]))/I[0];
179  Wd[1] = (p->f.torque[1] + p->m.omega[2]*p->m.omega[0]*(I[2]-I[0]))/I[1];
180  Wd[2] = (p->f.torque[2] + p->m.omega[0]*p->m.omega[1]*(I[0]-I[1]))/I[2];
181 #endif
182 
183  S1 = Qd[0]*Qd[0] + Qd[1]*Qd[1] + Qd[2]*Qd[2] + Qd[3]*Qd[3];
184 
185  Qdd[0] = 0.5 * ( -p->r.quat[1] * Wd[0] -
186  p->r.quat[2] * Wd[1] -
187  p->r.quat[3] * Wd[2] ) - p->r.quat[0] * S1;
188 
189  Qdd[1] = 0.5 * ( p->r.quat[0] * Wd[0] -
190  p->r.quat[3] * Wd[1] +
191  p->r.quat[2] * Wd[2] ) - p->r.quat[1] * S1;
192 
193  Qdd[2] = 0.5 * ( p->r.quat[3] * Wd[0] +
194  p->r.quat[0] * Wd[1] -
195  p->r.quat[1] * Wd[2] ) - p->r.quat[2] * S1;
196 
197  Qdd[3] = 0.5 * ( -p->r.quat[2] * Wd[0] +
198  p->r.quat[1] * Wd[1] +
199  p->r.quat[0] * Wd[2] ) - p->r.quat[3] * S1;
200 
201  S[0] = S1;
202  S[1] = Qd[0]*Qdd[0] + Qd[1]*Qdd[1] + Qd[2]*Qdd[2] + Qd[3]*Qdd[3];
203  S[2] = Qdd[0]*Qdd[0] + Qdd[1]*Qdd[1] + Qdd[2]*Qdd[2] + Qdd[3]*Qdd[3];
204 }

206 /** propagate angular velocities and quaternions \todo implement for
207  fixed_coord_flag */
208 void propagate_omega_quat_particle(Particle* p)
209 {
210  double lambda;
211 
212  double Qd[4], Qdd[4], S[3], Wd[3];
213 #ifdef ROTATION_PER_PARTICLE
214  if (!p->p.rotation) return;
215 #endif
216 
217  define_Qdd(p, Qd, Qdd, S, Wd);
218 
219  lambda = 1 - S[0]*time_step_squared_half - sqrt(1 - time_step_squared*(S[0] + time_step*(S[1] + time_step_half/2.*(S[2]-S[0]*S[0]))));
220 
221  for(int j=0; j < 3; j++){
222  p->m.omega[j]+= time_step_half*Wd[j];
223  }
224  ONEPART_TRACE(if(p->p.identity==check_id) fprintf(stderr,"%d: OPT: PV_1 v_new = (%.3e,%.3e,%.3e)\n",this_node,p->m.v[0],p->m.v[1],p->m.v[2]));
225 
226  p->r.quat[0]+= time_step*(Qd[0] + time_step_half*Qdd[0]) - lambda*p->r.quat[0];
227  p->r.quat[1]+= time_step*(Qd[1] + time_step_half*Qdd[1]) - lambda*p->r.quat[1];
228  p->r.quat[2]+= time_step*(Qd[2] + time_step_half*Qdd[2]) - lambda*p->r.quat[2];
229  p->r.quat[3]+= time_step*(Qd[3] + time_step_half*Qdd[3]) - lambda*p->r.quat[3];
230  // Update the director
231  convert_quat_to_quatu(p->r.quat, p->r.quatu);
232 #ifdef DIPOLES
233  // When dipoles are enabled, update dipole moment
234  convert_quatu_to_dip(p->r.quatu, p->p.dipm, p->r.dip);
235 #endif
236 
237 
238  ONEPART_TRACE(if(p->p.identity==check_id) fprintf(stderr,"%d: OPT: PPOS p = (%.3f,%.3f,%.3f)\n",this_node,p->r.p[0],p->r.p[1],p->r.p[2]));

241 /** convert the torques to the body-fixed frames and propagate angular velocities */
242 void convert_torques_propagate_omega()
243 {
244  Particle *p;
245  Cell *cell;
246  int np;
247  double tx, ty, tz;
248 
249  INTEG_TRACE(fprintf(stderr,"%d: convert_torques_propagate_omega:\n",this_node));
250 
251 #if defined(LB_GPU) && defined(ENGINE)
252  if (lattice_switch & LATTICE_LB_GPU) {
253  copy_v_cs_from_GPU();
254  }
255 #endif
256 
257  for (int c = 0; c < local_cells.n; c++)
258  {
259  cell = local_cells.cell[c];
260  p = cell->part;
261  np = cell->n;
262  for(int i = 0; i < np; i++)
263  {
264 #ifdef ROTATION_PER_PARTICLE
265  if (!p[i].p.rotation)
266  continue;
267 #endif
268  double A[9];
269  define_rotation_matrix(&p[i], A);
270 
271  tx = A[0 + 3*0]*p[i].f.torque[0] + A[0 + 3*1]*p[i].f.torque[1] + A[0 + 3*2]*p[i].f.torque[2];
272  ty = A[1 + 3*0]*p[i].f.torque[0] + A[1 + 3*1]*p[i].f.torque[1] + A[1 + 3*2]*p[i].f.torque[2];
273  tz = A[2 + 3*0]*p[i].f.torque[0] + A[2 + 3*1]*p[i].f.torque[1] + A[2 + 3*2]*p[i].f.torque[2];
274 
275  if ( thermo_switch & THERMO_LANGEVIN )
276  {
277 #if defined (VIRTUAL_SITES) && defined(THERMOSTAT_IGNORE_NON_VIRTUAL)
278  if (!ifParticleIsVirtual(&p[i]))
279 #endif
280  {
281  friction_thermo_langevin_rotation(&p[i]);
282 
283  p[i].f.torque[0]+= tx;
284  p[i].f.torque[1]+= ty;
285  p[i].f.torque[2]+= tz;
286  }
287  }
288  else
289  {
290  p[i].f.torque[0] = tx;
291  p[i].f.torque[1] = ty;
292  p[i].f.torque[2] = tz;
293  }
294 
295 #if defined(ENGINE) && (defined(LB) || defined(LB_GPU))
296  double omega_swim[3] = {0, 0, 0};
297  double omega_swim_body[3] = {0, 0, 0};
298  if ( p[i].swim.swimming && lattice_switch != 0 )
299  {
300  double dip[3];
301  double diff[3];
302  double cross[3];
303  double l_diff, l_cross;
304 
305  dip[0] = p[i].swim.dipole_length * p[i].r.quatu[0];
306  dip[1] = p[i].swim.dipole_length * p[i].r.quatu[1];
307  dip[2] = p[i].swim.dipole_length * p[i].r.quatu[2];
308 
309  diff[0] = ( p[i].swim.v_center[0] - p[i].swim.v_source[0] );
310  diff[1] = ( p[i].swim.v_center[1] - p[i].swim.v_source[1] );
311  diff[2] = ( p[i].swim.v_center[2] - p[i].swim.v_source[2] );
312 
313  cross[0] = diff[1]*dip[2] - diff[2]*dip[1];
314  cross[1] = diff[0]*dip[2] - diff[2]*dip[0];
315  cross[2] = diff[0]*dip[1] - diff[1]*dip[0];
316 
317  l_diff = sqrt(diff[0]*diff[0] + diff[1]*diff[1] + diff[2]*diff[2]);
318  l_cross = sqrt(cross[0]*cross[0] + cross[1]*cross[1] + cross[2]*cross[2]);
319 
320  if ( l_cross > 0 && p[i].swim.dipole_length > 0 )
321  {
322  omega_swim[0] = l_diff * cross[0] / ( l_cross * p[i].swim.dipole_length );
323  omega_swim[1] = l_diff * cross[1] / ( l_cross * p[i].swim.dipole_length );
324  omega_swim[2] = l_diff * cross[2] / ( l_cross * p[i].swim.dipole_length );
325 
326  omega_swim_body[0] = A[0 + 3*0]*omega_swim[0] + A[0 + 3*1]*omega_swim[1] + A[0 + 3*2]*omega_swim[2];
327  omega_swim_body[1] = A[1 + 3*0]*omega_swim[0] + A[1 + 3*1]*omega_swim[1] + A[1 + 3*2]*omega_swim[2];
328  omega_swim_body[2] = A[2 + 3*0]*omega_swim[0] + A[2 + 3*1]*omega_swim[1] + A[2 + 3*2]*omega_swim[2];
329 
330  p[i].f.torque[0] += p[i].swim.rotational_friction * ( omega_swim_body[0] - p[i].m.omega[0] );
331  p[i].f.torque[1] += p[i].swim.rotational_friction * ( omega_swim_body[1] - p[i].m.omega[1] );
332  p[i].f.torque[2] += p[i].swim.rotational_friction * ( omega_swim_body[2] - p[i].m.omega[2] );
333  }
334  }
335 #endif
336 
337  ONEPART_TRACE(if(p[i].p.identity==check_id) fprintf(stderr,"%d: OPT: SCAL f = (%.3e,%.3e,%.3e) v_old = (%.3e,%.3e,%.3e)\n",this_node,p[i].f.f[0],p[i].f.f[1],p[i].f.f[2],p[i].m.v[0],p[i].m.v[1],p[i].m.v[2]));
338 
339 #ifdef ROTATIONAL_INERTIA
340  p[i].m.omega[0]+= time_step_half*p[i].f.torque[0]/p[i].p.rinertia[0]/I[0];
341  p[i].m.omega[1]+= time_step_half*p[i].f.torque[1]/p[i].p.rinertia[1]/I[1];
342  p[i].m.omega[2]+= time_step_half*p[i].f.torque[2]/p[i].p.rinertia[2]/I[2];
343 #else
344  p[i].m.omega[0]+= time_step_half*p[i].f.torque[0]/I[0];
345  p[i].m.omega[1]+= time_step_half*p[i].f.torque[1]/I[1];
346  p[i].m.omega[2]+= time_step_half*p[i].f.torque[2]/I[2];
347 #endif
348  /* if the tensor of inertia is isotropic, the following refinement is not needed.
349  Otherwise repeat this loop 2-3 times depending on the required accuracy */
350  for(int times=0; times <= 5; times++)
351  {
352  double Wd[3];
353 
354 #ifdef ROTATIONAL_INERTIA
355  Wd[0] = (p[i].m.omega[1]*p[i].m.omega[2]*(I[1]-I[2]))/I[0]/p[i].p.rinertia[0];
356  Wd[1] = (p[i].m.omega[2]*p[i].m.omega[0]*(I[2]-I[0]))/I[1]/p[i].p.rinertia[1];
357  Wd[2] = (p[i].m.omega[0]*p[i].m.omega[1]*(I[0]-I[1]))/I[2]/p[i].p.rinertia[2];
358 #else
359  Wd[0] = (p[i].m.omega[1]*p[i].m.omega[2]*(I[1]-I[2]))/I[0];
360  Wd[1] = (p[i].m.omega[2]*p[i].m.omega[0]*(I[2]-I[0]))/I[1];
361  Wd[2] = (p[i].m.omega[0]*p[i].m.omega[1]*(I[0]-I[1]))/I[2];
362 #endif
363 
364  p[i].m.omega[0]+= time_step_half*Wd[0];
365  p[i].m.omega[1]+= time_step_half*Wd[1];
366  p[i].m.omega[2]+= time_step_half*Wd[2];
367  }
368 
369  ONEPART_TRACE(if(p[i].p.identity==check_id) fprintf(stderr,"%d: OPT: PV_2 v_new = (%.3e,%.3e,%.3e)\n",this_node,p[i].m.v[0],p[i].m.v[1],p[i].m.v[2]));
370 
371  }
372  }
