232 {
233  /* Prepare the Integrator */
234  on_integration_start();
235 
236  #ifdef IMMERSED_BOUNDARY
237  // Here we initialize volume conservation
238  // This function checks if the reference volumes have been set and if necessary calculates them
239  IBM_InitVolumeConservation();
240  #endif
241 
242  /* if any method vetoes (P3M not initialized), immediately bail out */
243  if (check_runtime_errors())
244  return;
245 
246 #ifdef MULTI_TIMESTEP
247  if (smaller_time_step > 0.){
248  mts_max = time_step/smaller_time_step;
249 #ifdef NPT
250  if (integ_switch == INTEG_METHOD_NPT_ISO) {
251  current_time_step_is_small = 1;
252  // Compute forces for small timestep -> get virial contribution.
253  if (recalc_forces)
254  thermo_heat_up();
255  force_calc();
256  thermo_cool_down();
257  ghost_communicator(&cell_structure.collect_ghost_force_comm);
258  current_time_step_is_small = 0;
259  // Store virial
260  for(int j=0;j<3;++j)
261  virial_store[j] = nptiso.p_vir[j];
262  thermo_heat_up();
263  force_calc();
264  thermo_cool_down();
265  ghost_communicator(&cell_structure.collect_ghost_force_comm);
266  rescale_forces();
267  }
268 #endif
269  }
270 #endif
271 
272  /* Verlet list criterion */
273  skin2 = SQR(0.5 * skin);
274 
275  INTEG_TRACE(fprintf(stderr,"%d: integrate_vv: integrating %d steps (recalc_forces=%d)\n",
276  this_node, n_steps, recalc_forces));
277 
278  /* Integration Step: Preparation for first integration step:
279  Calculate forces f(t) as function of positions p(t) ( and velocities v(t) ) */
280  /* reuse_forces logic:
281  -1: recalculate forces unconditionally, mostly used for timing
282  0: recalculate forces if recalc_forces is set, meaning it is probably necessary
283  1: do not recalculate forces. Mostly when reading checkpoints with forces
284  */
285  if (reuse_forces == -1 || (recalc_forces && reuse_forces != 1)) {
286  thermo_heat_up();
287 
288 #ifdef LB
289  transfer_momentum = 0;
290  if (lattice_switch & LATTICE_LB && this_node == 0)
291  if (warnings) fprintf (stderr, "Warning: Recalculating forces, so the LB coupling forces are not included in the particle force the first time step. This only matters if it happens frequently during sampling.\n");
292 #endif
293 #ifdef LB_GPU
294  transfer_momentum_gpu = 0;
295  if (lattice_switch & LATTICE_LB_GPU && this_node == 0)
296  if (warnings) fprintf (stderr, "Warning: Recalculating forces, so the LB coupling forces are not included in the particle force the first time step. This only matters if it happens frequently during sampling.\n");
297 #endif
298 
299  force_calc();
300 
301  if(integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
302  rescale_forces();
303 #ifdef ROTATION
304  convert_initial_torques();
305 #endif
306  }
307 
308  thermo_cool_down();
309 
310 #ifdef MULTI_TIMESTEP
311 #ifdef NPT
312  if (smaller_time_step > 0. && integ_switch == INTEG_METHOD_NPT_ISO)
313  for(j=0;j<3;++j)
314  nptiso.p_vir[j] += virial_store[j];
315 #endif
316 #endif
317 
318 #ifdef COLLISION_DETECTION
319  handle_collisions();
320 #endif
321  }
322 
323 #ifdef GHMC
324  if(thermo_switch & THERMO_GHMC)
325  ghmc_init();
326 #endif
327 
328  if (thermo_switch & THERMO_CPU)
329  mpi_thermalize_cpu(temperature);
330 
331  if (check_runtime_errors())
332  return;
333 
334  n_verlet_updates = 0;
335 
336  /* Integration loop */
337  for (int step=0; step<n_steps; step++) {
338  INTEG_TRACE(fprintf(stderr,"%d: STEP %d\n", this_node, step));
339 
340 #ifdef BOND_CONSTRAINT
341  save_old_pos();
342 #endif
343 
344 #ifdef GHMC
345  if(thermo_switch & THERMO_GHMC) {
346  if (step % ghmc_nmd == 0)
347  ghmc_momentum_update();
348  }
349 #endif
350

356  fprintf(stderr,"Warning: Use integrate_sd to use Brownian Dynamics Thermalizer.");
357  }
358 #endif
359 
360  /* Integration Steps: Step 1 and 2 of Velocity Verlet scheme:
361  v(t+0.5*dt) = v(t) + 0.5*dt * f(t)
362  p(t + dt) = p(t) + dt * v(t+0.5*dt)
363  NOTE 1: Prefactors do not occur in formulas since we use
364  rescaled forces and velocities.
365  NOTE 2: Depending on the integration method Step 1 and Step 2
366  cannot be combined for the translation.
367  */
368  if (integ_switch == INTEG_METHOD_NPT_ISO || nemd_method != NEMD_METHOD_OFF) {
369  propagate_vel();
370  propagate_pos();
371  } else if(integ_switch == INTEG_METHOD_STEEPEST_DESCENT) {
372  if(steepest_descent_step())
373  break;
374  } else {
375  propagate_vel_pos();
376  }
377 
378 #ifdef BOND_CONSTRAINT
379  /**Correct those particle positions that participate in a rigid/constrained bond */
380  cells_update_ghosts();
381 
382  correct_pos_shake();
383 #endif
384 
385 #ifdef ELECTROSTATICS
386  if(coulomb.method == COULOMB_MAGGS) {
387  maggs_propagate_B_field(0.5*time_step);
388  }
389 #endif
390 
391 #ifdef NPT
392  if (check_runtime_errors())
393  break;
394 #endif
395 
396 #ifdef MULTI_TIMESTEP
397  if (smaller_time_step > 0){
398  current_time_step_is_small = 1;
399  /* Calculate the forces */
400  thermo_heat_up();
401  force_calc();
402  thermo_cool_down();
403  ghost_communicator(&cell_structure.collect_ghost_force_comm);
404  rescale_forces();
405  for (mts_index=0;mts_index<mts_max;++mts_index) {
406  /* Small integration steps */
407  /* Propagate velocities and positions */
408  /* Assumes: not NEMD_METHOD_OFF; NPT not updated during small steps */
409  if(integ_switch == INTEG_METHOD_NPT_ISO || nemd_method != NEMD_METHOD_OFF) {
410  propagate_vel(); propagate_pos(); }
411  else
412  propagate_vel_pos();
413  cells_update_ghosts();
414  force_calc();
415  ghost_communicator(&cell_structure.collect_ghost_force_comm);
416 #ifdef NPT
417  // Store virial
418  for(j=0;j<3;++j)
419  virial_store[j] = nptiso.p_vir[j];
420 #endif
421  rescale_forces_propagate_vel();
422  }
423  current_time_step_is_small = 0;
424  thermo_heat_up();
425  force_calc();
426  thermo_cool_down();
427  ghost_communicator(&cell_structure.collect_ghost_force_comm);
428  rescale_forces();
429  recalc_forces=0;
430  }
431 #endif
432 
433 
434  /* Integration Step: Step 3 of Velocity Verlet scheme:
435  Calculate f(t+dt) as function of positions p(t+dt) ( and velocities v(t+0.5*dt) ) */
436 
437 #ifdef LB
438  transfer_momentum = 1;
439 #endif
440 #ifdef LB_GPU
441  transfer_momentum_gpu = 1;
442 #endif
443 
444  force_calc();
445 
446 // IMMERSED_BOUNDARY
447 #ifdef IMMERSED_BOUNDARY
448  // Now the forces are computed and need to go into the LB fluid
449  if (lattice_switch & LATTICE_LB) IBM_ForcesIntoFluid_CPU();
450 #ifdef LB_GPU
451  if (lattice_switch & LATTICE_LB_GPU) IBM_ForcesIntoFluid_GPU();
452 #endif
453 #endif
454 
455 #ifdef CATALYTIC_REACTIONS
456  integrate_reaction();
457 #endif
458 
459  if (check_runtime_errors())
460  break;
461 
462 #ifdef MULTI_TIMESTEP
463 #ifdef NPT
464  if (smaller_time_step > 0. && integ_switch == INTEG_METHOD_NPT_ISO)
465  for(j=0;j<3;++j)
466  nptiso.p_vir[j] += virial_store[j];
467 #endif
468 #endif
469  /* Integration Step: Step 4 of Velocity Verlet scheme:
470  v(t+dt) = v(t+0.5*dt) + 0.5*dt * f(t+dt) */
471  if(integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
472  rescale_forces_propagate_vel();
473 #ifdef ROTATION
474  convert_torques_propagate_omega();
475 #endif
476  }
477  // SHAKE velocity updates
478 #ifdef BOND_CONSTRAINT
479  ghost_communicator(&cell_structure.update_ghost_pos_comm);
480  correct_vel_shake();
481 #endif
482  //VIRTUAL_SITES update vel
483 #ifdef VIRTUAL_SITES
484  ghost_communicator(&cell_structure.update_ghost_pos_comm);
485  update_mol_vel();
486  if (check_runtime_errors()) break;
487 #endif
488 
489  // progagate one-step functionalities
490 #ifdef LB
491  if (lattice_switch & LATTICE_LB)
492  lattice_boltzmann_update();
493 
494  if (check_runtime_errors())
495  break;
496 #endif
497 
498 #ifdef LB_GPU
499  if(this_node == 0){
500 #ifdef ELECTROKINETICS
501  if (ek_initialized) {
502  ek_integrate();
503  }
504  else {
505 #endif
506  if (lattice_switch & LATTICE_LB_GPU)
507  lattice_boltzmann_update_gpu();
508 #ifdef ELECTROKINETICS
509  }
510 #endif
511  }
512 #endif //LB_GPU
513 
514 // IMMERSED_BOUNDARY
515 #ifdef IMMERSED_BOUNDARY
516 
517  IBM_UpdateParticlePositions();
518  // We reset all since otherwise the halo nodes may not be reset
519  // NB: the normal Espresso reset is also done after applying the forces
520 // if (lattice_switch & LATTICE_LB) IBM_ResetLBForces_CPU();
521 #ifdef LB_GPU
522  //if (lattice_switch & LATTICE_LB_GPU) IBM_ResetLBForces_GPU();
523 #endif
524 
525  if (check_runtime_errors()) break;
526 
527  // Ghost positions are now out-of-date
528  // We should update.
529  // Actually we seem to get the same results whether we do this here or not, but it is safer to do it
530  ghost_communicator(&cell_structure.update_ghost_pos_comm);
531 
532 #endif // IMMERSED_BOUNDARY
533 
534 #ifdef ELECTROSTATICS
535  if(coulomb.method == COULOMB_MAGGS) {
536  maggs_propagate_B_field(0.5*time_step);
537  }
538 #endif
539 
540 #ifdef NPT
541  if((this_node==0) && (integ_switch == INTEG_METHOD_NPT_ISO))
542  nptiso.p_inst_av += nptiso.p_inst;
543 #endif
544 
545 #ifdef GHMC
546  if(thermo_switch & THERMO_GHMC) {
547  if (step % ghmc_nmd == ghmc_nmd-1)
548  ghmc_mc();
549  }
550 #endif
551 
552  if(integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
553  /* Propagate time: t = t+dt */
554  sim_time += time_step;
555  }
556  }
557 
558  /* verlet list statistics */
559  if(n_verlet_updates>0) verlet_reuse = n_steps/(double) n_verlet_updates;
560  else verlet_reuse = 0;
561 
562 #ifdef NPT
563  if(integ_switch == INTEG_METHOD_NPT_ISO) {
564  nptiso.invalidate_p_vel = 0;
565  MPI_Bcast(&nptiso.p_inst, 1, MPI_DOUBLE, 0, comm_cart);

569  MPI_Bcast(&nptiso.p_inst_av, 1, MPI_DOUBLE, 0, comm_cart);
570  }
571 #endif
572 
573 #ifdef GHMC
574  if(thermo_switch & THERMO_GHMC)
575  ghmc_close();
576 #endif
