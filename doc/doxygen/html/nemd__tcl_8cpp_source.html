60 /** Hand over nemd status information to tcl interpreter. */

62 {
63  char buffer[TCL_INTEGER_SPACE+TCL_DOUBLE_SPACE];
64  switch (nemd_method) {
65  case NEMD_METHOD_OFF:
66  Tcl_AppendResult(interp, "off", (char *)NULL);

69  case NEMD_METHOD_EXCHANGE:
70  sprintf(buffer, "%d", nemddata.n_slabs);

80  Tcl_AppendResult(interp, " ",buffer, (char *)NULL);

89 /** Set nemd method to exchange and set nemd parameters */
90 int tclcommand_nemd_parse_exchange(Tcl_Interp *interp, int argc, char **argv)
91 {
92  int n_slabs, n_exchange;
93  if (argc < 4) {
94  Tcl_AppendResult(interp, "wrong # args: ", (char *)NULL);
95  return tclcommand_nemd_print_usage(interp);
96  }
97  if ( !ARG_IS_I(2, n_slabs) || !ARG_IS_I(3, n_exchange) ) {
98  Tcl_AppendResult(interp, "wrong argument type: ", (char *)NULL);
99  return tclcommand_nemd_print_usage(interp);
100  }
101  /* parameter sanity */
102  if ( n_slabs<0 || n_slabs%2!=0 ) {
103  Tcl_AppendResult(interp, "nemd <n_slabs> must be non negative and even!",(char *)NULL);
104  return (TCL_ERROR);
105  }
106  if ( n_slabs > 0 && n_exchange < 0 ) {
107  Tcl_AppendResult(interp, "nemd <n_exchange> must be positive!",(char *)NULL);
108  return (TCL_ERROR);
109  }
110 
111  nemd_method = NEMD_METHOD_EXCHANGE;
112  nemd_init(n_slabs, n_exchange, 0.0);
113  return (TCL_OK);
114 }

139 /** Hand over velocity profile to tcl interpreter */
140 int tclcommand_nemd_parse_and_print_profile(Tcl_Interp *interp)
141 {
142  int i;
143  double val;
144  char buffer[TCL_DOUBLE_SPACE];
145 
146  INTEG_TRACE(fprintf(stderr,"%d: tclcommand_nemd_parse_and_print_profile:\n",this_node));
147  if(nemd_method == NEMD_METHOD_OFF) {
148  Tcl_AppendResult(interp, "nemd is off", (char *)NULL);
149  return (TCL_OK);
150  }
151 
152  for(i=0;i<nemddata.n_slabs;i++) {
153  /* note: output velocities as usual have to be resacled by 1/time_step! */
154  val = nemddata.velocity_profile[i]/(nemddata.profile_norm*time_step);
155  Tcl_PrintDouble(interp, val, buffer);
156  Tcl_AppendResult(interp," ", buffer, (char *)NULL);
157 
158  nemddata.velocity_profile[i] = 0.0;
159  }
160 
161  nemddata.profile_norm = 0;
162  return (TCL_OK);
163 }

165 /** Gives back the calculated viscosity */
166 int tclcommand_nemd_parse_and_print_viscosity(Tcl_Interp *interp)
167 {
168  double shear_rate=0.0, mean_force, viscosity;
169  char buffer[TCL_DOUBLE_SPACE];
170  INTEG_TRACE(fprintf(stderr,"%d: tclcommand_nemd_parse_and_print_viscosity:\n",this_node));
171 
172  /* calculate shear_rate */
173  switch (nemd_method) {
174  case NEMD_METHOD_OFF:
175  Tcl_AppendResult(interp, "nemd is off", (char *)NULL);
176  return (TCL_OK);
177  case NEMD_METHOD_EXCHANGE:
178  shear_rate = 1.0;
179  case NEMD_METHOD_SHEARRATE:
180  shear_rate = nemddata.shear_rate;
181  }
182  /* rescale momentum exchange (vel_internal = time_step * vel) */
183  nemddata.momentum /= time_step;
184  /* Calculate average Force := Momentum transfer per time unit */
185  mean_force = nemddata.momentum / (nemddata.momentum_norm*time_step);
186  /* Calculate viscosity := mean_force / (shearrate * area) */
187  viscosity = mean_force / (shear_rate*4.0*box_l[0]*box_l[1]);
188  Tcl_PrintDouble(interp, viscosity, buffer);
189  Tcl_AppendResult(interp,buffer, (char *)NULL);
190 
191  nemddata.momentum = 0.0;
192  nemddata.momentum_norm = 0;
193  return (TCL_OK);
194 }
