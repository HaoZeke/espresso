51 inline void add_ljgen_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
52  double d[3], double dist, double force[3])
53 {
54  if (CUTOFF_CHECK(dist < ia_params->LJGEN_cut+ia_params->LJGEN_offset)) {
55  int j;
56  double r_off, frac, fac=0.0;
57  r_off = dist - ia_params->LJGEN_offset;
58 
59 #ifdef LJGEN_SOFTCORE
60  r_off *= r_off;
61  r_off += pow(ia_params->LJGEN_sig,2) * (1.0-ia_params->LJGEN_lambda)
62  * ia_params->LJGEN_softrad;
63  /* Taking a square root is not optimal, but we can't prevent the user from
64  using an odd m, n coefficient. */
65  r_off = sqrt(r_off);
66 #endif
67  /* normal case: resulting force/energy smaller than capping. */
68  if((ia_params->LJGEN_capradius == 0) || (r_off > ia_params->LJGEN_capradius)) {
69  frac = ia_params->LJGEN_sig/r_off;
70  fac = ia_params->LJGEN_eps
71 #ifdef LJGEN_SOFTCORE
72  * ia_params->LJGEN_lambda
73 #endif
74  * (ia_params->LJGEN_b1 * ia_params->LJGEN_a1 * pow(frac, ia_params->LJGEN_a1)
75  - ia_params->LJGEN_b2 * ia_params->LJGEN_a2 * pow(frac, ia_params->LJGEN_a2))
76  / (r_off * dist);
77  for(j=0;j<3;j++)
78  force[j] += fac * d[j];
79 
80 
81 
82 #ifdef LJ_WARN_WHEN_CLOSE
83  if(fac*dist > 1000) fprintf(stderr,"%d: LJ-Gen-Warning: Pair (%d-%d) force=%f dist=%f\n",
84  this_node,p1->p.identity,p2->p.identity,fac*dist,dist);
85 #endif
86 
87 #ifdef CONFIGTEMP
88  extern double configtemp[2];
89  int numfac = 0;

91  if (p2->p.configtemp) numfac+=1;

93  - ia_params->LJGEN_b2 * ia_params->LJGEN_a2 * pow(frac, ia_params->LJGEN_a2)) / r_off);
94  configtemp[1] += numfac* ia_params->LJGEN_eps * (-ia_params->LJGEN_b1 * ia_params->LJGEN_a1 * (ia_params->LJGEN_a1-1) * pow(frac, ia_params->LJGEN_a1)
95  + ia_params->LJGEN_b2 * ia_params->LJGEN_a2 * (ia_params->LJGEN_a2-1) * pow(frac, ia_params->LJGEN_a2)) / (SQR(r_off));
96 #endif
97  }
98  /* capped part of lj potential. */
99  else if(dist > 0.0) {
100  frac = ia_params->LJGEN_sig/ia_params->LJGEN_capradius;
101  fac = ia_params->LJGEN_eps
102  * (ia_params->LJGEN_b1 * ia_params->LJGEN_a1 * pow(frac, ia_params->LJGEN_a1)
103  - ia_params->LJGEN_b2 * ia_params->LJGEN_a2 * pow(frac, ia_params->LJGEN_a2))
104  / (ia_params->LJGEN_capradius * dist);
105  for(j=0;j<3;j++)
106  /* vector d is rescaled to length LJGEN_capradius */
107  force[j] += fac * d[j];
108  }
109  /* this should not happen! */
110  else {
111  LJ_TRACE(fprintf(stderr, "%d: Lennard-Jones-Generic warning: Particles id1=%d id2=%d exactly on top of each other\n",this_node,p1->p.identity,p2->p.identity));
112

114  fac = ia_params->LJGEN_eps
115 #ifdef LJGEN_SOFTCORE
116  * ia_params->LJGEN_lambda
117 #endif
118  * (ia_params->LJGEN_b1 * ia_params->LJGEN_a1 * pow(frac, ia_params->LJGEN_a1)
119  - ia_params->LJGEN_b2 * ia_params->LJGEN_a2 * pow(frac, ia_params->LJGEN_a2))
120  / ia_params->LJGEN_capradius;
121

123  }

134 inline double ljgen_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
135  double d[3], double dist)
136 {
137  double r_off, frac;
138 
139  if (CUTOFF_CHECK(dist < ia_params->LJGEN_cut+ia_params->LJGEN_offset)) {
140  r_off = dist - ia_params->LJGEN_offset;
141 #ifdef LJGEN_SOFTCORE
142  r_off *= r_off;
143  r_off += pow(ia_params->LJGEN_sig,2) * (1.0-ia_params->LJGEN_lambda)
144  * ia_params->LJGEN_softrad;
145  /* Taking a square root is not optimal, but we can't prevent the user from
146  using an odd m, n coefficient. */
147  r_off = sqrt(r_off);
148 #endif
149  /* normal case: resulting force/energy smaller than capping. */
150  if(r_off > ia_params->LJGEN_capradius) {
151  frac = ia_params->LJGEN_sig/r_off;
152  return ia_params->LJGEN_eps
153 #ifdef LJGEN_SOFTCORE
154  * ia_params->LJGEN_lambda
155 #endif
156  *(ia_params->LJGEN_b1 * pow(frac, ia_params->LJGEN_a1)
157  - ia_params->LJGEN_b2 * pow(frac, ia_params->LJGEN_a2)
158  + ia_params->LJGEN_shift);
159  }
160  /* capped part of lj potential. */
161  else if(dist > 0.0) {
162  frac = ia_params->LJGEN_sig/ia_params->LJGEN_capradius;
163  return ia_params->LJGEN_eps
164 #ifdef LJGEN_SOFTCORE
165  * ia_params->LJGEN_lambda
166 #endif
167  *( ia_params->LJGEN_b1 * pow(frac, ia_params->LJGEN_a1)
168  - ia_params->LJGEN_b2 * pow(frac, ia_params->LJGEN_a2)
169  + ia_params->LJGEN_shift);
170  }
171  /* this should not happen! */
172  else {
173  frac = ia_params->LJGEN_sig/ia_params->LJGEN_capradius;
174  return ia_params->LJGEN_eps*(
175  ia_params->LJGEN_b1 * pow(frac, ia_params->LJGEN_a1)
176  - ia_params->LJGEN_b2 * pow(frac, ia_params->LJGEN_a2)
177  + ia_params->LJGEN_shift);
178  }
179  }
180  return 0.0;
181 }
