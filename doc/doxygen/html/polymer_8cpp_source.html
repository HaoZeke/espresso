56 int mindist3(int part_id, double r_catch, int *ids) {
57  Particle *partCfgMD;
58  double dx,dy,dz;
59  int i, me, caught=0;
60 
61  partCfgMD = (Particle*)malloc(n_part*sizeof(Particle));
62  mpi_get_particles(partCfgMD, NULL);
63  me = -1; /* Since 'mpi_get_particles' returns the particles unsorted, it's most likely that 'partCfgMD[i].p.identity != i'
64  --> prevent that! */
65  for(i=0; i<n_part; i++) if (partCfgMD[i].p.identity == part_id) me = i;
66  if (me == -1) {
67  ostringstream msg;
68  msg <<"failed to find desired particle " << part_id;
69  runtimeError(msg);
70  return 0;
71  }
72  for (i=0; i<n_part; i++) {
73  if (i != me) {
74  dx = partCfgMD[me].r.p[0] - partCfgMD[i].r.p[0]; dx -= dround(dx/box_l[0])*box_l[0];
75  dy = partCfgMD[me].r.p[1] - partCfgMD[i].r.p[1]; dy -= dround(dy/box_l[1])*box_l[1];
76  dz = partCfgMD[me].r.p[2] - partCfgMD[i].r.p[2]; dz -= dround(dz/box_l[2])*box_l[2];
77  if (sqrt(SQR(dx)+SQR(dy)+SQR(dz)) < r_catch) ids[caught++]=partCfgMD[i].p.identity;
78  }
79  }
80  free(partCfgMD);
81  return (caught);
82 }

86 double mindist4(double pos[3]) {
87  Particle *partCfgMD;
88  double mindist=30000.0, dx,dy,dz;
89  int i;
90 
91  if (n_part ==0) return (dmin(dmin(box_l[0],box_l[1]),box_l[2]));
92  partCfgMD = (Particle*)malloc(n_part*sizeof(Particle));
93  mpi_get_particles(partCfgMD, NULL);
94  for (i=0; i<n_part; i++) {
95  dx = pos[0] - partCfgMD[i].r.p[0]; dx -= dround(dx/box_l[0])*box_l[0];
96  dy = pos[1] - partCfgMD[i].r.p[1]; dy -= dround(dy/box_l[1])*box_l[1];
97  dz = pos[2] - partCfgMD[i].r.p[2]; dz -= dround(dz/box_l[2])*box_l[2];
98  mindist = dmin(mindist, SQR(dx)+SQR(dy)+SQR(dz));
99  }
100  free(partCfgMD);
101  if (mindist<30000.0)
102  return (sqrt(mindist));
103  return (-1.0);
104 }
