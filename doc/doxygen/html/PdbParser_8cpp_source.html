113

115  while(file.good()) {
116  buf = file.get();
117 
118  /* Ignore leading whitespace, check for end of file (standard says EOF is "generaly" -1) */
119  if(std::isspace(buf[0]) || (buf[0] == -1)) {
120  continue;
121  }
122  /* End of atoms section */
123  if(buf[0] == '[') {
