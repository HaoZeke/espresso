137 int tclcommand_analyze_parse_and_print_energy(Tcl_Interp *interp, int argc, char **argv)
138 {
139  /* 'analyze energy [{ fene <type_num> | harmonic <type_num> | subt_lj_harm <type_num> | subt_lj_fene <type_num> | subt_lj <type_num> | lj <type1> <type2> | ljcos <type1> <type2> | ljcos2 <type1> <type2> | gb <type1> <type2> | coulomb | kinetic | total }]' */
140  char buffer[TCL_DOUBLE_SPACE + TCL_INTEGER_SPACE + 2];
141  int i, j;
142  double value;
143  value = 0.0;
144  if (n_part == 0) {
145  Tcl_AppendResult(interp, "(no particles)",
146  (char *)NULL);
147  return (TCL_OK);
148  }
149 
150  if (total_energy.init_status == 0) {
151  init_energies(&total_energy);
152  master_energy_calc();
153  }
154 
155  if (argc == 0)
156  tclcommand_analyze_print_all(interp);
157  else {
158 
159  if (ARG0_IS_S("kinetic"))
160  value = total_energy.data.e[0];
161  else if (ARG0_IS_S("bonded") ||
162  ARG0_IS_S("fene") ||
163  ARG0_IS_S("subt_lj_harm") ||
164  ARG0_IS_S("subt_lj_fene") ||
165  ARG0_IS_S("subt_lj") ||
166  ARG0_IS_S("harmonic") ||
167  ARG0_IS_S("umbrella") ||
168  ARG0_IS_S("endangledist")) {
169  if(argc<2 || ! ARG1_IS_I(i)) {
170  Tcl_ResetResult(interp);
171  Tcl_AppendResult(interp, "wrong # or type of arguments for: analyze energy bonded <type_num>",
172  (char *)NULL);
173  return (TCL_ERROR);
174  }
175  if(i < 0 || i >= n_bonded_ia) {
176  Tcl_AppendResult(interp, "bond type does not exist", (char *)NULL);
177  return (TCL_ERROR);
178  }
179  value = *obsstat_bonded(&total_energy, i);
180  }
181  else if (ARG0_IS_S("nonbonded") ||
182  ARG0_IS_S("lj") ||
183  ARG0_IS_S("buckingham") ||
184  ARG0_IS_S("lj-cos") ||
185  ARG0_IS_S("lj-cos2") ||
186  ARG0_IS_S("cos2") ||
187  ARG0_IS_S("gb") ||
188  ARG0_IS_S("tabulated")) {
189  if(argc<3 || ! ARG_IS_I(1, i) || ! ARG_IS_I(2, j)) {
190  Tcl_ResetResult(interp);
191  Tcl_AppendResult(interp, "wrong # or type of arguments for: analyze energy nonbonded <type1> <type2>",
192  (char *)NULL);
193  return (TCL_ERROR);
194  }
195  if(i < 0 || i >= n_particle_types || j < 0 || j >= n_particle_types) {
196  Tcl_AppendResult(interp, "particle type does not exist", (char *)NULL);
197  return (TCL_ERROR);
198  }
199  value = *obsstat_nonbonded(&total_energy, i, j);
200  }
201 
202  else if( ARG0_IS_S("coulomb")) {
203 #ifdef ELECTROSTATICS
204  value = 0;
205  for (i = 0; i < total_energy.n_coulomb; i++)
206  value += total_energy.coulomb[i];
207 #else
208  Tcl_AppendResult(interp, "ELECTROSTATICS not compiled (see myconfig.hpp)\n", (char *)NULL);
209 #endif
210  }
211  else if( ARG0_IS_S("magnetic")) {
212 #ifdef DIPOLES
213  value = 0;
214  for (i = 0; i < total_energy.n_dipolar; i++)
215  value += total_energy.dipolar[i];
216 #else
217  Tcl_AppendResult(interp, "DIPOLES not compiled (see myconfig.hpp)\n", (char *)NULL);
218 #endif
219  }
220 
221  else if (ARG0_IS_S("total")) {
222  value = total_energy.data.e[0];
223  for (i = 1; i < total_energy.data.n; i++)
224  value += total_energy.data.e[i];
225  for (i = 0; i < n_external_potentials; i++) {
226  value += external_potentials[i].energy;
227  }
228 
229  }
230  else {
231  Tcl_AppendResult(interp, "unknown feature of: analyze energy",
232  (char *)NULL);
233  return (TCL_ERROR);
234  }
235  Tcl_PrintDouble(interp, value, buffer);
236  Tcl_AppendResult(interp, buffer, (char *)NULL);
237  }
238 
239  return (TCL_OK);
240 }
