32 /** \file molforces.cpp
33  * Routines for calculating and applying trap forces upon molecules.
34  * This trap force can be set to
35  * - a harmonic potential with a restlength of zero on the molecular centre of mass
36  * - a drag on the molecular velocity
37  * - a cancelation of the total force on the molecule (including thermostat forces)
38  * The centre of mass can be fixed to an absolute position or to a relative position in the
39  * simulation box.
40  * The molecular trap forces is distributed evenly upon all particles in a molecule.
41  * (see \ref topology.cpp and \ref molforces.cpp)
42  */

72 /* calculates the force applies by traps on all molecules */
73 void calc_trap_force()
74 {
75  Molecule *m;
76  int mi,j;
77 #ifdef EXTERNAL_FORCES
78  double trappos;
79 #endif
80 
81 
82  if ( !topo_part_info_synced ) {
83  ostringstream msg;
84  msg <<"can't calculate moltrap: must execute analyse set topo_part_sync first";
85  runtimeError(msg);
86  return;
87  } else {
88 
89  m = &topology[0];
90 
91  for (mi = 0; mi < n_molecules; mi++) {
92  m = &topology[mi];
93  for(j = 0; j < 3; j++) {
94 #ifdef EXTERNAL_FORCES
95  if (m->trap_flag & COORD_FIXED(j)) {
96  /* Is the molecule trapped at a certain position in space? */
97  if (m->isrelative == 1) {
98  /* Is the trap set to absolute coordinates... */
99  trappos = m->trap_center[j]*box_l[j];
100  } else {
101  /* or to relative ones? */
102  trappos = m->trap_center[j];
103  }
104  m->trap_force[j] = 0;
105  /* the trap_force holding the molecule to the set position in calculated */
106  m->trap_force[j] += -((m->com[j]-trappos)*m->trap_spring_constant)/(double)(m->part.n);
107  /* the drag force applied to the molecule is calculated */
108  m->trap_force[j] += -(m->v[j]*m->drag_constant)/(double)(m->part.n);
109  /* the force applies by the traps is added to fav */
110  /* favcounter counts how many times we have added the force to fav since last time "analyze mol force" was called */
111  /* upon Espresso initialization it is set to -1 because in the first call of "integrate" there is an extra initial time step */
112  /* calling "analyze mol force" resets favcounter to 0 */
113  if (m->favcounter > -1) m->fav[j] -= m->v[j]*m->drag_constant + (m->com[j]-trappos)*m->trap_spring_constant;
114  }
115  if (m->noforce_flag & COORD_FIXED(j)) {
116  /* the trap force required to cancel out the total force acting on the molecule is calculated */
117  m->trap_force[j] -= m->f[j]/(double)m->part.n;
118  if (m->favcounter > -1) m->fav[j] -= m->f[j];
119  }
120 #endif
121  }
122  m->favcounter++;
123  }
124  }
125 }

163 /* This is only done for the trapped molecules to save time */
164 void calc_local_mol_info (IntList *local_trapped_mols)
165 {
166  int mi, i,j, mol;
167  Particle *p;
168  int np, c;
169  Cell *cell;
170  int lm;
171  int fixed;
172 
173  /* First reset all molecule masses,forces,centers of mass*/
174  for ( mi = 0 ; mi < n_molecules ; mi++ ) {
175  topology[mi].mass = 0;
176  for ( i = 0 ; i < 3 ; i++) {
177  topology[mi].f[i] = 0.0;
178  topology[mi].com[i] = 0.0;
179  topology[mi].v[i] = 0.0;
180  }
181  }
182 
183  for (c = 0; c < local_cells.n; c++) {
184  cell = local_cells.cell[c];
185  p = cell->part;
186  np = cell->n;
187  for(i = 0; i < np; i++) {
188  mol = p[i].p.mol_id;
189  if ( mol >= n_molecules ) {
190  ostringstream msg;
191  msg <<"can't calculate molforces no such molecule as " << mol;
192  runtimeError(msg);
193  return;
194  }
195 
196  /* Check to see if this molecule is fixed */
197  fixed =0;
198  for(j = 0; j < 3; j++) {
199 #ifdef EXTERNAL_FORCES
200  if (topology[mol].trap_flag & COORD_FIXED(j)) fixed = 1;
201  if (topology[mol].noforce_flag & COORD_FIXED(j)) fixed = 1;
202 #endif
203  }
204  if (fixed) {
205  topology[mol].mass += PMASS(p[i]);
206  /* Unfold the particle */
207  unfold_position(p[i].r.p, p[i].m.v, p[i].l.i);
208 
209  for ( j = 0 ; j < 3 ; j++ ) {
210  topology[mol].f[j] += p[i].f.f[j];
211  topology[mol].com[j] += p[i].r.p[j]*PMASS(p[i]);
212  topology[mol].v[j] += p[i].m.v[j]*PMASS(p[i]);
213  }
214  /* Fold the particle back */
215  fold_position(p[i].r.p,p[i].l.i);
216 
217  }
218  }
219  }
220 
221  /* Final normalisation of centers of mass and velocity*/
222  for ( lm = 0 ; lm < local_trapped_mols->n; lm++ ) {
223  mi = local_trapped_mols->e[lm];
224  for ( i = 0 ; i < 3 ; i++) {
225  topology[mi].com[i] = topology[mi].com[i]/(double)(topology[mi].mass);
226  topology[mi].v[i] = topology[mi].v[i]/(double)(topology[mi].mass);
227  }
228  }

235 void mpi_comm_mol_info(IntList *local_trapped_mols) {
236  int i, j, k, mol, count;
237  double com[3] = {0,0,0};
238  double v[3] = {0,0,0};
239  double f[3] = {0,0,0};
240  double mass = 0;
241  /* number of trapped molecules on each node */
242  int *n_local_mols;
243  /* sum of all elements of n_local_mols */
244  int sum_n_local_mols;
245  /* lists of which molecules are on each node in order of ascending node number */
246  int *local_mols;
247  MPI_Status status;
248 
249  n_local_mols = (int *) malloc(n_nodes*sizeof(int));
250  sum_n_local_mols = 0;
251 
252  /* Everyone tells me how many trapped molecules are on their node */
253  for (i=1; i <n_nodes; i++) {
254  MPI_Recv(&(n_local_mols[i]),1,MPI_INT,i,99,comm_cart,&status);
255  }
256 
257  for (i=1; i <n_nodes; i++) {
258  sum_n_local_mols += n_local_mols[i];
259  }
260  local_mols = (int *) malloc(sum_n_local_mols*sizeof(int));
261 
262  /* Everyone tells me which trapped molecules are on their node */
263  count = 0;
264  for (i=1; i <n_nodes; i++) {
265  MPI_Recv(&(local_mols[count]),n_local_mols[i],MPI_INT,i,99,comm_cart,&status);
266  count += n_local_mols[i];
267  }
268 
269  /* Initialise the centre of masses, velocities and forces to 0
270  except for molecules present on master node which are initialized to the local values on the master node
271  The centre of masses and velocities are weighted by the total mass on the master node */
272  for (i = 0; i < n_molecules; i++) {
273  mol =i;
274  if (intlist_contains(local_trapped_mols,i)) {
275  for (j = 0; j < 3; j++) {
276  topology[mol].com[j] = topology[mol].com[j] * topology[mol].mass;
277  topology[mol].v[j] = topology[mol].v[j] * topology[mol].mass;
278  topology[mol].f[j] = topology[mol].f[j];
279  }
280  } else {
281  topology[mol].mass = 0;
282  for (j = 0; j < 3; j++) {
283  topology[mol].com[j] = 0;
284  topology[mol].v[j] = 0;
285  topology[mol].f[j] = 0;
286  }
287  }
288  }
289 
290  /* The masses, coms, velocities and forces for trapped molecules are received from the slave nodes.
291  They are added into the running sums in topology[mol] */
292  count = 0;
293  for (i = 1; i < n_nodes; i++) {
294  for (j = 0; j < n_local_mols[i]; j++) {
295  mol = local_mols[count];
296  count += 1;
297  MPI_Recv(&mass,1,MPI_DOUBLE,i,99,comm_cart,&status);
298  MPI_Recv(com,3,MPI_DOUBLE,i,99,comm_cart,&status);
299  MPI_Recv(v,3,MPI_DOUBLE,i,99,comm_cart,&status);
300  MPI_Recv(f,3,MPI_DOUBLE,i,99,comm_cart,&status);
301  topology[mol].mass = topology[mol].mass + mass;
302  for (k = 0; k< 3; k++) {
303  topology[mol].com[k] += com[k]*mass;
304  topology[mol].v[k] += v[k]*mass;
305  topology[mol].f[k] += f[k];
306  }
307  }
308  }
309 
310  /* The centre of masses and velocities are renormalized by the total molecular weights */
311  for (mol = 0; mol < n_molecules; mol++) {
312  for (k=0;k <3; k ++) {
313  topology[mol].com[k] = topology[mol].com[k]/topology[mol].mass;
314  topology[mol].v[k] = topology[mol].v[k]/topology[mol].mass;
315  }
316  }
317 
318  /* The force exerted by the traps on the molecules are calculated */
319  calc_trap_force();
320 
321  /* The molecule information and trap forces are sent back to the slave nodes. */
322  count = 0;
323  for (i = 1; i < n_nodes ; i++) {
324  for (j = 0; j < n_local_mols[i]; j++) {
325  mol = local_mols[count];
326  count += 1;
327  MPI_Send(&(topology[mol].mass),1,MPI_DOUBLE,i,99,comm_cart);
328  MPI_Send(topology[mol].com,3,MPI_DOUBLE,i,99,comm_cart);
329  MPI_Send(topology[mol].v,3,MPI_DOUBLE,i,99,comm_cart);
330  MPI_Send(topology[mol].f,3,MPI_DOUBLE,i,99,comm_cart);
331  MPI_Send(topology[mol].trap_force,3,MPI_DOUBLE,i,99,comm_cart);
332  }
333  }
334 
335  free(local_mols);
336  free(n_local_mols);
337 
338 }
