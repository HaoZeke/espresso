317 int map_3don2d_grid(int g3d[3],int g2d[3], int mult[3])
318 {
319  int i,row_dir=-1;
320  /* trivial case */
321  if(g3d[2]==1) {
322  for(i=0;i<3;i++) mult[i]=1;
323  return 2;
324  }
325  if(g2d[0]%g3d[0] == 0) {
326  if(g2d[1]%g3d[1] == 0) {row_dir=2; }
327  else if(g2d[1]%g3d[2] == 0) {row_dir=1; g2d[2]=g2d[1]; g2d[1]=1; }
328  }
329  else if(g2d[0]%g3d[1] == 0) {
330  if(g2d[1]%g3d[0]==0) {row_dir=2; i=g2d[0]; g2d[0]=g2d[1]; g2d[1]=i; }
331  else if(g2d[1]%g3d[2]==0) {row_dir=0; g2d[2]=g2d[1]; g2d[1]=g2d[0]; g2d[0]=1; }
332  }
333  else if(g2d[0]%g3d[2] == 0) {
334  if(g2d[1]%g3d[0]==0) {row_dir=1; g2d[2]=g2d[0]; g2d[0]=g2d[1]; g2d[1]=1; }
335  else if(g2d[1]%g3d[1]==0) {row_dir=0; g2d[2]=g2d[0]; g2d[0]=1; }
336  }
337  for(i=0;i<3;i++) mult[i]=g2d[i]/g3d[i];
338  return row_dir;
339 }
