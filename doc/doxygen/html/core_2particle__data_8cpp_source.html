432 int realloc_particlelist(ParticleList *l, int size)
433 {
434  int old_max = l->max;
435  Particle *old_start = l->part;
436 
437  PART_TRACE(fprintf(stderr, "%d: realloc_particlelist %p: %d/%d->%d\n", this_node,
438  l, l->n, l->max, size));
439 
440  if (size < l->max) {
441  if (size == 0)
442  /* to be able to free an array again */
443  l->max = 0;
444  else
445  /* shrink not as fast, just lose half, rounded up */
446  l->max = PART_INCREMENT*(((l->max + size + 1)/2 +
447  PART_INCREMENT - 1)/PART_INCREMENT);
448  }
449  else
450  /* round up */
451  l->max = PART_INCREMENT*((size + PART_INCREMENT - 1)/PART_INCREMENT);
452  if (l->max != old_max)
453  l->part = (Particle *) realloc(l->part, sizeof(Particle)*l->max);
454  return l->part != old_start;
455 }
