53 void calc_comforce()
54 {
55  IA_parameters *ia_params;
56  double com0[3], com1[3], MofImatrix[9], diff[3];
57  double vect0[3], vect1[3], eva[3], eve[3], fvect[3];
58  Particle *p;
59  int np;
60  Cell *cell;
61 
62  for (int t0=0; t0<n_particle_types-1; t0++) {
63  for (int t1=t0+1; t1<n_particle_types; t1++) {
64  ia_params = get_ia_param(t0,t1);
65  if (ia_params->COMFORCE_flag == 1) {
66  centermass(t0,com0);
67  centermass(t1,com1);
68  for (int i = 0; i < 3; i++) {
69  diff[i]=com1[i]-com0[i];
70  }
71  momentofinertiamatrix(t0, MofImatrix);
72  calc_eigenvalues_3x3(MofImatrix, eva);
73  /* perpendicular force */
74  if(ia_params->COMFORCE_dir == 1) {
75  calc_eigenvector_3x3(MofImatrix,eva[0],eve);
76  /*By doing two vector products find radial axis along the target system */
77  vector_product(eve,diff,vect0);
78  vector_product(vect0,eve,vect1);
79 
80  /* normalize vect1, return is fvect */
81  unit_vector(vect1,fvect);
82  } else {
83  /* parallel force */
84  calc_eigenvector_3x3(MofImatrix,eva[0],fvect);
85  }
86 
87  /* orient it along the com vector */
88  if (scalar(fvect,diff) < 0.) {
89  for (int i = 0; i < 3; i++) {
90  fvect[i] = -fvect[i];
91  }
92  }
93 
94  /* Now apply the force */
95  for (int c = 0; c < local_cells.n; c++) {
96  cell = local_cells.cell[c];
97  p = cell->part;
98  np = cell->n;
99  for(int i = 0; i < np; i++) {
100  if(p[i].p.type==t0) {
101  for(int j = 0; j < 3; j++) {
102  p[i].f.f[j] -= ia_params->COMFORCE_fratio * ia_params->COMFORCE_force * fvect[j];
103  }
104  }
105  if(p[i].p.type==t1) {
106  for (int j = 0; j < 3; j++) {
107  p[i].f.f[j] += ia_params->COMFORCE_force * fvect[j];
108  }
109  }
110  }
111  }
112  /*end of force application */
113  }
114  }
115  }
116 
117 }
