369 /** Return all bond partners of a particle including bonds that are not stored at the particle itself up to a certain distance in numbers of bonds. Return a tcl list to the interpreter c*/
370 void tclcommand_part_print_bond_partners(Particle *part, char *buffer, Tcl_Interp *interp, int distance)
371 {
372  int c, p, i, p1, p2;
373  Bonded_ia_parameters *ia_params;
374  Particle *part1;
375  /* partners is a list containing the currently found bond partners and their distance in number of
376  bonds for each particle */
377  IntList *partners;
378  /* link list for all linear connections up to a certain number of bonds for the particle requiested */
379  IntList *links;
380 
381  /* setup bond partners and distance list. Since we need to identify particles via their identity,
382  we use a full sized array */
383  partners = (IntList*)malloc((max_seen_particle + 1)*sizeof(IntList));
384  for (p = 0; p <= max_seen_particle; p++) init_intlist(&partners[p]);
385  updatePartCfg(WITH_BONDS);
386 
387  /* determine initial connectivity */
388  for (p = 0; p < n_part; p++) {
389  part1 = &partCfg[p];
390  p1 = part1->p.identity;
391  for (i = 0; i < part1->bl.n;) {
392  ia_params = &bonded_ia_params[part1->bl.e[i++]];
393  if (ia_params->num == 1) {
394  p2 = part1->bl.e[i++];
395  add_partner(&partners[p1], p1, p2, 1);
396  add_partner(&partners[p2], p2, p1, 1);
397  }
398  else
399  i += ia_params->num;
400  }
401  }
402 
403  /* Create links to particle */
404  distance++;
405  links = (IntList*)malloc((distance+1)*sizeof(IntList));
406  for( c = 0; c <= distance; c++) init_intlist(&links[c]);
407 
408  p1 = part->p.identity;
409  add_link(&links[0], &links[0],0, p1, 1);
410 
411  for (p = 0; p < partners[p1].n; p+=2) {
412  add_link(&links[1], &links[0], 0, partners[p1].e[p], 2);
413  }
414 
415  for (c = 2; c <= distance; c++) {
416  for (i = 0; i < links[c-1].n; i+=c) {
417  p2 = links[c-1].e[c+i-1];
418  for (p = 0; p < partners[p2].n; p+=2) {
419  if( partners[p2].e[p] != links[c-1].e[c+i-2]) {
420  add_link(&links[c], &links[c-1], i, partners[p2].e[p],(c+1));
421  }
422  }
423  }
424  }
425 
426  p1 = part->p.identity;
427 
428  for (c = 2; c <= distance; c++) {
429  Tcl_AppendResult(interp, " {", (char *)NULL);
430  for (i = 0; i < links[c-1].n; i+=c ) {
431  Tcl_AppendResult(interp, " { ", (char *)NULL);
432  for (p = 1; p < c; p++) {
433  sprintf(buffer, "%d ",links[c-1].e[i+p]);
434  Tcl_AppendResult(interp, buffer, (char *)NULL);
435  }
436  Tcl_AppendResult(interp, "}", (char *)NULL);
437  }
438  Tcl_AppendResult(interp, " }", (char *)NULL);
439  }
440 
441  /* free memory */
442  for (p = 0; p <= max_seen_particle; p++) realloc_intlist(&partners[p], 0);
443  free(partners);
444  for(i=0;i<distance+1; i++) realloc_intlist(&links[i], 0);
445  free(links);
446 }
