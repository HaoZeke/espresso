40 /** Computes the BONDED_COULOMB pair force and adds this
41  force to the particle forces (see \ref interaction_data.cpp).
42  @param p1 Pointer to first particle.
43  @param p2 Pointer to second/middle particle.
44  @param iaparams bond type number of the angle interaction (see \ref interaction_data.cpp).
45  @param dx particle distance vector
46  @param force returns force of particle 1
47  @return 0.
48 */
49 inline int calc_bonded_coulomb_pair_force(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, double dx[3], double force[3])
50 {
51  int i;
52  double fac;
53  double dist2 = sqrlen(dx);
54  double dist = sqrt(dist2);
55 
56  fac = iaparams->p.bonded_coulomb.prefactor * p1->p.q * p2->p.q / (dist*dist2);
57 
58  for(i=0;i<3;i++)
59  force[i] = fac*dx[i];
60  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: BONDED_COULOMB f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist2,fac));
61  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: BONDED_COULOMB f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist2,fac));
62 
63  return 0;
64 }
