62 inline void add_rf_coulomb_pair_force_no_cutoff(Particle *p1, Particle *p2, double d[3], double dist, double force[3])
63 {
64  int j;
65  double fac;
66  fac = 1.0 / (dist*dist*dist) + rf_params.B / (rf_params.r_cut*rf_params.r_cut*rf_params.r_cut);
67  fac *= coulomb.prefactor * p1->p.q * p2->p.q;
68 
69  for (j=0;j<3;j++)
70  force[j] += fac * d[j];
71 
72  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: RF f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist,fac));
73  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: RF f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist,fac));
74 }

92 inline double rf_coulomb_pair_energy_no_cutoff(Particle *p1, Particle *p2, double dist)
93 {
94  double fac;
95  fac = 1.0 / dist - (rf_params.B*dist*dist) / (2*rf_params.r_cut*rf_params.r_cut*rf_params.r_cut);
96  //cut off part
97  fac -= (1-rf_params.B/2) / rf_params.r_cut;
98  fac *= coulomb.prefactor * p1->p.q * p2->p.q;
99  return fac;
100 }

102 inline double rf_coulomb_pair_energy(Particle *p1, Particle *p2, double dist)
103 {

105  {
106  return rf_coulomb_pair_energy_no_cutoff(p1,p2,dist);
107  }
108  return 0.0;
109 }
