190 int tclcommand_iccp3m_parse_normals(Tcl_Interp *interp,int n_ic, char *string) {
191  iccp3m_cfg.nvectorx = (double*) realloc(iccp3m_cfg.nvectorx,sizeof(double)*(iccp3m_cfg.n_ic));
192  iccp3m_cfg.nvectory = (double*) realloc(iccp3m_cfg.nvectory,sizeof(double)*(iccp3m_cfg.n_ic));
193  iccp3m_cfg.nvectorz = (double*) realloc(iccp3m_cfg.nvectorz,sizeof(double)*(iccp3m_cfg.n_ic));
194  const char opening_bracket = '{';
195  const char closing_bracket = '}';
196 
197  std::string arg(string);
198  size_t beginVector;
199  size_t endVector;
200  std::string sVector;
201  std::stringstream ssVector;
202 
203  double x,y,z;
204  for (int i = 0; i<n_ic; i++) {
205  beginVector = arg.find_first_of(opening_bracket);
206  endVector = arg.find_first_of(closing_bracket);
207  sVector = arg.substr(beginVector+1, endVector-2);
208  sVector.append(" "); // I could not figure out why -2 and append " " but it works!
209  ssVector.str(sVector);
210  ssVector >> x;
211  ssVector >> y;
212  ssVector >> z;
213  if (!ssVector.good()) {
214  Tcl_AppendResult(interp, "Could not understand ", ssVector.str().c_str(), (char *)NULL);
215  return TCL_ERROR;
216  }
217  iccp3m_cfg.nvectorx[i] = x;
218  iccp3m_cfg.nvectory[i] = y;
219  iccp3m_cfg.nvectorz[i] = z;
220  arg.erase(0, endVector+1);
221  }
222 
223  return TCL_OK;
224 }
