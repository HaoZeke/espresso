122 double bit_random_generator() {
123  /* Creates random numbers by XOR-ing two lines in a big matrix of linear independent rows/columns -> 'R250'
124  It's extremely fast, but has some Triplett-Correlations. */
125  // extern int rand_w_array[MERS_BIT_RANDOM];
126  // extern int random_pointer_1;
127  // extern int random_pointer_2;
128  double random_number;
129 
130  rand_w_array[random_pointer_1] = rand_w_array[random_pointer_1] ^ rand_w_array[random_pointer_2];
131  random_number = FACTOR * rand_w_array[random_pointer_1];
132  ++random_pointer_1;
133  ++random_pointer_2;
134  if(random_pointer_1 == MERS_BIT_RANDOM) random_pointer_1 = 0;
135  else if(random_pointer_2 == MERS_BIT_RANDOM) random_pointer_2 = 0;
136 
137  return(random_number);
138 }

153 void init_bit_random_generator(int iseed) {
154  /* Initializes the matrix for the bit_random_generator with a random but linear independent bit-pattern */
155  // extern double bit_random_generator();
156  // extern int rand_w_array[MERS_BIT_RANDOM];
157  int i = 0;
158  int imask1 = 0;
159  int imask2 = 0;
160  double rmod = (double) iseed;
161  bit_seed = iseed;
162  random_pointer_1 = 0;
163  random_pointer_2 = MERS1;
164 
165  /* Warm up the modulo generator */
166  for(i = 0; i < NWARM; ++i) {
167  rmod = MULTIPLY * rmod;
168  rmod = rmod - (double) ( (int) (rmod * FACTOR) ) * BIGFLOAT;
169  rmod = (double) ( (int) (rmod + 0.1) );
170  }
171 
172  /* Put random numbers on the working array */
173  for(i = 0; i < MERS_BIT_RANDOM; ++i) {
174  rmod = MULTIPLY * rmod;
175  rmod = rmod - (double) ( (int) (rmod * FACTOR) ) * BIGFLOAT;
176  rand_w_array[i] = (int) (rmod + 0.1);
177  rmod = (double) ( rand_w_array[i] );
178  }
179 
180  /* Ensure linear independence of the array/matrix */
181  imask1 = 1;
182  imask2 = BIGINTEGER;
183  for(i = NBIT - 2; i > 0; --i) {
184  rand_w_array[i] = ( rand_w_array[i] | imask1 ) & imask2;
185  imask2 = imask2 ^ imask1;
186  imask1 = imask1 * 2;
187  }
188  rand_w_array[0] = imask1;
189 
190  /* Warm up */
191  for(i = 0; i < NWARM; ++i) bit_random_generator();
192 }
