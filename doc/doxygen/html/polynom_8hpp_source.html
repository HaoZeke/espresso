34 /** evaluate the polynomial interpreted as a Taylor series via the Horner scheme */
35 inline double evaluateAsTaylorSeriesAt(Polynom *series, double x)
36 {
37  int cnt = series->n - 1;
38  double *c = series->e;
39  double r = c[cnt];
40  while (--cnt >= 0)
41  r = r*x + c[cnt];
42  return r;
43 }

45 /** evaluate the polynomial interpreted as a Chebychev series. Requires a series with at least
46  three coefficients, i.e. no linear approximations! */
47 inline double evaluateAsChebychevSeriesAt(Polynom *series, double x)
48 {
49  int j;
50  double *c = series->e;
51  double x2 = 2.0 * x;
52  double dd = c[series->n - 1];
53  double d = x2*dd + c[series->n - 2];
54  for(j = series->n - 3; j >= 1; j--) {
55  double tmp = d;
56  d = x2*d - dd + c[j];
57  dd = tmp;
58  }
59  return x*d - dd + 0.5 * c[0];
60 }
