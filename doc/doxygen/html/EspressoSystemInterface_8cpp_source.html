73 void EspressoSystemInterface::gatherParticles() {
74  Cell *cell;
75  Particle *p;
76  int i,c,np;
77 
78  // get particles from other nodes
79 #ifdef CUDA
80  if (m_gpu)
81  {
82  if(gpu_get_global_particle_vars_pointer_host()->communication_enabled) {
83  ESIF_TRACE(puts("Calling copy_part_data_to_gpu()"));
84  copy_part_data_to_gpu();
85  reallocDeviceMemory(gpu_get_global_particle_vars_pointer_host()->number_of_particles);
86  if(m_splitParticleStructGpu && (this_node == 0))
87  split_particle_struct();
88  }
89  }
90 #endif
91 
92  if (needsQ() || needsR() || needsQuatu()) {
93  R.clear();
94 
95  #ifdef ELECTROSTATICS
96  Q.clear();
97  #endif
98 
99  #ifdef ROTATION
100  Quatu.clear();
101  #endif
102 
103  for (c = 0; c < local_cells.n; c++) {
104  cell = local_cells.cell[c];
105  p = cell->part;
106  np = cell->n;
107 
108  if(needsR())
109  R.reserve(R.size()+np);
110 
111 #ifdef ELECTROSTATICS
112  if(needsQ())
113  Q.reserve(Q.size()+np);
114 #endif
115 
116 #ifdef ROTATION
117  if(needsQuatu())
118  Quatu.reserve(Quatu.size()+np);
119 #endif
120 
121  for(i = 0; i < np; i++) {
122 
123  if(needsR())
124  R.push_back(Vector3(p[i].r.p));
125 
126 #ifdef ELECTROSTATICS
127  if(needsQ())
128  Q.push_back(p[i].p.q);
129 #endif
130 
131 #ifdef ROTATION
132  if(needsQuatu())
133  Quatu.push_back(Vector3(p[i].r.quatu));
134 #endif
135  }
136  }
137  }
138 }
