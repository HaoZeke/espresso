41 /** Calculate smooth step force between particle p1 and p2 */
42 inline void add_BMHTF_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
43  double d[3], double dist, double dist2, double force[3])
44 {
45  int j;
46  double pw8, fac = 0.0;
47  if(CUTOFF_CHECK(dist < ia_params->BMHTF_cut)) {
48  pw8 = dist2*dist2*dist2*dist2;
49  fac = ia_params->BMHTF_A*ia_params->BMHTF_B*
50  exp(ia_params->BMHTF_B*(ia_params->BMHTF_sig - dist))/dist -
51  6*ia_params->BMHTF_C/pw8 - 8*ia_params->BMHTF_D/pw8/dist2;
52 
53  for(j=0;j<3;j++) force[j] += fac * d[j];
54  }
55 }

57 /** calculate smooth step potential energy between particle p1 and p2. */
58 inline double BMHTF_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
59  double d[3], double dist, double dist2)
60 {
61  double pw6;
62 
63  if(CUTOFF_CHECK(dist < ia_params->BMHTF_cut)) {
64  pw6 = dist2*dist2*dist2;
65  return ia_params->BMHTF_A*
66  exp(ia_params->BMHTF_B*(ia_params->BMHTF_sig - dist)) -
67  ia_params->BMHTF_C/pw6 - ia_params->BMHTF_D/pw6/dist2 + ia_params->BMHTF_computed_shift;
68  }
69  return 0.0;
