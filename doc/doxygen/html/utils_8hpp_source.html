366 /** Calculates the sinc-function as sin(PI*x)/(PI*x).
367  *
368  * (same convention as in Hockney/Eastwood). In order to avoid
369  * divisions by 0, arguments, whose modulus is smaller than epsi, will
370  * be evaluated by an 8th order Taylor expansion of the sinc
371  * function. Note that the difference between sinc(x) and this
372  * expansion is smaller than 0.235e-12, if x is smaller than 0.1. (The
373  * next term in the expansion is the 10th order contribution
374  * PI^10/39916800 * x^10 = 0.2346...*x^12). This expansion should
375  * also save time, since it reduces the number of function calls to
376  * sin().
377 */
378 inline double sinc(double d)
379 {
380 #define epsi 0.1
381 
382 #define c2 -0.1666666666667e-0
383 #define c4 0.8333333333333e-2
384 #define c6 -0.1984126984127e-3
385 #define c8 0.2755731922399e-5
386 
387  double PId = PI*d, PId2;
388 
389  if (fabs(d)>epsi)
390  return sin(PId)/PId;
391  else {
392  PId2 = SQR(PId);
393  return 1.0 + PId2*(c2+PId2*(c4+PId2*(c6+PId2*c8)));
394  }
395 }

499 /** Calc eigevalues of a 3x3 matrix stored in q as a 9x1 array*/
500 inline int calc_eigenvalues_3x3(double *q, double *eva) {
501  double q11,q22,q33,q12,q13,q23;
502  double a,b,c;
503  double QQ,R,R2,QQ3;
504  double theta,root1,root2,x1,x2,x3,help;
505  int anzdiff=3;
506 
507  /* copy tensor to local variables (This is really a fuc off code!) */
508  q11 = *q++; q12 = *q++; q13 = *q;
509  q22 = *(q+=2); q23 = *(++q); q33 = *(q+=3);
510 
511  /* solve the cubic equation of the eigenvalue problem */
512  a = -(q11 + q22 + q33);
513  b = q11*q22 + q11*q33 + q22*q33 - q12*q12 - q13*q13 - q23*q23;
514  c = -(q11*q22*q33 + 2*q12*q23*q13 - q12*q12*q33 - q13*q13*q22 - q23*q23*q11);
515  QQ = (a*a-3*b)/9.0;
516  R = (2*a*a*a - 9*a*b + 27*c)/54;
517  QQ3 = QQ*QQ*QQ;
518  R2 = R*R;
519 
520  if (R2 < QQ3) {
521  theta = acos(R/sqrt(QQ3));
522  root1 = sqrt(QQ)*cos(theta/3.0);
523  root2 = sqrt(QQ*3.0)*sin(theta/3.0);
524  x1 = - 2*root1 - a/3.0;
525  x2 = root1 - root2 - a/3.0;
526  x3 = root1 + root2 - a/3.0;
527  }
528  else {
529  double AA,BB,signum = 1.0;
530 
531  if (R < 0) { signum = -1.0; R = -R; }
532  AA = - signum*exp( log(R + sqrt(R2-QQ3))/3.0);
533  if (AA==0.0) BB = 0.0;
534  else BB = QQ/AA;
535 
536  /* compute the second way of the diagonalization
537  * remark : a diagonal matrix has to have real eigenvalues (=>x2=x3)
538  * but (!), the general solution of this case can have in general
539  * imaginary solutions for x2,x3 (see numerical recipies p. 185) */
540  x1 = (AA+BB) - a/3.0;
541  x2 = 0.5*(AA+BB) - a/3.0;
542  x3 = x2;
543  }
544 
545  /* order the eigenvalues in decreasing order */
546  if (x1<x2) { help = x1; x1 = x2; x2 = help; }
547  if (x1<x3) { help = x1; x1 = x3; x3 = help; }
548  if (x2<x3) { help = x2; x2 = x3; x3 = help; }
549  eva[0] = x1; eva[1] = x2; eva[2] = x3;
550 
551  /* calculate number of different eigenvalues */
552  if(x1==x2) anzdiff--;
553  if(x2==x3) anzdiff--;
554 
555  return anzdiff;
556 }

558 /** Calc eigevectors of a 3x3 matrix stored in a as a 9x1 array*/
559 /** Given an eigenvalue (eva) returns the corresponding eigenvector (eve)*/
560 inline int calc_eigenvector_3x3(double *a,double eva,double *eve) {
561  int i,j,ind1,ind2,ind3,row1,row2;
562  double A_x1[3][3],coeff1,coeff2,norm;
563 
564  /* build (matrix - eva*unity) */
565  for(i=0;i<3;i++) {
566  for(j=0;j<3;j++) A_x1[i][j]=a[3*i+j];
567  A_x1[i][i]-=eva;
568  }
569 
570  /* solve the linear equation system */
571  for(ind1=0;ind1<3;ind1++) {
572  ind2=(ind1+1) % 3; ind3=(ind1+2) % 3;
573  row1=0;
574  do {
575  row1++;
576  } while ((row1<3) && (A_x1[ind3][row1]==0));
577 
578  /* okay, if one whole column is empty then we can take this
579  direction as the eigenvector
580  remember : {eigenvectors} = kernel(A_x1) */
581  if (row1 == 3) {
582  eve[ind3]=1.0;
583  eve[(ind3+1) % 3]=0.0;
584  eve[(ind3+2) % 3]=0.0;
585  return 1;
586  }
587 
588  for(i=1;i<3;i++) {
589  row2=(row1+i) % 3;
590  coeff1=A_x1[ind1][row1]*A_x1[ind3][row2]-
591  A_x1[ind1][row2]*A_x1[ind3][row1];
592  coeff2=A_x1[ind2][row1]*A_x1[ind3][row2]-
593  A_x1[ind2][row2]*A_x1[ind3][row1];
594  if (coeff1!=0.0) {
595  eve[ind2]=1.0;
596  eve[ind1]=-coeff2/coeff1;
597  eve[ind3]=-( A_x1[ind2][row1]+eve[ind1]* A_x1[ind1][row1])
598  /A_x1[ind3][row1];
599  norm = sqrt(eve[0]*eve[0] + eve[1]*eve[1] + eve[2]*eve[2]);
600  eve[0]/=norm; eve[1]/=norm; eve[2]/=norm;
601  return 1;
602  }
603  else {
604  if (coeff2!=0.0) {
605  eve[ind1]=1.0;
606  eve[ind2]=-coeff1/coeff2;
607  eve[ind3]=-( A_x1[ind1][row1]+eve[ind2]* A_x1[ind2][row1])
608  /A_x1[ind3][row1];
609  norm = sqrt(eve[0]*eve[0] + eve[1]*eve[1] + eve[2]*eve[2]);
610  eve[0]/=norm; eve[1]/=norm; eve[2]/=norm;
611  return(1);
612  }
613  }
614  } /* loop over the different rows */
615  } /* loop over the different columns */
616 
617  /* the try failed => not a singular matrix: only solution is (0,0,0) */
618  return 0;
619 }

629  * by its LU decomposition. Due to the partial pivoting the result
630  * may contain row permutations which are recorded in perms.
631  * @return 0 for success, -1 otherwise (i.e. matrix is singular)
632  * @param A Matrix to be decomposed (Input/Output)
633  * @param n Dimension of the matrix (Input)
634  * @param perms Records row permutations effected by pivoting (Output)
635  */
636 inline int lu_decompose_matrix(double **A, int n, int *perms) {
637  int i, j, k, ip;
638  double max, sum, tmp;
639 
640  double *scal = (double *)malloc(n*sizeof(double));
641 
642  /* loop over rows and store implicit scaling factors */
643  for (i=0; i<n; i++) {
644  max = 0.0;
645  for (j=0; j<n; j++) {
646  if ( (tmp=fabs(A[i][j])) > max) max=tmp;
647  }
648  if (max == 0.0) {
649  /* matrix has a zero row and is singular */
650  return -1;
651  }
652  scal[i] = 1.0/max;
653  }
654 
655  /** Crout's algorithm: Calculate L and U columnwise from top to
656  * bottom. The diagonal elements of L are chosen to be 1. Only
657  * previously determined entries are used in the calculation. The
658  * original matrix elements are used only once and can be
659  * overwritten with the elements of L and U, the diagonal of L not
660  * being stored. Rows may be permuted according to the largest
661  * element (pivot) in the lower part, where rows are normalized to
662  * have the largest element scaled to unity.
663  */
664 
665  /* loop over columns */
666  for (j=0; j<n; j++) {
667 
668  /* calculate upper triangle part (without diagonal) */
669  for (i=0; i<j; i++) {
670  sum = A[i][j];
671  for (k=0;k<=i-1;k++) sum -= A[i][k]*A[k][j];
672  A[i][j] = sum;
673  }
674 
675  /* calculate diagonal and lower triangle part */
676  /* pivot is determined on the fly, but not yet divided by */
677  ip = j;
678  max = 0.0;
679  for (i=j; i<n; i++) {
680  sum = A[i][j];
681  for (k=0; k<=j-1; k++) sum -= A[i][k]*A[k][j];
682  A[i][j] = sum;
683  if ((tmp=scal[i]*fabs(sum)) > max) {
684  max = tmp;
685  ip = i;
686  }
687  }
688 
689  /* swap rows according to pivot index */
690  if (j != ip) {
691  for (k=0; k<n; k++) {
692  tmp = A[j][k];
693  A[j][k] = A[ip][k];
694  A[ip][k] = tmp;
695  }
696  scal[ip] = scal[j];
697  }
698  perms[j] = ip;
699 
700  if (A[j][j] == 0.0) {
701  /* zero pivot indicates singular matrix */
702  return -1;
703  }
704 
705  /* now divide by pivot element */
706  if (j != n) {
707  tmp = 1.0/A[j][j];
708  for (i=j+1; i<n; i++)
709  A[i][j] *= tmp;
710  }
711 
712  }
713 
714  free(scal);
715 
716  return 0;
717 }
718 
719 /** Solve the linear equation system for a LU decomposed matrix A.
720  * Uses forward substitution for the lower triangle part and
721  * backsubstitution for the upper triangle part. Row permutations as
722  * indicated in perms are applied to b accordingly. The solution is
723  * written to b in place.
724  * @param A Matrix in LU decomposed form (Input)
725  * @param n Dimension of the matrix (Input)
726  * @param perms Indicates row permutations due to pivoting (Input)
727  * @param b Right-hand side of equation system (Input).
728  * Is destroyed and contains the solution x afterwards (Output).
729  */
730 inline void lu_solve_system(double **A, int n, int *perms, double *b) {
731  int i, j;
732  double sum;
733 
734  /* Step 1: Solve Ly=b */
735 
736  /* forward substitution for lower triangle part */
737  for (i=0; i<n; i++) {
738  /* take care of the correct permutations */
739  sum=b[perms[i]];
740  b[perms[i]]=b[i];
741  for (j=0; j<=i-1; j++) sum -= A[i][j]*b[j];
742  b[i] = sum;
743  }
744 
745  /* Step 2: Solve Ux=y */
746 
747  /* backsubstitution for upper triangle part */
748  for (i=n-1; i>=0; i--) {
749  sum = b[i];
750  for (j=i+1; j<n; j++) sum -= A[i][j]*b[j];
751  b[i]=sum/A[i][i];
752  }
753 
754 }
