176 static void mpifake_dtblock(MPI_Datatype newtype, MPI_Datatype oldtype, int count, int disp) {
177 
178  int extent, upper, lower;
179 
180  if (count > 0) {
181 
182  extent = (oldtype->upper - oldtype->lower) * (count - 1);
183  if (extent > 0) {
184  upper = oldtype->upper + extent + disp;
185  lower = oldtype->lower + disp;
186  } else {
187  upper = oldtype->upper + disp;
188  lower = oldtype->lower + extent + disp;
189  }
190 
191  } else {
192 
193  upper = 0;
194  lower = 0;
195 
196  }
197 
198  if (upper > newtype->upper) newtype->upper = upper;
199  if (lower < newtype->lower) newtype->lower = lower;
200 
201  newtype->size += count * oldtype->size;
202
