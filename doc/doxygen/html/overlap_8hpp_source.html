26 /** \file overlap.hpp
27  * Routines to calculate the energy and/or force
28  * for bonds, angles and dihedrals as a sum of N functions in the forms:
29  * bonds --- parameter: [N, a_i, b_i, c_i], function: U(bond) = sum_(i=1,N) {a_i*exp[-(bond-b_i)^2 /(c_i^2)]}.
30  * angles--- parameter: [N, a_i, b_i, c_i], function: U(cos(angl)) = sum_(i=1,N) {a_i*exp[-(cos(angl)-b_i)^2 / (2 * c_i^2)]}.
31  * dihedrals---parameter: [N, a_i, b_i, c_i], function: U(dihe)= sum_(i=1,N) {a_i*(1+Math.cos(c_i*dihe+b_i))}.
32  * Require feature OVERLAPPED compiled in myconfig.hpp (for more info of FEATURES, see \ref config.hpp ).

43 /** Computes the two body overlapped bonded force.
44  Adds this force to the particle forces in forces.hpp (see \ref tclcommand_inter).
45  @param p1 Pointer to first particle.
46  @param p2 Pointer to second/middle particle.
47  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).
48  @param dx particle distance vector
49  @param force returns force of particle 1
50  @return 0.
51  Needs feature OVERLAPPED compiled in (see \ref config.hpp).
52 */
53 inline int calc_overlap_bond_force(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, double dx[3], double force[3])
54 {
55  int i;
56  double fac;
57 
58  int ig;
59  double ev;
60  double prob=0.;
61  double Apart=0.;
62 
63  double dist2 = sqrlen(dx);
64  double dist = sqrt(dist2);
65 
66  /* compute fac = - 1/r * dU/dr */
67  /* prob = sum_(i=1,N) [ a_i * exp(-(r -b_i)^2/c_i^2))] */
68  /* Apart = sum_(i=1,N) [ a_i * exp(-(r -b_i)^2/c_i^2)) * (x-b_i) / c_i^2] */
69  /* fac = - 2.0 * ( 1/r + Apart / prob ) */
70  for(ig=0; ig<iaparams->p.overlap.noverlaps; ig++) {
71  ev = (dist - iaparams->p.overlap.para_b[ig]) / iaparams->p.overlap.para_c[ig];
72  prob = prob + iaparams->p.overlap.para_a[ig] * exp (-1.0*ev*ev);
73  Apart = Apart + iaparams->p.overlap.para_a[ig] * exp (-1.0*ev*ev) * (dist-iaparams->p.overlap.para_b[ig]) / (iaparams->p.overlap.para_c[ig] * iaparams->p.overlap.para_c[ig]);
74  }
75  fac = -2. * ( 1 / dist + Apart / prob);
76  fac /= dist;
77 
78  /* compute force */
79  for(i=0;i<3;i++)
80  force[i] = fac*dx[i];
81 
82  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: HARMONIC f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist2,fac));
83  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: HARMONIC f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist2,fac));
84 
85  return 0;
86 }

97 inline int overlap_bond_energy(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, double dx[3], double *_energy)

103 
104  double dist2 = sqrlen(dx);
105  double dist = sqrt(dist2);
106 
107  /*compute overlapped bond energy */
108  /*prob = sum_(i=1,N) [ a_i * exp(-(r -b_i)^2/c_i^2))] */
109  /*pot = -1.0 * Log { prob / r^2 } */

119 /** Computes the three body overlapped angle interaction force.
120  Adds this force to the particle forces in forces.hpp (see \ref tclcommand_inter).
121  @param p_mid Pointer to second/middle particle.
122  @param p_left Pointer to first/left particle.
123  @param p_right Pointer to third/right particle.
124  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).
125  @param force1 returns force of particle 1
126  @param force2 returns force of particle 2
127  @return 0
128  Needs feature OVERLAPPED compiled in (see \ref config.hpp).
129 */
130 inline int calc_overlap_angle_force(Particle *p_mid, Particle *p_left,
131  Particle *p_right, Bonded_ia_parameters *iaparams,
132  double force1[3], double force2[3])
133 {
134  double cosine, vec1[3], vec2[3], d1i, d2i, dist2, fac, f1=0.0, f2=0.0;
135  int j;
136 
137  int ig;
138  double ev;
139  double prob=0.;
140  double Apart=0.;
141 
142  cosine=0.0;
143  /* vector from p_left to p_mid */
144  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
145  dist2 = sqrlen(vec1);
146  d1i = 1.0 / sqrt(dist2);
147  for(j=0;j<3;j++) vec1[j] *= d1i;
148  /* vector from p_mid to p_right */
149  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
150  dist2 = sqrlen(vec2);
151  d2i = 1.0 / sqrt(dist2);
152  for(j=0;j<3;j++) vec2[j] *= d2i;
153  /* scalar produvt of vec1 and vec2 */
154  /* Notice: cosine = - costheta */
155  cosine = scalar(vec1, vec2);
156 
157  if ( cosine > TINY_COS_VALUE ) cosine = TINY_COS_VALUE;
158  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
159 
160  /* compute fac = - dU/d(costheta) */
161  /* prob = sum_(i=1,N) [ a_i * exp(-(costheta -b_i)^2/c_i^2))] */
162  /* Apart = sum_(i=1,N) [ a_i * exp(-(costheta -b_i)^2/c_i^2)) * (costheta-b_i) / c_i^2] */
163  /* fac = -2.0 * ( Apart / prob ) */
164 
165  for(ig=0; ig<iaparams->p.overlap.noverlaps; ig++) {
166  ev = (-cosine - iaparams->p.overlap.para_b[ig]) / iaparams->p.overlap.para_c[ig];
167  prob = prob + iaparams->p.overlap.para_a[ig] * exp (-1.0*ev*ev);
168  Apart = Apart + iaparams->p.overlap.para_a[ig] * exp (-1.0*ev*ev) * (-cosine-iaparams->p.overlap.para_b[ig]) / (iaparams->p.overlap.para_c[ig] * iaparams->p.overlap.para_c[ig]);
169  }
170  fac = -2. * ( Apart / prob);
171 
172  /* compute force */
173  for(j=0;j<3;j++) {
174  f1 = fac * (cosine * vec1[j] - vec2[j]) * d1i;
175  f2 = fac * (cosine * vec2[j] - vec1[j]) * d2i;
176 
177  force1[j] = (f1-f2);
178  force2[j] = -f1;
179  }
180  return 0;
181 }

234 /** Computes the four body overlapped dihedral interaction force.
235  Adds this force to the particle forces in forces.hpp (see \ref tclcommand_inter).
236  @param p1, p2, p3, p4 define the angle between the planes p1,p2,p3 and p2,p3,p4
237  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).
238  @param force1 returns force of particle 1
239  @param force2 returns force of particle 2
240  @param force3 returns force of particle 3
241  @return 0
242  Needs feature OVERLAPPED compiled in (see \ref config.hpp).
243 */
244 inline int calc_overlap_dihedral_force(Particle *p2, Particle *p1,
245  Particle *p3, Particle *p4, Bonded_ia_parameters *iaparams,
246  double force2[3], double force1[3], double force3[3])
247 {
248  int i;
249  /* vectors for dihedral angle calculation */
250  double v12[3], v23[3], v34[3], v12Xv23[3], v23Xv34[3], l_v12Xv23, l_v23Xv34;
251  double v23Xf1[3], v23Xf4[3], v34Xf4[3], v12Xf1[3];
252  /* dihedral angle, cosine of the dihedral angle */
253  double phi, cosphi, sinmphi_sinphi;
254  /* force factors */
255  double fac, f1[3], f4[3];
256 
257  int ig;
258  double f_dihe = 0.;
259 
260  /* dihedral angle */
261  calc_dihedral_angle(p1, p2, p3, p4, v12, v23, v34, v12Xv23, &l_v12Xv23, v23Xv34, &l_v23Xv34, &cosphi, &phi);
262  /* dihedral angle not defined - force zero */
263  if ( phi == -1.0 ) {
264  for(i=0;i<3;i++) { force1[i] = 0.0; force2[i] = 0.0; force3[i] = 0.0; }
265  return 0;
266  }
267 
268  /* calculate force components (directions) */
269  for(i=0;i<3;i++) {
270  f1[i] = (v23Xv34[i] - cosphi*v12Xv23[i])/l_v12Xv23;;
271  f4[i] = (v12Xv23[i] - cosphi*v23Xv34[i])/l_v23Xv34;
272  }
273  vector_product(v23, f1, v23Xf1);
274  vector_product(v23, f4, v23Xf4);
275  vector_product(v34, f4, v34Xf4);
276  vector_product(v12, f1, v12Xf1);
277 
278  /* calculate force magnitude */
279  //fac = sum_(i=1,N) { a_i * c_i * c_i * cos(c_i*x + b_i)/cos(phi) }
280  //fac = sum_(i=1,N) { a_i * c_i * sin(c_i*phi + b_i) /sin(phi)}
281  for(ig=0; ig<iaparams->p.overlap.noverlaps; ig++) {
282  fac = iaparams->p.overlap.para_a[ig] * iaparams->p.overlap.para_c[ig];
283 
284  if(fabs(sin(phi)) < TINY_SIN_VALUE) {
285  /*(comes from taking the first term of the MacLaurin expansion of
286  sin(n*phi - phi0) and sin(phi) and then making the division).
287  The original code had a 2PI term in the cosine (cos(2PI - nPhi))
288  but I removed it because it wasn't doing anything. AnaVV*/
289  sinmphi_sinphi = iaparams->p.overlap.para_c[ig]*
290  cos(iaparams->p.overlap.para_c[ig] * phi + iaparams->p.overlap.para_b[ig])/cosphi;
291  }
292  else {
293  sinmphi_sinphi = sin(iaparams->p.overlap.para_c[ig] * phi - iaparams->p.overlap.para_b[ig])/sin(phi);
294  }
295  fac *= sinmphi_sinphi;
296  f_dihe += fac;
297  }
298 
299  /* store dihedral forces */
300  for(i=0;i<3;i++) {
301  force1[i] = f_dihe*v23Xf1[i];
302  force2[i] = f_dihe*(v34Xf4[i] - v12Xf1[i] - v23Xf1[i]);
303  force3[i] = f_dihe*(v12Xf1[i] - v23Xf4[i] - v34Xf4[i]);
304  }
305  return 0;
306 }
