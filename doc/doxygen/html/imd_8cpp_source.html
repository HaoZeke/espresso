80 static void swap4(char *data, int ndata) {
81  int i;
82  char *dataptr;
83  char b0, b1;
84 
85  dataptr = data;
86  for (i=0; i<ndata; i+=4) {
87  b0 = dataptr[0];
88  b1 = dataptr[1];
89  dataptr[0] = dataptr[3];
90  dataptr[1] = dataptr[2];
91  dataptr[2] = b1;
92  dataptr[3] = b0;
93  dataptr += 4;
94  }
95 }

97 static int32_t imd_htonl(int32_t h) {
98  int32_t n;
99  ((char *)&n)[0] = (h >> 24) & 0x0FF;
100  ((char *)&n)[1] = (h >> 16) & 0x0FF;
101  ((char *)&n)[2] = (h >> 8) & 0x0FF;
102  ((char *)&n)[3] = h & 0x0FF;
103  return n;
104 }

106 static int32_t imd_ntohl(int32_t n) {
107  int32_t h = 0;
108 
109  h = ((char *)&n)[0] << 24;
110  h |= ((char *)&n)[1] << 16;
111  h |= ((char *)&n)[2] << 8;
112  h |= ((char *)&n)[3];
113 
114  return h;
115 }

127 static int32_t imd_readn(void *s, char *ptr, int32_t n) {
128  int32_t nleft;
129  int32_t nread;
130 
131  nleft = n;
132  while (nleft > 0) {
133  if ((nread = vmdsock_read(s, ptr, nleft)) < 0) {
134  if (errno == EINTR)
135  nread = 0; /* and call read() again */
136  else
137  return -1;
138  } else if (nread == 0)
139  break; /* EOF */
140  nleft -= nread;
141  ptr += nread;
142  }
143  return n-nleft;
144 }

253 int imd_recv_handshake(void *s) {
254  int32_t buf;
255  IMDType type;
256 
257  /* Wait 5 seconds for the handshake to come */
258  if (vmdsock_selread(s, 5) != 1) return -1;
259 
260  /* Check to see that a valid handshake was received */
261  type = imd_recv_header_nolengthswap(s, &buf);
262  if (type != IMD_HANDSHAKE) return -1;
263 
264  /* Check its endianness, as well as the IMD version. */
265  if (buf == IMDVERSION) {
266  if (!imd_go(s)) return 0;
267  return -1;
268  }
269  swap4((char *)&buf, 4);
270  if (buf == IMDVERSION) {
271  if (!imd_go(s)) return 1;
272  }
273 
274  /* We failed to determine endianness. */
275  return -1;
276 }
