433 /* the data transfer routine for the lclcblks itself */
434 void distribute(int e_size, double fac)
435 {
436  int c, node, inv_node;
437  double sendbuf[8];
438  double recvbuf[8] = { 0, 0, 0, 0, 0, 0, 0, 0};
439  MPI_Status status;
440 
441  /* send/recv to/from other nodes. Also builds up the gblcblk. */
442  for (node = 0; node < n_nodes; node++) {
443  inv_node = n_nodes - node - 1;
444  /* up */
445  if (node == this_node) {
446  /* calculate sums of cells below */
447  for (c = 1; c < n_layers; c++)
448  addscale_vec(blwentry(gblcblk, c, e_size), fac, blwentry(gblcblk, c - 1, e_size), blwentry(lclcblk, c - 1, e_size), e_size);
449 
450  /* calculate my ghost contribution only if a node above exists */
451  if (node + 1 < n_nodes) {
452  addscale_vec(sendbuf, fac, blwentry(gblcblk, n_layers - 1, e_size), blwentry(lclcblk, n_layers - 1, e_size), e_size);
453  copy_vec(sendbuf + e_size, blwentry(lclcblk, n_layers, e_size), e_size);
454  MPI_Send(sendbuf, 2*e_size, MPI_DOUBLE, node + 1, 0, comm_cart);
455  }
456  }
457  else if (node + 1 == this_node) {
458  MPI_Recv(recvbuf, 2*e_size, MPI_DOUBLE, node, 0, comm_cart, &status);
459  copy_vec(blwentry(gblcblk, 0, e_size), recvbuf, e_size);
460  copy_vec(blwentry(lclcblk, 0, e_size), recvbuf + e_size, e_size);
461  }
462 
463  /* down */
464  if (inv_node == this_node) {
465  /* calculate sums of all cells above */
466  for (c = n_layers + 1; c > 2; c--)
467  addscale_vec(abventry(gblcblk, c - 3, e_size), fac, abventry(gblcblk, c - 2, e_size), abventry(lclcblk, c, e_size), e_size);
468 
469  /* calculate my ghost contribution only if a node below exists */
470  if (inv_node - 1 >= 0) {
471  addscale_vec(sendbuf, fac, abventry(gblcblk, 0, e_size), abventry(lclcblk, 2, e_size), e_size);
472  copy_vec(sendbuf + e_size, abventry(lclcblk, 1, e_size), e_size);
473  MPI_Send(sendbuf, 2*e_size, MPI_DOUBLE, inv_node - 1, 0, comm_cart);
474  }
475  }
476  else if (inv_node - 1 == this_node) {
477  MPI_Recv(recvbuf, 2*e_size, MPI_DOUBLE, inv_node, 0, comm_cart, &status);
478  copy_vec(abventry(gblcblk, n_layers - 1, e_size), recvbuf, e_size);
479  copy_vec(abventry(lclcblk, n_layers + 1, e_size), recvbuf + e_size, e_size);
480  }
481  }

579 static void add_z_force()
580 {
581  int np, c, i;
582  double add;
583  Particle *part;
584  double *othcblk;
585  int size = 2;
586  double field_tot=0;
587 
588  /* Const. potential: subtract global dipole moment */
589  if (mmm2d_params.const_pot_on) {
590  double gbl_dm_z = 0;
591  double lcl_dm_z = 0;
592 
593  for (c = 0; c < local_cells.n; c++) {
594  int npl = local_cells.cell[c]->n;
595  Particle *pl = local_cells.cell[c]->part;
596  for (i = 0; i < npl; i++) {
597  lcl_dm_z += pl[i].p.q*(pl[i].r.p[2] + pl[i].l.i[2]*box_l[2]);
598  }
599  }
600  MPI_Allreduce(&lcl_dm_z, &gbl_dm_z, 1, MPI_DOUBLE, MPI_SUM, comm_cart);
601 
602  field_induced = gbl_dm_z * coulomb.prefactor*4*M_PI*ux*uy*uz;
603  field_applied = mmm2d_params.pot_diff * uz;
604  field_tot = field_induced + field_applied;
605  }
606 
607  for (c = 1; c <= n_layers; c++) {
608  othcblk = block(gblcblk, c - 1, size);
609  add = othcblk[QQEQQP] - othcblk[QQEQQM];
610  np = cells[c].n;
611  part = cells[c].part;
612  for (i = 0; i < np; i++) {
613  part[i].f.f[2] += part[i].p.q*(add+field_tot);
614  LOG_FORCES(fprintf(stderr, "%d: part %d force %10.3g %10.3g %10.3g\n",
615  this_node, part[i].p.identity, part[i].f.f[0],
