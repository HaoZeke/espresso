45 /** Computes the three body angle interaction force and adds this
46  force to the particle forces (see \ref tclcommand_inter).
47  @param p_mid Pointer to second/middle particle.
48  @param p_left Pointer to first/left particle.
49  @param p_right Pointer to third/right particle.
50  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).
51  @param force1 returns force of particle 1
52  @param force2 returns force of particle 2
53  @return 0
54 */
55 inline int calc_angle_harmonic_force(Particle *p_mid, Particle *p_left, Particle *p_right,
56  Bonded_ia_parameters *iaparams, double force1[3], double force2[3])
57 {
58  double cosine, vec1[3], vec2[3], d1i, d2i, dist2, fac, f1=0.0, f2=0.0;
59  int j;
60 
61  cosine=0.0;
62  /* vector from p_left to p_mid */
63  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
64  dist2 = sqrlen(vec1);
65  d1i = 1.0 / sqrt(dist2);
66  for(j=0;j<3;j++) vec1[j] *= d1i;
67  /* vector from p_mid to p_right */
68  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
69  dist2 = sqrlen(vec2);
70  d2i = 1.0 / sqrt(dist2);
71  for(j=0;j<3;j++) vec2[j] *= d2i;
72  /* scalar produvt of vec1 and vec2 */
73  cosine = scalar(vec1, vec2);
74  fac = iaparams->p.angle_harmonic.bend;
75 
76  {
77  double phi,sinphi;
78  if ( cosine > TINY_COS_VALUE) cosine = TINY_COS_VALUE;
79  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
80  phi = acos(-cosine);
81 
82  sinphi = sin(phi);
83  if ( sinphi < TINY_SIN_VALUE ) sinphi = TINY_SIN_VALUE;
84  fac *= (phi - iaparams->p.angle_harmonic.phi0)/sinphi;
85 
86 #ifdef CONFIGTEMP
87  extern double configtemp[2];
88  if (p_left->p.configtemp) {
89  configtemp[0] += SQR(fac*sinphi*d1i);
90  configtemp[1] -= iaparams->p.angle_harmonic.bend*(1+cosine/sinphi*(phi - iaparams->p.angle_harmonic.phi0))*SQR(d1i);
91  }
92  if (p_mid->p.configtemp) {
93  configtemp[0] += SQR(fac*sinphi) * (1./sqrlen(vec1) + 1./sqrlen(vec2) - 2*cosine*d1i*d2i);
94  configtemp[1] -= iaparams->p.angle_harmonic.bend*((1/sqrlen(vec1)+1/sqrlen(vec2)-cosine*d1i*d2i)
95  +cosine/sinphi * (1./sqrlen(vec1)+1./sqrlen(vec2)-d1i*d2i/cosine) *(phi - iaparams->p.angle_harmonic.phi0));
96  }
97  if (p_right->p.configtemp) {
98  configtemp[0] += SQR(fac*sinphi*d2i);
99  configtemp[1] -= iaparams->p.angle_harmonic.bend*(1+cosine/sinphi*(phi - iaparams->p.angle_harmonic.phi0))*SQR(d2i);
100  }
101 
102 #endif
103 
104  }
105  for(j=0;j<3;j++) {
106  f1 = fac * (cosine * vec1[j] - vec2[j]) * d1i;
107  f2 = fac * (cosine * vec2[j] - vec1[j]) * d2i;
108 
109  force1[j] = (f1-f2);
110  force2[j] = -f1;
111  }
112 
113  return 0;
114 }

116 /* The force on each particle due to a three-body bonded potential
117  is computed. */
118 inline void calc_angle_harmonic_3body_forces(Particle *p_mid, Particle *p_left,
119  Particle *p_right, Bonded_ia_parameters *iaparams,
120  double force1[3], double force2[3], double force3[3]) {
121 
122  int j;
123  double pot_dep;
124  double cos_phi;
125  double sin_phi;
126  double vec31[3];
127  double vec21[3];
128  double vec12[3]; // espresso convention
129  double vec21_sqr;
130  double vec31_sqr;
131  double vec21_magn;
132  double vec31_magn;
133  double fj[3];
134  double fk[3];
135  double fac;
136 
137  get_mi_vector(vec12, p_mid->r.p, p_left->r.p);
138  for(j = 0; j < 3; j++)
139  vec21[j] = -vec12[j];
140 
141  get_mi_vector(vec31, p_right->r.p, p_mid->r.p);
142  vec21_sqr = sqrlen(vec21);
143  vec21_magn = sqrt(vec21_sqr);
144  vec31_sqr = sqrlen(vec31);
145  vec31_magn = sqrt(vec31_sqr);
146  cos_phi = scalar(vec21, vec31) / (vec21_magn * vec31_magn);
147  sin_phi = sqrt(1.0 - SQR(cos_phi));
148 
149  /* uncomment this block if interested in the angle
150  if(cos_phi < -1.0) cos_phi = -TINY_COS_VALUE;
151  if(cos_phi > 1.0) cos_phi = TINY_COS_VALUE;
152  phi = acos(cos_phi);
153  */
154  {
155  double K, phi, phi0;
156  if(cos_phi < -1.0) cos_phi = -TINY_COS_VALUE;
157  if(cos_phi > 1.0) cos_phi = TINY_COS_VALUE;
158  phi = acos(cos_phi);
159 
160  K = iaparams->p.angle_harmonic.bend;
161  phi0 = iaparams->p.angle_harmonic.phi0;
162 
163  // potential dependent term [dU/dphi = K * (phi - phi0)]
164  pot_dep = K * (phi - phi0);
165  }
166 
167  fac = pot_dep / sin_phi;
168 
169  for(j = 0; j < 3; j++) {
170  fj[j] = vec31[j] / (vec21_magn * vec31_magn) - cos_phi * vec21[j] / vec21_sqr;
171  fk[j] = vec21[j] / (vec21_magn * vec31_magn) - cos_phi * vec31[j] / vec31_sqr;
172  }
173 
174  // note that F1 = -(F2 + F3)
175  for(j = 0; j < 3; j++) {
176  force1[j] = force1[j] - fac * (fj[j] + fk[j]);
177  force2[j] = force2[j] + fac * fj[j];
178  force3[j] = force3[j] + fac * fk[j];
179  }
180 }

183 /** Computes the three body angle interaction energy (see \ref tclcommand_inter, \ref tclcommand_analyze).
184  @param p_mid Pointer to first particle.
185  @param p_left Pointer to second/middle particle.
186  @param p_right Pointer to third particle.
187  @param iaparams bond type number of the angle interaction (see \ref tclcommand_inter).
188  @param _energy return energy pointer.
189  @return 0.
190 */
191 inline int angle_harmonic_energy(Particle *p_mid, Particle *p_left, Particle *p_right,
192  Bonded_ia_parameters *iaparams, double *_energy)
193 {
194  double cosine, vec1[3], vec2[3], d1i, d2i, dist2;
195  int j;
196 
197  cosine=0.0;
198  /* vector from p_mid to p_left */
199  get_mi_vector(vec1, p_mid->r.p, p_left->r.p);
200  dist2 = sqrlen(vec1);
201  d1i = 1.0 / sqrt(dist2);
202  for(j=0;j<3;j++) vec1[j] *= d1i;
203  /* vector from p_right to p_mid */
204  get_mi_vector(vec2, p_right->r.p, p_mid->r.p);
205  dist2 = sqrlen(vec2);
206  d2i = 1.0 / sqrt(dist2);
207  for(j=0;j<3;j++) vec2[j] *= d2i;
208  /* scalar produvt of vec1 and vec2 */
209  cosine = scalar(vec1, vec2);
210  if ( cosine > TINY_COS_VALUE) cosine = TINY_COS_VALUE;
211  if ( cosine < -TINY_COS_VALUE) cosine = -TINY_COS_VALUE;
212  /* bond angle energy */
213  {
214  double phi;
215  phi = acos(-cosine);
216  *_energy = 0.5*iaparams->p.angle_harmonic.bend*SQR(phi - iaparams->p.angle_harmonic.phi0);
217  }
218  return 0;
219 }
