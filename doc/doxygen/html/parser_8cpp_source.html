34 int parse_int_list(Tcl_Interp *interp, char *list, IntList *il) {
35  int tmp_argc, res = 1;
36  char **tmp_argv;
37  Tcl_SplitList(interp, list, &tmp_argc, &tmp_argv);
38  realloc_intlist(il, il->n = tmp_argc);
39  for (int i = 0 ; i < tmp_argc; i++)
40  if (Tcl_GetInt(interp, tmp_argv[i], &(il->e[i])) == TCL_ERROR) {
41  res = 0;
42  break;
43  }
44  Tcl_Free((char *)tmp_argv);

62 int gather_runtime_errors(Tcl_Interp *interp, int error_code) {
63  list<string> errors = mpiRuntimeErrorCollectorGather();
64 
65  if (errors.size() == 0) return error_code;
66 
67  /* reset any results of the previous command, since we got an error
68  during evaluation, they are at best bogus. But any normal error
69  messages should be kept, they might help to track down the
70  problem.
71  */
72  if (error_code != TCL_ERROR)
73  Tcl_ResetResult(interp);
74  else
75  Tcl_AppendResult(interp, " ", (char *) NULL);
76 
77  Tcl_AppendResult(interp, "background_errors ", (char *) NULL);
78 
79  for (list<string>::const_iterator it = errors.begin();
80  it != errors.end(); ++it)
81  Tcl_AppendResult(interp, it->c_str(), (char*)NULL);
82 
83  return TCL_ERROR;
84 }
