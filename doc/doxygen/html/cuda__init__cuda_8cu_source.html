55 int cuda_check_gpu(int dev)
56 {
57  cudaDeviceProp deviceProp;
58  cudaError_t error = cudaGetDeviceProperties(&deviceProp, dev);
59  if (error != cudaSuccess) {
60  cuda_error = cudaGetErrorString(error);
61  return ES_ERROR;
62  }
63  if (deviceProp.major < computeCapabilityMinMajor ||
64  (deviceProp.major == computeCapabilityMinMajor &&
65  deviceProp.minor < computeCapabilityMinMinor)) {
66  cuda_error = "compute capability insufficient";
67  return ES_ERROR;
68  }
69  return ES_OK;
