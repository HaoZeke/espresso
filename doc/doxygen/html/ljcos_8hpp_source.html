40 inline void add_ljcos_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
41  double d[3], double dist, double force[3])
42 {
43  int j;
44  double r_off, frac2, frac6, fac=0.0;
45 
46  if(CUTOFF_CHECK(dist < ia_params->LJCOS_cut+ia_params->LJCOS_offset)) {
47  r_off = dist - ia_params->LJCOS_offset;
48  /* cos part of ljcos potential. */
49  if(dist > ia_params->LJCOS_rmin+ia_params->LJCOS_offset) {
50  fac = (r_off/dist) * ia_params->LJCOS_alfa * ia_params->LJCOS_eps * (sin(ia_params->LJCOS_alfa * SQR(r_off) + ia_params->LJCOS_beta));
51  for(j=0;j<3;j++)
52  force[j] += fac * d[j];
53  }
54  /* lennard-jones part of the potential. */
55  else if(dist > 0) {
56  frac2 = SQR(ia_params->LJCOS_sig/r_off);
57  frac6 = frac2*frac2*frac2;
58  fac = 48.0 * ia_params->LJCOS_eps * frac6*(frac6 - 0.5) / (r_off * dist);
59 
60  for(j=0;j<3;j++)
61  force[j] += fac * d[j];
62 
63 #ifdef LJ_WARN_WHEN_CLOSE
64  if(fac*dist > 1000) fprintf(stderr,"%d: LJCOS-Warning: Pair (%d-%d) force=%f dist=%f\n",
65  this_node,p1->p.identity,p2->p.identity,fac*dist,dist);
66 #endif
67  }
68  /* this should not happen! */
69  else {
70  LJ_TRACE(fprintf(stderr, "%d: Lennard-Jones warning: Particles id1=%d id2=%d exactly on top of each other\n",this_node,p1->p.identity,p2->p.identity));
71 
72  frac2 = SQR(ia_params->LJ_sig/ia_params->LJ_capradius);
73  frac6 = frac2*frac2*frac2;
74  fac = 48.0 * ia_params->LJ_eps * frac6*(frac6 - 0.5) / ia_params->LJ_capradius;
75 
76  force[0] += fac * ia_params->LJ_capradius;
77  }
78 
79  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: LJ f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist,fac));
80  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: LJ f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist,fac));
81 
82  LJ_TRACE(fprintf(stderr,"%d: LJ: Pair (%d-%d) dist=%.3f: force+-: (%.3e,%.3e,%.3e)\n",
83  this_node,p1->p.identity,p2->p.identity,dist,fac*d[0],fac*d[1],fac*d[2]));
84  }
85 }

88 inline double ljcos_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
89  double d[3], double dist)
90 {
91  double r_off, frac2, frac6;
92 
93  if(CUTOFF_CHECK(dist < ia_params->LJCOS_cut+ia_params->LJCOS_offset)) {
94  r_off = dist-ia_params->LJCOS_offset;
95  /* lennard-jones part of the potential. */
96  if (dist < (ia_params->LJCOS_rmin+ia_params->LJCOS_offset)) {
97  //printf("this is nomal , %.3e \n",r_off);
98  frac2 = SQR(ia_params->LJCOS_sig/r_off);
99  frac6 = frac2*frac2*frac2;
100  return 4.0*ia_params->LJCOS_eps*(SQR(frac6)-frac6);
101  }
102  /* cosine part of the potential. */
103  else if (dist < (ia_params->LJCOS_cut+ia_params->LJCOS_offset)) {
104  return .5*ia_params->LJCOS_eps*(cos(ia_params->LJCOS_alfa*SQR(r_off)+ia_params->LJCOS_beta)-1.);
105  }
106  /* this should not happen! */
107  else {
108  fprintf(stderr,"this is the distance, which is negative %.3e\n",r_off);
109  }
110  }
111  return 0.0;
112 }
