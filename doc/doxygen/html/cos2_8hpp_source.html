44 /** Calculate cos2 force between particle p1 and p2 */
45 inline void add_cos2_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
46  double d[3], double dist, double force[3])
47 {
48  int j;
49  double fac=0.0;
50  if(CUTOFF_CHECK(dist < ia_params->COS2_cut)) {
51  if (dist > ia_params->COS2_offset) {
52  fac = -ia_params->COS2_eps*M_PI/2/ia_params->COS2_w/dist * sin(M_PI*(dist-ia_params->COS2_offset)/ia_params->COS2_w);
53  }
54  for(j=0;j<3;j++)
55  force[j] += fac * d[j];
56  }
57 }

59 /** calculate cos2 energy between particle p1 and p2. */
60 inline double cos2_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
61  double d[3], double dist)
62 {
63 
64  if(CUTOFF_CHECK(dist < ia_params->COS2_cut)) {
65  if (dist < ia_params->COS2_offset)
66  return -ia_params->COS2_eps;
67  else
68  return -ia_params->COS2_eps/2 * (cos(M_PI*(dist-ia_params->COS2_offset)/ia_params->COS2_w)+1);
69  }
70  return 0.0;
71 }
