185 int fft_find_comm_groups(int grid1[3], int grid2[3], int *node_list1, int *node_list2,
186  int *group, int *pos, int *my_pos)
187 {
188  int i;
189  /* communication group cell size on grid1 and grid2 */
190  int s1[3], s2[3];
191  /* The communication group cells build the same super grid on grid1 and grid2 */
192  int ds[3];
193  /* communication group size */
194  int g_size=1;
195  /* comm. group cell index */
196  int gi[3];
197  /* position of a node in a grid */
198  int p1[3], p2[3];
199  /* node identity */
200  int n;
201  /* this_node position in the communication group. */
202  int c_pos=-1;
203  /* flag for group identification */
204  int my_group=0;
205 
206  FFT_TRACE(fprintf(stderr,"%d: fft_find_comm_groups:\n",this_node));
207  FFT_TRACE(fprintf(stderr,"%d: for grid1=(%d,%d,%d) and grids=(%d,%d,%d)\n",
208  this_node,grid1[0],grid1[1],grid1[2],grid2[0],grid2[1],grid2[2]));
209 
210  /* calculate dimension of comm. group cells for both grids */
211  if( (grid1[0]*grid1[1]*grid1[2]) != (grid2[0]*grid2[1]*grid2[2]) ) return -1; /* unlike number of nodes */
212  for(i=0;i<3;i++) {
213  s1[i] = grid1[i] / grid2[i];
214  if(s1[i] == 0) s1[i] = 1;
215  else if(grid1[i] != grid2[i]*s1[i]) return -1; /* grids do not match!!! */
216 
217  s2[i] = grid2[i] / grid1[i];
218  if(s2[i] == 0) s2[i] = 1;
219  else if(grid2[i] != grid1[i]*s2[i]) return -1; /* grids do not match!!! */
220 
221  ds[i] = grid2[i] / s2[i];
222  g_size *= s2[i];
223  }
224 
225  /* calc node_list2 */
226  /* loop through all comm. group cells */
227  for(gi[2] = 0; gi[2] < ds[2]; gi[2]++)
228  for(gi[1] = 0; gi[1] < ds[1]; gi[1]++)
229  for(gi[0] = 0; gi[0] < ds[0]; gi[0]++) {
230  /* loop through all nodes in that comm. group cell */
231  for(i=0;i<g_size;i++) {
232  p1[0] = (gi[0]*s1[0]) + (i%s1[0]);
233  p1[1] = (gi[1]*s1[1]) + ((i/s1[0])%s1[1]);
234  p1[2] = (gi[2]*s1[2]) + (i/(s1[0]*s1[1]));
235 
236  p2[0] = (gi[0]*s2[0]) + (i%s2[0]);
237  p2[1] = (gi[1]*s2[1]) + ((i/s2[0])%s2[1]);
238  p2[2] = (gi[2]*s2[2]) + (i/(s2[0]*s2[1]));
239 
240  n = node_list1[ get_linear_index(p1[0],p1[1],p1[2],grid1) ];
241  node_list2[ get_linear_index(p2[0],p2[1],p2[2],grid2) ] = n ;
242 
243  pos[3*n+0] = p2[0]; pos[3*n+1] = p2[1]; pos[3*n+2] = p2[2];
244  if(my_group==1) group[i] = n;
245  if(n==this_node && my_group==0) {
246  my_group = 1;
247  c_pos = i;
248  my_pos[0] = p2[0]; my_pos[1] = p2[1]; my_pos[2] = p2[2];
249  i=-1; /* restart the loop */
250  }
251  }
252  my_group=0;
253  }
254 
255  /* permute comm. group according to the nodes position in the group */
256  /* This is necessary to have matching node pairs during communication! */
257  while( c_pos>0 ) {
258  n=group[g_size-1];
259  for(i=g_size-1; i>0; i--) group[i] = group[i-1];
260  group[0] = n;
261  c_pos--;
262  }
263  return g_size;
264 }

266 int fft_calc_local_mesh(int n_pos[3], int n_grid[3], int mesh[3], double mesh_off[3],
267  int loc_mesh[3], int start[3])
268 {
269  int i, last[3], size=1;
270 
271  for(i=0;i<3;i++) {
272  start[i] = (int)ceil( (mesh[i]/(double)n_grid[i])*n_pos[i] - mesh_off[i] );
273  last[i] = (int)floor((mesh[i]/(double)n_grid[i])*(n_pos[i]+1) - mesh_off[i] );
274  /* correct round off errors */
275  if( (mesh[i]/(double)n_grid[i])*(n_pos[i]+1) - mesh_off[i] - last[i] < 1.0e-15 ) last[i]--;
276  if(1.0+ (mesh[i]/(double)n_grid[i])*n_pos[i]-mesh_off[i]-start[i] < 1.0e-15 ) start[i]--;
277  loc_mesh[i] = last[i]-start[i]+1;
278  size *= loc_mesh[i];
279  }
280  return size;
281 }
