44  The neighbor index is into \ref node_neighbors is stored,

46  will have 6 entries in \ref node_neighbours, all pointing to itself).*/
47 void le_dd_comms_manager :: init(int neighbor_count){
48 
49  int i, cnt;
50 
51  comms_order = new unsigned short int [neighbor_count];
52 
53  i = 0;
54  comms_order[i++] = 2; /* regular y neighbors come first */
55  comms_order[i++] = 3;

74 /** Change some state based on a new lees-edwards offset,
75  variables saved are \ref total_xcells,
76  \ref le_offset_cells_down, \ref le_offset_cells_up*/
77 int le_dd_comms_manager :: update_on_le_offset_change(){
78 
79  double le_dx;
80  int old_off_up, old_off_down;
81 
82  old_off_up = le_offset_cells_up;
83  old_off_down = le_offset_cells_down;
84 
85  total_xcells = dd.cell_grid[0] * node_grid[0];
86 
87  le_dx = lees_edwards_offset;
88  while( le_dx < 0.0 ) {le_dx += box_l[0]; }
89  while( le_dx >= box_l[0] ){le_dx -= box_l[0]; }
90 
91  /* round to the right (up) on transfer from lower to upper
92  because the extra neighbor cell is provided to the right. */
93  le_offset_cells_up = int(le_dx * dd.inv_cell_size[0]);
94  if( le_dx * dd.inv_cell_size[0] > (float)le_offset_cells_up )
95  le_offset_cells_up++;
96 
97  /* round to the right (down) on transfer from upper to lower as well
98  because the extra neighbor cell is also provided to the right. */
99  le_offset_cells_down = -1 * int(le_dx * dd.inv_cell_size[0]);
100  while(le_offset_cells_down < 0) le_offset_cells_down += total_xcells;
101 
102  if( le_offset_cells_down == old_off_down
103  && le_offset_cells_up == old_off_up ){
104  cells_shifted = LE_CELLS_SAME;
105  return( LE_CELLS_SAME );
106  }else{
107  cells_shifted = LE_CELLS_SHIFTED;
108  return( LE_CELLS_SHIFTED );
109  }
110 }
