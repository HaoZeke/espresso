40 /** Calculate Gaussian force between particle p1 and p2 */
41 inline void add_gaussian_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params, double d[3], double dist, double dist2, double force[3])
42 {
43  double fac;
44  int j;
45  if (CUTOFF_CHECK(dist < ia_params->Gaussian_cut)) {
46  fac = ia_params->Gaussian_eps/pow(ia_params->Gaussian_sig, 2) *
47  exp(-0.5* SQR(dist / ia_params->Gaussian_sig));
48 
49  for(j=0;j<3;j++)
50  force[j] += fac * d[j];
51  }
52 }

54 /** calculate Lennard jones energy between particle p1 and p2. */
55 inline double gaussian_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params, double d[3], double dist, double dist2)
56 {
57  if (CUTOFF_CHECK(dist < ia_params->Gaussian_cut)) {
58  return ia_params->Gaussian_eps *
59  exp(-0.5* SQR(dist / ia_params->Gaussian_sig));
60  }
61  return 0.0;
