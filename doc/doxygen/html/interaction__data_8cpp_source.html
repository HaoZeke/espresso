356 static void recalc_maximal_cutoff_bonded()
357 {
358  int i;
359  double max_cut_tmp;
360 
361  max_cut_bonded = 0.0;
362 
363  for (i = 0; i < n_bonded_ia; i++) {
364  switch (bonded_ia_params[i].type) {
365  case BONDED_IA_FENE:
366  max_cut_tmp = bonded_ia_params[i].p.fene.r0+bonded_ia_params[i].p.fene.drmax;
367  if(max_cut_bonded < max_cut_tmp)
368  max_cut_bonded = max_cut_tmp;
369  break;
370  case BONDED_IA_HARMONIC:
371  if((bonded_ia_params[i].p.harmonic.r_cut>0)&&(max_cut_bonded < bonded_ia_params[i].p.harmonic.r_cut))
372  max_cut_bonded = bonded_ia_params[i].p.harmonic.r_cut;
373  break;
374  case BONDED_IA_SUBT_LJ:
375  if(max_cut_bonded < bonded_ia_params[i].p.subt_lj.r)
376  max_cut_bonded = bonded_ia_params[i].p.subt_lj.r;
377  break;
378  case BONDED_IA_RIGID_BOND:
379  if(max_cut_bonded < sqrt(bonded_ia_params[i].p.rigid_bond.d2))
380  max_cut_bonded = sqrt(bonded_ia_params[i].p.rigid_bond.d2);
381  break;
382 #ifdef TABULATED
383  case BONDED_IA_TABULATED:
384  if(bonded_ia_params[i].p.tab.type == TAB_BOND_LENGTH &&
385  max_cut_bonded < bonded_ia_params[i].p.tab.maxval)
386  max_cut_bonded = bonded_ia_params[i].p.tab.maxval;
387  break;
388 #endif
389 #ifdef OVERLAPPED
390  case BONDED_IA_OVERLAPPED:
391  /* in UNIT Angstrom */
392  if(bonded_ia_params[i].p.overlap.type == OVERLAP_BOND_LENGTH &&
393  max_cut_bonded < bonded_ia_params[i].p.overlap.maxval)
394  max_cut_bonded = bonded_ia_params[i].p.overlap.maxval;
395  break;
396 #endif
397 #ifdef IMMERSED_BOUNDARY
398  case BONDED_IA_IBM_TRIEL:
399  if(max_cut_bonded < bonded_ia_params[i].p.ibm_triel.maxdist)
400  max_cut_bonded = bonded_ia_params[i].p.ibm_triel.maxdist;
401  break;
402 #endif
403  default:
404  break;
405  }
406  }
407 
408  /* Bond angle and dihedral potentials do not contain a cutoff
409  intrinsically. The cutoff for these potentials depends on the
410  bond length potentials. For bond angle potentials nothing has to
411  be done (it is assumed, that particles participating in a bond
412  angle or dihedral potential are bound to each other by some bond
413  length potential (FENE, Harmonic or tabulated)). For dihedral
414  potentials (both normal and tabulated ones) it follows, that the
415  cutoff is TWO TIMES the maximal cutoff! That's what the following
416  lines assure. */
417  max_cut_tmp = 2.0*max_cut_bonded;
418  for (i = 0; i < n_bonded_ia; i++) {
419  switch (bonded_ia_params[i].type) {
420  case BONDED_IA_DIHEDRAL:
421  max_cut_bonded = max_cut_tmp;
422  break;
423 #ifdef TABULATED
424  case BONDED_IA_TABULATED:
425  if(bonded_ia_params[i].p.tab.type == TAB_BOND_DIHEDRAL)
426  max_cut_bonded = max_cut_tmp;
427  break;
428 #endif
429 #ifdef OVERLAPPED
430  case BONDED_IA_OVERLAPPED:
431  if(bonded_ia_params[i].p.overlap.type == OVERLAP_BOND_DIHEDRAL)
432  max_cut_bonded = max_cut_tmp;
433  break;
434 #endif
435  default:
436  break;
437  }
438  }

441 static void recalc_global_maximal_nonbonded_cutoff()
442 {
443  /* user defined minimal global cut. This makes sure that data of
444  pairs of particles with a distance smaller than this are always
445  available on the same node (through ghosts). Required for example
446  for the relative virtual sites algorithm. */
447  max_cut_global = min_global_cut;
448 
449 #ifdef ELECTROSTATICS
450  /* Cutoff for the real space electrostatics.
451  Note that the box length may have changed,
452  but the method not yet reinitialized.
453  */
454  switch (coulomb.method) {
455 #ifdef P3M
456  case COULOMB_ELC_P3M:
457  if (max_cut_global < elc_params.space_layer)
458  max_cut_global = elc_params.space_layer;
459  // fall through
460  case COULOMB_P3M_GPU:
461  case COULOMB_P3M: {
462  /* do not use precalculated r_cut here, might not be set yet */
463  double r_cut = p3m.params.r_cut_iL* box_l[0];
464  if (max_cut_global < r_cut)
465  max_cut_global = r_cut;
466  break;
467  }
468 #endif
469 #ifdef EWALD_GPU
470  case COULOMB_EWALD_GPU:
471  if (max_cut_global < ewaldgpu_params.rcut)
472  max_cut_global = ewaldgpu_params.rcut;
473  break;
474 #endif
475  case COULOMB_DH:
476  if (max_cut_global < dh_params.r_cut)
477  max_cut_global = dh_params.r_cut;
478  break;
479  case COULOMB_RF:
480  case COULOMB_INTER_RF:
481  if (max_cut_global < rf_params.r_cut)
482  max_cut_global = rf_params.r_cut;
483  break;
484  default:
485  break;
486  }
487 #endif /*ifdef ELECTROSTATICS */
488 
489 #ifdef DIPOLES
490  switch (coulomb.Dmethod) {
491 #ifdef DP3M
492  case DIPOLAR_MDLC_P3M:
493  // fall through
494  case DIPOLAR_P3M: {
495  /* do not use precalculated r_cut here, might not be set yet */
496  double r_cut = dp3m.params.r_cut_iL* box_l[0];
497  if (max_cut_global < r_cut)
498  max_cut_global = r_cut;
499  break;
500  }
501 #endif /*ifdef DP3M */
502  default:
503  break;
504  }
505 #endif
506 
507 #ifdef DPD
508  if (dpd_r_cut != 0) {
509  if(max_cut_global < dpd_r_cut)
510  max_cut_global = dpd_r_cut;
511  }
512 #endif
513 
514 #ifdef TRANS_DPD
515  if (dpd_tr_cut != 0) {
516  if(max_cut_global < dpd_tr_cut)
517  max_cut_global = dpd_tr_cut;
518  }
519 #endif
520 
521 }
