117 void tclcommand_analyze_print_vel_distr(Tcl_Interp *interp, int type, int bins, double given_max) {

123  max = given_max*time_step;

130  centermass_vel(type, com);

136  vel = partCfg[i].m.v[j] - com[j];

138  min = vel;
139  }
140  if (max < vel) {
141  max = vel;
142  }
143  }
144  }
145  }
146 
147  if (p_count == 0) {
148  return;
149  }
150

152  max = -min;
153  } else {

155  }

164  ind = (int) ((vel - min) * inv_bin_width);

172  Tcl_AppendResult(interp, " {\n", (char *) NULL);

176  vel += bin_width;
177  }
178  Tcl_AppendResult(interp, "}\n", (char *) NULL);
