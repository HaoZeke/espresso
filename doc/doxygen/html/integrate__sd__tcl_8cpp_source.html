122 int tclcommand_integrate_sd(ClientData data, Tcl_Interp *interp, int argc, char **argv)
123 {
124  #ifdef SD
125  int n_steps;
126 
127  INTEG_TRACE(fprintf(stderr,"%d: integrate:\n",this_node));
128 
129  if (argc < 1) {
130  Tcl_AppendResult(interp, "wrong # args: \n\"", (char *) NULL);
131  return tclcommand_integrate_sd_print_usage(interp);
132  }
133  else if (argc < 2) {
134  return tclcommand_integrate_sd_print_status(interp);
135  }
136 
137  if (ARG1_IS_S("set")) {
138  if (argc < 3){
139  return tclcommand_integrate_sd_print_status(interp);
140  }
141 
142  Tcl_AppendResult(interp, "unknown integrator method:\n", (char *)NULL);
143  return tclcommand_integrate_sd_print_usage(interp);
144 
145  } else if ( !ARG_IS_I(1,n_steps) ) return tclcommand_integrate_sd_print_usage(interp);
146 
147  /* go on with integrate <n_steps> */
148  if(n_steps <= 0) {
149  Tcl_AppendResult(interp, "illegal number of steps (must be >0) \n", (char *) NULL);
150  return tclcommand_integrate_sd_print_usage(interp);
151  }
152  /* perform integration */
153  if (!correlations_autoupdate && !observables_autoupdate) {
154  integrate_sd(n_steps);
155  } else {
156  for (int i=0; i<n_steps; i++) {
157  integrate_sd(1);
158  autoupdate_observables();
159  autoupdate_correlations();
160  }
161  if (n_steps == 0){
162  integrate_sd(0);
163  }
164  }
165  ;
166  return gather_runtime_errors(interp, TCL_OK);
167  #else
168  return TCL_ERROR;
169  #endif
170 }
