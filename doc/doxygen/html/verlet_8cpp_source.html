53 /** Add a particle pair to a verlet pair list.
54  Checks verlet pair list size and reallocates memory if necessary.
55  * \param p1 Pointer to particle one.
56  * \param p2 Pointer to particle two.
57  * \param pl Pointer to the verlet pair list.
58  */
59 inline void add_pair(PairList *pl, Particle *p1, Particle *p2)
60 {
61  /* check size of verlet List */
62  if(pl->n+1 >= pl->max) {
63  pl->max += LIST_INCREMENT;
64  pl->pair = (Particle **)realloc(pl->pair, 2*pl->max*sizeof(Particle *));
65  }
66  /* add pair */
67  pl->pair[(2*pl->n) ] = p1;
68  pl->pair[(2*pl->n)+1] = p2;
69  /* increase number of pairs */
70  pl->n++;
71 }

172  /* Loop local cells */

174  cell = local_cells.cell[c];
175  p1 = cell->part;
176  np = cell->n;
177  /* calculate bonded interactions (loop local particles) */
178  for(i = 0; i < np; i++) {
179 #ifdef MULTI_TIMESTEP
180  if (p1[i].p.smaller_timestep==current_time_step_is_small || smaller_time_step < 0.)
181 #endif
182  {
183  add_bonded_force(&p1[i]);
184 #ifdef CONSTRAINTS
185  add_constraints_forces(&p1[i]);
186 #endif
187  add_external_potential_forces(&p1[i]);
188  }
189  }
190 
191  /* Loop cell neighbors */
192  for (n = 0; n < dd.cell_inter[c].n_neighbors; n++) {
193  pairs = dd.cell_inter[c].nList[n].vList.pair;
194  np = dd.cell_inter[c].nList[n].vList.n;
195  /* verlet list loop */
196  for(i=0; i<2*np; i+=2) {
197  p1 = pairs[i]; /* pointer to particle 1 */
198  p2 = pairs[i+1]; /* pointer to particle 2 */
199 #ifdef MULTI_TIMESTEP
200  if (smaller_time_step < 0.
201  || (p1->p.smaller_timestep==0 && p2->p.smaller_timestep==0 && current_time_step_is_small==0)
202  || (!(p1->p.smaller_timestep==0 && p2->p.smaller_timestep==0) && current_time_step_is_small==1))
203 #endif
204  {
205  dist2 = distance2vec(p1->r.p, p2->r.p, vec21);
206  add_non_bonded_pair_force(p1, p2, vec21, sqrt(dist2), dist2);
207  }
208  }
209  }
210  }
