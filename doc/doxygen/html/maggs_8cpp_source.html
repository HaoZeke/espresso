23 /** \file maggs.cpp
24  * Maxwell Equations Molecular Dynamics (MEMD) method for electrostatic
25  * interactions.
26  *
27  * We use a local update scheme to propagate artificial B-fields on a
28  * lattice in the system. In principal, the algorithm simulates full
29  * electrodynamics, but with a tunable speed of light.
30  *
31  * The method is very usable for large particle numbers or highly
32  * parallel architectures, since it is local and scales linearly.
33  * It is not suited for high-precision calculation of forces, since
34  * the simple interpolation scheme produces errors in the order of
35  * 10^-5 in the force.
36  *
37  * The chosen mesh should roughly be of the size of the particles.
38  *
39  * Further reading on the algorithm:
40  * I. Pasichnyk and B. Dunweg, Coulomb interaction via local
41  * dynamics: a molecular-dynamics algorithm. J. Phys:
42  * Condens. Matter, 16 ,p. 1399-4020, (2004).
43  *
44  */

340 /** Calculates the finite differences rotation in real space in mue-nue plane:
341  \f$\frac{\partial}{\partial t}{D} = \nabla \times B\f$ (and prefactors plus current)
342  The given "double* field" should be a B-Field!!
343  @return rotation result
344  @param mue direction 1 of plane to rotate in
345  @param nue direction 2 of plane to rotate in
346  @param field input B-field
347  @param Neighbor neighbor lattice site
348  @param index index of current lattice site
349 */
350 double maggs_calc_curl(int mue, int nue, double* field, int* Neighbor, int index)
351 {
352  double result;
353 
354  result = field[index+mue] + field[3*Neighbor[OPP_DIR(mue)]+nue] -
355  field[3*Neighbor[OPP_DIR(nue)]+mue] - field[index+nue];
356 
357  return result;
358 }

488 void maggs_compute_dipole_correction()
489 {
490 /*
491 maggs.prefactor = sqrt(4. * M_PI * maggs.bjerrum * temperature);
492  = sqrt(1/epsilon)
493 maggs.pref2 = maggs.bjerrum * temperature;
494  = 1/4*pi*epsilon
495 */
496 
497  /* Local dipole moment */
498  double local_dipole_moment[3] = {0.0, 0.0, 0.0};
499  /* Global dipole moment */
500  double dipole_moment[3];
501 
502  int dim,c,np,i;
503  Particle* p;
504  Cell* cell;
505 
506  /* Compute the global dipole moment */
507  for (c = 0; c < local_cells.n; c++) {
508  cell = local_cells.cell[c];
509  p = cell->part;
510  np = cell->n;
511  for(i=0; i<np; i++)
512  for (dim=0;dim<3;dim++)
513  local_dipole_moment[dim] += p[i].r.p[dim] * p[i].p.q;
514  }
515 
516  MPI_Allreduce(local_dipole_moment, dipole_moment, 3, MPI_DOUBLE, MPI_SUM, comm_cart);
517 
518  double volume = box_l[0] * box_l[1] * box_l[2];
519  double dipole_prefactor = 4.0*M_PI / (2.0*volume*(maggs.epsilon_infty + 1.0));
520 
521  /* apply correction to all particles: */
522  for (c = 0; c < local_cells.n; c++) {
523  cell = local_cells.cell[c];
524  p = cell->part;
525  np = cell->n;
526  for(i=0; i<np; i++)
527  for (dim=0;dim<3;dim++)
528  p[i].f.f[dim] += p[i].p.q * dipole_prefactor * dipole_moment[dim];
529  }
530 }

1172  int py = 0;

1176  double sy = 0.;

1179  double sxyz = 0.;

1184  // {0.,0.,1.}, {1.,0.,1.}, {0.,1.,1.}, {1.,1.,1.}};

1187  l=0;
1188  index = 0;
1189  for(i=0;i<2;i++)
1190  for(j=0;j<2;j++)
1191  for(k=0;k<2;k++) {
1192  n[index][2] = m;
1193  n[index][1] = l;
1194  n[index][0] = k;
1195  index++;
1196  }
1197

1199  prefac = 1. / maggs.mesh;
1200  prefac = 0.5 * prefac * prefac * prefac * SQR(maggs.prefactor);
1201 
1202  for(i=0;i<8;i++)
1203  {
1204  for(j=0;j<8;j++)

1208  {

1210  sx = sx * sx;
1211  nomx = 2.*factor*px*(n[i][0] - n[j][0]);
1212  for(py = 0; py < maggs.mesh; ++py)

1218  for(pz = 0; pz < maggs.mesh; ++pz)

1228  }
1229  }
1230  }
1231  }
