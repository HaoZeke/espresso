66 /** Calculate non bonded energies between a pair of particles.
67  @param p1 pointer to particle 1.
68  @param p2 pointer to particle 2.
69  @param d vector between p1 and p2.
70  @param dist distance between p1 and p2.
71  @param dist2 distance squared between p1 and p2. */
72 inline void add_non_bonded_pair_virials(Particle *p1, Particle *p2, double d[3],
73  double dist, double dist2)
74 {
75  int p1molid, p2molid, k, l;
76  double force[3] = {0, 0, 0};
77 
78  calc_non_bonded_pair_force(p1, p2,d, dist, dist2, force);
79 
80  *obsstat_nonbonded(&virials, p1->p.type, p2->p.type) += d[0]*force[0] + d[1]*force[1] + d[2]*force[2];
81 
82  /* stress tensor part */
83  for(k=0;k<3;k++)
84  for(l=0;l<3;l++)
85  obsstat_nonbonded(&p_tensor, p1->p.type, p2->p.type)[k*3 + l] += force[k]*d[l];
86 
87  p1molid = p1->p.mol_id;
88  p2molid = p2->p.mol_id;
89  if ( p1molid == p2molid ) {
90  *obsstat_nonbonded_intra(&virials_non_bonded, p1->p.type, p2->p.type) += d[0]*force[0] + d[1]*force[1] + d[2]*force[2];
91 
92  for(k=0;k<3;k++)
93  for(l=0;l<3;l++)
94  obsstat_nonbonded_intra(&p_tensor_non_bonded, p1->p.type, p2->p.type)[k*3 + l] += force[k]*d[l];
95  }
96  if ( p1molid != p2molid ) {
97  *obsstat_nonbonded_inter(&virials_non_bonded, p1->p.type, p2->p.type) += d[0]*force[0] + d[1]*force[1] + d[2]*force[2];
98 
99  for(k=0;k<3;k++)
100  for(l=0;l<3;l++)
101  obsstat_nonbonded_inter(&p_tensor_non_bonded, p1->p.type, p2->p.type)[k*3 + l] += force[k]*d[l];
102  }
103 
104 #ifdef ELECTROSTATICS
105  /* real space coulomb */
106  if (coulomb.method != COULOMB_NONE) {
107  switch (coulomb.method) {
108 #ifdef P3M
109  case COULOMB_P3M_GPU:
110  case COULOMB_P3M:
111  virials.coulomb[0] += p3m_pair_energy(p1->p.q*p2->p.q,d,dist2,dist);
112  break;
113 #endif
114 
115  /* short range potentials, where we use the virial */
116  /***************************************************/
117  case COULOMB_DH: {
118  double force[3] = {0, 0, 0};
119 
120  add_dh_coulomb_pair_force(p1,p2,d,dist, force);
121  for(k=0;k<3;k++)
122  for(l=0;l<3;l++)
123  p_tensor.coulomb[k*3 + l] += force[k]*d[l];
124  virials.coulomb[0] += force[0]*d[0] + force[1]*d[1] + force[2]*d[2];
125  break;
126  }
127  case COULOMB_RF: {
128  double force[3] = {0, 0, 0};
129 
130  add_rf_coulomb_pair_force(p1,p2,d,dist, force);
131  for(k=0;k<3;k++)
132  for(l=0;l<3;l++)
133  p_tensor.coulomb[k*3 + l] += force[k]*d[l];
134  virials.coulomb[0] += force[0]*d[0] + force[1]*d[1] + force[2]*d[2];
135  break;
136  }
137  case COULOMB_INTER_RF:
138  // this is done together with the other short range interactions
139  break;
140  default:
141  fprintf(stderr,"calculating pressure for electrostatics method that doesn't have it implemented\n");
142  break;
143  }
144  }
145 #endif /*ifdef ELECTROSTATICS */
146 
147 #ifdef DIPOLES
148  /* real space magnetic dipole-dipole */
149  if (coulomb.Dmethod != DIPOLAR_NONE) {
150  fprintf(stderr,"calculating pressure for magnetostatics which doesn't have it implemented\n");
151  }
152 #endif /*ifdef DIPOLES */
153 }

156 #ifdef TABULATED
157  //char* errtxt;

159 
160  /* Calculates the bonded force between two particles */
161  switch(iaparams->type) {
162  case BONDED_IA_FENE:
163  calc_fene_pair_force(p1,p2,iaparams,dx,force);
164  break;
165 #ifdef ROTATION
166  case BONDED_IA_HARMONIC_DUMBBELL:
167  calc_harmonic_dumbbell_pair_force(p1,p2,iaparams,dx,force);
168  break;

174  case BONDED_IA_SUBT_LJ:
175  calc_subt_lj_pair_force(p1,p2,iaparams,dx,force);
176  break;
177 #endif
178  /* since it is not clear at the moment how to handle a many body interaction here, I skip it */
179  case BONDED_IA_ANGLE_HARMONIC:
180  (*i)++; force[0] = force[1] = force[2] = 0; break;
181  case BONDED_IA_ANGLE_COSINE:
182  (*i)++; force[0] = force[1] = force[2] = 0; break;
183  case BONDED_IA_ANGLE_COSSQUARE:
184  (*i)++; force[0] = force[1] = force[2] = 0; break;
185  case BONDED_IA_ANGLEDIST:
186  (*i)++; force[0] = force[1] = force[2] = 0; break;
187  case BONDED_IA_DIHEDRAL:
188  (*i)+=2; force[0] = force[1] = force[2] = 0; break;
189 
190 #ifdef TABULATED

194  case TAB_BOND_LENGTH:
195  calc_tab_bond_force(p1, p2, iaparams, dx, force); break;
196  case TAB_BOND_ANGLE:
197  (*i)++; force[0] = force[1] = force[2] = 0; break;
198  case TAB_BOND_DIHEDRAL:
199  (*i)+=2; force[0] = force[1] = force[2] = 0; break;
200  default:
201  ostringstream msg;
202  msg <<"calc_bonded_force: tabulated bond type of atom " << p1->p.identity << " unknown\n";
203  runtimeError(msg);
204  return;
205  }
206  break;
207 #endif
208 #ifdef OVERLAPPED
209  case BONDED_IA_OVERLAPPED:
210  // printf("BONDED OVERLAP, Particle: %d, P2: %d TYPE_OVERLAP: %d\n",p1->p.identity,p2->p.identity,iparams->p.tab.type);
211  //char *errtxt;
212  switch(iaparams->p.overlap.type) {
213  case OVERLAP_BOND_LENGTH:
214  calc_overlap_bond_force(p1, p2, iaparams, dx, force); break;
215  case OVERLAP_BOND_ANGLE:
216  (*i)++; force[0] = force[1] = force[2] = 0; break;
217  case OVERLAP_BOND_DIHEDRAL:
218  (*i)+=2; force[0] = force[1] = force[2] = 0; break;
219  default:
220  ostringstream msg;
221  msg <<"calc_bonded_force: overlapped bond type of atom " << p1->p.identity << " unknown\n";
222  runtimeError(msg);
223  return;
224  }
225  break;
226 #endif
227 #ifdef BOND_CONSTRAINT
228  case BONDED_IA_RIGID_BOND:
229  force[0] = force[1] = force[2] = 0; break;
230 #endif
231 #ifdef BOND_VIRTUAL
232  case BONDED_IA_VIRTUAL_BOND:
233  force[0] = force[1] = force[2] = 0; break;
234 #endif
235  default :
236  // fprintf(stderr,"add_bonded_virials: WARNING: Bond type %d of atom %d unhandled\n",bonded_ia_params[type_num].type,p1->p.identity);
237  fprintf(stderr,"add_bonded_virials: WARNING: Bond type %d , atom %d unhandled, Atom 2: %d\n",iaparams->type,p1->p.identity,p2->p.identity);
238  force[0] = force[1] = force[2] = 0;
239  break;
240  }
241 }

298 /** Calculate bonded virials for one particle.
299  For performance reasons the force routines add their values directly to the particles.
300  So here we do some tricks to get the value out without changing the forces.
301  @param p1 particle for which to calculate virials
302 */

304 {

309 
310  int i, k, l;
311  int type_num;
312 
313  i = 0;
314  while(i<p1->bl.n) {
315  type_num = p1->bl.e[i++];
316  iaparams = &bonded_ia_params[type_num];
317 
318  /* fetch particle 2 */
319  p2 = local_particles[p1->bl.e[i++]];
320  if ( ! p2 ) {

336  for(l=0;l<3;l++)

343  The central particle of the three-particle interaction is responsible
344  for the contribution of the entire interaction - this is the coding
345  not the physics.
346 */
347 inline void add_three_body_bonded_stress(Particle *p1) {
348  double dx12[3]; // espresso notation
349  double dx21[3];
350  double dx31[3];
351  double force1[3];
352  double force2[3];
353  double force3[3];
354 
355  //char *errtxt;
356  Particle *p2;
357  Particle *p3;
358  Bonded_ia_parameters *iaparams;
359 
360  int i, k, j, l;
361  int type_num;
362  BondedInteraction type;
363 
364  i = 0;
365  while(i < p1->bl.n) {
366  /* scan bond list for angular interactions */
367  type_num = p1->bl.e[i];
368  iaparams = &bonded_ia_params[type_num];
369  type = iaparams->type;
370 
371  if(type == BONDED_IA_ANGLE_HARMONIC || type == BONDED_IA_ANGLE_COSINE || type == BONDED_IA_ANGLE_COSSQUARE){
372  p2 = local_particles[p1->bl.e[++i]];
373  p3 = local_particles[p1->bl.e[++i]];
374 
375  get_mi_vector(dx12, p1->r.p, p2->r.p);
376  for(j = 0; j < 3; j++)
377  dx21[j] = -dx12[j];
378 
379  get_mi_vector(dx31, p3->r.p, p1->r.p);
380 
381  for(j = 0; j < 3; j++) {
382  force1[j] = 0.0;
383  force2[j] = 0.0;
384  force3[j] = 0.0;
385  }
386 
387  calc_three_body_bonded_forces(p1, p2, p3, iaparams, force1, force2, force3);
388 
389  /* uncomment the next line to see that the virial is indeed zero */
390  //printf("W = %g\n", scalar(force2, dx21) + scalar(force3, dx31));
391 
392  /* three-body bonded interactions contribute to the stress but not the scalar pressure */
393  for(k = 0; k < 3; k++) {
394  for(l = 0; l < 3; l++) {
395  obsstat_bonded(&p_tensor, type_num)[3 * k + l] += force2[k] * dx21[l] + force3[k] * dx31[l];
396  }
397  }
398  i = i + 1;
399  }
400  // skip over non-angular interactions
401  else if(type == BONDED_IA_FENE) {
402  i = i + 2;
403  }
404  else if(type == BONDED_IA_STRETCHING_FORCE) {
405  i = i + 2;
406  }
407  else if(type == BONDED_IA_STRETCHLIN_FORCE) {
408  i = i + 2;
409  }
410  else if(type == BONDED_IA_AREA_FORCE_LOCAL) {
411  i = i + 3;
412  }
413  else if(type == BONDED_IA_AREA_FORCE_GLOBAL) {
414  i = i + 3;
415  }
416  else if(type == BONDED_IA_BENDING_FORCE) {
417  i = i + 4;
418  }
419  else if(type == BONDED_IA_VOLUME_FORCE) {
420  i = i + 3;
421  }
422  else if(type == BONDED_IA_HARMONIC) {
423  i = i + 2;
424  }
425 #ifdef LENNARD_JONES
426  else if(type == BONDED_IA_SUBT_LJ) {
427  i = i + 2;
428  }
429 #endif
430  else if(type == BONDED_IA_ANGLEDIST) {
431  i = i + 3;
432  }
433  else if(type == BONDED_IA_DIHEDRAL) {
434  i = i + 4;
435  }
436 #ifdef TABULATED
437  else if(type == BONDED_IA_TABULATED) {
438  if(iaparams->p.tab.type == TAB_BOND_LENGTH) {
439  i = i + 2;
440  }
441  else if(iaparams->p.tab.type == TAB_BOND_ANGLE) {
442  p2 = local_particles[p1->bl.e[++i]];
443  p3 = local_particles[p1->bl.e[++i]];
444 
445  get_mi_vector(dx12, p1->r.p, p2->r.p);
446  for(j = 0; j < 3; j++)
447  dx21[j] = -dx12[j];
448 
449  get_mi_vector(dx31, p3->r.p, p1->r.p);
450 
451  for(j = 0; j < 3; j++) {
452  force1[j] = 0.0;
453  force2[j] = 0.0;
454  force3[j] = 0.0;
455  }
456 
457  calc_three_body_bonded_forces(p1, p2, p3, iaparams, force1, force2, force3);
458 
459  /* uncomment the next line to see that the virial is indeed zero */
460  //printf("W = %g\n", scalar(force2, dx21) + scalar(force3, dx31));
461 
462  /* three-body bonded interactions contribute to the stress but not the scalar pressure */
463  for(k = 0; k < 3; k++) {
464  for(l = 0; l < 3; l++) {
465  obsstat_bonded(&p_tensor, type_num)[3 * k + l] += force2[k] * dx21[l] + force3[k] * dx31[l];
466  }
467  }
468  i = i + 1;
469  }

471  i = i + 4;

476  runtimeError(msg);
477  }

479 #endif
480 #ifdef BOND_CONSTRAINT

482  i = i + 2;
483  }
484 #endif
485 #ifdef BOND_VIRTUAL

487  i = i + 2;
488  }

498 /** Calculate kinetic pressure (aka energy) for one particle.
499  @param p1 particle for which to calculate pressure
500  @param v_comp flag which enables (1) compensation of the velocities required
501  for deriving a pressure reflecting \ref nptiso_struct::p_inst
502  (hence it only works with domain decomposition); naturally it
503  therefore doesn't make sense to use it without NpT.
504 */
505 inline void add_kinetic_virials(Particle *p1,int v_comp)
506 {
507  int k, l;
508  /* kinetic energy */
509 #ifdef MULTI_TIMESTEP
510  if (smaller_time_step > 0.) {
511  if(v_comp)
512  virials.data.e[0] += SQR(time_step/smaller_time_step)*(SQR(p1->m.v[0] - p1->f.f[0]) + SQR(p1->m.v[1] - p1->f.f[1]) + SQR(p1->m.v[2] - p1->f.f[2]))*PMASS(*p1);
513  else
514  virials.data.e[0] += SQR(time_step/smaller_time_step)*(SQR(p1->m.v[0]) + SQR(p1->m.v[1]) + SQR(p1->m.v[2]))*PMASS(*p1);
515  } else
516 #endif
517  {
518  if(v_comp)
519  virials.data.e[0] += (SQR(p1->m.v[0] - p1->f.f[0]) + SQR(p1->m.v[1] - p1->f.f[1]) + SQR(p1->m.v[2] - p1->f.f[2]))*PMASS(*p1);
520  else
521  virials.data.e[0] += (SQR(p1->m.v[0]) + SQR(p1->m.v[1]) + SQR(p1->m.v[2]))*PMASS(*p1);
522  }

524  /* ideal gas contribution (the rescaling of the velocities by '/=time_step' each will be done later) */
