198 int tclcommand_analyze_parse_and_print_pressure(Tcl_Interp *interp, int v_comp, int argc, char **argv)
199 {
200  /* 'analyze pressure [{ bond <type_num> | nonbonded <type1> <type2> | coulomb | ideal | total }]' */
201  char buffer[TCL_DOUBLE_SPACE + TCL_INTEGER_SPACE + 2];
202  int i, j;
203  double value, p_vel[3];
204  value = 0.0;
205 
206  if (n_part == 0) {
207  Tcl_AppendResult(interp, "(no particles)",
208  (char *)NULL);
209  return (TCL_OK);
210  }
211 
212  /* if desired (v_comp==1) replace ideal component with instantaneous one */
213  if (total_pressure.init_status != 1+v_comp ) {
214  init_virials(&total_pressure);
215  init_p_tensor(&total_p_tensor);
216 
217  init_virials_non_bonded(&total_pressure_non_bonded);
218  init_p_tensor_non_bonded(&total_p_tensor_non_bonded);
219 
220  if(v_comp && (integ_switch == INTEG_METHOD_NPT_ISO) && !(nptiso.invalidate_p_vel)) {
221  if (total_pressure.init_status == 0)
222  master_pressure_calc(0);
223  total_pressure.data.e[0] = 0.0;
224  MPI_Reduce(nptiso.p_vel, p_vel, 3, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
225  for(i=0; i<3; i++)
226  if(nptiso.geometry & nptiso.nptgeom_dir[i])
227  total_pressure.data.e[0] += p_vel[i];
228  total_pressure.data.e[0] /= (nptiso.dimension*nptiso.volume);
229  total_pressure.init_status = 1+v_comp; }
230  else
231  master_pressure_calc(v_comp);
232  }
233 
234  if (argc == 0)
235  tclcommand_analyze_print_pressure_all(interp);
236  else {
237 
238  if (ARG0_IS_S("ideal"))
239  value = total_pressure.data.e[0];
240  else if (ARG0_IS_S("bonded") ||
241  ARG0_IS_S("fene") ||
242  ARG0_IS_S("subt_lj_harm") ||
243  ARG0_IS_S("subt_lj_fene") ||
244  ARG0_IS_S("subt_lj") ||
245  ARG0_IS_S("harmonic") ||
246  ARG0_IS_S("umbrella") ||
247  ARG0_IS_S("endangledist")) {
248  if(argc<2 || ! ARG1_IS_I(i)) {
249  Tcl_ResetResult(interp);
250  Tcl_AppendResult(interp, "wrong # or type of arguments for: analyze pressure bonded <type_num>",
251  (char *)NULL);
252  return (TCL_ERROR);
253  }
254  if(i < 0 || i >= n_bonded_ia) {
255  Tcl_AppendResult(interp, "bond type does not exist", (char *)NULL);
256  return (TCL_ERROR);
257  }
258  value = *obsstat_bonded(&total_pressure, i);
259  }
260  else if (ARG0_IS_S("nonbonded") ||
261  ARG0_IS_S("lj") ||
262  ARG0_IS_S("buckingham") ||
263  ARG0_IS_S("morse") ||
264  ARG0_IS_S("soft-sphere") ||
265  ARG0_IS_S("lj-cos") ||
266  ARG0_IS_S("lj-cos2") ||
267  ARG0_IS_S("cos2") ||
268  ARG0_IS_S("tabulated") ||
269  ARG0_IS_S("gb")) {
270  if(argc<3 || ! ARG_IS_I(1, i) || ! ARG_IS_I(2, j)) {
271  Tcl_ResetResult(interp);
272  Tcl_AppendResult(interp, "wrong # or type of arguments for: analyze pressure nonbonded <type1> <type2>",
273  (char *)NULL);
274  return (TCL_ERROR);
275  }
276  if(i < 0 || i >= n_particle_types || j < 0 || j >= n_particle_types) {
277  Tcl_AppendResult(interp, "particle type does not exist", (char *)NULL);
278  return (TCL_ERROR);
279  }
280  value = *obsstat_nonbonded(&total_pressure, i, j);
281  }
282  else if( ARG0_IS_S("tot_nb_intra") ||
283  ARG0_IS_S("tot_nonbonded_intra")) {
284  value = 0.0;
285  for (i = 0; i < n_particle_types; i++)
286  for (j = i; j < n_particle_types; j++)
287  value += *obsstat_nonbonded_intra(&total_pressure_non_bonded, i, j);
288  }
289  else if( ARG0_IS_S("tot_nb_inter") ||
290  ARG0_IS_S("tot_nonbonded_inter")) {
291  value = 0.0;
292  for (i = 0; i < n_particle_types; i++)
293  for (j = i; j < n_particle_types; j++)
294  value += *obsstat_nonbonded_inter(&total_pressure_non_bonded, i, j);
295  }
296  else if( ARG0_IS_S("nb_intra") ||
297  ARG0_IS_S("nonbonded_intra")) {
298  if(argc<3 || ! ARG_IS_I(1, i) || ! ARG_IS_I(2, j)) {
299  Tcl_ResetResult(interp);
300  Tcl_AppendResult(interp, "wrong # or type of arguments for: analyze pressure nb_intra <type1> <type2>",
301  (char *)NULL);
302  return (TCL_ERROR);
303  }
304  if(i < 0 || i >= n_particle_types || j < 0 || j >= n_particle_types) {
305  Tcl_AppendResult(interp, "particle type does not exist", (char *)NULL);
306  return (TCL_ERROR);
307  }
308  value = *obsstat_nonbonded_intra(&total_pressure_non_bonded, i, j);
309  }
310  else if( ARG0_IS_S("nb_inter") ||
311  ARG0_IS_S("nonbonded_inter")) {
312  if(argc<3 || ! ARG_IS_I(1, i) || ! ARG_IS_I(2, j)) {

319  Tcl_AppendResult(interp, "particle type does not exist", (char *)NULL);
320  return (TCL_ERROR);
321  }
322  value = *obsstat_nonbonded_inter(&total_pressure_non_bonded, i, j);
323  }
324  else if( ARG0_IS_S("coulomb")) {
325 #ifdef ELECTROSTATICS
326  value = 0;
327  for (i = 0; i < total_pressure.n_coulomb; i++)
328  value += total_pressure.coulomb[i];
329 #else
330  Tcl_AppendResult(interp, "ELECTROSTATICS not compiled (see config.hpp)\n", (char *)NULL);
331 #endif
332  }
333  else if( ARG0_IS_S("dipolar")) {
334 #ifdef DIPOLES
335  value = 0;
336  for (i = total_pressure.n_coulomb-1; i < total_pressure.n_coulomb; i++) /*when DLC will be installed this has to be changed */
337  value += total_pressure.coulomb[i];
338 #else
339  Tcl_AppendResult(interp, "DIPOLES not compiled (see config.hpp)\n", (char *)NULL);
340 #endif
341  }
342 #ifdef VIRTUAL_SITES_RELATIVE
343  else if (ARG0_IS_S("vs_relative")) {
344  value =total_pressure.vs_relative[0];
345  }
346 #endif
347  else if (ARG0_IS_S("total")) {
348  value = total_pressure.data.e[0];
349  for (i = 1; i < total_pressure.data.n; i++) {
350  value += total_pressure.data.e[i];
351  }
352  }
353  else {
354  Tcl_AppendResult(interp, "unknown feature of: analyze pressure",
355  (char *)NULL);
356  return (TCL_ERROR);
357  }
358  Tcl_PrintDouble(interp, value, buffer);
359  Tcl_AppendResult(interp, buffer, (char *)NULL);
360  }
361 
362  return (TCL_OK);
363 }
