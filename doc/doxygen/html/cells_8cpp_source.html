68 /** Extensive Debug function to check the consistency of the cells and
69  the particles therein. Use with care! */
70 void check_cells_consistency()
71 {
72  int c, index;
73  IntList used;
74  alloc_intlist(&used, n_cells);
75  memset(used.e, 0, n_cells*sizeof(int));
76 
77  for (c = 0; c < local_cells.n; c++) {
78  index = (char *)local_cells.cell[c] - (char *)cells;
79  if ((index % sizeof(Cell)) != 0) {
80  fprintf(stderr, "%d: local cell pointer not even aligned, certainly wrong (local_cell[%d], index=%d).\n", this_node, c, index);
81  errexit();
82  }
83  index /= sizeof(Cell);
84  if (index < 0 || index >= n_cells) {
85  fprintf(stderr, "%d: local cell pointer out of range, maybe old leftover (local_cell[%d]).\n", this_node, c);
86  errexit();
87  }
88  if (used.e[index]) {
89  fprintf(stderr, "%d: local cell is already pointed to (local_cell[%d]).\n", this_node, c);
90  errexit();
91  }
92  used.e[index] = 1;
93  }
94 
95  for (c = 0; c < ghost_cells.n; c++) {
96  index = (char *)ghost_cells.cell[c] - (char *)cells;
97  if ((index % sizeof(Cell)) != 0) {
98  fprintf(stderr, "%d: ghost cell pointer not even aligned, certainly wrong (ghost_cell[%d], index=%d).\n", this_node, c, index);
99  errexit();
100  }
101  index /= sizeof(Cell);
102  if (index < 0 || index >= n_cells) {
103  fprintf(stderr, "%d: ghost cell pointer out of range, maybe old leftover (ghost_cell[%d]).\n", this_node, c);
104  errexit();
105  }
106  if (used.e[index]) {
107  fprintf(stderr, "%d: ghost cell is already pointed to (ghost_cell[%d]).\n", this_node, c);
108  errexit();
109  }
110  used.e[index] = 1;
111  }
112  for (c = 0; c < n_cells; c++)
113  if (!used.e[c]) {
114  fprintf(stderr, "%d: cell %d is not used anywhere.\n", this_node, c);
115  errexit();
116  }
117  realloc_intlist(&used, 0);
118 }

378 void local_sort_particles()
379 {
380  CELL_TRACE(fprintf(stderr, "%d: entering local_sort_particles\n", this_node));
381 
382  /* first distribute strictly on nodes */
383  cells_resort_particles(CELL_GLOBAL_EXCHANGE);
384 
385  CELL_TRACE(fprintf(stderr, "%d: sorting local cells\n", this_node));
386 
387  /* now sort the local cells */
388  for (int c = 0; c < local_cells.n; c++) {
389  Cell *cell = local_cells.cell[c];
390  Particle *p = cell->part;
391  int np = cell->n;
392 
393 #ifdef CELL_DEBUG
394  for (int id = 0; id < np; ++id) {
395  Cell *tgt_cell = cell_structure.position_to_cell(p[id].r.p);
396  if (tgt_cell != cell) {
397  fprintf(stderr, "%d: particle %d at position %lf %lf %lf is not in its expected cell. Have %ld, expected %ld\n", this_node, p[id].p.identity, p[id].r.p[0], p[id].r.p[1], p[id].r.p[2], (cell - cells)/sizeof(Cell*), (tgt_cell - cells) /sizeof(Cell*));
398  }
399  }
400 #endif
401 
402  qsort(p, np, sizeof(Particle), compare_particles);
403  update_local_particles(cell);
404  }
405 
406  CELL_TRACE(dump_particle_ordering());
407  CELL_TRACE(fprintf(stderr, "%d: leaving local_sort_particles\n", this_node));
408 }
