279 int tcl_appinit(Tcl_Interp *interp)
280 {
281  if (Tcl_Init(interp) == TCL_ERROR)
282  return TCL_ERROR;
283 
284 #ifdef TK
285  if (Tk_Init(interp) == TCL_ERROR)
286  return TCL_ERROR;
287 #endif
288 
289  /*
290  installation of tcl commands
291  */
292  tcl_register_commands(interp);
293  tcl_register_global_variables(interp);
294 
295  /* evaluate the Tcl initialization script */
296  char *scriptdir = getenv("ESPRESSO_SCRIPTS");
297  if (!scriptdir)
298  scriptdir = get_default_scriptsdir();
299 
300  /* fprintf(stderr,"Script directory: %s\n", scriptdir);*/
301 
302  char cwd[1024];
303  if ((getcwd(cwd, 1024) == NULL) || (chdir(scriptdir) != 0)) {
304  fprintf(stderr,
305  "\n\ncould not change to script dir %s, please check ESPRESSO_SCRIPTS.\n\n\n",
306  scriptdir);
307  exit(1);
308  }
309  if (Tcl_EvalFile(interp, "init.tcl") == TCL_ERROR) {
310  fprintf(stderr, "\n\nerror in initialization script: %s\n\n\n",
311  Tcl_GetStringResult(interp));
312  exit(1);
313  }
314  if (chdir(cwd) != 0) {
315  fprintf(stderr,
316  "\n\ncould not change back to execution dir %s ????\n\n\n",
317  cwd);
318  exit(1);
319  }
320 
321  return (TCL_OK);
322 }
