184 int tclcommand_print_average_errorbars(double_correlation* self, Tcl_Interp* interp) {
185  char buffer[TCL_DOUBLE_SPACE];
186  double* correlation_time;
187  double variance;
188  double errorbar;
189  if (self->dim_A != self->dim_corr) {
190  Tcl_AppendResult(interp, buffer, "Error in print average_errorbars: Only makes sense when the dimensions of \
191  the observables and correlation match (Isn't it?) ", (char *)NULL);
192  return TCL_ERROR;
193  }
194  if (!self->autocorrelation) {
195  Tcl_AppendResult(interp, buffer, "Error in print_average_errorbars: Must be an autocorrelation ", (char *)NULL);
196  return TCL_ERROR;
197  }
198  correlation_time = (double*) malloc(self->dim_corr*sizeof(double));
199  correlation_get_correlation_time(self, correlation_time);
200 
201  for (unsigned j=0; j<self->dim_corr; j++) {
202  variance=(self->A_accumulated_variance[j]/self->n_data - self->A_accumulated_average[j]*self->A_accumulated_average[j]/self->n_data/self->n_data);
203  errorbar=sqrt(variance*(correlation_time[j]/self->dt / self->n_data));
204  Tcl_PrintDouble(interp, errorbar, buffer);
205  Tcl_AppendResult(interp, buffer, " ",(char *)NULL);
206  }
207 
208  free(correlation_time);
209  return TCL_OK;
210 }

665 int double_correlation_print_spherically_averaged_sf(double_correlation* self, Tcl_Interp* interp) {
666 
667  int qi,qj,qk,qn, dim_sf, order2;
668  double dt=self->dt;
669  observable_sf_params* params=(observable_sf_params*)self->A_obs->container;
670  char buffer[TCL_DOUBLE_SPACE];
671  int *q_vals;
672  double *q_density;
673  double *av_sf_Re;
674  double *av_sf_Im;
675 
676  q_vals=params->q_vals;
677  q_density=params->q_density;
678  dim_sf=params->dim_sf;
679  order2=params->order*params->order;
680 
681  av_sf_Re=(double*)malloc(order2*sizeof(double));
682  av_sf_Im=(double*)malloc(order2*sizeof(double));
683 
684  // compute spherically averaged sf
685  for (unsigned j=0; j<self->n_result; j++) {
686  // compute the spherically averaged sf for current dt
687  for (int k=0;k<order2;k++) av_sf_Re[k]=av_sf_Im[k]=0.0;
688  for (int k=0;k<dim_sf;k++) {
689  qi=q_vals[3*k ];
690  qj=q_vals[3*k+1];
691  qk=q_vals[3*k+2];
692  qn= qi*qi + qj*qj + qk*qk;
693  av_sf_Re[qn-1]+=self->result[j][2*k ];
694  av_sf_Im[qn-1]+=self->result[j][2*k+1];
695  }
696  for (int k=0;k<order2;k++) {
697  if (q_density[k]>0.0) {
698  av_sf_Re[k]/=q_density[k];
699  av_sf_Im[k]/=q_density[k];
700  }
701  // note: if q_density[k]==0, we did not add anything to av_sf_Xx[k], so it is 0.0
702  }
703  // now print what we obtained
704  for (int k=0;k<order2;k++) {
705  Tcl_AppendResult(interp, " { ", (char *)NULL);
706  Tcl_PrintDouble(interp, self->tau[j]*dt, buffer);
707  Tcl_AppendResult(interp, buffer, " } { ",(char *)NULL);
708  Tcl_PrintDouble(interp, (double) self->n_sweeps[j], buffer);
709  Tcl_AppendResult(interp, buffer, " } { ", (char *)NULL);
710  Tcl_PrintDouble(interp, sqrt((double)k+1.0), buffer);
711  Tcl_AppendResult(interp, buffer, " } { ", (char *)NULL);
712  if (self->n_sweeps[j] == 0 ) {
713  Tcl_PrintDouble(interp, 0., buffer);
714  Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
715  Tcl_PrintDouble(interp, 0., buffer);
716  Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
717  }
718  else {
719  Tcl_PrintDouble(interp, av_sf_Re[k]/(double) self->n_sweeps[j], buffer);
720  Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
721  Tcl_PrintDouble(interp, av_sf_Im[k]/(double) self->n_sweeps[j], buffer);
722  Tcl_AppendResult(interp, buffer, " ", (char *)NULL);
723  }
724  Tcl_AppendResult(interp, " } \n", (char *)NULL);
725  }
726  }
