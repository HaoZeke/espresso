39  double eps, double sig, double offset,

41 {
42  IA_parameters *data = get_ia_param_safe(part_type_a, part_type_b);

46  data->LJCOS2_eps = eps;

63 /** calculate ljcos2_capradius from force_cap */
64 void calc_ljcos2_cap_radii()
65 {
66  /* do not compute cap radii if force capping is "individual" */
67  if( force_cap != -1.0){
68  int i,j,cnt=0;
69  IA_parameters *params;
70  double force=0.0, rad=0.0, step, frac2, frac6;
71 
72  for(i=0; i<n_particle_types; i++) {
73  for(j=0; j<n_particle_types; j++) {
74  params = get_ia_param(i,j);
75  if(force_cap > 0.0 && params->LJCOS2_eps > 0.0) {
76  /* I think we have to solve this numerically... and very crude as well */
77  cnt=0;
78  rad = params->LJCOS2_sig;
79  step = -0.1 * params->LJCOS2_sig;
80  force=0.0;
81 
82  while(step != 0) {
83  frac2 = SQR(params->LJCOS2_sig/rad);
84  frac6 = frac2*frac2*frac2;
85  if (rad < params->LJCOS2_rchange) {
86  force = 48.0 * params->LJCOS2_eps * frac6*(frac6 - 0.5)/rad;
87  }
88  else {
89  force = 0;
90  }
91  if((step < 0 && force_cap < force) || (step > 0 && force_cap > force)) {
92  step = - (step/2.0);
93  }
94  if(fabs(force-force_cap) < 1.0e-6) step=0;
95  rad += step; cnt++;
96  }
97  params->LJCOS2_capradius = rad;
98  }
99  else {
100  params->LJCOS2_capradius = 0.0;
101  }
102  FORCE_TRACE(fprintf(stderr,"%d: Ptypes %d-%d have cap_radius %f and cap_force %f (iterations: %d)\n",
103  this_node,i,j,rad,force,cnt));
104  }
105  }
106  }
107 }
