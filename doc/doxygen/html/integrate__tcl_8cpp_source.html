101 /** Parse integrate npt_isotropic command */
102 int tclcommand_integrate_set_npt_isotropic(Tcl_Interp *interp, int argc, char **argv)
103 {
104  int xdir, ydir, zdir;
105  xdir = ydir = zdir = nptiso.cubic_box = 0;
106 
107  if (argc < 4) {
108  Tcl_AppendResult(interp, "wrong # args: \n", (char *)NULL);
109  return tclcommand_integrate_print_usage(interp);
110  }
111  /* set parameters p_ext and piston */
112  if ( !ARG_IS_D(3, nptiso.p_ext) ) return tclcommand_integrate_print_usage(interp);
113  tclcallback_p_ext(interp, &nptiso.p_ext);
114  if ( argc > 4 ) {
115  if(!ARG_IS_D(4, nptiso.piston) ) return tclcommand_integrate_print_usage(interp);
116  tclcallback_npt_piston(interp, &nptiso.piston); }
117  else if ( nptiso.piston <= 0.0 ) {
118  Tcl_AppendResult(interp, "You must set <piston> as well before you can use this integrator! \n", (char *)NULL);
119  return tclcommand_integrate_print_usage(interp);
120  }
121 
122  if ( argc > 5 ) {
123  if (!ARG_IS_I(5,xdir) || !ARG_IS_I(6,ydir) || !ARG_IS_I(7,zdir) ) {
124  return tclcommand_integrate_print_usage(interp);}
125  else {
126  /* set the geometry to include rescaling specified directions only*/
127  nptiso.geometry = 0; nptiso.dimension = 0; nptiso.non_const_dim = -1;
128  if ( xdir ) {
129  nptiso.geometry = ( nptiso.geometry | NPTGEOM_XDIR );
130  nptiso.dimension += 1;
131  nptiso.non_const_dim = 0;
132  }
133  if ( ydir ) {
134  nptiso.geometry = ( nptiso.geometry | NPTGEOM_YDIR );
135  nptiso.dimension += 1;
136  nptiso.non_const_dim = 1;
137  }
138  if ( zdir ) {
139  nptiso.geometry = ( nptiso.geometry | NPTGEOM_ZDIR );
140  nptiso.dimension += 1;
141  nptiso.non_const_dim = 2;
142  }
143  }
144  } else {
145  /* set the geometry to include rescaling in all directions; the default*/
146  nptiso.geometry = 0;
147  nptiso.geometry = ( nptiso.geometry | NPTGEOM_XDIR );
148  nptiso.geometry = ( nptiso.geometry | NPTGEOM_YDIR );
149  nptiso.geometry = ( nptiso.geometry | NPTGEOM_ZDIR );
150  nptiso.dimension = 3; nptiso.non_const_dim = 2;
151  }
152 
153  if ( argc > 8 ) {
154  /* enable if the volume fluctuations should also apply to dimensions which are switched off by the above flags
155  and which do not contribute to the pressure (3D) / tension (2D, 1D) */
156  if (!ARG_IS_S(8,"-cubic_box")) {
157  return tclcommand_integrate_print_usage(interp);
158  } else {
159  nptiso.cubic_box = 1;
160  }
161  }
162 
163  /* Sanity Checks */
164 #ifdef ELECTROSTATICS
165  if ( nptiso.dimension < 3 && !nptiso.cubic_box && coulomb.bjerrum > 0 ){
166  fprintf(stderr,"WARNING: If electrostatics is being used you must use the -cubic_box option!\n");
167  fprintf(stderr,"Automatically reverting to a cubic box for npt integration.\n");
168  fprintf(stderr,"Be aware though that all of the coulombic pressure is added to the x-direction only!\n");
169  nptiso.cubic_box = 1;
170  }
171 #endif
172 
173 #ifdef DIPOLES
174  if ( nptiso.dimension < 3 && !nptiso.cubic_box && coulomb.Dbjerrum > 0 ){
175  fprintf(stderr,"WARNING: If magnetostatics is being used you must use the -cubic_box option!\n");
176  fprintf(stderr,"Automatically reverting to a cubic box for npt integration.\n");
177  fprintf(stderr,"Be aware though that all of the magnetostatic pressure is added to the x-direction only!\n");
178  nptiso.cubic_box = 1;
179  }
180 #endif
181 
182 
183  if( nptiso.dimension == 0 || nptiso.non_const_dim == -1) {
184  Tcl_AppendResult(interp, "You must enable at least one of the x y z components as fluctuating dimension(s) for box length motion!", (char *)NULL);
185  Tcl_AppendResult(interp, "Cannot proceed with npt_isotropic, reverting to nvt integration... \n", (char *)NULL);
186  integ_switch = INTEG_METHOD_NVT;
187  mpi_bcast_parameter(FIELD_INTEG_SWITCH);
188  return (TCL_ERROR);
189  }
190 
191  /* set integrator switch */
192  integ_switch = INTEG_METHOD_NPT_ISO;
193  mpi_bcast_parameter(FIELD_INTEG_SWITCH);
194 
195  /* broadcast npt geometry information to all nodes */
196  mpi_bcast_nptiso_geom();
197  return (TCL_OK);
198 }

200 int tclcommand_integrate(ClientData data, Tcl_Interp *interp, int argc, char **argv)
201 {
202  int n_steps, reuse_forces = 0;
203 
204  INTEG_TRACE(fprintf(stderr,"%d: integrate:\n",this_node));
205 
206  if (argc < 1) {
207  Tcl_AppendResult(interp, "wrong # args: \n\"", (char *) NULL);
208  return tclcommand_integrate_print_usage(interp); }
209  else if (argc < 2) { return tclcommand_integrate_print_status(interp); }
210 
211  if (ARG1_IS_S("set")) {
212  if (argc < 3) return tclcommand_integrate_print_status(interp);
213  if (ARG_IS_S(2,"nvt")) return tclcommand_integrate_set_nvt(interp, argc, argv);
214 #ifdef NPT
215  else if (ARG_IS_S(2,"npt_isotropic")) return tclcommand_integrate_set_npt_isotropic(interp, argc, argv);
216 #endif
217  else {
218  Tcl_AppendResult(interp, "unknown integrator method:\n", (char *)NULL);
219  return tclcommand_integrate_print_usage(interp);
220  }
221  } else {
222  if ( !ARG_IS_I(1,n_steps) ) return tclcommand_integrate_print_usage(interp);
223 
224  // actual integration
225  if ((argc == 3) && ARG_IS_S(2, "reuse_forces")) {
226  reuse_forces = 1;
227  }
228  else if ((argc == 3) && ARG_IS_S(2, "recalc_forces")) {
229  reuse_forces = -1;
230  }
231  else if (argc != 2) return tclcommand_integrate_print_usage(interp);
232  }
233  /* go on with integrate <n_steps> */
234  if(n_steps < 0) {
235  Tcl_AppendResult(interp, "illegal number of steps (must be >0) \n", (char *) NULL);
236  return tclcommand_integrate_print_usage(interp);;
237  }
238 
239  /* if skin wasn't set, do an educated guess now */
240  if (!skin_set) {
241  if (max_cut == 0.0) {
242  Tcl_AppendResult(interp, "cannot automatically determine skin, please set it manually via \"setmd skin\"\n", (char *) NULL);
243  return TCL_ERROR;
244  }
245  skin = 0.4*max_cut;
246  mpi_bcast_parameter(FIELD_SKIN);
247  }
248 
249  /* perform integration */
250  if (!correlations_autoupdate && !observables_autoupdate) {
251  if (mpi_integrate(n_steps, reuse_forces))
252  return gather_runtime_errors(interp, TCL_OK);
253  } else {
254  for (int i=0; i<n_steps; i++) {
255  if (mpi_integrate(1, reuse_forces))
256  return gather_runtime_errors(interp, TCL_OK);
257  reuse_forces=1;
258  autoupdate_observables();
259  autoupdate_correlations();
260  }
261  if (n_steps == 0){
262  if (mpi_integrate(0, reuse_forces))
263  return gather_runtime_errors(interp, TCL_OK);
264  }
265  }
266  return TCL_OK;
267 }
