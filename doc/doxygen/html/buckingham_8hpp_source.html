44 /**Resultant Force due to a buckingham potential between two particles at interatomic separation r greater than or equal to discont*/
45 inline double buck_force_r(double A, double B, double C, double D, double r )
46 {
47  return (A*B*exp(-B*r) - 6.0*C/pow(r, 7) - 4.0*D/pow(r, 5));
48 }
49 /**Potential Energy due to a buckingham potential between two particles at interatomic separation r greater than or equal to discont*/
50 inline double buck_energy_r(double A, double B, double C, double D, double shift, double r )
51 {
52  return (A*exp(-B*r) - C/pow(r, 6) - D/pow(r, 4) + shift);
53 }

57 inline void add_buck_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
58  double d[3], double dist, double force[3])
59 {
60  int j;
61  double fac;
62  if(CUTOFF_CHECK(dist < ia_params->BUCK_cut)) {
63  if(ia_params->BUCK_capradius==0.0)
64  {
65  /* case: resulting force/energy greater than discontinuity and
66  less than cutoff (true buckingham region) */
67  if(dist > ia_params->BUCK_discont) {
68  fac = buck_force_r(ia_params->BUCK_A, ia_params->BUCK_B, ia_params->BUCK_C, ia_params->BUCK_D, dist )/dist;
69  for(j=0;j<3;j++) {
70  force[j] += fac * d[j];
71  }
72 #ifdef LJ_WARN_WHEN_CLOSE
73  if(fac*dist > 1000) fprintf(stderr,"%d: BUCK-Warning: Pair (%d-%d) force=%f dist=%f\n",
74  this_node,p1->p.identity,p2->p.identity,fac*dist,dist);
75 #endif
76  }
77  else
78  {
79  /* resulting force/energy in the linear region*/
80  fac = -ia_params->BUCK_F2/dist;
81  for(j=0;j<3;j++) {
82  force[j] += fac * d[j];
83  }
84  }
85  }
86  /* if Buckingham potential is capped. */
87  else
88  {
89  if(dist>ia_params->BUCK_capradius) {

112 /** calculate Buckingham energy between particle p1 and p2. */
113 inline double buck_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
114  double d[3], double dist)
115 {
116 
117  if(CUTOFF_CHECK(dist < ia_params->BUCK_cut)) {
118  if(ia_params->BUCK_capradius==0.0)
119  {
120  /* case: resulting force/energy greater than discont and
121  less than cutoff (true buckingham region) */
122  if(dist > ia_params->BUCK_discont)
123  return buck_energy_r(ia_params->BUCK_A, ia_params->BUCK_B, ia_params->BUCK_C, ia_params->BUCK_D, ia_params->BUCK_shift, dist);
124  else
125  /* resulting force/energy in the linear region*/
126  return (ia_params->BUCK_F1+ia_params->BUCK_F2*dist);
127  }
128  else
129  {
130  if(dist>ia_params->BUCK_capradius)
131  /*true buckigham region*/
132  return buck_energy_r(ia_params->BUCK_A, ia_params->BUCK_B, ia_params->BUCK_C, ia_params->BUCK_D, ia_params->BUCK_shift, dist);
133  else
134  /*capped region*/
135  return buck_energy_r(ia_params->BUCK_A, ia_params->BUCK_B, ia_params->BUCK_C, ia_params->BUCK_D, ia_params->BUCK_shift, ia_params->BUCK_capradius);
136  }
137  }
138  return 0.0;
139 }
