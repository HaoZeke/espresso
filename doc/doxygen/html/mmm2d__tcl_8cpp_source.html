65 int tclcommand_inter_coulomb_parse_mmm2d(Tcl_Interp * interp, int argc, char ** argv)
66 {
67  int err;
68  double maxPWerror;
69  double far_cut = -1;
70  double top = 1, mid = 1, bot = 1;
71  double delta_top = 0, delta_bot = 0;
72  double pot_diff = 0;
73  int const_pot_on = 0;
74 
75  if (argc < 1) {
76  Tcl_AppendResult(interp, "wrong # arguments: inter coulomb mmm2d <maximal pairwise error> "
77  "{<fixed far cutoff>} <{dielectric <e1> <e2> <e3>} | {dielectric-contrasts <d1> <d2>} | {capacitor <dU>}>", (char *) NULL);
78  return TCL_ERROR;
79  }
80 
81  if (! ARG0_IS_D(maxPWerror))
82  return TCL_ERROR;
83  --argc; ++argv;
84 
85  if (argc >= 1) {
86  if (ARG0_IS_D(far_cut)){
87  --argc; ++argv;
88  } else {
89  Tcl_ResetResult(interp);
90  }
91  }
92 
93  if (argc != 0) {
94  if (argc == 4 && ARG0_IS_S("dielectric")) {
95  if (!ARG_IS_D(1,top) || !ARG_IS_D(2,mid) || !ARG_IS_D(3,bot))
96  return TCL_ERROR;
97 
98  delta_top = (mid - top)/(mid + top);
99  delta_bot = (mid - bot)/(mid + bot);
100  }
101  else if (argc == 3 && ARG0_IS_S("dielectric-contrasts")) {
102  if (!ARG_IS_D(1,delta_top) || !ARG_IS_D(2,delta_bot))
103  return TCL_ERROR;
104  }
105  else if (argc == 2 && ARG0_IS_S("capacitor")) {
106  if (!ARG_IS_D(1,pot_diff))
107  return TCL_ERROR;
108  delta_top = -1;
109  delta_bot = -1;
110  const_pot_on = 1;
111  }
112  else {
113  Tcl_AppendResult(interp, "wrong # arguments: inter coulomb mmm2d <maximal pairwise error> "
114  "{<fixed far cutoff>} <{dielectric <e1> <e2> <e3>} | {dielectric-contrasts <d1> <d2>} | {capacitor <dU>}>", (char *) NULL);
115  return TCL_ERROR;
116  }
117  }
118 
119  if (cell_structure.type != CELL_STRUCTURE_NSQUARE &&
120  cell_structure.type != CELL_STRUCTURE_LAYERED) {
121  Tcl_AppendResult(interp, "MMM2D requires layered of nsquare cell structure", (char *)NULL);
122  return TCL_ERROR;
123  }
124 
125  if ((err = MMM2D_set_params(maxPWerror, far_cut, delta_top, delta_bot, const_pot_on, pot_diff)) > 0) {
126  Tcl_AppendResult(interp, mmm2d_errors[err], (char *)NULL);
127  return TCL_ERROR;
128  }
129  return TCL_OK;
130 }
