251 int tclcommand_thermostat_parse_inter_dpd(Tcl_Interp *interp, int argc, char ** argv)
252 {
253  double temp;
254 
255  if (argc < 2) {
256  Tcl_AppendResult(interp, "thermostat needs 1 parameter: "
257  "<temperature>",
258  (char *) NULL);
259  return TCL_ERROR;
260  }
261 
262  if (argc>2 && ARG_IS_S(2, "ignore_fixed_particles")) {
263  if (argc == 3)
264  dpd_ignore_fixed_particles=1;
265  else if (argc!= 4 || (!ARG_IS_I(3, dpd_ignore_fixed_particles)))
266  return TCL_ERROR;
267  mpi_bcast_parameter(FIELD_DPD_IGNORE_FIXED_PARTICLES);
268  return TCL_OK;
269  }
270 
271  /* copy lattice-boltzmann parameters */
272  if (! ARG_IS_D(2, temp)) { return TCL_ERROR; }
273 
274  if ( temp < 0.0 ) {
275  Tcl_AppendResult(interp, "temperature must be non-negative", (char *) NULL);
276  return TCL_ERROR;
277  }
278  temperature = temp;
279  thermo_switch = ( thermo_switch | THERMO_INTER_DPD );
280  mpi_bcast_parameter(FIELD_THERMO_SWITCH);
281  mpi_bcast_parameter(FIELD_TEMPERATURE);
282  return (TCL_OK);
283 }

286  int part_type_a, int part_type_b,
287  int argc, char ** argv)

289  /* parameters needed for LJ */

292  int wf,twf;

294 
295  /* get inter_dpd interaction type */
296  if (argc < 7) {
297  Tcl_AppendResult(interp, "inter_dpd needs 6 parameters: "
298  "<gamma> <r_cut> <wf> <tgamma> <tr_cut> <twf>",
299  (char *) NULL);
300  return 0;
301  }
302  if (temperature == -1) {
303  Tcl_AppendResult(interp, "Please set temperature first: temperature inter_dpd temp",(char *) NULL);
304  return 0;
305  }
306 
307  /* copy lennard-jones parameters */
308  if ((! ARG_IS_D(1, gamma)) ||
309  (! ARG_IS_D(2, r_c)) ||
310  (! ARG_IS_I(3, wf)) ||
311  (! ARG_IS_D(4, tgamma)) ||
312  (! ARG_IS_D(5, tr_c)) ||
313  (! ARG_IS_I(6, twf)) ) {
314  Tcl_AppendResult(interp, "inter_dpd needs 6 parameters: "
315  "<gamma> <r_cut> <wf> <tgamma> <tr_cut> <twf> ",
316  (char *) NULL);
317  return 0;
318  }
319  change = 7;
320 
321  if (inter_dpd_set_params(part_type_a, part_type_b,
322  gamma,r_c,wf,tgamma,tr_c,twf) == ES_ERROR) {
323  Tcl_AppendResult(interp, "particle types must be non-negative", (char *) NULL);
324  return 0;
325  }

327
