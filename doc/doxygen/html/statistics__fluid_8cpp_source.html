63 void lb_calc_fluid_momentum(double *result) {
64 
65  int x, y, z, index;
66  double j[3], momentum[3] = { 0.0, 0.0, 0.0 };
67 
68  for (x=1; x<=lblattice.grid[0]; x++) {
69  for (y=1; y<=lblattice.grid[1]; y++) {
70  for (z=1; z<=lblattice.grid[2]; z++) {
71  index = get_linear_index(x,y,z,lblattice.halo_grid);
72 
73  lb_calc_local_j(index,j);
74  momentum[0] += j[0] + lbfields[index].force[0];
75  momentum[1] += j[1] + lbfields[index].force[1];
76  momentum[2] += j[2] + lbfields[index].force[2];
77 
78  }
79  }
80  }
81 
82  momentum[0] *= lbpar.agrid/lbpar.tau;
83  momentum[1] *= lbpar.agrid/lbpar.tau;
84  momentum[2] *= lbpar.agrid/lbpar.tau;
85 
86  MPI_Reduce(momentum, result, 3, MPI_DOUBLE, MPI_SUM, 0, comm_cart);
87 
88 }

93 void lb_calc_fluid_temp(double *result) {
94  int x, y, z, index;
95  double rho, j[3];
96  double temp = 0.0;
97  int number_of_non_boundary_nodes = 0;
98 
99  for (x=1; x<=lblattice.grid[0]; x++) {
100  for (y=1; y<=lblattice.grid[1]; y++) {
101  for (z=1; z<=lblattice.grid[2]; z++) {
102 
103  index = get_linear_index(x,y,z,lblattice.halo_grid);
104 
105 #ifdef LB_BOUNDARIES
106  if ( !lbfields[index].boundary )
107 #endif
108  {
109  lb_calc_local_fields(index, &rho, j, NULL);
110  temp += scalar(j,j);
111  number_of_non_boundary_nodes++;
112  }
113  }
114  }
115  }
116 
117  // @Todo: lblattice.agrid is 3d. What to use here?
118  temp *= 1./(3.*lbpar.rho[0]*number_of_non_boundary_nodes*
119  lbpar.tau*lbpar.tau*lbpar.agrid)/n_nodes;
120 
121  MPI_Reduce(&temp, result, 1, MPI_DOUBLE, MPI_SUM, 0, comm_cart);
122 }
