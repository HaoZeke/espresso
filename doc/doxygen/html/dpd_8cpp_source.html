102 {

108 #endif
109  /* prefactor friction force */
110  /* NOTE: velocities are scaled with time_step, so divide by time_step here*/

112  /* prefactor random force */

120  dpd_pref4 = sqrt(24.0*temperature*dpd_tgamma/time_step);
121  dpd_tr_cut_inv = 1.0/dpd_tr_cut;

162 void add_dpd_thermo_pair_force(Particle *p1, Particle *p2, double d[3], double dist, double dist2)
163 {
164  extern double dpd_gamma,dpd_pref1, dpd_pref2,dpd_r_cut,dpd_r_cut_inv;
165  extern int dpd_wf;
166 #ifdef TRANS_DPD
167  extern double dpd_tgamma, dpd_pref3, dpd_pref4,dpd_tr_cut,dpd_tr_cut_inv;
168  extern int dpd_twf;
169 #endif
170  int j;
171  // velocity difference between p1 and p2
172  double vel12_dot_d12=0.0;
173  // inverse distance
174  double dist_inv;
175  // weighting functions for friction and random force
176  double omega,omega2;// omega = w_R/dist
177  double friction, noise;
178  //Projection martix
179 #ifdef TRANS_DPD
180  int i;
181  double P_times_dist_sqr[3][3]={{dist2,0,0},{0,dist2,0},{0,0,dist2}},noise_vec[3];
182  double f_D[3],f_R[3];
183 #endif
184  double tmp;
185 #ifdef DPD_MASS
186  double massf;
187 #endif
188 
189  if( le_chatterjee_test_pair(p1, p2) ) return;
190 
191 #ifdef EXTERNAL_FORCES
192  // if any of the two particles is fixed in some direction then
193  // do not add any dissipative or stochastic dpd force part
194  // because dissipation-fluctuation theorem is violated
195  if (dpd_ignore_fixed_particles)
196  if ( (p1->p.ext_flag | p2->p.ext_flag) & COORDS_FIX_MASK) return;
197 #endif
198 
199 #ifdef VIRTUAL_SITES
200  if (ifParticleIsVirtual(p1) || ifParticleIsVirtual(p2)) return;
201 #endif
202 
203 #ifdef DPD_MASS_RED
204  massf=2*PMASS(*p1)*PMASS(*p2)/(PMASS(*p1)+PMASS(*p2));
205 #endif
206 
207 #ifdef DPD_MASS_LIN
208  massf=0.5*(PMASS(*p1)+PMASS(*p2));
209 #endif
210 
211 
212  dist_inv = 1.0/dist;
213 
214  if((dist < dpd_r_cut)&&(dpd_gamma > 0.0)) {
215  if ( dpd_wf == 1 ) //w_R=1
216  {
217  omega = dist_inv;
218  }
219  else //w_R=(1-r/r_c)
220  {
221  omega = dist_inv- dpd_r_cut_inv;
222  }
223 #ifdef DPD_MASS
224  omega*=sqrt(massf);
225 #endif
226  omega2 = SQR(omega);
227  //DPD part
228  // friction force prefactor
229  for(j=0; j<3; j++) vel12_dot_d12 += (p1->m.v[j] - p2->m.v[j]) * d[j];
230  friction = dpd_pref1 * omega2 * vel12_dot_d12;
231  // random force prefactor
232  noise = dpd_pref2 * omega * (d_random()-0.5);
233  for(j=0; j<3; j++) {
234  p1->f.f[j] += ( tmp = (noise - friction)*d[j] );
235  p2->f.f[j] -= tmp;
236  }
237  }
238 #ifdef TRANS_DPD
239  //DPD2 part
240  if ((dist < dpd_tr_cut)&&(dpd_tgamma > 0.0)){
241  if ( dpd_twf == 1 )
242  {
243  omega = dist_inv;
244  }
245  else
246  {
247  omega = dist_inv- dpd_tr_cut_inv;
248  }
249 #ifdef DPD_MASS
250  omega*=sqrt(massf);
251 #endif
252  omega2 = SQR(omega);
253  for (i=0;i<3;i++){
254  //noise vector
255  noise_vec[i]=d_random()-0.5;
256  // Projection Matrix
257  for (j=0;j<3;j++){
258  P_times_dist_sqr[i][j]-=d[i]*d[j];
259  }
260  }
261  for (i=0;i<3;i++){
262  //Damping force
263  f_D[i]=0;
264  //Random force
265  f_R[i]=0;
266  for (j=0;j<3;j++){
267  f_D[i]+=P_times_dist_sqr[i][j]*(p1->m.v[j] - p2->m.v[j]);
268  f_R[i]+=P_times_dist_sqr[i][j]*noise_vec[j];
269  }
270  //NOTE: velocity are scaled with time_step
271  f_D[i]*=dpd_pref3*omega2;
272  //NOTE: noise force scales with 1/sqrt(time_step
273  f_R[i]*=dpd_pref4*omega*dist_inv;
274  }
275  for(j=0; j<3; j++) {
276  tmp=f_R[j]-f_D[j];
277  p1->f.f[j] += tmp;
278  p2->f.f[j] -= tmp;
279  }
280  }
281 #endif
282 }
283 
284 #endif

300 int inter_dpd_set_params(int part_type_a, int part_type_b,
301  double gamma, double r_c, int wf,
302  double tgamma, double tr_c,
303  int twf)
304 {
305  extern double temperature;
306  IA_parameters *data = get_ia_param_safe(part_type_a, part_type_b);
307 
308  if (!data) return ES_ERROR;
309 
310  data->dpd_gamma = gamma;
311  data->dpd_r_cut = r_c;
312  data->dpd_wf = wf;
313  data->dpd_pref1 = gamma/time_step;
314  data->dpd_pref2 = sqrt(24.0*temperature*gamma/time_step);
315  data->dpd_tgamma = tgamma;
316  data->dpd_tr_cut = tr_c;
317  data->dpd_twf = twf;
318  data->dpd_pref3 = tgamma/time_step;
319  data->dpd_pref4 = sqrt(24.0*temperature*tgamma/time_step);
320 
321  /* broadcast interaction parameters */
322  mpi_bcast_ia_params(part_type_a, part_type_b);
323 
324  return ES_OK;
325 }
