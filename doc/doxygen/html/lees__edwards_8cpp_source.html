38 /** React to news of a broadcast change in the \ref lees_edwards_offset */
39 void lees_edwards_step_boundaries(){
40 
41  double delta;
42 
43  delta = lees_edwards_offset - lees_edwards_prev_offset;
44  while( delta > 0.5 * box_l[0] ) delta -= box_l[0];
45  while( delta < -0.5 * box_l[0] ) delta += box_l[0];
46 
47  while( lees_edwards_offset > 0.5*box_l[0]) { lees_edwards_count++; lees_edwards_offset -= box_l[0];}
48  while( lees_edwards_offset < -0.5*box_l[0]) { lees_edwards_count--; lees_edwards_offset += box_l[0];}
49 
50  /* update the apparent shear rate */
51  if( sim_time - lees_edwards_prev_set_at > 0.0 )
52  lees_edwards_rate = delta * time_step / (sim_time - lees_edwards_prev_set_at);
53  else
54  lees_edwards_rate = 0.0;
55 
56  /* save some state */
57  lees_edwards_prev_set_at = sim_time;
58  lees_edwards_prev_offset = lees_edwards_offset;
