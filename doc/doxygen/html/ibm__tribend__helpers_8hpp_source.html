171  const int jlDelta = (derivativeNodeID == xj->p.identity) ? 1 : 0;
172 
173  // Gradient of cosine theta.
174  for (int i=0; i < 3; i++)
175  {
176  cosThetaDeriv.el[i]= 1.0/(dximLen*dxjmLen) * ((ilDelta - mlDelta) * dxjm.el[i] +
177  (jlDelta - mlDelta) * dxim.el[i] -
178  dxjmLen/dximLen * cosTheta * (ilDelta - mlDelta) * dxim.el[i] -
179  dximLen/dxjmLen * cosTheta * (jlDelta - mlDelta) * dxjm.el[i] );
180  }
