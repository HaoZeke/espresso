38 inline double KS(double lambda){ // Defined by (19) from Dupin2007
39  double res;
40  res = (pow(lambda,0.5) + pow(lambda,-2.5))/(lambda + pow(lambda,-3.));
41  return res;
42 }

44 /** Computes the STRETCHING_FORCE pair force and adds this
45  force to the particle forces (see \ref tclcommand_inter).
46  @param p1 Pointer to first particle.
47  @param p2 Pointer to second particle.
48  @param iaparams spring stiffness ks, initial distance between particles (see \ref tclcommand_inter).
49  @param dx particle distance vector
50  @param force returns force of particle 1
51  @return true if the bond is broken
52  */
53 inline int calc_stretching_force_pair_force(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, double dx[3], double force[3])
54 {
55  int i;
56  double fac, dr, len2, len;
57  double lambda;
58 
59  len2 = sqrlen(dx);
60 
61  len = sqrt(len2);
62  dr = len - iaparams->p.stretching_force.r0;
63 
64  lambda = 1.0*len/iaparams->p.stretching_force.r0;
65  //fac = -iaparams->p.stretching_force.ks * KS(lambda) * dr / iaparams->p.stretching_force.r0;
66  fac = -iaparams->p.stretching_force.ks * KS(lambda) * dr ; // no normalization
67 
68  for(i=0;i<3;i++)
69  force[i] = fac*dx[i]/len;
70 
71  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: FENE f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,sqrt(len2),fac));
72  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: FENE f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,sqrt(len2),fac));
73 
74  return 0;
75 }
