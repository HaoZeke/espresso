34 int tclcommand_analyze_set_parse_chain_topology(Tcl_Interp *interp, int argc, char **argv)
35 {
36  /* parses a chain topology (e.g. in 'analyze ( rg | <rg> ) [chain start n chains chain length]' , or
37  in 'analyze set chains <chain_start> <n_chains> <chain_length>') */
38  int m, i, pc;
39 
40  if (argc < 3) {
41  Tcl_AppendResult(interp, "chain structure info consists of <start> <n> <length>", (char *)NULL);
42  return TCL_ERROR;
43  }
44 
45  if (! (ARG0_IS_I(chain_start) && ARG1_IS_I(chain_n_chains) && ARG_IS_I(2, chain_length)))
46  return TCL_ERROR;
47 
48  realloc_topology(chain_n_chains);
49  pc = 0;
50  for (m = 0; m < n_molecules; m++) {
51  topology[m].type = 0;
52  realloc_intlist(&topology[m].part, topology[m].part.n = chain_length);
53  for (i = 0; i < chain_length; i++)
54  topology[m].part.e[i] = pc++;
55  }
56 
57  return TCL_OK;
58  return tclcommand_analyze_set_parse_topo_part_sync(interp);
59 }

291 int tclcommand_analyze_parse_g_av(Tcl_Interp *interp, int what, int argc, char **argv)
292 {
293  /* 'analyze { <g1> | <g2> | <g3> } [<chain_start> <n_chains> <chain_length>]' */
294  /******************************************************************************/
295  int i, window = 1;
296  char buffer[TCL_DOUBLE_SPACE+2];
297  double *gx = NULL;
298  double weights[3]; weights[0] = weights[1] = weights[2] = 1;
299 
300  while (argc >= 1) {
301  if (ARG0_IS_S("-sliding")) {
302  window = 0;
303  argc--; argv++;
304  }
305  else if (ARG0_IS_S("-weights")) {
306  if (argc < 4 || !ARG_IS_D(1,weights[0]) || !ARG_IS_D(2,weights[1]) || !ARG_IS_D(3,weights[2]) )
307  return (TCL_ERROR);
308  argc-=4; argv+=4;
309  }
310  else break;
311  }
312  if (tclcommand_analyze_set_parse_chain_topology_check(interp, argc, argv) == TCL_ERROR) return TCL_ERROR;
313  if ((argc != 0) && (argc != 3)) { Tcl_AppendResult(interp, "only chain structure info or -sliding allowed", (char *)NULL); return TCL_ERROR; }
314  if (n_configs == 0) { Tcl_AppendResult(interp, "no configurations found! Use 'analyze append' to save some!", (char *)NULL); return TCL_ERROR; }
315  switch (what) {
316  case 1:
317  calc_g1_av(&gx, window, weights); break;
318  case 2:
319  calc_g2_av(&gx, window, weights); break;
320  case 3:
321  calc_g3_av(&gx, window, weights); break;
322  default: ;
323  }
324  for (i=0; i<n_configs; i++) {
325  sprintf(buffer,"%f ",gx[i]); Tcl_AppendResult(interp, buffer, (char *)NULL);
326  }
327  free(gx);
328  return (TCL_OK);
329 }

332 int tclcommand_analyze_parse_formfactor(Tcl_Interp *interp, int average, int argc, char **argv)
333 {
334  /* 'analyze { formfactor | <formfactor> } <qmin> <qmax> <qbins> [<chain_start> <n_chains> <chain_length>]' */

337  int i;
338  double qmin,qmax, q,qfak, *ff; int qbins;
339  if (argc < 3) {
340  Tcl_AppendResult(interp, "Wrong # of args! Usage: analyze formfactor <qmin> <qmax> <qbins> [<chain_start> <n_chains> <chain_length>]",
341  (char *)NULL);
342  return (TCL_ERROR);
343  } else {
344  if (!ARG0_IS_D(qmin))
345  return (TCL_ERROR);
346  if (!ARG1_IS_D(qmax))
347  return (TCL_ERROR);
348  if (!ARG_IS_I(2, qbins))
349  return (TCL_ERROR);
350  argc-=3; argv+=3;
351  }
352  if (tclcommand_analyze_set_parse_chain_topology_check(interp, argc, argv) == TCL_ERROR) return TCL_ERROR;
353 
354  if ((chain_n_chains == 0) || (chain_length == 0)) {
355  Tcl_AppendResult(interp, "The chain topology has not been set",(char *)NULL); return TCL_ERROR;
356  }
357 
358  if (qbins <=0) {
359  Tcl_AppendResult(interp, "Nothing to be done - choose <qbins> greater zero to get S(q)!",(char *)NULL); return TCL_ERROR;
360  }
361 
362  if (qmin <= 0.) {
363  Tcl_AppendResult(interp, "formfactor S(q) requires qmin > 0", (char *)NULL);
364  return TCL_ERROR;
365  }
366  if (qmax <= qmin) {
367  Tcl_AppendResult(interp, "formfactor S(q) requires qmin < qmax", (char *)NULL);
368  return TCL_ERROR;
369  }
370 
371  if (!average) analyze_formfactor(qmin, qmax, qbins, &ff);
372  else if (n_configs == 0) {
373  Tcl_AppendResult(interp, "no configurations found! ", (char *)NULL);
374  Tcl_AppendResult(interp, "Use 'analyze append' to save some, or 'analyze formfactor ...' to only look at current state!",
375  (char *)NULL);
376  return TCL_ERROR; }
377  else analyze_formfactor_av(qmin, qmax, qbins, &ff);
378 
379  q = qmin; qfak = pow((qmax/qmin),(1.0/qbins));
380  for(i=0; i<=qbins; i++) { sprintf(buffer,"{%f %f} ",q,ff[i]); q*=qfak; Tcl_AppendResult(interp, buffer, (char *)NULL); }
381  free(ff);
382  return (TCL_OK);
383 }
