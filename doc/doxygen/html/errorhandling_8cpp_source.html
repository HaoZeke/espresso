35 static void sigint_handler(int sig) {
36  /* without this exit handler the nodes might exit asynchronously
37  * without calling MPI_Finalize, which may cause MPI to hang
38  * (e.g. this handler makes CTRL-c work properly with poe)
39  *
40  * NOTE: mpirun installs its own handler for SIGINT/SIGTERM
41  * and takes care of proper cleanup and exit */
42 
43 
44  static int numcalls = 0;
45  if (numcalls++ > 0) exit(sig); // catch sig only once
46 
47  /* we use runtime_error to indicate that sig was called;
48  * upon next call of mpi_gather_runtime_errors all nodes
49  * will clean up and exit. */
50  ostringstream msg;
51  msg <<"caught signal "<< sig ;
52  runtimeError(msg);
53 }
54
