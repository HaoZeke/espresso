420 double hzeta(double s, double q)
421 {
422  double max_bits = 54.0;
423  int jmax = 12, kmax = 10;
424  int j, k;
425  double pmax, scp, pcp, ans;
426 
427  if((s > max_bits && q < 1.0) || (s > 0.5*max_bits && q < 0.25))
428  return pow(q, -s);
429  if(s > 0.5*max_bits && q < 1.0) {
430  double p1 = pow(q, -s);
431  double p2 = pow(q/(1.0+q), s);
432  double p3 = pow(q/(2.0+q), s);
433  return p1 * (1.0 + p2 + p3);
434  }
435  /* Euler-Maclaurin summation formula
436  * [Moshier, p. 400, with several typo corrections]
437  */
438  pmax = pow(kmax + q, -s);
439  scp = s;
440  pcp = pmax / (kmax + q);
441  ans = pmax*((kmax+q)/(s-1.0) + 0.5);
442 
443  for(k=0; k<kmax; k++)
444  ans += pow(k + q, -s);
445 
446 
447  for(j=0; j<=jmax; j++) {
448  double delta = hzeta_c[j+1] * scp * pcp;
449  ans += delta;
450  scp *= (s+2*j+1)*(s+2*j+2);
451  pcp /= (kmax + q)*(kmax + q);
452  }
453 
454  return ans;
455 }

531 double LPK0(double x)
532 {
533  if (x >= 27.) {
534  double tmp = .5*exp(-x)/sqrt(x);
535  return tmp*ak0_data[0];
536  }
537  if (x >= 23.) {
538  double tmp = exp(-x)/sqrt(x), xx = (16./3.)/x - 5./3.;
539  return tmp*(xx*ak0_data[1] + 0.5*ak0_data[0]);
540  }
541  if (x > 2) {
542  int j = ak01_orders[((int)x) - 2];
543  double tmp, x2;
544  double dd0, d0;
545  double *s0;
546  if (x <= 8) {
547  s0 = ak0_data;
548  x2 = (2.*16./3.)/x - 2.*5./3.;
549  } else {
550  s0 = ak02_data;
551  x2 = (2.*16.)/x - 2.;
552  }
553  dd0 = s0[j];
554  d0 = x2*dd0 + s0[j - 1];
555  for(j -= 2; j >= 1; j--) {
556  double tmp0 = d0;
557  d0 = x2*d0 - dd0 + s0[j];
558  dd0 = tmp0;
559  }
560  tmp = exp(-x)/sqrt(x);
561  return tmp*(0.5*(s0[0] + x2*d0) - dd0);
562  }
563  /* x <= 2 */
564  {
565  /* I0/1 series */
566  int j = 10;
567  double ret, tmp, x2 = (2./4.5)*x*x - 2.;
568  double dd0, d0;
569  dd0 = bi0_data[j];
570  d0 = x2*dd0 + bi0_data[j - 1];
571  for(j -= 2; j >= 1; j--) {
572  double tmp0 = d0;
573  d0 = x2*d0 - dd0 + bi0_data[j];
574  dd0 = tmp0;
575  }
576  tmp = log(x) - M_LN2;
577  ret = -tmp*(0.5*(bi0_data[0] + x2*d0)- dd0);
578 
579  /* K0/K1 correction */
580  j = 9;
581  x2 = x*x - 2.;
582  dd0 = bk0_data[j];
583  d0 = x2*dd0 + bk0_data[j - 1];
584  for(j -= 2; j >= 1; j--) {
585  double tmp0 = d0;
586  d0 = x2*d0 - dd0 + bk0_data[j];
587  dd0 = tmp0;
588  }
589  return ret + (0.5*(x2*d0 + bk0_data[0]) - dd0);
590  }
591 }
