32  int m,p,i, tmp;
33  int mol_size, ind;

35 
36  #ifdef LEES_EDWARDS

39  errexit();
40  }
41  #endif
42 
43  /* check molecule information */
44  if ( n_molecules < 0 ) return ES_ERROR;
45 
46  if (!sortPartCfg()) {
47  ostringstream msg;
48  msg <<"analyze_fold_molecules: could not sort particle config, particle ids not consecutive?";
49  runtimeError(msg);
50  return ES_ERROR;
51  }
52 
53  /* loop molecules */
54  for(m=0; m<n_molecules; m++) {
55  mol_size = topology[m].part.n;
56  if(mol_size > 0) {
57  /* calc center of mass */
58  calc_mol_center_of_mass(topology[m],com);
59  /* fold coordinates */
60  for(i=0; i<3; i++) {
61  if ( PERIODIC(i) ) {
62  tmp = (int)floor((com[i]+shift[i])*box_l_i[i]);
63  cm_tmp =0.0;
64  for(p=0; p<mol_size; p++) {
65  ind = 3*topology[m].part.e[p] + i;
66  coord[ind] -= tmp*box_l[i];
67  coord[ind] += shift[i];
68  cm_tmp += coord[ind];
69  }
70  cm_tmp /= (double)mol_size;
71  if(cm_tmp < -10e-6 || cm_tmp > box_l[i]+10e-6) {
72  ostringstream msg;
73  msg <<"analyze_fold_molecules: chain center of mass is out of range (coord " << i << ": " << cm_tmp << " not in box_l " << box_l[i] << ")";
74  runtimeError(msg);
75  return ES_ERROR;
76  }
77  }
78  }
79  }
80  }
81  return ES_OK;
82 }

105 void calc_mol_center_of_mass(Molecule mol, double com[3])
106 {
107  int i,j,id;
108  double M = 0.0;
109  for(j=0; j<3; j++) com[j]=0.0;
110

112  id = mol.part.e[i];

116  for(j=0; j<3; j++) com[j] /= M;
