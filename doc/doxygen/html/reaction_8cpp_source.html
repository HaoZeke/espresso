121  }
122

124  if ( check_catalyzer != 0 ) {
125 
126  /* Loop cell neighbors */
127  for (n = 0; n < dd.cell_inter[c].n_neighbors; n++) {
128  pairs = dd.cell_inter[c].nList[n].vList.pair;
129  np = dd.cell_inter[c].nList[n].vList.n;
130 
131  /* Verlet list loop */

135

138  dist2 = sqrlen(vec21);

143

145  p1->p.catalyzer_count++;

150  }
151  }
152  }
153  }
154  }
155  }
