117 void calc_rg(double **_rg)
118 {
119  int i, j, p;
120  double dx,dy,dz, r_CM_x,r_CM_y,r_CM_z;
121  double r_G=0.0,r_G2=0.0,r_G4=0.0;
122  double *rg=NULL,IdoubMPC, tmp;
123  double M;
124  *_rg = rg = (double*)realloc(rg,4*sizeof(double));
125 
126 
127  for (i=0; i<chain_n_chains; i++) {
128  M=0.0;
129  r_CM_x = r_CM_y = r_CM_z = 0.0;
130  IdoubMPC = 1./(double)chain_length;
131  for (j=0; j<chain_length; j++) {
132  p = chain_start+i*chain_length + j;
133  r_CM_x += partCfg[p].r.p[0]*PMASS(partCfg[p]);
134  r_CM_y += partCfg[p].r.p[1]*PMASS(partCfg[p]);
135  r_CM_z += partCfg[p].r.p[2]*PMASS(partCfg[p]);
136  M += PMASS(partCfg[p]);
137  }
138  r_CM_x /= M;
139  r_CM_y /= M;
140  r_CM_z /= M;
141  tmp = 0.0;
142  for (j=0; j<chain_length; ++j) {
143  p = chain_start+i*chain_length + j;
144  dx = partCfg[p].r.p[0] - r_CM_x;
145  dy = partCfg[p].r.p[1] - r_CM_y;
146  dz = partCfg[p].r.p[2] - r_CM_z;
147  tmp += (SQR(dx) + SQR(dy) + SQR(dz));
148  }
149  tmp *= IdoubMPC;
150  r_G += sqrt(tmp);
151  r_G2 += tmp;
152  r_G4 += tmp*tmp;
153  }
154  tmp = (double)chain_n_chains;
155  rg[0] = r_G/tmp;
156  rg[2] = r_G2/tmp;
157  rg[1] = sqrt(rg[2] - rg[0]*rg[0]);
158  rg[3] = sqrt(r_G4/tmp - rg[2]*rg[2]);

205 {
206  int i, j, p;
207  double dx,dy,dz, r_H=0.0,r_H2=0.0, *rh=NULL, ri=0.0, prefac, tmp;
208  *_rh = rh = (double*)realloc(rh,2*sizeof(double));
209 
210  prefac = 0.5*chain_length*chain_length; /* 1/N^2 is not a normalization factor */
211  for(p=0;p<chain_n_chains;p++) {
212  ri=0.0;
213  for(i=chain_start+chain_length*p;i<chain_start+chain_length*(p+1);i++) {
214  for(j=i+1;j<chain_start+chain_length*(p+1);j++) {
215  dx = partCfg[i].r.p[0]-partCfg[j].r.p[0]; dx*=dx;
216  dy = partCfg[i].r.p[1]-partCfg[j].r.p[1]; dy*=dy;
217  dz = partCfg[i].r.p[2]-partCfg[j].r.p[2]; dz*=dz;
218  ri += 1.0/sqrt(dx+dy+dz);
219  }
220  }
221  tmp = prefac/ri;
222  r_H += tmp;
223  r_H2 += tmp*tmp;
224  }
225  tmp = (double)chain_n_chains;
226  rh[0] = r_H/tmp;
227  rh[1] = sqrt(r_H2/tmp - rh[0]*rh[0]);

439 void calc_g123(double *_g1, double *_g2, double *_g3)
440 {
441  /* - Mean square displacement of a monomer
442  - Mean square displacement in the center of gravity of the chain itself
443  - Motion of the center of mass */
444  int i, j, p;
445  double g1=0.0, g2=0.0, g3=0.0, cm_tmp[3];
446  double M;
447 
448  for(j=0; j<chain_n_chains; j++) {
449  cm_tmp[0] = cm_tmp[1] = cm_tmp[2] = 0.0;
450  M=0.0;
451  for(i=0; i<chain_length; i++) {
452  p = chain_start+j*chain_length + i;
453  cm_tmp[0]+=partCfg[p].r.p[0]*PMASS(partCfg[p]);
454  cm_tmp[1]+=partCfg[p].r.p[1]*PMASS(partCfg[p]);
455  cm_tmp[2]+=partCfg[p].r.p[2]*PMASS(partCfg[p]);
456  M += PMASS(partCfg[p]);
457  }
458  cm_tmp[0] /= M;
459  cm_tmp[1] /= M;
460  cm_tmp[2] /= M;
461  for(i=0; i<chain_length; i++) {
462  p = chain_start+j*chain_length + i;
463  g1 += SQR(partCfg[p].r.p[0]-partCoord_g[3*p])
464  + SQR(partCfg[p].r.p[1]-partCoord_g[3*p+1])
465  + SQR(partCfg[p].r.p[2]-partCoord_g[3*p+2]);
466  g2 += SQR( (partCfg[p].r.p[0]-partCoord_g[3*p])
467  - (cm_tmp[0]-partCM_g[3*j] ) )
468  + SQR( (partCfg[p].r.p[1]-partCoord_g[3*p+1])
469  - (cm_tmp[1]-partCM_g[3*j+1]) )
470  + SQR( (partCfg[p].r.p[2]-partCoord_g[3*p+2])
471  - (cm_tmp[2]-partCM_g[3*j+2]) );
472  }
473  g3 += SQR(cm_tmp[0]-partCM_g[3*j])
474  + SQR(cm_tmp[1]-partCM_g[3*j+1])
475  + SQR(cm_tmp[2]-partCM_g[3*j+2]);
476  }
477  *_g1 = g1 / (1.*chain_n_chains*chain_length);
478  *_g2 = g2 / (1.*chain_n_chains*chain_length);
479  *_g3 = g3 / (1.*chain_n_chains);

565 void analyze_formfactor(double qmin, double qmax, int qbins, double **_ff) {
566  int i,j,k,qi, cnt,cnt_max;
567  double q,qfak, qr, dx,dy,dz, *r_ij=NULL, *ff=NULL;
568  *_ff = ff = (double*)realloc(ff,(qbins+1)*sizeof(double));
569  r_ij = (double*)realloc(r_ij,chain_length*(chain_length-1)/2*sizeof(double));
570 
571  qfak = pow((qmax/qmin),(1.0/qbins));
572  for(qi=0; qi<=qbins; qi++) ff[qi] = 0.0;
573  for(k=0; k < chain_n_chains; k++) {
574  cnt = 0;
575  /* Prepare distance matrice r_ij for current chain k */
576  for(i=chain_start+k*chain_length; i<chain_start+(k+1)*chain_length; i++) {
577  for(j=i+1; j<chain_start+(k+1)*chain_length;j++) {
578  dx = partCfg[i].r.p[0]-partCfg[j].r.p[0]; dx*=dx;
579  dy = partCfg[i].r.p[1]-partCfg[j].r.p[1]; dy*=dy;
580  dz = partCfg[i].r.p[2]-partCfg[j].r.p[2]; dz*=dz;
581  r_ij[cnt++] = sqrt(dx + dy + dz);
582  }
583  }
584  q = qmin; cnt_max = cnt;
585  /* Derive spherically averaged S(q) = 1/chain_length * Sum(i,j=1..chain_length)[sin(q*r_ij)/q*r_ij] for current chain k */
586  for(qi=0; qi<=qbins; qi++) {
587  ff[qi] += chain_length;
588  for(cnt=0; cnt<cnt_max; cnt++) {
589  qr = q*r_ij[cnt];
590  ff[qi] += 2*sin(qr)/qr;
591  }
592  q *= qfak;
593  }
594  }
595  for(qi=0; qi<=qbins; qi++) ff[qi] /= ((double)chain_length*chain_n_chains);
596  free(r_ij);
597 }

637 void analyze_rdfchain(double r_min, double r_max, int r_bins, double **_f1, double **_f2, double **_f3) {
638  int i, j, ind, c_i, c_j, mon;
639  double bin_width, inv_bin_width, factor, r_in, r_out, bin_volume, dist, chain_mass,
640  *cm=NULL, *min_d=NULL, *f1=NULL, *f2=NULL, *f3=NULL;
641 
642  *_f1 = f1 = (double*)realloc(f1,r_bins*sizeof(double));
643  *_f2 = f2 = (double*)realloc(f2,r_bins*sizeof(double));
644  *_f3 = f3 = (double*)realloc(f3,r_bins*sizeof(double));
645  cm = (double*)realloc(cm,(chain_n_chains*3)*sizeof(double));
646  min_d = (double*)realloc(min_d,(chain_n_chains*chain_n_chains)*sizeof(double));
647  for(i=0; i<r_bins; i++) {
648  f1[i] = f2[i] = f3[i] = 0.0;
649  }
650  bin_width = (r_max-r_min) / (double)r_bins;
651  inv_bin_width = 1.0 / bin_width;
652  for(c_i=0; c_i<chain_n_chains; c_i++) {
653  cm[3*c_i] = cm[3*c_i+1] = cm[3*c_i+2] = 0.0;
654  for(c_j=(c_i+1); c_j<chain_n_chains; c_j++) {
655  min_d[c_i*chain_n_chains+c_j] = box_l[0] + box_l[1] + box_l[2];
656  }
657  }
658  for(c_i=0; c_i<chain_n_chains; c_i++) {
659  for(i=0; i<chain_length; i++) {
660  mon = chain_start + c_i*chain_length + i;
661  cm[3*c_i]+= partCfg[mon].r.p[0]*PMASS(partCfg[mon]);
662  cm[3*c_i+1]+= partCfg[mon].r.p[1]*PMASS(partCfg[mon]);
663  cm[3*c_i+2]+= partCfg[mon].r.p[2]*PMASS(partCfg[mon]);
664  for(c_j=(c_i+1); c_j<chain_n_chains; c_j++) {
665  for(j=0; j<chain_length; j++) {
666  dist = min_distance(partCfg[mon].r.p, partCfg[chain_start + c_j*chain_length+j].r.p);
667  if ((dist > r_min) && (dist < r_max)) {
668  ind = (int) ( (dist - r_min)*inv_bin_width );
669  f1[ind]+= 1.0;
670  }
671  if (dist<min_d[c_i*chain_n_chains+c_j]) min_d[c_i*chain_n_chains+c_j] = dist;
672  }
673  }
674  }
675  }
676  chain_mass = 0;
677  for(i=0; i<chain_length; i++) chain_mass+= PMASS(partCfg[i]);
678  for(c_i=0; c_i<chain_n_chains; c_i++) {
679  cm[3*c_i]/= chain_mass;
680  cm[3*c_i+1]/= chain_mass;
681  cm[3*c_i+2]/= chain_mass;
682  for(c_j=(c_i+1); c_j<chain_n_chains; c_j++) {
683  dist = min_d[c_i*chain_n_chains+c_j];
684  if ((dist > r_min) && (dist < r_max)) {
685  ind = (int) ( (dist - r_min)*inv_bin_width );
686  f3[ind]+= 1.0;
687  }
688  }
689  }
690  for(c_i=0; c_i<chain_n_chains; c_i++) {
691  for(c_j=(c_i+1); c_j<chain_n_chains; c_j++) {
692  dist = min_distance(&cm[3*c_i],&cm[3*c_j]);
693  if ((dist > r_min) && (dist < r_max)) {
694  ind = (int) ( (dist - r_min)*inv_bin_width );
695  f2[ind]+= 1.0;
696  }
697  }
698  }
699  /* normalization */
700  factor = box_l[0]*box_l[1]*box_l[2] *1.5/PI/(chain_n_chains*(chain_n_chains-1));
701  for(i=0; i<r_bins; i++) {
702  r_in = i*bin_width + r_min;
703  r_out = r_in + bin_width;
704  bin_volume = r_out*r_out*r_out - r_in*r_in*r_in;
705  f1[i] *= factor / (bin_volume * chain_length*chain_length);
706  f2[i] *= factor / bin_volume;
707  f3[i] *= factor / bin_volume;
708  }
709 }
