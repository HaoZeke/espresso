40  force to the particle forces (see \ref interaction_data.cpp).
41  @param p1 Pointer to first particle.
42  @param p2 Pointer to second/middle particle.
43  @param iaparams bond type number of the angle interaction (see \ref interaction_data.cpp).
44  @param dx particle distance vector
45  @param force returns force of particle 1
46  @return 0.
47 */
48 inline int calc_harmonic_pair_force(Particle *p1, Particle *p2, Bonded_ia_parameters *iaparams, double dx[3], double force[3])
49 {
50  int i;
51  double fac;
52  double dist2 = sqrlen(dx);
53  double dist = sqrt(dist2);
54  double dr;
55 
56  if ((iaparams->p.harmonic.r_cut > 0.0) &&
57  (dist > iaparams->p.harmonic.r_cut))
58  return 1;
59 
60  dr = dist - iaparams->p.harmonic.r;
61  fac = -iaparams->p.harmonic.k * dr;
62  if (fabs(dr) > ROUND_ERROR_PREC) {
63  if(dist>ROUND_ERROR_PREC) { /* Regular case */
64  fac /= dist;
65  } else { /* dx[] == 0: the force is undefined. Let's use a random direction */
66  for(i=0;i<3;i++) dx[i] = d_random()-0.5;
67  fac /= sqrt(sqrlen(dx));
68  }
69  } else {
70  fac=0;
71  }
72 
73  for(i=0;i<3;i++)
74  force[i] = fac*dx[i];
75  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: HARMONIC f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist2,fac));
76  ONEPART_TRACE(if(p2->p.identity==check_id) fprintf(stderr,"%d: OPT: HARMONIC f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p2->f.f[0],p2->f.f[1],p2->f.f[2],p1->p.identity,dist2,fac));
77 
78 #ifdef CONFIGTEMP
79  extern double configtemp[2];
80  int numfac = 0;
81  if (p1->p.configtemp) numfac+=1;
82  if (p2->p.configtemp) numfac+=1;
83  configtemp[0] += numfac*SQR(iaparams->p.harmonic.k * dr);
84  configtemp[1] -= numfac*iaparams->p.harmonic.k*(3-2.*iaparams->p.harmonic.r/dist);
85 #endif
86 
87  return 0;
88 }
