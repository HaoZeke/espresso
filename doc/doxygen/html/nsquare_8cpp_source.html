56 static void nsq_prepare_comm(GhostCommunicator *comm, int data_parts)
57 {
58  int n;
59  /* no need for comm for only 1 node */
60  if (n_nodes == 1) {
61  prepare_comm(comm, data_parts, 0);
62  return;
63  }
64 
65  prepare_comm(comm, data_parts, n_nodes);
66  /* every node has its dedicated comm step */
67  for(n = 0; n < n_nodes; n++) {

70  comm->comm[n].n_part_lists = 1;
71  comm->comm[n].node = n;
72  comm->comm[n].mpi_comm = comm_cart;
73  }

76 void nsq_topology_init(CellPList *old)
77 {
78  Particle *part;
79  int n, c, p, np, ntodo, diff;
80 
81  CELL_TRACE(fprintf(stderr, "%d: nsq_topology_init, %d\n", this_node, old->n));
82 
83  cell_structure.type = CELL_STRUCTURE_NSQUARE;
84  cell_structure.position_to_node = map_position_node_array;
85  cell_structure.position_to_cell = nsq_position_to_cell;
86 
87  realloc_cells(n_nodes);
88 
89  /* mark cells */
90  local = &cells[this_node];
91  realloc_cellplist(&local_cells, local_cells.n = 1);
92  local_cells.cell[0] = local;
93 
94  realloc_cellplist(&ghost_cells, ghost_cells.n = n_nodes - 1);
95  c = 0;
96  for (n = 0; n < n_nodes; n++)
97  if (n != this_node)
98  ghost_cells.cell[c++] = &cells[n];
99 
100  /* distribute force calculation work */
101  ntodo = (n_nodes + 3)/2;
102  init_cellplist(&me_do_ghosts);
103  realloc_cellplist(&me_do_ghosts, ntodo);
104  for (n = 0; n < n_nodes; n++) {
105  diff = n - this_node;
106  /* simple load balancing formula. Basically diff % n, where n >= n_nodes, n odd.
107  The node itself is also left out, as it is treated differently */
108  if (((diff > 0 && (diff % 2) == 0) ||
109  (diff < 0 && ((-diff) % 2) == 1))) {
110  CELL_TRACE(fprintf(stderr, "%d: doing interactions with %d\n", this_node, n));
111  me_do_ghosts.cell[me_do_ghosts.n++] = &cells[n];
112  }
113  }
114 
115  /* create communicators */
116  nsq_prepare_comm(&cell_structure.ghost_cells_comm, GHOSTTRANS_PARTNUM);
117  nsq_prepare_comm(&cell_structure.exchange_ghosts_comm, GHOSTTRANS_PROPRTS | GHOSTTRANS_POSITION);
118  nsq_prepare_comm(&cell_structure.update_ghost_pos_comm, GHOSTTRANS_POSITION);
119  nsq_prepare_comm(&cell_structure.collect_ghost_force_comm, GHOSTTRANS_FORCE);
120 
121  /* here we just decide what to transfer where */
122  if (n_nodes > 1) {
123  for (n = 0; n < n_nodes; n++) {
124  /* use the prefetched send buffers. Node 0 transmits first and never prefetches. */
125  if (this_node == 0 || this_node != n) {
126  cell_structure.ghost_cells_comm.comm[n].type = GHOST_BCST;
127  cell_structure.exchange_ghosts_comm.comm[n].type = GHOST_BCST;
128  cell_structure.update_ghost_pos_comm.comm[n].type = GHOST_BCST;
129  }
130  else {
131  cell_structure.ghost_cells_comm.comm[n].type = GHOST_BCST | GHOST_PREFETCH;
132  cell_structure.exchange_ghosts_comm.comm[n].type = GHOST_BCST | GHOST_PREFETCH;
133  cell_structure.update_ghost_pos_comm.comm[n].type = GHOST_BCST | GHOST_PREFETCH;
134  }
135  cell_structure.collect_ghost_force_comm.comm[n].type = GHOST_RDCE;
136  }
137  /* first round: all nodes except the first one prefetch their send data */
138  if (this_node != 0) {
139  cell_structure.ghost_cells_comm.comm[0].type |= GHOST_PREFETCH;
140  cell_structure.exchange_ghosts_comm.comm[0].type |= GHOST_PREFETCH;
141  cell_structure.update_ghost_pos_comm.comm[0].type |= GHOST_PREFETCH;
142  }
143  }
144 
145  /* copy particles */
146  for (c = 0; c < old->n; c++) {
147  part = old->cell[c]->part;
148  np = old->cell[c]->n;
149  for (p = 0; p < np; p++)
150  append_unindexed_particle(local, &part[p]);
151  }
152  update_local_particles(local);
153 }

155 void nsq_balance_particles(int global_flag)
156 {
157  int i, n, surplus, s_node, tmp, lack, l_node, transfer;
158 
159  /* we don't have the concept of neighbors, and therefore don't need that.
160  However, if global particle changes happen, we might want to rebalance. */
161  if (global_flag != CELL_GLOBAL_EXCHANGE)
162  return;
163 
164  int pp = cells_get_n_particles();
165  int *ppnode = (int*)malloc(n_nodes*sizeof(int));
166  /* minimal difference between node shares */
167  int minshare = n_part/n_nodes;
168  int maxshare = minshare + 1;
169 
170  CELL_TRACE(fprintf(stderr, "%d: nsq_balance_particles: load %d-%d\n", this_node, minshare, maxshare));
171 
172  MPI_Allgather(&pp, 1, MPI_INT, ppnode, 1, MPI_INT, comm_cart);
173  for (;;) {
174  /* find node with most excessive particles */
175  surplus = -1;
176  s_node = -1;
177  for (n = 0; n < n_nodes; n++) {
178  tmp = ppnode[n] - minshare;
179  CELL_TRACE(fprintf(stderr, "%d: nsq_balance_particles: node %d has %d\n", this_node, n, ppnode[n]));
180  if (tmp > surplus) {
181  surplus = tmp;
182  s_node = n;
183  }
184  }
185  CELL_TRACE(fprintf(stderr, "%d: nsq_balance_particles: excess %d on node %d\n", this_node, surplus, s_node));
186 
187  /* find node with most lacking particles */
188  lack = -1;
189  l_node = -1;
190  for (n = 0; n < n_nodes; n++) {
191  tmp = maxshare - ppnode[n];
192  if (tmp > lack) {
193  lack = tmp;
194  l_node = n;
195  }
196  }
197  CELL_TRACE(fprintf(stderr, "%d: nsq_balance_particles: lack %d on node %d\n", this_node, lack, l_node));
198 
199  /* should not happen: minshare or maxshare wrong or more likely,
200  the algorithm */
201  if (s_node == -1 || l_node == -1) {
202  fprintf(stderr, "%d: Particle load balancing failed\n", this_node);
203  break;
204  }
205 
206  /* exit if all nodes load is withing min and max share */
207  if (lack <= 1 && surplus <= 1)
208  break;
209 
210  transfer = lack < surplus ? lack : surplus;
211 
212  if (s_node == this_node) {
213  ParticleList send_buf;
214  init_particlelist(&send_buf);
215  realloc_particlelist(&send_buf, send_buf.n = transfer);
216  for (i = 0; i < transfer; i++) {
217  memcpy(&send_buf.part[i], &local->part[--local->n], sizeof(Particle));
218  }
219  realloc_particlelist(local, local->n);
220  update_local_particles(local);
221 
222  send_particles(&send_buf, l_node);
223 #ifdef ADDITIONAL_CHECKS
224  check_particle_consistency();
225 #endif
226  }
227  else if (l_node == this_node) {
228  recv_particles(local, s_node);
229 #ifdef ADDITIONAL_CHECKS
230  check_particle_consistency();
231 #endif
232  }
233  ppnode[s_node] -= transfer;
234  ppnode[l_node] += transfer;
235  }
236  CELL_TRACE(fprintf(stderr, "%d: nsq_balance_particles: done\n", this_node));
237 
238  free(ppnode);
239 }

244  Particle *partl, *partg;

248 
249  npl = local->n;
250  partl = local->part;
251 
252  /* calculate bonded interactions and non bonded node-node */
253  for (p = 0; p < npl; p++) {
254  pt1 = &partl[p];
255  add_bonded_force(pt1);
256 #ifdef CONSTRAINTS
257  add_constraints_forces(pt1);
258 #endif
259  add_external_potential_forces(pt1);
260 
261  if (rebuild_verletlist)
262  memcpy(pt1->l.p_old, pt1->r.p, 3*sizeof(double));
263 
264  /* other particles, same node */
265  for (p2 = p + 1; p2 < npl; p2++) {
266  pt2 = &partl[p2];
267  get_mi_vector(d, pt1->r.p, pt2->r.p);
268  dist2 = sqrlen(d);
269  dist = sqrt(dist2);
270 #ifdef EXCLUSIONS
271  if (do_nonbonded(pt1, pt2))
272 #endif
273  add_non_bonded_pair_force(pt1, pt2, d, dist, dist2);
274  }
