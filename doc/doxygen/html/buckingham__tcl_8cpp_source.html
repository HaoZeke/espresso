60 int tclcommand_inter_parse_buckingham(Tcl_Interp * interp,
61  int part_type_a, int part_type_b,
62  int argc, char ** argv)
63 {
64  /* parameters needed for buckingham */
65  double buck_A,buck_B,buck_C,buck_D,buck_cut,buck_discont,buck_shift,buck_cap_radius,F1,F2;
66  int change;
67 
68 
69  /* get buckingham interaction type */
70  if (argc < 8) {
71  Tcl_AppendResult(interp, "buckingham needs 7 parameters: "
72  "<buck_A> <buck_B> <buck_C> <buck_D> <buck_cut> <buck_discontinuity> <buck_shift> ",
73  (char *) NULL);
74  return 0;
75  }
76  /* copy buckingham parameters */
77  if ((! ARG_IS_D(1, buck_A)) ||
78  (! ARG_IS_D(2, buck_B)) ||
79  (! ARG_IS_D(3, buck_C)) ||
80  (! ARG_IS_D(4, buck_D)) ||
81  (! ARG_IS_D(5, buck_cut)) ||
82  (! ARG_IS_D(6, buck_discont)) ||
83  (! ARG_IS_D(7, buck_shift))) {
84  Tcl_AppendResult(interp, "buckingham needs 7 DOUBLE parameters: "
85  "<buck_A> <buck_B> <buck_C> <buck_D> <buck_cut> <buck_discontinuity> <buck_shift> ",
86  (char *) NULL);
87  return 0;
88  }
89  change = 8;
90  buck_cap_radius = -1.0;
91  F1 = 0.0;
92  F2 = 0.0;
93  /* check whether there are additional doubles, cap radius, F1 and F2*/
94  if (argc >= 9 && ARG_IS_D(8, buck_cap_radius))
95  {
96  change++;
97  if(argc >= 10 && ARG_IS_D(9, F1))
98  {
99  change++;
100  if(argc >= 11 && ARG_IS_D(10, F1))
101  change++;
102  }
103  }
104  else
105  Tcl_ResetResult(interp);
106  if(buck_discont>buck_cut)
107  {
108  Tcl_AppendResult(interp, "ERROR: <buck_cut> must be greater than <buck_discontinuity>",
109  (char *) NULL);
110  return 0;
111  }
112  if (buckingham_set_params(part_type_a, part_type_b,
113  buck_A, buck_B, buck_C, buck_D,
114  buck_cut, buck_discont, buck_shift,
115  buck_cap_radius, F1, F2) == ES_ERROR) {
116  Tcl_AppendResult(interp, "particle types must be non-negative", (char *) NULL);
117  return 0;
118  }
119  return change;
120  }
