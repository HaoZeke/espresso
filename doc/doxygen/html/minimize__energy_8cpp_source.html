48 bool steepest_descent_step(void) {
49  Cell *cell;
50  Particle *p;
51  int c, i, j, np;
52  double f_max = -std::numeric_limits<double>::max();
53  double f;
54  /* Verlet list criterion */
55  const double skin2 = SQR(0.5*skin);
56  double f_max_global;
57  double dx[3], dx2;
58  const double max_dx2 = SQR(params->max_displacement);
59 
60  for (c = 0; c < local_cells.n; c++) {
61  cell = local_cells.cell[c];
62  p = cell->part;
63  np = cell->n;
64  for(i = 0; i < np; i++) {
65  f = 0.0;
66  dx2 = 0.0;
67 #ifdef VIRTUAL_SITES
68  if (ifParticleIsVirtual(&p[i])) continue;
69 #endif
70  for(j=0; j < 3; j++){
71 #ifdef EXTERNAL_FORCES
72  if (!(p[i].p.ext_flag & COORD_FIXED(j)))
73 #endif
74  {
75  f += SQR(p[i].f.f[j]);
76  dx[j] = params->gamma * p[i].f.f[j];
77  dx2 += SQR(dx[j]);
78  MINIMIZE_ENERGY_TRACE(printf("part %d dim %d dx %e gamma*f %e\n", i, j, dx[j], params->gamma * p[i].f.f[j]));
79  }
80 #ifdef EXTERNAL_FORCES
81  else {
82  dx[j] = 0.0;
83  }
84 #endif
85  }
86 
87  if(dx2 <= max_dx2) {
88  p[i].r.p[0] += dx[0];
89  p[i].r.p[1] += dx[1];
90  p[i].r.p[2] += dx[2];
91  } else {
92  const double c = params->max_displacement/std::sqrt(dx2);
93  p[i].r.p[0] += c*dx[0];
94  p[i].r.p[1] += c*dx[1];
95  p[i].r.p[2] += c*dx[2];
96  }
97  if(distance2(p[i].r.p,p[i].l.p_old) > skin2 ) resort_particles = 1;
98  f_max = std::max(f_max, f);
99  }
100  }
101  MINIMIZE_ENERGY_TRACE(printf("f_max %e resort_particles %d\n", f_max, resort_particles));
102  announce_resort_particles();
103  MPI_Allreduce(&f_max, &f_max_global, 1, MPI_DOUBLE, MPI_MAX, comm_cart);
104  return (sqrt(f_max_global) < params->f_max);
105 }
