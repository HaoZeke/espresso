46 inline void add_morse_pair_force(Particle *p1, Particle *p2, IA_parameters *ia_params,
47  double d[3], double dist, double force[3])
48 {
49  double add1, add2, fac=0.0;
50  int j;
51  if(CUTOFF_CHECK(dist < ia_params->MORSE_cut)) {
52  /* normal case: resulting force/energy smaller than capping. */
53 
54  if(dist > ia_params->MORSE_capradius ) {
55  add1 = exp(-2.0 * ia_params->MORSE_alpha * (dist - ia_params->MORSE_rmin));
56  add2 = exp( -ia_params->MORSE_alpha * (dist - ia_params->MORSE_rmin));
57  fac = -ia_params->MORSE_eps * 2.0 * ia_params->MORSE_alpha * (add2 - add1) / dist;
58 
59  for(j=0;j<3;j++)
60  force[j] += fac * d[j];
61 #ifdef MORSE_WARN_WHEN_CLOSE
62  if(fac*dist > 1000) fprintf(stderr,"%d: Morse-Warning: Pair (%d-%d) force=%f dist=%f\n",
63  this_node,p1->p.identity,p2->p.identity,fac*dist,dist);
64 #endif
65 
66  }
67  /* capped part of morse potential. */
68 
69  else if(dist > 0.0) {
70 
71  add1 = exp(-2.0 * ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
72  add2 = exp( -ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
73  fac = -ia_params->MORSE_eps * 2.0 * ia_params->MORSE_alpha * (add2 - add1) / ia_params->MORSE_capradius;
74 
75  /* vector d is rescaled to length MORSE_capradius */
76  for(j=0;j<3;j++)
77  force[j] += fac * d[j];
78  }
79  /* this should not happen! */
80 
81  else {
82  MORSE_TRACE(fprintf(stderr, "%d: Morse warning: Particles id1=%d id2=%d exactly on top of each other\n",this_node,p1->p.identity,p2->p.identity));
83 
84  add1 = exp(-2.0 * ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
85  add2 = exp( -ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
86  fac = -ia_params->MORSE_eps * 2.0 * ia_params->MORSE_alpha * (add2 - add1) / ia_params->MORSE_capradius;
87 
88  force[0] += fac * ia_params->MORSE_capradius;
89  }
90 
91  ONEPART_TRACE(if(p1->p.identity==check_id) fprintf(stderr,"%d: OPT: MORSE f = (%.3e,%.3e,%.3e) with part id=%d at dist %f fac %.3e\n",this_node,p1->f.f[0],p1->f.f[1],p1->f.f[2],p2->p.identity,dist,fac));

101 inline double morse_pair_energy(Particle *p1, Particle *p2, IA_parameters *ia_params,
102  double d[3], double dist)
103 {
104 
105  double add1, add2, fac;
106 
107  if(CUTOFF_CHECK(dist < ia_params->MORSE_cut)) {
108 
109  /* normal case: resulting force/energy smaller than capping. */
110  if(dist > ia_params->MORSE_capradius) {
111  add1 = exp(-2.0 * ia_params->MORSE_alpha * (dist - ia_params->MORSE_rmin));
112  add2 = 2.0 * exp( -ia_params->MORSE_alpha * (dist - ia_params->MORSE_rmin));
113  fac = ia_params->MORSE_eps * (add1 - add2) - ia_params->MORSE_rest;
114  return fac;
115  }
116 
117  /* capped part of morse potential. */
118  else if(dist > 0.0) {
119  add1 = exp(-2.0 * ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
120  add2 = 2.0 * exp( -ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
121  fac = ia_params->MORSE_eps * (add1 - add2) - ia_params->MORSE_rest;
122  return fac;
123  }
124  /* this should not happen! */
125 
126  else {
127  add1 = exp(-2.0 * ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
128  add2 = 2.0 * exp( -ia_params->MORSE_alpha * (ia_params->MORSE_capradius - ia_params->MORSE_rmin));
129  fac = ia_params->MORSE_eps * (add1 - add2) - ia_params->MORSE_rest;
130  return fac;
131  }
132 
133  }
134
