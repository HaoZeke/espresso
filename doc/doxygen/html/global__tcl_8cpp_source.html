66 void register_global_callback(int field, SetCallback *callback)
67 {
68  /* resize, if necessary, and initialize fields inbetween as read-only */
69  if (n_callbacks <= field) {
70  callbacks = (SetCallback **)realloc(callbacks, (field + 1)*sizeof(SetCallback *));
71  for (int f = n_callbacks; f < field; ++f)
72  callbacks[f] = tclcallback_ro;
73  n_callbacks = field + 1;
74  }
75  callbacks[field] = callback;
