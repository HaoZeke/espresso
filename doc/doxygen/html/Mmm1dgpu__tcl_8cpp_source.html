55 int tclcommand_inter_coulomb_parse_mmm1dgpu (Tcl_Interp * interp, int argc, char **argv)
56 {
57  double switch_rad, maxPWerror;
58  int bessel_cutoff;
59 
60  if (argc < 2)
61  {
62  Tcl_AppendResult (interp, "wrong # arguments: inter coulomb mmm1dgpu <switch radius> "
63  "{<bessel cutoff>} <maximal error for near formula> | tune <maximal pairwise error>", (char *) NULL);
64  return TCL_ERROR;
65  }
66 
67  if (ARG0_IS_S ("tune"))
68  {
69  /* autodetermine bessel cutoff AND switching radius */
70  if (!ARG_IS_D (1, maxPWerror))
71  return TCL_ERROR;
72  bessel_cutoff = -1;
73  switch_rad = -1;
74  }
75  else
76  {
77  if (argc == 2)
78  {
79  /* autodetermine bessel cutoff */
80  if ((!ARG_IS_D (0, switch_rad)) || (!ARG_IS_D (1, maxPWerror)))
81  return TCL_ERROR;
82  bessel_cutoff = -1;
83  }
84  else if (argc == 3)
85  {
86  /* fully manual */
87  if ((!ARG_IS_D (0, switch_rad)) || (!ARG_IS_I (1, bessel_cutoff)) || (!ARG_IS_D (2, maxPWerror)))
88  return TCL_ERROR;
89 
90  if (bessel_cutoff <= 0)
91  {
92  Tcl_AppendResult (interp, "bessel cutoff too small", (char *) NULL);
93  return TCL_ERROR;
94  }
95  }
96  else
97  {
98  Tcl_AppendResult (interp, "wrong # arguments: inter coulomb mmm1dgpu <switch radius> " "{<bessel cutoff>} <maximal error for near formula> | tune <maximal pairwise error>", (char *) NULL);
99  return TCL_ERROR;
100  }
101 
102  if (switch_rad <= 0 || switch_rad > box_l[2])
103  {
104  Tcl_AppendResult (interp, "switching radius is not between 0 and box_l[2]", (char *) NULL);
105  return TCL_ERROR;
106  }
107  }
108 
109  // turn on MMM1DGPU
110 
111  /* coulomb.prefactor apparently is not available yet at this point */
112  if (!mmm1dgpuForce) // inter coulomb mmm1dgpu was never called before
113  {
114  mmm1dgpuForce = new Mmm1dgpuForce(espressoSystemInterface, 0, maxPWerror,
115  switch_rad, bessel_cutoff);
116  forceActors.add(mmm1dgpuForce);
117  energyActors.add(mmm1dgpuForce);
118  }
119  /* set_params needs to be called both upon create and upon update because it is responsible for writing
120  to the struct from which the TCL command "inter coulomb" retrieves the current parameter set */
121  mmm1dgpuForce->set_params(0, 0, maxPWerror, switch_rad, bessel_cutoff, true);
122 
123  coulomb.method = COULOMB_MMM1D_GPU;
124  mpi_bcast_coulomb_params();
125 
126  return TCL_OK;
127 }
